50 double
51 DipolarDirectSumWithReplica::kernel(bool force_flag, bool energy_flag,
52  ParticleRange const &particles) const {
53 
54  assert(n_nodes == 1);
55  assert(force_flag || energy_flag);
56 
57  std::vector<double> x, y, z;
58  std::vector<double> mx, my, mz;
59  std::vector<double> fx, fy, fz;
60  std::vector<double> tx, ty, tz;
61 
62  auto const n_part = particles.size();
63 
64  x.resize(n_part);
65  y.resize(n_part);
66  z.resize(n_part);
67 
68  mx.resize(n_part);
69  my.resize(n_part);
70  mz.resize(n_part);
71 
72  if (force_flag) {
73  fx.resize(n_part);
74  fy.resize(n_part);
75  fz.resize(n_part);
76  tx.resize(n_part);
77  ty.resize(n_part);
78  tz.resize(n_part);
79  }
80 
81  int dip_particles = 0;
82  for (auto const &p : particles) {
83  if (p.dipm() != 0.0) {
84  const Utils::Vector3d dip = p.calc_dip();
85 
86  mx[dip_particles] = dip[0];
87  my[dip_particles] = dip[1];
88  mz[dip_particles] = dip[2];
89 
90  /* here we wish the coordinates to be folded into the primary box */
91  auto const ppos = folded_position(p.pos(), box_geo);
92  x[dip_particles] = ppos[0];
93  y[dip_particles] = ppos[1];
94  z[dip_particles] = ppos[2];
95 
96  if (force_flag) {
97  fx[dip_particles] = 0;
98  fy[dip_particles] = 0;
99  fz[dip_particles] = 0;
100  tx[dip_particles] = 0;
101  ty[dip_particles] = 0;
102  tz[dip_particles] = 0;
103  }
104 
105  dip_particles++;
106  }
107  }
108 
109  /* energy calculation */
110  double energy = 0.;
111 
112  int NCUT[3];
113  for (int i = 0; i < 3; i++) {
114  NCUT[i] = box_geo.periodic(i) ? n_replica : 0;
115  }
116  auto const NCUT2 = Utils::sqr(n_replica);
117 
118  for (int i = 0; i < dip_particles; i++) {
119  for (int j = 0; j < dip_particles; j++) {
120  auto const pe1 = mx[i] * mx[j] + my[i] * my[j] + mz[i] * mz[j];
121  auto const rx = x[i] - x[j];
122  auto const ry = y[i] - y[j];
123  auto const rz = z[i] - z[j];
124 
125  for (int nx = -NCUT[0]; nx <= NCUT[0]; nx++) {
126  auto const rnx = rx + nx * box_geo.length()[0];
127  auto const rnx2 = rnx * rnx;
128  for (int ny = -NCUT[1]; ny <= NCUT[1]; ny++) {
129  auto const rny = ry + ny * box_geo.length()[1];
130  auto const rny2 = rny * rny;
131  for (int nz = -NCUT[2]; nz <= NCUT[2]; nz++) {
132  if (!(i == j && nx == 0 && ny == 0 && nz == 0) and
133  (nx * nx + ny * ny + nz * nz <= NCUT2)) {
134  auto const rnz = rz + nz * box_geo.length()[2];
135  auto const r2 = rnx2 + rny2 + rnz * rnz;
136  auto const r = sqrt(r2);
137  auto const r3 = r2 * r;
138  auto const r5 = r3 * r2;
139  auto const r7 = r5 * r2;
140 
141  auto const pe2 = mx[i] * rnx + my[i] * rny + mz[i] * rnz;
142  auto const pe3 = mx[j] * rnx + my[j] * rny + mz[j] * rnz;
143  auto const pe4 = 3.0 / r5;
144 
145  // Energy
146  energy += pe1 / r3 - pe4 * pe2 * pe3;
147 
148  if (force_flag) {
149  // Forces
150  auto const a = pe4 * pe1;
151  auto const b = -15.0 * pe2 * pe3 / r7;
152  auto const c = pe4 * pe3;
153  auto const d = pe4 * pe2;
154 
155  fx[i] += (a + b) * rnx + c * mx[i] + d * mx[j];
156  fy[i] += (a + b) * rny + c * my[i] + d * my[j];
157  fz[i] += (a + b) * rnz + c * mz[i] + d * mz[j];
158 
159  // Torques
160  auto const ax = my[i] * mz[j] - my[j] * mz[i];
161  auto const ay = mx[j] * mz[i] - mx[i] * mz[j];
162  auto const az = mx[i] * my[j] - mx[j] * my[i];
163 
164  auto const bx = my[i] * rnz - rny * mz[i];
165  auto const by = rnx * mz[i] - mx[i] * rnz;
166  auto const bz = mx[i] * rny - rnx * my[i];
167 
168  tx[i] += -ax / r3 + bx * c;
169  ty[i] += -ay / r3 + by * c;
170  tz[i] += -az / r3 + bz * c;
171  } /* if force_flag */
172  } /* if distance criterion */
173  } /* for nz */
174  } /* for ny */
175  } /* for nx */
176  } /* for j */
177  } /* for i */
178 
179  /* update particle forces and torques */
180  if (force_flag) {
181 
182  dip_particles = 0;
183 
184  for (auto &p : particles) {
185  if (p.dipm() != 0.) {
186  auto &force = p.force();
187  auto &torque = p.torque();
188 
189  force[0] += prefactor * fx[dip_particles];
190  force[1] += prefactor * fy[dip_particles];
191  force[2] += prefactor * fz[dip_particles];
192 
193  torque[0] += prefactor * tx[dip_particles];
194  torque[1] += prefactor * ty[dip_particles];
195  torque[2] += prefactor * tz[dip_particles];
196 
197  dip_particles++;
198  }
199  }
200  } /* if force_flag */
201 
202  return 0.5 * prefactor * energy;
203 }

205 DipolarDirectSumWithReplica::DipolarDirectSumWithReplica(double prefactor,
206  int n_replica)
207  : prefactor{prefactor}, n_replica{n_replica} {
208  if (n_nodes > 1) {
209  throw std::runtime_error(
210  "MPI parallelization not supported by DipolarDirectSumWithReplicaCpu.");
211  }
212  if (prefactor <= 0.) {
213  throw std::domain_error("Parameter 'prefactor' must be > 0");
214  }
215  if (n_replica < 0) {
216  throw std::domain_error("Parameter 'n_replica' must be >= 0");
217  }
218  sanity_checks();
219  if (n_replica == 0) {
220  fprintf(stderr, "Careful: the number of extra replicas to take into "
221  "account during the direct sum calculation is zero\n");
222  }
223 }
