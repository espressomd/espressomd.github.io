42 /**
43  * Calculate dipolar energy and optionally force between two particles.
44  * @param[in,out] p1 First particle
45  * @param[in] dip1 Cached dipole moment of the first particle
46  * @param[in,out] p2 Second particle
47  * @param[in] force_flag If true, update the particle forces and torques
48  */
49 double DipolarDirectSum::calc_dipole_dipole_ia(Particle &p1,
50  Utils::Vector3d const &dip1,
51  Particle &p2,
52  bool force_flag) const {
53 
54  // Cache dipole moment
55  auto const dip2 = p2.calc_dip();
56 
57  // Distance between particles
58  auto const dr = box_geo.get_mi_vector(p1.pos(), p2.pos());
59 
60  // Powers of distance
61  auto const r2 = dr.norm2();
62  auto const r = sqrt(r2);
63  auto const r3 = r2 * r;
64  auto const r5 = r3 * r2;
65  auto const r7 = r5 * r2;
66 
67  // Dot products
68  auto const pe1 = dip1 * dip2;
69  auto const pe2 = dip1 * dr;
70  auto const pe3 = dip2 * dr;
71  auto const pe4 = 3.0 / r5;
72 
73  // Energy
74  auto const energy = prefactor * (pe1 / r3 - pe4 * pe2 * pe3);
75 
76  // Forces, if requested
77  if (force_flag) {
78  auto const a = pe4 * pe1;
79  auto const b = -15.0 * pe2 * pe3 / r7;
80  auto const ab = a + b;
81  auto const cc = pe4 * pe3;
82  auto const dd = pe4 * pe2;
83 
84  // Forces
85  auto const ff = ab * dr + cc * dip1 + dd * dip2;
86  p1.force() += prefactor * ff;
87  p2.force() -= prefactor * ff;
88 
89  // Torques
90  auto const aa = vector_product(dip1, dip2);
91  auto const b1 = vector_product(dip1, dr);
92  auto const b2 = vector_product(dip2, dr);
93  p1.torque() += prefactor * (-aa / r3 + b1 * cc);
94  p2.torque() += prefactor * (aa / r3 + b2 * dd);
95  }
96 
97  return energy;
98 }

100 double DipolarDirectSum::kernel(bool force_flag, bool energy_flag,
101  ParticleRange const &particles) const {
102 
103  assert(n_nodes == 1);
104  assert(force_flag || energy_flag);
105 
106  double energy = 0.0;
107  // Iterate over all particles
108  for (auto it = particles.begin(), end = particles.end(); it != end; ++it) {
109  // If the particle has no dipole moment, ignore it
110  if (it->dipm() == 0.0)
111  continue;
112 
113  auto const dip1 = it->calc_dip();
114  auto jt = it;
115  /* Skip diagonal */
116  ++jt;
117  for (; jt != end; ++jt) {
118  // If the particle has no dipole moment, ignore it
119  if (jt->dipm() == 0.0)
120  continue;
121  // Calculate energy and/or force between the particles
122  energy += calc_dipole_dipole_ia(*it, dip1, *jt, force_flag);
123  }
124  }
125 
126  return energy;
127 }
