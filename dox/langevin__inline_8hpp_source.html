35 /** Langevin thermostat for particle translational velocities.
36  * Collects the particle velocity (different for ENGINE, PARTICLE_ANISOTROPY).
37  * Collects the langevin parameters kT, gamma (different for
38  * THERMOSTAT_PER_PARTICLE). Applies the noise and friction term.
39  * @param[in] langevin Parameters
40  * @param[in] p %Particle
41  * @param[in] time_step Time step
42  * @param[in] kT Temperature
43  */
44 inline Utils::Vector3d
45 friction_thermo_langevin(LangevinThermostat const &langevin, Particle const &p,
46  double time_step, double kT) {
47  // Early exit for virtual particles without thermostat
48  if (p.is_virtual() and !thermo_virtual) {
49  return {};
50  }
51 
52  // Determine prefactors for the friction and the noise term
53  Thermostat::GammaType pref_friction = langevin.pref_friction;
54  Thermostat::GammaType pref_noise = langevin.pref_noise;
55 #ifdef THERMOSTAT_PER_PARTICLE
56  // override default if particle-specific gamma
57  if (p.gamma() >= Thermostat::GammaType{}) {
58  auto const gamma =
59  p.gamma() >= Thermostat::GammaType{} ? p.gamma() : langevin.gamma;
60  pref_friction = -gamma;
61  pref_noise = LangevinThermostat::sigma(kT, time_step, gamma);
62  }
63 #endif // THERMOSTAT_PER_PARTICLE
64 
65  // Get effective velocity in the thermostatting
66 #ifdef ENGINE
67  auto const &velocity = (p.swimming().v_swim != 0)
68  ? p.v() - p.swimming().v_swim * p.calc_director()
69  : p.v();
70 #else
71  auto const &velocity = p.v();
72 #endif // ENGINE
73 #ifdef PARTICLE_ANISOTROPY
74  // Particle frictional isotropy check
75  auto const aniso_flag = (pref_friction[0] != pref_friction[1]) ||
76  (pref_friction[1] != pref_friction[2]);
77 
78  // In case of anisotropic particle: body-fixed reference frame. Otherwise:
79  // lab-fixed reference frame.
80  const Utils::Matrix<double, 3, 3> fric_mat =
81  boost::qvm::diag_mat(pref_friction);
82  auto const friction_op =
83  aniso_flag ? convert_body_to_space(p, fric_mat) : fric_mat;
84  const Utils::Matrix<double, 3, 3> noise_op = boost::qvm::diag_mat(pref_noise);
85 #else
86  auto const &friction_op = pref_friction;
87  auto const &noise_op = pref_noise;
88 #endif // PARTICLE_ANISOTROPY
89 
90  return friction_op * velocity +
91  noise_op * Random::noise_uniform<RNGSalt::LANGEVIN>(
92  langevin.rng_counter(), langevin.rng_seed(), p.id());
93 }

105 inline Utils::Vector3d
106 friction_thermo_langevin_rotation(LangevinThermostat const &langevin,
107  Particle const &p, double time_step,
108  double kT) {
109 
110  auto pref_friction = -langevin.gamma_rotation;
111  auto pref_noise = langevin.pref_noise_rotation;
112 
113 #ifdef THERMOSTAT_PER_PARTICLE
114  // override default if particle-specific gamma
115  if (p.gamma_rot() >= Thermostat::GammaType{}) {
116  auto const gamma = p.gamma_rot() >= Thermostat::GammaType{}
117  ? p.gamma_rot()
118  : langevin.gamma_rotation;
119  pref_friction = -gamma;
120  pref_noise = LangevinThermostat::sigma(kT, time_step, gamma);
121  }
122 #endif // THERMOSTAT_PER_PARTICLE
123

125  langevin.rng_counter(), langevin.rng_seed(), p.id());

127  hadamard_product(pref_noise, noise);
128 }
