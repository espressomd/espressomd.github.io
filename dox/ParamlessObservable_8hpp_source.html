41/** Cython interface for parameter-free observables.
42 * Create new observables with @ref NEW_PARAMLESS_OBSERVABLE.
43 * @tparam CoreObs The core class exposed in Cython by this class.
44 */
45template <class CoreObs>
46class ParamlessObservableInterface : public Observable {
47public:
48 ParamlessObservableInterface() : m_observable(std::make_shared<CoreObs>()) {}
49
50 std::shared_ptr<::Observables::Observable> observable() const override {
51 return m_observable;
52 }
53
54private:
55 std::shared_ptr<CoreObs> m_observable;
56};
