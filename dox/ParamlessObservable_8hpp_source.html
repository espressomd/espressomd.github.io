44/** Cython interface for parameter-free observables.
45 * Create new observables with @ref NEW_PARAMLESS_OBSERVABLE.
46 * @tparam CoreObs The core class exposed in Cython by this class.
47 */
48template <class CoreObs>
49class ParamlessObservableInterface : public Observable {
50public:
51 ParamlessObservableInterface() : m_observable(std::make_shared<CoreObs>()) {}
52
53 std::shared_ptr<::Observables::Observable> observable() const override {
54 return m_observable;
55 }
56
57private:
58 std::shared_ptr<CoreObs> m_observable;
59};
