51 /** Put the calculated force stored on the ibm particles into the fluid by
52  * updating the @ref lbfields structure.
53  * Called from the integration loop right after the forces have been
54  * calculated.
55  */
56 void IBM_ForcesIntoFluid_CPU() {
57  // Update the forces on the ghost particles
58  cell_structure.ghosts_update(Cells::DATA_PART_FORCE);
59 
60  // Loop over local cells
61  for (auto &p : cell_structure.local_particles()) {
62  if (p.is_virtual()) {
63  CoupleIBMParticleToFluid(p, p.pos());
64  }
65  }
66 
67  for (auto &p : cell_structure.ghost_particles()) {
68  // for ghost particles we have to check if they lie
69  // in the range of the local lattice nodes
70  if (in_local_halo(p.pos())) {
71  if (p.is_virtual()) {
72  CoupleIBMParticleToFluid(p, p.pos());
73  }
74  }
75  }
76 }

82 void IBM_UpdateParticlePositions(ParticleRange const &particles,
83  double time_step, int this_node) {

92  // Euler integrator
93  for (auto &p : particles) {
94  if (p.is_virtual()) {

102  }
103 
104  if (cell_structure.check_resort_required(particles, skin)) {
105  cell_structure.set_resort_particles(Cells::RESORT_LOCAL);

109 /** Put the momentum of a given particle into the LB fluid. */
110 void CoupleIBMParticleToFluid(Particle const &p, Utils::Vector3d const &pos) {
111  // Convert units from MD to LB
112  auto const delta_j = p.force() * Utils::int_pow<4>(lbpar.tau) / lbpar.agrid;
113 
114  // Get indices and weights of affected nodes using discrete delta function
115  Utils::Vector<std::size_t, 8> node_index{};
116  Utils::Vector6d delta{};
117  lblattice.map_position_to_lattice(pos, node_index, delta);
118 
119  // Loop over all affected nodes
120  for (int z = 0; z < 2; z++) {
121  for (int y = 0; y < 2; y++) {
122  for (int x = 0; x < 2; x++) {
123  // Do not put force into a halo node
124  if (!IsHalo(static_cast<int>(node_index[(z * 2 + y) * 2 + x]))) {
125  // Add force into the lbfields structure
126  auto &local_f =
127  lbfields[node_index[(z * 2 + y) * 2 + x]].force_density;
128 
129  local_f +=
130  delta[3 * x + 0] * delta[3 * y + 1] * delta[3 * z + 2] * delta_j;
131  }
132  }
133  }
134  }
135 }

137 /** Calculate the LB fluid velocity at a particle position.
138  * Very similar to the velocity interpolation done in standard ESPResSo,
139  * except that we add the f/2 contribution, cf. @cite guo02a.
140  * The fluid velocity is obtained by linear interpolation,
141  * cf. eq. (11) in @cite ahlrichs99a.
142  */
143 template <bool ReturnVelocity>
144 Utils::Vector3d GetIBMInterpolatedVelocity(Utils::Vector3d const &pos) {
145  auto const f_ext =
146  lbpar.ext_force_density * Utils::sqr(lbpar.agrid * lbpar.tau);
147 
148  /* determine elementary lattice cell surrounding the particle
149  and the relative position of the particle in this cell */
150  Utils::Vector<std::size_t, 8> node_index{};
151  Utils::Vector6d delta{};
152  lblattice.map_position_to_lattice(pos, node_index, delta);
153 
154  // This for the f/2 contribution to the velocity
155  Utils::Vector3d force_added = {};
156  Utils::Vector3d interpolated_u = {};
157 
158  for (int z = 0; z < 2; z++) {
159  for (int y = 0; y < 2; y++) {
160  for (int x = 0; x < 2; x++) {
161  auto const index = node_index[(z * 2 + y) * 2 + x];
162  auto const local_delta =
163  delta[3 * x + 0] * delta[3 * y + 1] * delta[3 * z + 2];
164  const auto &f = lbfields[index].force_density_buf;
165 
166  double local_density;
167  Utils::Vector3d local_j;
168 
169  // This can be done more easily without copying the code twice.
170  // We probably can even set the boundary velocity directly.
171 #ifdef LB_BOUNDARIES
172  if (lbfields[index].boundary) {
173  if (ReturnVelocity) {
174  local_density = lbpar.density;
175  auto const i = lbfields[index].boundary - 1;
176  local_j = lbpar.density * LBBoundaries::lbboundaries[i]->velocity();
177  }
178  } else
179 #endif
180  {
181  auto const modes = lb_calc_modes(static_cast<int>(index), lbfluid);
182  local_density = lbpar.density + modes[0];
183 
184  if (ReturnVelocity) {
185  // Add the +f/2 contribution!!
186  local_j[0] = modes[1] + f[0] / 2.;
187  local_j[1] = modes[2] + f[1] / 2.;
188  local_j[2] = modes[3] + f[2] / 2.;
189  } else {
190  // Keep track of the forces that we added to the fluid.
191  // This is necessary for communication because this part is executed
192  // for real and ghost particles.
193  // Later on we sum the real and ghost contributions.
194  force_added += local_delta * (f - f_ext) / (2. * local_density);
195  }
196  }
197 
198  // Interpolate velocity
199  if (ReturnVelocity) {
200  interpolated_u += local_j * (local_delta / local_density);
201  }
202  }
203  }
204  }
205 
206  auto const unit_conversion = lbpar.agrid / lbpar.tau;
207  if (ReturnVelocity) {
208  return interpolated_u * unit_conversion;
209  }
210  return force_added * unit_conversion;
211 }

213 /** Build a cache structure which contains a flag for each LB node whether
214  * that node is a halo node or not.
215  */
216 bool IsHalo(std::size_t indexCheck) {
217  // First call --> build cache
218  if (isHaloCache == nullptr) {
219  isHaloCache = new bool[lblattice.halo_grid_volume];
220  // Assume everything is a halo and correct in the next step
221  for (int i = 0; i < lblattice.halo_grid_volume; i++)
222  isHaloCache[i] = true;
223  // Loop through and check where indexCheck occurs
224  auto index = lblattice.halo_offset;
225  for (int z = 1; z <= lblattice.grid[2]; z++) {
226  for (int y = 1; y <= lblattice.grid[1]; y++) {
227  for (int x = 1; x <= lblattice.grid[0]; x++) {
228  isHaloCache[index] = false;
229  ++index;
230  }
231  index += 2; /* skip halo region */
232  }
233  index += 2 * lblattice.halo_grid[0]; /* skip halo region */
234  }
235  }
236 
237  // Return
238  return isHaloCache[indexCheck];
239 }

256 /** Get particle velocities from LB and set the velocity field in the
257  * particles data structure.
258  */
259 void ParticleVelocitiesFromLB_CPU() {
260  std::unordered_set<int> coupled_ghost_particles;
261 
262  // Loop over particles in local cells.
263  // Here all contributions are included: velocity, external force and
264  // particle force.
265  for (auto &p : cell_structure.local_particles()) {
266  if (p.is_virtual() and should_be_coupled(p, coupled_ghost_particles)) {
267  for (auto pos : positions_in_halo(p.pos(), box_geo)) {
268  if (in_local_domain(pos)) {
269  p.force() = GetIBMInterpolatedVelocity<true>(pos);
270  break;
271  }
272  }
273  }
274  }
275  // Loop over particles in ghost cells
276  // Here we only add the particle forces stemming from the ghosts
277  for (auto &p : cell_structure.ghost_particles()) {
278  if (p.is_virtual() and should_be_coupled(p, coupled_ghost_particles)) {
279  for (auto pos : positions_in_halo(p.pos(), box_geo)) {
280  if (in_local_domain(pos)) {
281  p.force() = GetIBMInterpolatedVelocity<true>(pos);
282  break;
283  }
284  }
285  } else {
286  p.force() = {};
287  }
288  }
289 
290  // Now the local particles contain a velocity (stored in the force field)
291  // and the ghosts contain the rest of the velocity in their respective force
292  // fields.
293  // We need to add these. Since we have stored them in the force, not the
294  // velocity fields, we can use the standard force communicator and then
295  // transfer to the velocity afterwards.
296  // Note that this overwrites the actual force which would be a problem for
297  // real particles.
298  // This could be solved by keeping a backup of the local forces before this
299  // operation is attempted.
300  cell_structure.ghosts_reduce_forces();
301 
302  // Transfer to velocity field
303  for (auto &p : cell_structure.local_particles()) {
304  if (p.is_virtual()) {
305  p.v() = p.force();
306  }
307  }
308 }
