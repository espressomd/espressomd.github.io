35 /** Sweep that swaps @c force_to_be_applied and @c last_applied_force
36  * and resets @c force_to_be_applied to the global external force.
37  */
38 template <typename PdfField, typename ForceField> class ResetForce {
39  using FloatType = typename PdfField::value_type;
40 
41 public:
42  ResetForce(BlockDataID const &last_applied_force_field_id,
43  BlockDataID const &force_to_be_applied_id)
44  : m_last_applied_force_field_id(last_applied_force_field_id),
45  m_force_to_be_applied_id(force_to_be_applied_id),
46  m_ext_force(Vector3<FloatType>{0, 0, 0}) {}
47 
48  void set_ext_force(Utils::Vector3d const &ext_force) {
49  m_ext_force = to_vector3<FloatType>(ext_force);
50  }
51 
52  Utils::Vector3d get_ext_force() const { return to_vector3d(m_ext_force); }
53 
54  void operator()(IBlock *block) {
55  auto force_field =
56  block->template getData<ForceField>(m_last_applied_force_field_id);
57  auto force_to_be_applied =
58  block->template getData<ForceField>(m_force_to_be_applied_id);
59 
60  force_field->swapDataPointers(force_to_be_applied);
61 
62  lbm::accessor::Vector::add_to_all(force_field, m_ext_force);
63  lbm::accessor::Vector::broadcast(force_to_be_applied,
64  Vector3<FloatType>{0});
65  }
66 
67 private:
68  const BlockDataID m_last_applied_force_field_id;
69  const BlockDataID m_force_to_be_applied_id;
70  Vector3<FloatType> m_ext_force;
71 };
