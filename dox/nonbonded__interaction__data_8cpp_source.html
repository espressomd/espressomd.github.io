63 static void mpi_realloc_ia_params_local(int new_size) {
64  if (new_size <= max_seen_particle_type)
65  return;
66 
67  auto new_params = std::vector<IA_parameters>(new_size * (new_size + 1) / 2);
68 
69  /* if there is an old field, move entries */
70  for (int i = 0; i < max_seen_particle_type; i++)
71  for (int j = i; j < max_seen_particle_type; j++) {
72  new_params.at(Utils::upper_triangular(i, j, new_size)) =
73  std::move(*get_ia_param(i, j));
74  }
75 
76  max_seen_particle_type = new_size;
77  std::swap(nonbonded_ia_params, new_params);
78 }

123 static double recalc_maximal_cutoff(const IA_parameters &data) {
124  auto max_cut_current = INACTIVE_CUTOFF;
125 
126 #ifdef LENNARD_JONES
127  max_cut_current = std::max(max_cut_current, (data.lj.cut + data.lj.offset));
128 #endif
129 
130 #ifdef WCA
131  max_cut_current = std::max(max_cut_current, data.wca.cut);
132 #endif
133 
134 #ifdef DPD
135  max_cut_current = std::max(
136  max_cut_current, std::max(data.dpd_radial.cutoff, data.dpd_trans.cutoff));
137 #endif
138 
139 #ifdef LENNARD_JONES_GENERIC
140  max_cut_current =
141  std::max(max_cut_current, (data.ljgen.cut + data.ljgen.offset));
142 #endif
143 
144 #ifdef SMOOTH_STEP
145  max_cut_current = std::max(max_cut_current, data.smooth_step.cut);
146 #endif
147 
148 #ifdef HERTZIAN
149  max_cut_current = std::max(max_cut_current, data.hertzian.sig);
150 #endif
151 
152 #ifdef GAUSSIAN
153  max_cut_current = std::max(max_cut_current, data.gaussian.cut);
154 #endif
155 
156 #ifdef BMHTF_NACL
157  max_cut_current = std::max(max_cut_current, data.bmhtf.cut);
158 #endif
159 
160 #ifdef MORSE
161  max_cut_current = std::max(max_cut_current, data.morse.cut);
162 #endif
163 
164 #ifdef BUCKINGHAM
165  max_cut_current = std::max(max_cut_current, data.buckingham.cut);
166 #endif
167 
168 #ifdef SOFT_SPHERE
169  max_cut_current = std::max(max_cut_current,
170  (data.soft_sphere.cut + data.soft_sphere.offset));
171 #endif
172 
173 #ifdef HAT
174  max_cut_current = std::max(max_cut_current, data.hat.r);
175 #endif
176 
177 #ifdef LJCOS
178  max_cut_current =
179  std::max(max_cut_current, (data.ljcos.cut + data.ljcos.offset));
180 #endif
181 
182 #ifdef LJCOS2
183  max_cut_current =
184  std::max(max_cut_current, (data.ljcos2.cut + data.ljcos2.offset));
185 #endif
186 
187 #ifdef GAY_BERNE
188  max_cut_current = std::max(max_cut_current, data.gay_berne.cut);
189 #endif
190 
191 #ifdef TABULATED
192  max_cut_current = std::max(max_cut_current, data.tab.cutoff());
193 #endif
194 
195 #ifdef THOLE
196  // If THOLE is active, use p3m cutoff
197  if (data.thole.scaling_coeff != 0)
198  max_cut_current = std::max(max_cut_current, Coulomb::cutoff());
199 #endif
200 
201  return max_cut_current;
202 }

204 double maximal_cutoff_nonbonded() {
205  auto max_cut_nonbonded = INACTIVE_CUTOFF;
206 
207  for (auto &data : nonbonded_ia_params) {
208  data.max_cut = recalc_maximal_cutoff(data);
209  max_cut_nonbonded = std::max(max_cut_nonbonded, data.max_cut);
210  }
211 
212  return max_cut_nonbonded;
213 }
