29class Shape {
30public:
31 /**
32 * @brief Calculate the minimum distance and the corresponding
33 * distance vector between a given position and the shape.
34 * @param[in] pos Position for which to calculate the distance.
35 * @param[out] dist Minimum distance between @p pos and the shape. Value
36 * is negative when @c pos is inside the shape, zero if
37 * @c pos is on the shape surface or positive if @c pos
38 * is outside the shape.
39 * @param[out] vec Distance vector.
40 */
41 virtual void calculate_dist(const Utils::Vector3d &pos, double &dist,
42 Utils::Vector3d &vec) const = 0;
43 /**
44 * @brief Check whether the given point is inside the shape or not.
45 * For the edge case where the point is on the surface (zero distance),
46 * it is considered to be inside the shape.
47 * @param pos Position to check.
48 */
49 virtual bool is_inside(Utils::Vector3d const &pos) const {
50 Utils::Vector3d vec;
51 double dist;
52 calculate_dist(pos, dist, vec);
53 return dist <= 0.0;
54 }
55 /**
56 * @brief Rasterize a shape on a regular grid.
57 * @param grid_size Number of grid points in every direction.
58 * @param grid_spacing Lattice distance.
59 * @param grid_offset Lattice offset.
60 * @return Flattened 3D matrix with 1's inside the shape and 0's outside.
61 */
62 std::vector<int> rasterize(Utils::Vector3i const &grid_size,
63 double grid_spacing, double grid_offset) const;
64 virtual ~Shape() = default;
65};
