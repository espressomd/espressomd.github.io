116struct MinimalImageDistance {
117 BoxGeometry const box;
118
119 Distance operator()(Particle const &p1, Particle const &p2) const {

211 /**

214 * @param pl List to add the particle to.

217 Particle &append_indexed_particle(ParticleList &pl, Particle &&p) {
218 /* Check if cell may reallocate, in which case the index
219 * entries for all particles in this cell have to be
220 * updated. */
221 auto const may_reallocate = pl.size() >= pl.capacity();
222 auto &new_part = pl.insert(std::move(p));

428 /**
429 * @brief Check whether a particle has moved further than half the skin
430 * since the last Verlet list update, thus requiring a resort.
431 * @param additional_offset Offset which is added to the distance the
432 * particle has travelled when comparing to half
433 * the Verlet skin (e.g., for Lees-Edwards BC).
434 * @return Whether a resort is needed.

436 bool
437 check_resort_required(Utils::Vector3d const &additional_offset = {}) const {
438 auto const particles = local_particles();
439 auto const lim = Utils::sqr(m_verlet_skin / 2.) - additional_offset.norm2();
440 return std::any_of(
441 particles.begin(), particles.end(), [lim](const auto &p) {
442 return ((p.pos() - p.pos_at_last_verlet_update()).norm2() > lim);
443 });

516 /**
517 * @brief Resolve ids to particles.
518 *
519 * @throws BondResolutionError if one of the ids
520 * was not found.
521 *
522 * @param partner_ids Ids to resolve.
523 * @return Vector of Particle pointers.
524 */
525 auto resolve_bond_partners(std::span<const int> partner_ids) {
526 boost::container::static_vector<Particle *, 4> partners;
527 get_local_particles(partner_ids, std::back_inserter(partners));
528
529 /* Check if id resolution failed for any partner */
530 if (std::ranges::find(partners, nullptr) != partners.end()) {
531 throw BondResolutionError{};
532 }
533
534 return partners;
535 }

538 /**
539 * @brief Execute kernel for every bond on particle.
540 * @tparam Handler Callable, which can be invoked with
541 * (Particle, int, std::span<Particle *>),
542 * returning a bool.
543 * @param p Particles for whom the bonds are evaluated.
544 * @param handler is called for every bond, and handed
545 * p, the bond id and a span with the bond
546 * partners as arguments. Its return value
547 * should indicate if the bond was broken.
548 */
549 template <class Handler>
550 void execute_bond_handler(Particle &p, Handler handler) {
551 for (const BondView bond : p.bonds()) {
552 auto const partner_ids = bond.partner_ids();
553
554 try {
555 auto partners = resolve_bond_partners(partner_ids);
556 auto const partners_span = std::span(partners.data(), partners.size());
557 auto const bond_broken = handler(p, bond.bond_id(), partners_span);
558 if (bond_broken) {
559 bond_broken_error(p.id(), partner_ids);
560 }
561 } catch (const BondResolutionError &) {
562 bond_broken_error(p.id(), partner_ids);
563 }
564 }
565 }

619 /**
620 * @brief Run link_cell algorithm for local cells.
621 *
622 * @tparam Kernel Needs to be callable with (Particle, Particle, Distance).
623 * @param kernel Pair kernel functor.
624 */
625 template <class Kernel> void link_cell(Kernel kernel) {
626 auto const maybe_box = decomposition().minimum_image_distance();
627 auto const local_cells_span = decomposition().local_cells();
628 auto const first = boost::make_indirect_iterator(local_cells_span.begin());
629 auto const last = boost::make_indirect_iterator(local_cells_span.end());
630
631 if (maybe_box) {
632 Algorithm::link_cell(
633 first, last,
634 [&kernel, df = detail::MinimalImageDistance{decomposition().box()}](
635 Particle &p1, Particle &p2) { kernel(p1, p2, df(p1, p2)); });
636 } else {
637 if (decomposition().box().type() != BoxType::CUBOID) {
638 throw std::runtime_error("Non-cuboid box type is not compatible with a "
639 "particle decomposition that relies on "
640 "EuclideanDistance for distance calculation.");
641 }
642 Algorithm::link_cell(
643 first, last,
644 [&kernel, df = detail::EuclidianDistance{}](
645 Particle &p1, Particle &p2) { kernel(p1, p2, df(p1, p2)); });
646 }
647 }

649 /** Non-bonded pair loop with verlet lists.
650 *
651 * @param pair_kernel Kernel to apply
652 * @param verlet_criterion Filter for verlet lists.
653 */
654 template <class PairKernel, class VerletCriterion>
655 void verlet_list_loop(PairKernel pair_kernel,
656 const VerletCriterion &verlet_criterion) {
657 /* In this case the verlet list update is attached to
658 * the pair kernel, and the verlet list is rebuilt as
659 * we go. */
660 if (m_rebuild_verlet_list) {
661 m_verlet_list.clear();
662
663 link_cell([&](Particle &p1, Particle &p2, Distance const &d) {
664 if (verlet_criterion(p1, p2, d)) {
665 m_verlet_list.emplace_back(&p1, &p2);
666 pair_kernel(p1, p2, d);
667 }
668 });
669
670 m_rebuild_verlet_list = false;
671 } else {
672 auto const maybe_box = decomposition().minimum_image_distance();
673 /* In this case the pair kernel is just run over the verlet list. */
674 if (maybe_box) {
675 auto const distance_function =
676 detail::MinimalImageDistance{decomposition().box()};
677 for (auto const &[p1, p2] : m_verlet_list) {
678 pair_kernel(*p1, *p2, distance_function(*p1, *p2));
679 }
680 } else {
681 auto const distance_function = detail::EuclidianDistance{};
682 for (auto const &[p1, p2] : m_verlet_list) {
683 pair_kernel(*p1, *p2, distance_function(*p1, *p2));
684 }
685 }
686 }
687 }

761 /**
762 * @brief Run kernel on all particles inside local cell and its neighbors.
763 *
764 * @param p Particle to find cell for
765 * @param kernel Function with signature <tt>double(Particle const&,
766 * Particle const&, Utils::Vector3d const&)</tt>
767 * @return false if cell is not found, otherwise true
768 */
769 template <class Kernel>
770 bool run_on_particle_short_range_neighbors(Particle const &p,
771 Kernel &kernel) {
772 auto const cell = find_current_cell(p);
773
774 if (cell == nullptr) {
775 return false;

793 void short_range_neighbor_loop(Particle const &p1, Cell *const cell,
794 Kernel &kernel, DistanceFunc const &df) {

796 for (auto const &p2 : cell->particles()) {
797 if (p1.id() != p2.id()) {
798 auto const vec = df(p1, p2).vec21;
799 kernel(p1, p2, vec);
800 }
801 }
802 /* Iterate over all neighbors */
803 for (auto const neighbor : cell->neighbors().all()) {
804 /* Iterate over particles in neighbors */
805 if (neighbor != cell) {
806 for (auto const &p2 : neighbor->particles()) {
807 auto const vec = df(p1, p2).vec21;
808 kernel(p1, p2, vec);
809 }
810 }
811 }
