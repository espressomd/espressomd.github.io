226 /**
227 * @brief Update local particle index.
228 *
229 * Update the entry for a particle in the local particle
230 * index.
231 *
232 * @param id Entry to update.
233 * @param p Pointer to the particle.
234 */
235 void update_particle_index(int id, Particle *p) {
236 assert(id >= 0);
237 // cppcheck-suppress assertWithSideEffect
238 assert(not p or p->id() == id);
239
240 if (static_cast<unsigned int>(id) >= m_particle_index.size())
241 m_particle_index.resize(static_cast<unsigned int>(id + 1));
242
243 m_particle_index[static_cast<unsigned int>(id)] = p;
244 }

275 /**
276 * @brief Append a particle to a list and update this
277 * particle index accordingly.
278 * @param pl List to add the particle to.
279 * @param p Particle to add.
280 */
281 Particle &append_indexed_particle(ParticleList &pl, Particle &&p) {
282 /* Check if cell may reallocate, in which case the index
283 * entries for all particles in this cell have to be
284 * updated. */
285 auto const may_reallocate = pl.size() >= pl.capacity();
286 auto &new_part = pl.insert(std::move(p));
287
288 if (may_reallocate)
289 update_particle_index(pl);
290 else {
291 update_particle_index(new_part);
292 }
293
294 return new_part;
295 }

587 /**
588 * @brief Resolve ids to particles.
589 *
590 * @throws BondResolutionError if one of the ids
591 * was not found.
592 *
593 * @param partner_ids Ids to resolve.
594 * @return Vector of Particle pointers.
595 */
596 auto resolve_bond_partners(std::span<const int> partner_ids) {
597 boost::container::static_vector<Particle *, 4> partners;
598 get_local_particles(partner_ids, std::back_inserter(partners));
599
600 /* Check if id resolution failed for any partner */
601 if (std::ranges::find(partners, nullptr) != partners.end()) {
602 throw BondResolutionError{};
603 }
604
605 return partners;
606 }

609 /**
610 * @brief Execute kernel for every bond on particle.
611 * @tparam Handler Callable, which can be invoked with
612 * (Particle, int, std::span<Particle *>),
613 * returning a bool.
614 * @param p Particles for whom the bonds are evaluated.
615 * @param handler is called for every bond, and handed
616 * p, the bond id and a span with the bond
617 * partners as arguments. Its return value
618 * should indicate if the bond was broken.
619 */
620 template <class Handler>
621 void execute_bond_handler(Particle &p, Handler handler) {
622 for (const BondView bond : p.bonds()) {
623 auto const partner_ids = bond.partner_ids();
624
625 try {
626 auto partners = resolve_bond_partners(partner_ids);
627 auto const partners_span = std::span(partners.data(), partners.size());
628 auto const bond_broken = handler(p, bond.bond_id(), partners_span);
629 if (bond_broken) {
630 bond_broken_error(p.id(), partner_ids);
631 }
632 } catch (const BondResolutionError &) {
633 bond_broken_error(p.id(), partner_ids);
634 }
635 }
636 }

690 /**
691 * @brief Run link_cell algorithm for local cells.
692 *
693 * @tparam Kernel Needs to be callable with (Particle, Particle, Distance).
694 * @param kernel Pair kernel functor.
695 */
696 template <class Kernel> void link_cell(Kernel kernel) {
697 auto const maybe_box = decomposition().minimum_image_distance();
698 auto const local_cells_span = decomposition().local_cells();
699 auto const first = boost::make_indirect_iterator(local_cells_span.begin());
700 auto const last = boost::make_indirect_iterator(local_cells_span.end());
701
702 if (maybe_box) {
703 Algorithm::link_cell(
704 first, last,
705 [&kernel, df = detail::MinimalImageDistance{decomposition().box()}](
706 Particle &p1, Particle &p2) { kernel(p1, p2, df(p1, p2)); });
707 } else {
708 if (decomposition().box().type() != BoxType::CUBOID) {
709 throw std::runtime_error("Non-cuboid box type is not compatible with a "
710 "particle decomposition that relies on "
711 "EuclideanDistance for distance calculation.");
712 }
713 Algorithm::link_cell(
714 first, last,
715 [&kernel, df = detail::EuclidianDistance{}](
716 Particle &p1, Particle &p2) { kernel(p1, p2, df(p1, p2)); });
717 }
718 }

782 /** Non-bonded pair loop with verlet lists.
783 *
784 * @param pair_kernel Kernel to apply
785 * @param verlet_criterion Filter for verlet lists.
786 */
787 template <class PairKernel, class VerletCriterion>
788 void verlet_list_loop(PairKernel pair_kernel,
789 const VerletCriterion &verlet_criterion) {
790 /* In this case the verlet list update is attached to
791 * the pair kernel, and the verlet list is rebuilt as
792 * we go. */
793 if (m_rebuild_verlet_list) {
794 m_verlet_list.clear();
795
796 link_cell([&](Particle &p1, Particle &p2, Distance const &d) {
797 if (verlet_criterion(p1, p2, d)) {
798 m_verlet_list.emplace_back(&p1, &p2);
799 pair_kernel(p1, p2, d);
800 }
801 });
802
803 m_rebuild_verlet_list = false;
804 m_rebuild_verlet_list_cabana = true;
805 } else {
806 auto const maybe_box = decomposition().minimum_image_distance();
807 /* In this case the pair kernel is just run over the verlet list. */
808 if (maybe_box) {
809 auto const distance_function =
810 detail::MinimalImageDistance{decomposition().box()};
811 for (auto const &[p1, p2] : m_verlet_list) {
812 pair_kernel(*p1, *p2, distance_function(*p1, *p2));
813 }
814 } else {
815 auto const distance_function = detail::EuclidianDistance{};
816 for (auto const &[p1, p2] : m_verlet_list) {
817 pair_kernel(*p1, *p2, distance_function(*p1, *p2));
818 }
819 }
820 }

897 *
898 * @param p Particle to find cell for
899 * @param kernel Function with signature <tt>double(Particle const&,
900 * Particle const&, Utils::Vector3d const&)</tt>
901 * @return false if cell is not found, otherwise true
902 */
903 template <class Kernel>
904 bool run_on_particle_short_range_neighbors(Particle const &p,
905 Kernel &kernel) {
906 auto const cell = find_current_cell(p);
907
908 if (cell == nullptr) {
909 return false;
910 }
911
912 auto const maybe_box = decomposition().minimum_image_distance();
913
914 if (maybe_box) {
915 auto const distance_function =
916 detail::MinimalImageDistance{decomposition().box()};
917 short_range_neighbor_loop(p, cell, kernel, distance_function);
918 } else {
919 auto const distance_function = detail::EuclidianDistance{};
920 short_range_neighbor_loop(p, cell, kernel, distance_function);
921 }
922 return true;
923 }

926 template <class Kernel, class DistanceFunc>
927 void short_range_neighbor_loop(Particle const &p1, Cell *const cell,
928 Kernel &kernel, DistanceFunc const &df) {
929 /* Iterate over particles inside cell */
930 for (auto const &p2 : cell->particles()) {
931 if (p1.id() != p2.id()) {
932 auto const vec = df(p1, p2).vec21;
933 kernel(p1, p2, vec);
934 }
935 }
936 /* Iterate over all neighbors */
937 for (auto const neighbor : cell->neighbors().all()) {
938 /* Iterate over particles in neighbors */
939 if (neighbor != cell) {
940 for (auto const &p2 : neighbor->particles()) {
941 auto const vec = df(p1, p2).vec21;
942 kernel(p1, p2, vec);
943 }
944 }
945 }
