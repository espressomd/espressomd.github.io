44 /**
45  * @brief Tuning algorithm for P3M.
46  *
47  * The algorithm basically determines the mesh, cao
48  * and then the real-space cutoff, in this order.
49  *
50  * For each mesh, the optimal cao for the previous mesh is re-used as an
51  * initial guess, and the algorithm checks whether increasing or decreasing
52  * it leads to a better solution. This is efficient, since the optimal cao
53  * only changes little with the meshes in general.
54  *
55  * The real-space cutoff for a given mesh and cao is determined via a
56  * bisection on the error estimate, which determines where the error
57  * estimate equals the required accuracy. Therefore the smallest possible,
58  * i.e. fastest real-space cutoff is determined.
59  *
60  * Both the search over mesh and cao stop to search in a specific
61  * direction once the computation time is significantly higher
62  * than the currently known optimum.
63  */

66  System::System &m_system;

154  void tune() {
155  // activate tuning mode
156  get_params().tuning = true;
157 
158  auto const tuned_params = get_time();
159 
160  // deactivate tuning mode
161  get_params().tuning = false;
162 
163  if (tuned_params.time == time_sentinel) {
164  throw std::runtime_error(m_logger->get_name() +
165  ": failed to reach requested accuracy");
166  }
167  // set tuned parameters
168  get_params().accuracy = tuned_params.accuracy;
169  commit(tuned_params.mesh, tuned_params.cao, tuned_params.r_cut_iL,
170  tuned_params.alpha_L);
171 
172  m_logger->tuning_results(tuned_params.mesh, tuned_params.cao,
173  tuned_params.r_cut_iL, tuned_params.alpha_L,
174  tuned_params.accuracy, tuned_params.time);
175  }
176 
177 protected:
178  auto get_n_trials() { return m_n_trials; }
179  void increment_n_trials() { ++m_n_trials; }
180  void reset_n_trials() { m_n_trials = 0ul; }
181  double get_m_time(Utils::Vector3i const &mesh, int &tuned_cao,
