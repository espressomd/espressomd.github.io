31template <typename FloatType, Arch Architecture>
32std::shared_ptr<CoulombP3M>
33new_coulomb_p3m_impl(P3MParameters &&p3m, TuningParameters const &tuning_params,
34 double prefactor) {
35 using DefaultFFTConfig = P3MFFTConfig<Utils::MemoryOrder::ROW_MAJOR,
36 Utils::MemoryOrder::ROW_MAJOR, true, 2>;
37 auto state_ptr =
38 std::make_unique<CoulombP3MState<FloatType, DefaultFFTConfig>>(
39 std::move(p3m));
40 auto obj = std::make_shared<
41 CoulombP3MHeffte<FloatType, Architecture, DefaultFFTConfig>>(
42 std::move(state_ptr), tuning_params, prefactor);
43 return obj;
44}

46std::shared_ptr<CoulombP3M>
47new_coulomb_p3m_heffte(P3MParameters &&p3m_params,
48 TuningParameters const &tuning_params, double prefactor,
49 bool single_precision, Arch arch) {
50 auto fptr = &new_coulomb_p3m_impl<float, Arch::CPU>;
51 if (single_precision) {
52 if (arch == Arch::CPU) {
53 fptr = new_coulomb_p3m_impl<float, Arch::CPU>;
54 } else {
55 fptr = new_coulomb_p3m_impl<float, Arch::CUDA>;
56 }
57 } else {
58 if (arch == Arch::CPU) {
59 fptr = new_coulomb_p3m_impl<double, Arch::CPU>;
60 } else {
61 throw std::invalid_argument(
62 "P3M GPU only implemented in single-precision mode");
63 }
64 }
65 return fptr(std::move(p3m_params), tuning_params, prefactor);
66}
