72 static LocalBox make_regular_decomposition(Utils::Vector3d const &box_l,
73 Utils::Vector3i const &node_index,
74 Utils::Vector3i const &node_grid) {
75
76 auto const local_length = Utils::hadamard_division(box_l, node_grid);
77 auto const my_left = Utils::hadamard_product(node_index, local_length);
78
79 decltype(LocalBox::m_boundaries) boundaries;
80 for (unsigned int dir = 0u; dir < 3u; dir++) {
81 /* left boundary ? */
82 boundaries[2u * dir] = (node_index[dir] == 0);
83 /* right boundary ? */
84 boundaries[2u * dir + 1u] = -(node_index[dir] + 1 == node_grid[dir]);
85 }
86
87 return {my_left, local_length, boundaries, CellStructureType::REGULAR};
88 }
