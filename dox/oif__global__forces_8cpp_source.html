42 Utils::Vector2d calc_oif_global(int molType, CellStructure &cs) {
43  // first-fold-then-the-same approach
44  double partArea = 0.0;
45  // z volume
46  double VOL_partVol = 0.;
47 
48  cs.bond_loop([&partArea, &VOL_partVol,
49  molType](Particle &p1, int bond_id,
50  Utils::Span<Particle *> partners) {
51  if (p1.mol_id() != molType)
52  return false;
53 
54  if (boost::get<OifGlobalForcesBond>(bonded_ia_params.at(bond_id).get()) !=
55  nullptr) {
56  // remaining neighbors fetched
57  auto const p11 =
58  unfolded_position(p1.pos(), p1.image_box(), box_geo.length());
59  auto const p22 = p11 + box_geo.get_mi_vector(partners[0]->pos(), p11);
60  auto const p33 = p11 + box_geo.get_mi_vector(partners[1]->pos(), p11);
61 
62  // unfolded positions correct
63  auto const VOL_A = area_triangle(p11, p22, p33);
64  partArea += VOL_A;
65 
66  auto const VOL_norm = get_n_triangle(p11, p22, p33);
67  auto const VOL_dn = VOL_norm.norm();
68  auto const VOL_hz = 1.0 / 3.0 * (p11[2] + p22[2] + p33[2]);
69  VOL_partVol += VOL_A * -1 * VOL_norm[2] / VOL_dn * VOL_hz;
70  }
71 
72  return false;
73  });
74 
75  auto const area_volume_local = Utils::Vector2d{{partArea, VOL_partVol}};
76  return boost::mpi::all_reduce(comm_cart, area_volume_local, std::plus<>());
77 }

79 void add_oif_global_forces(Utils::Vector2d const &area_volume, int molType,
80  CellStructure &cs) {
81  // first-fold-then-the-same approach
82  double area = area_volume[0];
83  double VOL_volume = area_volume[1];
84 
85  cs.bond_loop([area, VOL_volume, molType](Particle &p1, int bond_id,
86  Utils::Span<Particle *> partners) {
87  if (p1.mol_id() != molType)
88  return false;
89 
90  if (auto const *iaparams = boost::get<OifGlobalForcesBond>(
91  bonded_ia_params.at(bond_id).get())) {
92  auto const p11 =
93  unfolded_position(p1.pos(), p1.image_box(), box_geo.length());
94  auto const p22 = p11 + box_geo.get_mi_vector(partners[0]->pos(), p11);
95  auto const p33 = p11 + box_geo.get_mi_vector(partners[1]->pos(), p11);
96 
97  // unfolded positions correct
98  // starting code from volume force
99  auto const VOL_norm = get_n_triangle(p11, p22, p33).normalize();
100  auto const VOL_A = area_triangle(p11, p22, p33);
101  auto const VOL_vv = (VOL_volume - iaparams->V0) / iaparams->V0;
102 
103  auto const VOL_force =
104  (1.0 / 3.0) * iaparams->kv * VOL_vv * VOL_A * VOL_norm;
105 
106  auto const h = (1. / 3.) * (p11 + p22 + p33);
107 
108  auto const deltaA = (area - iaparams->A0_g) / iaparams->A0_g;
109 
110  auto const m1 = h - p11;
111  auto const m2 = h - p22;
112  auto const m3 = h - p33;
113 
114  auto const m1_length = m1.norm();
115  auto const m2_length = m2.norm();
116  auto const m3_length = m3.norm();
117 
118  auto const fac = iaparams->ka_g * VOL_A * deltaA /
119  (m1_length * m1_length + m2_length * m2_length +
120  m3_length * m3_length);
121 
122  p1.force() += fac * m1 + VOL_force;
123  partners[0]->force() += fac * m2 + VOL_force;
124  partners[1]->force() += fac * m3 + VOL_force;
125  }
126 
127  return false;
128  });
129 }
