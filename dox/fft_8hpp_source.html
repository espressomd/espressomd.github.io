24/** \file
25 *
26 * Routines, row decomposition, data structures and communication for the
27 * 3D-FFT.
28 *
29 * The 3D-FFT is split into 3 ond dimensional FFTs. The data is
30 * distributed in such a way, that for the actual direction of the
31 * FFT each node has a certain number of rows for which it performs a
32 * 1D-FFT. After performing the FFT on that direction the data is
33 * redistributed.
34 *
35 * For simplicity at the moment I have implemented a full complex to
36 * complex FFT (even though a real to complex FFT would be sufficient).
37 *
38 * For more information about FFT usage, see \ref fft.cpp "fft.cpp".
39 */

56/** Aligned allocator for fft data. */
57template <class T> struct fft_allocator {
58 typedef T value_type;
59 fft_allocator() noexcept = default; // default ctor not required
60 template <class U> explicit fft_allocator(const fft_allocator<U> &) {}
61 template <class U> bool operator==(const fft_allocator<U> &) const {
62 return true;
63 }
64 template <class U> bool operator!=(const fft_allocator<U> &) const {
65 return false;
66 }
67
68 T *allocate(const std::size_t n) const {
69 if (n == 0) {
70 return nullptr;
71 }
72 if (n > std::numeric_limits<std::size_t>::max() / sizeof(T)) {
73 throw std::bad_array_new_length();
74 }
75 void *const pv = fftw_malloc(n * sizeof(T));
76 if (!pv) {
77 throw std::bad_alloc();
78 }
79 return static_cast<T *>(pv);
80 }
81 void deallocate(T *const p, std::size_t) const noexcept { fftw_free(p); }
82};

200/** Pack a block (<tt>size[3]</tt> starting at <tt>start[3]</tt>) of an input
201 * 3d-grid with dimension <tt>dim[3]</tt> into an output 3d-block with
202 * dimension <tt>size[3]</tt>.
203 *
204 * The block with dimensions <tt>(size[0], size[1], size[2])</tt> is stored
205 * in 'row-major-order' or 'C-order', that means the first index is
206 * changing slowest when running through the linear array. The
207 * element (@c i0 (slow), @c i1 (mid), @c i2 (fast)) has the linear index
208 * <tt>li = i2 + size[2] * (i1 + (size[1] * i0))</tt>.
209 *
210 * \param[in] in pointer to input 3d-grid.
211 * \param[out] out pointer to output 3d-grid (block).
212 * \param[in] start start index of the block in the in-grid.
213 * \param[in] size size of the block (=dimension of the out-grid).
214 * \param[in] dim size of the in-grid.
215 * \param[in] element size of a grid element (e.g. 1 for Real, 2 for Complex).
216 */
217void fft_pack_block(double const *in, double *out, int const start[3],
218 int const size[3], int const dim[3], int element);
