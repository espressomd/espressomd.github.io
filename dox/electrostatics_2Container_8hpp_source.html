61  void on_bind_system(::System::System &system) override {

71 public:
72  Container() {
73  add_parameters({
74  {"solver",
75  [this](Variant const &v) {
76  if (m_extension) {
77  if (context()->is_head_node()) {
78  throw std::runtime_error(
