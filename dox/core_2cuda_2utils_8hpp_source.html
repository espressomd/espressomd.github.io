40/**
41 * @brief Fatal CUDA exception.
42 * Best course of action is to terminate the program immediately.
43 */
44
45class cuda_fatal_error {
46 std::string m_msg;
47 std::terminate_handler m_terminate_handler;
48
49public:
50 explicit cuda_fatal_error(std::string msg);
51
52 ~cuda_fatal_error() { terminate(); }
53
54 auto get_terminate() noexcept { return m_terminate_handler; }
55
56 auto set_terminate(std::terminate_handler callback) noexcept {
57 auto old_handler = m_terminate_handler;
58 m_terminate_handler = callback;
59 return old_handler;
60 }
61
62 void terminate() noexcept;
63
64 char const *what() const noexcept { return m_msg.c_str(); }
65};

67/**
68 * @brief Invoke a function and silently ignore any thrown
69 * @ref cuda_runtime_error error.
70 * This is useful when querying the properties of CUDA devices that may
71 * not have a suitable CUDA version, or when there is no compatible CUDA
72 * device available.
73 */
74template <class F, class... Args>
75void invoke_skip_cuda_exceptions(F &&f, Args &&...args) {
76 try {
77 return f(args...);
78 } catch (cuda_runtime_error const &) { // NOLINT(bugprone-empty-catch)
79 // pass
80 }
81}
