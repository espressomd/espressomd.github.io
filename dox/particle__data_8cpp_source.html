54 /**
55  * @brief A generic particle update.
56  *
57  * Here the sub-struct structure of Particle is
58  * used: the specification of the data member to update
59  * consists of two parts, the pointer to the substruct @p s
60  * and a pointer to a member of that substruct @p m.
61  *
62  * @tparam S Substruct type of Particle
63  * @tparam s Pointer to a member of Particle
64  * @tparam T Type of the member to update, must be serializable
65  * @tparam m Pointer to the member.
66  */
67 template <typename S, S Particle::*s, typename T, T S::*m>
68 struct UpdateParticle {
69  T value;
70 
71  void operator()(Particle &p) const { (p.*s).*m = value; }
72 
73  template <class Archive> void serialize(Archive &ar, long int) { ar &value; }
74 };

171  * @brief Delete specific bond.
172  */
173 struct RemoveBond {
174  std::vector<int> bond;
175 
176  void operator()(Particle &p) const {
177  assert(not bond.empty());
178  auto const view =
179  BondView(bond.front(), {bond.data() + 1, bond.size() - 1});
180  auto it = boost::find(p.bonds(), view);
181 
182  if (it != p.bonds().end()) {
183  p.bonds().erase(it);
184  }
185  }
186 
187  template <class Archive> void serialize(Archive &ar, long int) { ar &bond; }
188 };
189 
190 /**
191  * @brief Delete pair bonds to a specific partner
192  */
193 struct RemovePairBondsTo {
194  int other_pid;
195 
196  void operator()(Particle &p) const {
197  using Bond = std::vector<int>;
198  std::vector<Bond> to_delete;
199  for (auto b : p.bonds()) {
200  if (b.partner_ids().size() == 1 and b.partner_ids()[0] == other_pid)
201  to_delete.push_back(Bond{b.bond_id(), other_pid});
202  }
203  for (auto b : to_delete) {
204  RemoveBond{b}(p);
205  }
206  }
207  template <class Archive> void serialize(Archive &ar, long int) {
208  ar &other_pid;
209  }
210 };

309 /**
310  * @brief Visitor for message evaluation.
311  *
312  * This visitor either recurses into the active type
313  * if it is a variant type, or otherwise calls
314  * operator()(Particle&) on the active type with
315  * the particle that ought to be updated. This construction
316  * allows to nest message variants to allow for sub-
317  * categories. Those are mostly used here to differentiate
318  * the updates for the substructs of Particle.
319  */
320 struct UpdateVisitor : public boost::static_visitor<void> {
321  explicit UpdateVisitor(int id) : id(id) {}
322 
323  const int id;
324 
325  /* Recurse into sub-variants */
326  template <class... Message>
327  void operator()(const boost::variant<Message...> &msg) const {
328  boost::apply_visitor(*this, msg);
329  }
330  /* Plain messages are just called. */
331  template <typename Message> void operator()(const Message &msg) const {
332  assert(cell_structure.get_local_particle(id));
333  msg(*cell_structure.get_local_particle(id));
334  }
335 };
336 } // namespace

346 static void mpi_send_update_message_local(int node, int id) {
347  if (node == comm_cart.rank()) {
348  UpdateMessage msg{};
349  comm_cart.recv(0, some_tag, msg);
350  boost::apply_visitor(UpdateVisitor{id}, msg);
351  }
352 
353  on_particle_change();
354 }

356 REGISTER_CALLBACK(mpi_send_update_message_local)

358 /**
359  * @brief Send a particle update message.
360  *
361  * This sends the message to the node that is responsible for the particle,
362  * where
363  * @p msg is called with the particle as argument. The message then performs the
364  * change to the particle that is encoded in it. The mechanism to call a functor
365  * based on the active type of a variant is called visitation. Here we can use
366  * @c UpdateVisitor with a single templated call operator on the visitor because
367  * all message (eventually) provide the same interface. Overall this is
368  * logically equivalent to nested switch statements over the message types,
369  * where the case statements play the role of the call of the messages in our
370  * case. A general introduction can be found in the documentation of
371  * boost::variant.
372  *
373  * @param id Id of the particle to update
374  * @param msg The message
375  */
376 static void mpi_send_update_message(int id, const UpdateMessage &msg) {
377  auto const pnode = get_particle_node(id);
378 
379  mpi_call(mpi_send_update_message_local, pnode, id);
380 
381  /* If the particle is remote, send the
382  * message to the target, otherwise we
383  * can just apply the update directly. */
384  if (pnode != comm_cart.rank()) {
385  comm_cart.send(pnode, some_tag, msg);
386  } else {
387  boost::apply_visitor(UpdateVisitor{id}, msg);
388  }
389 
390  on_particle_change();
391 }

767 void auto_exclusions(int distance) {
768  /* partners is a list containing the currently found excluded particles for
769  each particle, and their distance, as an interleaved list */
770  std::unordered_map<int, std::vector<int>> partners;
771 
772  /* determine initial connectivity */
773  for (auto const &part1 : partCfg()) {
774  auto const p1 = part1.id();
775  for (auto const bond : part1.bonds()) {
776  if ((bond.partner_ids().size() == 1) and (bond.partner_ids()[0] != p1)) {
777  auto const p2 = bond.partner_ids()[0];
778  add_partner(partners[p1], p1, p2, 1);
779  add_partner(partners[p2], p2, p1, 1);
780  }
781  }
782  }
783 
784  /* calculate transient connectivity. For each of the current neighbors,
785  also exclude their close enough neighbors.
786  */
787  for (int count = 1; count < distance; count++) {
788  for (auto const &p : partCfg()) {
789  auto const p1 = p.id();
790  for (int i = 0; i < partners[p1].size(); i += 2) {
791  auto const p2 = partners[p1][i];
792  auto const dist1 = partners[p1][i + 1];
793  if (dist1 > distance)
794  continue;
795  /* loop over all partners of the partner */
796  for (int j = 0; j < partners[p2].size(); j += 2) {
797  auto const p3 = partners[p2][j];
798  auto const dist2 = dist1 + partners[p2][j + 1];
799  if (dist2 > distance)
800  continue;
801  add_partner(partners[p1], p1, p3, dist2);
802  add_partner(partners[p3], p3, p1, dist2);
803  }
804  }
805  }
806  }
807 
808  /* setup the exclusions and clear the arrays. We do not setup the exclusions
809  up there, since on_part_change clears the partCfg, so that we would have to
810  restore it continuously. Of course this could be optimized by bundling the
811  exclusions, but this is only done once and the overhead is as much as for
812  setting the bonds, which the user apparently accepted.
813  */
814  for (auto &kv : partners) {
815  auto const id = kv.first;
816  auto const partner_list = kv.second;
817  for (int j : partner_list)
818  if (id < j)
819  add_particle_exclusion(id, j);
820  }
821 }
