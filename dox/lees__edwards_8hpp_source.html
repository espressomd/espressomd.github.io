46 class Push : public UpdateOffset {
47  Utils::Vector3d const &m_box_l;
48 
49 public:
50  Push(BoxGeometry const &box, double time_step)
51  : UpdateOffset{box, time_step}, m_box_l{box.length()} {}
52 
53  void operator()(Particle &p) const {
54  // Lees-Edwards acts at the zero step for the velocity half update and at
55  // the midstep for the position.
56  //
57  // The update of the velocity at the end of the time step is triggered by
58  // the flag and occurs in @ref propagate_vel_finalize_p_inst
59  if (p.pos()[m_le.shear_plane_normal] >= m_box_l[m_le.shear_plane_normal]) {
60  p.lees_edwards_flag() = -1; // perform a negative half velocity shift
61  } else if (p.pos()[m_le.shear_plane_normal] < 0) {
62  p.lees_edwards_flag() = 1; // perform a positive half velocity shift
63  } else {
64  p.lees_edwards_flag() = 0;
65  }
66 
67  auto const dir = static_cast<double>(p.lees_edwards_flag());
68  p.v()[m_le.shear_direction] += dir * m_le.shear_velocity;
69  p.pos()[m_le.shear_direction] += dir * m_le.pos_offset;
70  p.lees_edwards_offset() -= dir * m_le.pos_offset;
71  // TODO: clarify whether the fold is needed
72  // p.pos()[m_le.shear_direction] = Algorithm::periodic_fold(
73  // p.pos()[m_le.shear_direction], m_box_l()[m_le.shear_direction]);
74  UpdateOffset::operator()(p);
75  }
76 };
