38 __global__ void split_kernel_rq(CUDA_particle_data *particles, float *r,
39  float *q, unsigned int n) {
40  auto const idx = blockDim.x * blockIdx.x + threadIdx.x;
41  if (idx >= n)
42  return;
43 
44  CUDA_particle_data p = particles[idx];
45 
46  r[3 * idx + 0] = p.p[0];
47  r[3 * idx + 1] = p.p[1];
48  r[3 * idx + 2] = p.p[2];
49 #ifdef ELECTROSTATICS
50  q[idx] = p.q;
51 #endif
52 }

126 void EspressoSystemInterface::split_particle_struct() {
127  auto const device_particles = gpu_get_particle_pointer();
128  auto const n = static_cast<unsigned int>(device_particles.size());
129  if (n == 0)
130  return;
131 
132  dim3 grid(n / 512 + 1, 1, 1);
133  dim3 block(512, 1, 1);
134 
135  if (m_needsQGpu && m_needsRGpu)
136  split_kernel_rq<<<dim3(grid), dim3(block), 0, nullptr>>>(
137  device_particles.data(), m_r_gpu_begin, m_q_gpu_begin, n);
138  else if (m_needsQGpu)
139  split_kernel_q<<<dim3(grid), dim3(block), 0, nullptr>>>(
140  device_particles.data(), m_q_gpu_begin, n);
141  else if (m_needsRGpu)
142  split_kernel_r<<<dim3(grid), dim3(block), 0, nullptr>>>(
143  device_particles.data(), m_r_gpu_begin, n);
144 #ifdef DIPOLES

147  device_particles.data(), m_dip_gpu_begin, n);

149 }
