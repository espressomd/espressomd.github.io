37 static void mpi_rotate_system_local(double phi, double theta, double alpha) {
38  auto const particles = cell_structure.local_particles();
39 
40  // Calculate center of mass
41  Utils::Vector3d local_com{};
42  double local_mass = 0.0;
43 
44  for (auto const &p : particles) {
45  if (not p.is_virtual()) {
46  local_com += p.mass() * p.pos();
47  local_mass += p.mass();
48  }
49  }
50 
51  auto const total_mass =
52  boost::mpi::all_reduce(comm_cart, local_mass, std::plus<>());
53  auto const com =
54  boost::mpi::all_reduce(comm_cart, local_com, std::plus<>()) / total_mass;
55 
56  // Rotation axis in Cartesian coordinates
57  Utils::Vector3d axis;
58  axis[0] = sin(theta) * cos(phi);
59  axis[1] = sin(theta) * sin(phi);
60  axis[2] = cos(theta);
61 
62  // Rotate particle coordinates
63  for (auto &p : particles) {
64  // Move the center of mass of the system to the origin
65  p.pos() = com + Utils::vec_rotate(axis, alpha, p.pos() - com);
66 #ifdef ROTATION
67  local_rotate_particle(p, axis, alpha);
68 #endif
69  }
70 
71  cell_structure.set_resort_particles(Cells::RESORT_GLOBAL);
72  on_particle_change();
73  update_dependent_particles();
74 }
