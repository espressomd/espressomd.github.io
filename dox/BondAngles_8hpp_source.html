39 /** Calculate bond angles between particles in a polymer.
40  * For @f$ n @f$ bonded particles, return the @f$ n-2 @f$ angles along the
41  * chain, in radians.
42  */
43 class BondAngles : public PidObservable {
44 public:
45  using PidObservable::PidObservable;
46  explicit BondAngles(std::vector<int> ids) : PidObservable(std::move(ids)) {
47  if (this->ids().size() < 3)
48  throw std::runtime_error("At least 3 particles are required");
49  }
50 
51  std::vector<double>
52  evaluate(ParticleReferenceRange particles,
53  const ParticleObservables::traits<Particle> &traits) const override {
54  std::vector<double> res(n_values());
55  auto v1 = box_geo.get_mi_vector(traits.position(particles[1]),
56  traits.position(particles[0]));
57  auto n1 = v1.norm();
58  for (std::size_t i = 0, end = n_values(); i < end; i++) {
59  auto v2 = box_geo.get_mi_vector(traits.position(particles[i + 2]),
60  traits.position(particles[i + 1]));
61  auto const n2 = v2.norm();
62  auto const cosine = boost::algorithm::clamp(
63  (v1 * v2) / (n1 * n2), -TINY_COS_VALUE, TINY_COS_VALUE);
64  /* to reduce computational time, after calculating an angle ijk, the
65  * vector r_ij takes the value r_jk, but to orient it correctly, it has
66  * to be multiplied -1; it's cheaper to do this operation on a double
67  * than on a vector of doubles
68  */
69  res[i] = acos(-cosine);
70  v1 = v2;
71  n1 = n2;
72  }
73  return res;
74  }
75  std::vector<std::size_t> shape() const override {
76  assert(ids().size() >= 2);
77  return {ids().size() - 2};
78  }
79 };
