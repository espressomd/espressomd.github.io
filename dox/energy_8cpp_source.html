47std::shared_ptr<Observable_stat> System::calculate_energy() {
48
49 auto obs_energy_ptr = std::make_shared<Observable_stat>(
50 1ul, static_cast<std::size_t>(bonded_ias->get_next_key()),
51 nonbonded_ias->get_max_seen_particle_type());
52
53 if (long_range_interactions_sanity_checks()) {
54 return obs_energy_ptr;
55 }
56
57 auto &obs_energy = *obs_energy_ptr;
58#if defined(ESPRESSO_CUDA) and \
59 (defined(ESPRESSO_ELECTROSTATICS) or defined(ESPRESSO_DIPOLES))
60 gpu->clear_energy_on_device();
61 gpu->update();
62#endif
63 on_observable_calc();
64
65 auto const local_parts = cell_structure->local_particles();
66
67 for (auto const &p : local_parts) {
68 obs_energy.kinetic_lin[0] += translational_kinetic_energy(p);
69 obs_energy.kinetic_rot[0] += rotational_kinetic_energy(p);
70 }
71
72 auto const coulomb_kernel = coulomb.pair_energy_kernel();
73 auto const dipoles_kernel = dipoles.pair_energy_kernel();
74
75 short_range_loop(
76 [this, coulomb_kernel_ptr = get_ptr(coulomb_kernel), &obs_energy](
77 Particle const &p1, int bond_id, std::span<Particle *> partners) {
78 auto const &iaparams = *bonded_ias->at(bond_id);
79 auto const result = calc_bonded_energy(iaparams, p1, partners, *box_geo,
80 coulomb_kernel_ptr);
81 if (result) {
82 obs_energy.bonded_contribution(bond_id)[0] += result.value();
83 return false;
84 }
85 return true;
86 },
87 [coulomb_kernel_ptr = get_ptr(coulomb_kernel),
88 dipoles_kernel_ptr = get_ptr(dipoles_kernel), this,
89 &obs_energy](Particle const &p1, Particle const &p2, Distance const &d) {
90 auto const &ia_params =
91 nonbonded_ias->get_ia_param(p1.type(), p2.type());
92 add_non_bonded_pair_energy(
93 p1, p2, d.vec21, sqrt(d.dist2), d.dist2, ia_params, *bonded_ias,
94 coulomb, coulomb_kernel_ptr, dipoles_kernel_ptr, obs_energy);
95 },
96 *cell_structure, maximal_cutoff(), bonded_ias->maximal_cutoff());
97
98#ifdef ESPRESSO_ELECTROSTATICS
99 /* calculate k-space part of electrostatic interaction. */
100 obs_energy.coulomb[1] = coulomb.calc_energy_long_range();
101#endif
102
103#ifdef ESPRESSO_DIPOLES
104 /* calculate k-space part of magnetostatic interaction. */
105 obs_energy.dipolar[1] = dipoles.calc_energy_long_range();
106#endif
107
108 constraints->add_energy(local_parts, get_sim_time(), obs_energy);
109
110#if defined(ESPRESSO_CUDA) and \
111 (defined(ESPRESSO_ELECTROSTATICS) or defined(ESPRESSO_DIPOLES))
112 auto const energy_host = gpu->copy_energy_to_host();
113 if (!obs_energy.coulomb.empty())
114 obs_energy.coulomb[1] += static_cast<double>(energy_host.coulomb);
115 if (!obs_energy.dipolar.empty())
116 obs_energy.dipolar[1] += static_cast<double>(energy_host.dipolar);
117#endif
118
119 obs_energy.mpi_reduce();
120 return obs_energy_ptr;
121 // NOLINTNEXTLINE(clang-analyzer-cplusplus.NewDeleteLeaks)
122}

124double System::particle_short_range_energy_contribution(int pid) {
125 if (cell_structure->get_resort_particles()) {
126 cell_structure->update_ghosts_and_resort_particle(get_global_ghost_flags());
127 }
128
129 auto ret = 0.0;
130 if (auto const p = cell_structure->get_local_particle(pid)) {
131 auto const coulomb_kernel = coulomb.pair_energy_kernel();
132 auto kernel = [&ret, this](Particle const &p, Particle const &p1,
133 Utils::Vector3d const &vec) {
134#ifdef ESPRESSO_EXCLUSIONS
135 if (not do_nonbonded(p, p1))
136 return;
137#endif
138 auto const &ia_params = nonbonded_ias->get_ia_param(p.type(), p1.type());
139 // Add energy for current particle pair to result
140 ret += calc_non_bonded_pair_energy(p, p1, ia_params, vec, vec.norm(),
141 *bonded_ias, coulomb, nullptr);
142 };
143 cell_structure->run_on_particle_short_range_neighbors(*p, kernel);
144 }
145 return ret;
146}

148std::optional<double> System::particle_bond_energy(int pid, int bond_id,
149 std::vector<int> partners) {
150 if (cell_structure->get_resort_particles()) {
151 cell_structure->update_ghosts_and_resort_particle(get_global_ghost_flags());
152 }
153 Particle const *p = cell_structure->get_local_particle(pid);
154 if (not p or p->is_ghost())
155 return {}; // not available on this MPI rank or ghost
156 auto const &iaparams = *bonded_ias->at(bond_id);
157 try {
158 auto resolved_partners = cell_structure->resolve_bond_partners(partners);
159 auto const coulomb_kernel = coulomb.pair_energy_kernel();
160 return calc_bonded_energy(
161 iaparams, *p,
162 std::span(resolved_partners.data(), resolved_partners.size()), *box_geo,
163 get_ptr(coulomb_kernel));
164 } catch (const BondResolutionError &) {
165 bond_broken_error(p->id(), partners);
166 return {};
167 }
168}
