59 /**
60  * @brief Resize a @c thrust::device_vector.
61  *
62  * Due to a bug in thrust (https://github.com/thrust/thrust/issues/939),
63  * resizing or appending to default constructed containers causes undefined
64  * behavior by dereferencing a null-pointer for certain types. This
65  * function is used instead of the resize member function to side-step
66  * the problem. This is done by replacing the existing vector by a new
67  * one constructed with the desired size if resizing from capacity zero.
68  * Behaves as-if @c vec.resize(n) was called.
69  * This is fixed in Thrust 1.11, shipped in CUDA 11.3
70  * (https://github.com/NVIDIA/thrust/commit/1c4f25d9).
71  *
72  * @tparam T Type contained in the vector.
73  * @param vec Vector to resize.
74  * @param n Desired new size of the vector.
75  */
76 template <class T>
77 void resize_or_replace(thrust::device_vector<T> &vec, std::size_t n) {
78  if (vec.capacity() == 0) {
79  vec = thrust::device_vector<T>(n);
80  } else {
81  vec.resize(n);
82  }
83 }

85 template <typename T> void free_device_vector(thrust::device_vector<T> &vec) {
86  vec.clear();
87  thrust::device_vector<T>().swap(vec);
88 }

224 void GpuParticleData::Storage::copy_particles_to_device() {
225  // resize buffers
226  auto const n_part = particle_data_host.size();
227  resize_or_replace(particle_data_device, n_part);
228  particle_forces_host.resize(3ul * n_part);
229  resize_or_replace(particle_forces_device, 3ul * n_part);
230 #ifdef ROTATION
231  particle_torques_host.resize(3ul * n_part);
232  resize_or_replace(particle_torques_device, 3ul * n_part);
233 #endif
234 
235  // zero out device memory for forces and torques
236  cudaMemsetAsync(raw_data_pointer(particle_forces_device), 0x0,
237  byte_size(particle_forces_device), stream[0]);
238 #ifdef ROTATION
239  cudaMemsetAsync(raw_data_pointer(particle_torques_device), 0x0,
240  byte_size(particle_torques_device), stream[0]);
241 #endif
242 
243  // copy particles to device
244  cudaMemcpyAsync(raw_data_pointer(particle_data_device),
245  particle_data_host.data(), byte_size(particle_data_host),
246  cudaMemcpyHostToDevice, stream[0]);
247 }

305 __global__ void split_kernel_r(GpuParticleData::GpuParticle *particles,
306  float *r, std::size_t n) {
307  auto idx = blockDim.x * blockIdx.x + threadIdx.x;
308  if (idx >= n)
309  return;
310 
311  auto const &p = particles[idx];
312  idx *= 3u;
313  r[idx + 0u] = p.p[0u];
314  r[idx + 1u] = p.p[1u];
315  r[idx + 2u] = p.p[2u];
316 }

363 void GpuParticleData::Storage::split_particle_struct() {
364  auto const n_part = particle_data_device.size();
365  if (n_part == 0ul)
366  return;
367 
368  using prop = GpuParticleData::prop;
369  dim3 const threadsPerBlock{512u, 1u, 1u};
370  dim3 const numBlocks{static_cast<unsigned>(n_part / threadsPerBlock.x + 1ul)};
371 
372 #ifdef ELECTROSTATICS
373  if (m_need[prop::q] and m_need[prop::pos]) {
374  split_kernel_rq<<<numBlocks, threadsPerBlock, 0, nullptr>>>(
375  raw_data_pointer(particle_data_device), particle_pos_device,
376  particle_q_device, n_part);
377  } else if (m_need[prop::q]) {
378  split_kernel_q<<<numBlocks, threadsPerBlock, 0, nullptr>>>(
379  raw_data_pointer(particle_data_device), particle_q_device, n_part);
380  } else
381 #endif
382  if (m_need[prop::pos]) {
383  split_kernel_r<<<numBlocks, threadsPerBlock, 0, nullptr>>>(
384  raw_data_pointer(particle_data_device), particle_pos_device, n_part);
385  }
386 #ifdef DIPOLES
387  if (m_need[prop::dip]) {
388  split_kernel_dip<<<numBlocks, threadsPerBlock, 0, nullptr>>>(
389  raw_data_pointer(particle_data_device), particle_dip_device, n_part);
390  }
391 #endif
392 }

394 void GpuParticleData::Storage::realloc_device_memory() {
395  using prop = GpuParticleData::prop;
396  auto const new_size = particle_data_device.size();
397  auto const resize_needed = new_size != current_size;
398  if (m_need[prop::pos] and (resize_needed or particle_pos_device == nullptr)) {
399  if (particle_pos_device != nullptr) {
400  cuda_safe_mem(cudaFree(particle_pos_device));
401  }
402  cuda_safe_mem(
403  cudaMalloc(&particle_pos_device, 3ul * new_size * sizeof(float)));
404  }
405 #ifdef DIPOLES
406  if (m_need[prop::dip] and (resize_needed or particle_dip_device == nullptr)) {
407  if (particle_dip_device != nullptr) {
408  cuda_safe_mem(cudaFree(particle_dip_device));
409  }
410  cuda_safe_mem(
411  cudaMalloc(&particle_dip_device, 3ul * new_size * sizeof(float)));
412  }
413 #endif
414 #ifdef ELECTROSTATICS
415  if (m_need[prop::q] and (resize_needed or particle_q_device == nullptr)) {
416  if (particle_q_device != nullptr) {
417  cuda_safe_mem(cudaFree(particle_q_device));
418  }
419  cuda_safe_mem(cudaMalloc(&particle_q_device, new_size * sizeof(float)));
420  }
421 #endif
422  current_size = new_size;
423 }
