63 /** @brief Parameters for the ELC method */
64 struct elc_data {
65  elc_data(double maxPWerror, double gap_size, double far_cut, bool neutralize,
66  double delta_top, double delta_bot, bool const_pot, double pot_diff);
67 
68  /**
69  * @brief Maximal allowed pairwise error for the potential and force.
70  * Note that this counts for the plain 1/r contribution
71  * alone, without the prefactor and the charge prefactor.
72  */
73  double maxPWerror;
74  /** Size of the empty gap. Note that ELC relies on the user to make sure
75  * that this condition is fulfilled.
76  */
77  double gap_size;
78  /** Up to where particles can be found. */
79  double box_h;
80  /**
81  * @brief Cutoff of the exponential sum.
82  * Since in all other MMM methods this is the far formula,
83  * it is given the same name here.
84  */
85  double far_cut;
86  /** Squared value of #far_cut. */
87  double far_cut2;
88  /** Flag whether #far_cut was set by the user, or calculated by ESPResSo.
89  * In the latter case, the cutoff will be adapted if important parameters,
90  * such as the box dimensions, change.
91  */
92  bool far_calculated;
93 
94  /// @brief Flag whether there is any dielectric contrast in the system.
95  bool dielectric_contrast_on;
96  /// @brief Flag whether a constant potential difference is applied.
97  bool const_pot;
98  /**
99  * @brief Flag whether the box is neutralized by a homogeneous background.
100  * If true, use a homogeneous neutralizing background for non-neutral
101  * systems. Unlike the 3D case, this background adds an additional
102  * force pointing towards the system center, and the gap size
103  * enters into the value of the forces, so be careful with this.
104  */
105  bool neutralize;
106 
107  /// dielectric contrast in the upper part of the simulation cell.
108  double delta_mid_top;
109  /// dielectric contrast in the lower part of the simulation cell.
110  double delta_mid_bot;
111  /// @brief Constant potential difference.
112  double pot_diff;
113 
114  /** Layer around the dielectric contrast in which we trick around. */
115  double space_layer;
116  /** The space that is finally left. */
117  double space_box;
118 
119  /// pairwise contributions from the lowest and top layers
120  template <typename Kernel>
121  void dielectric_layers_contribution(CoulombP3M const &p3m,
122  Utils::Vector3d const &pos1,
123  Utils::Vector3d const &pos2, double q1q2,
124  Kernel &&kernel) const {
125  if (pos1[2] < space_layer) {
126  auto const q_eff = delta_mid_bot * q1q2;
127  auto const d = get_mi_vector(pos2, {pos1[0], pos1[1], -pos1[2]});
128  kernel(q_eff, d);
129  }
130  if (pos1[2] > (box_h - space_layer)) {
131  auto const q_eff = delta_mid_top * q1q2;
132  auto const l = 2. * box_h;
133  auto const d = get_mi_vector(pos2, {pos1[0], pos1[1], l - pos1[2]});
134  kernel(q_eff, d);
135  }
136  }
137 
138  /// self energies of top and bottom layers with their virtual images
139  double dielectric_layers_self_energy(CoulombP3M const &p3m,
140  ParticleRange const &particles) const {
141  auto energy = 0.;
142  for (auto const &p : particles) {
143  dielectric_layers_contribution(
144  p3m, p.pos(), p.pos(), Utils::sqr(p.q()),
145  [&](double q1q2, Utils::Vector3d const &d) {
146  energy += p3m.pair_energy(q1q2, d.norm());
147  });
148  }
149  return energy;
150  }
151 
152  /// forces of particles in border layers with themselves
153  void dielectric_layers_self_forces(CoulombP3M const &p3m,
154  ParticleRange const &particles) const {
155  for (auto &p : particles) {
156  dielectric_layers_contribution(
157  p3m, p.pos(), p.pos(), Utils::sqr(p.q()),
158  [&](double q1q2, Utils::Vector3d const &d) {
159  p.force() += p3m.pair_force(q1q2, d, d.norm());
160  });
161  }
162  }
163 
164 private:
165  Utils::Vector3d get_mi_vector(Utils::Vector3d const &a,
166  Utils::Vector3d const &b) const;
167 };

184  void on_activation() {
185  sanity_checks_periodicity();
186  sanity_checks_cell_structure();
187  sanity_checks_charge_neutrality();
188  sanity_checks_dielectric_contrasts();
189  /* Most ELC parameters do not depend on the P3M parameters,
190  * but the P3M parameters depend on the ELC parameters during tuning,
191  * therefore ELC needs to be tuned before P3M. */
192  recalc_box_h();
193  recalc_far_cut();
194  visit_base_solver([](auto &solver) { solver->on_activation(); });
195  /* With dielectric contrasts, the ELC space layer depends
196  * on the P3M real-space cutoff, and the P3M FFT parameters
197  * depend on the ELC space layer */
198  if (elc.dielectric_contrast_on) {
199  recalc_space_layer();
200  visit_base_solver([](auto &actor) { actor->init(); });
201  }
202  }
203  /** @brief Recalculate all box-length-dependent parameters. */
204  void on_boxl_change() {
205  visit_base_solver([](auto &actor) { actor->on_boxl_change(); });
206  recalc_box_h();
207  recalc_far_cut();
208  recalc_space_layer();
209  }

243  /**
244  * @brief Veto real-space cutoff values that are incompatible with ELC.
245  * When ELC is used with dielectric contrasts, the short-range cutoff needs
246  * to be smaller than the gap size to allow placement of the image charges.
247  */
248  boost::optional<std::string> veto_r_cut(double r_cut) const {
249  if (elc.dielectric_contrast_on and r_cut >= elc.gap_size) {
250  return {std::string("conflict with ELC w/ dielectric contrasts")};
251  }
252  return {};
253  }

255  /** @brief Calculate short-range pair energy correction. */
256  double pair_energy_correction(double q1q2, Particle const &p1,
257  Particle const &p2) const {
258  double energy = 0.;
259  if (elc.dielectric_contrast_on) {
260  energy = boost::apply_visitor(
261  [this, &p1, &p2, q1q2](auto &p3m_ptr) {
262  auto const &pos1 = p1.pos();
263  auto const &pos2 = p2.pos();
264  auto const &p3m = *p3m_ptr;
265  auto energy = 0.;
266  elc.dielectric_layers_contribution(
267  p3m, pos1, pos2, q1q2,
268  [&](double q_eff, Utils::Vector3d const &d) {
269  energy += p3m.pair_energy(q_eff, d.norm());
270  });
271  elc.dielectric_layers_contribution(
272  p3m, pos2, pos1, q1q2,
273  [&](double q_eff, Utils::Vector3d const &d) {
274  energy += p3m.pair_energy(q_eff, d.norm());
275  });
276  return energy / 2.;
277  },
278  base_solver);
279  }
280  return energy;
281  }
282 
283  /** @brief Add short-range pair force corrections. */
284  void add_pair_force_corrections(Particle &p1, Particle &p2,
285  double q1q2) const {
286  if (elc.dielectric_contrast_on) {
287  boost::apply_visitor(
288  [this, &p1, &p2, q1q2](auto &p3m_ptr) {
289  auto const &pos1 = p1.pos();
290  auto const &pos2 = p2.pos();
291  auto const &p3m = *p3m_ptr;
292  elc.dielectric_layers_contribution(
293  p3m, pos1, pos2, q1q2,
294  [&](double q_eff, Utils::Vector3d const &d) {
295  p1.force() += p3m.pair_force(q_eff, d, d.norm());
296  });
297  elc.dielectric_layers_contribution(
298  p3m, pos2, pos1, q1q2,
299  [&](double q_eff, Utils::Vector3d const &d) {
300  p2.force() += p3m.pair_force(q_eff, d, d.norm());
301  });
302  },
303  base_solver);
304  }
305  }
