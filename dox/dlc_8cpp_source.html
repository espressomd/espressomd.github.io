68 /** Calculate the maximal dipole moment in the system */
69 static double calc_mu_max() {
70  auto const local_particles = cell_structure.local_particles();
71  auto const mu_max_local = std::accumulate(
72  local_particles.begin(), local_particles.end(), 0.,
73  [](double mu, Particle const &p) { return std::max(mu, p.dipm()); });
74 
75  return boost::mpi::all_reduce(comm_cart, mu_max_local,
76  boost::mpi::maximum<double>());

79 static double g1_DLC_dip(double g, double x) {
80  auto const c = g / x;
81  auto const cc2 = c * c;
82  auto const x3 = x * x * x;
83  auto const a = g * g * g / x + 1.5 * cc2 + 1.5 * g / x3 + 0.75 / (x3 * x);
84  return a;
85 }

87 static double g2_DLC_dip(double g, double x) {
88  auto const x2 = x * x;
89  auto const a = g * g / x + 2.0 * g / x2 + 2.0 / (x2 * x);
90  return a;
91 }

106 /**
107  * @brief Compute the dipolar force and torque corrections.
108  * %Algorithm implemented accordingly to @cite brodka04a.
109  */
110 static void dipolar_force_corrections(int kcut,
111  std::vector<Utils::Vector3d> &fs,
112  std::vector<Utils::Vector3d> &ts,
113  ParticleRange const &particles) {
114  auto const facux = 2. * Utils::pi() * box_geo.length_inv()[0];
115  auto const facuy = 2. * Utils::pi() * box_geo.length_inv()[1];
116 
117  auto const n_local_particles = particles.size();
118  std::vector<double> ReSjp(n_local_particles);
119  std::vector<double> ReSjm(n_local_particles);
120  std::vector<double> ImSjp(n_local_particles);
121  std::vector<double> ImSjm(n_local_particles);
122  std::vector<double> ReGrad_Mup(n_local_particles);
123  std::vector<double> ImGrad_Mup(n_local_particles);
124  std::vector<double> ReGrad_Mum(n_local_particles);
125  std::vector<double> ImGrad_Mum(n_local_particles);
126 
127  for (int ix = -kcut; ix <= +kcut; ix++) {
128  for (int iy = -kcut; iy <= +kcut; iy++) {
129  if (ix == 0 and iy == 0) {
130  continue;
131  }
132  auto const gx = static_cast<double>(ix) * facux;
133  auto const gy = static_cast<double>(iy) * facuy;
134  auto const gr = sqrt(gx * gx + gy * gy);
135 
136  // We assume short slab direction is in the z-direction
137  auto const fa1 = 1. / (gr * (exp(gr * box_geo.length()[2]) - 1.));
138 
139  // ... Compute S+,(S+)*,S-,(S-)*, and Spj,Smj for the current g
140 
141  std::size_t ip = 0;
142  double S[4] = {0., 0., 0., 0.}; // S of Brodka method, or is S[4] =
143  // {Re(S+), Im(S+), Re(S-), Im(S-)}
144  for (auto const &p : particles) {
145  if (p.dipm() != 0.) {
146  auto const &pos = p.pos();
147  auto const dip = p.calc_dip();
148 
149  auto const a = gx * dip[0] + gy * dip[1];
150  auto const b = gr * dip[2];
151  auto const er = gx * pos[0] + gy * pos[1];
152  auto const ez = gr * pos[2];
153  auto const c = cos(er);
154  auto const d = sin(er);
155  auto const f = exp(ez);
156 
157  ReSjp[ip] = (b * c - a * d) * f;
158  ImSjp[ip] = (c * a + b * d) * f;
159  ReSjm[ip] = (-b * c - a * d) / f;
160  ImSjm[ip] = (c * a - b * d) / f;
161  ReGrad_Mup[ip] = c * f;
162  ReGrad_Mum[ip] = c / f;
163  ImGrad_Mup[ip] = d * f;
164  ImGrad_Mum[ip] = d / f;
165 
166  S[0] += ReSjp[ip];
167  S[1] += ImSjp[ip];
168  S[2] += ReSjm[ip];
169  S[3] += ImSjm[ip];
170  }
171  ip++;
172  }
173 
174  MPI_Allreduce(MPI_IN_PLACE, S, 4, MPI_DOUBLE, MPI_SUM, comm_cart);
175 
176  // ... Now we can compute the contributions to E,Fj,Ej for the current
177  // g-value
178  ip = 0;
179  for (auto &p : particles) {
180  if (p.dipm() != 0.) {
181  {
182  // compute contributions to the forces
183  auto const s1 = -(-ReSjp[ip] * S[3] + ImSjp[ip] * S[2]);
184  auto const s2 = +(ReSjm[ip] * S[1] - ImSjm[ip] * S[0]);
185  auto const s3 = -(-ReSjm[ip] * S[1] + ImSjm[ip] * S[0]);
186  auto const s4 = +(ReSjp[ip] * S[3] - ImSjp[ip] * S[2]);
187 
188  auto const s1z = +(ReSjp[ip] * S[2] + ImSjp[ip] * S[3]);
189  auto const s2z = -(ReSjm[ip] * S[0] + ImSjm[ip] * S[1]);
190  auto const s3z = -(ReSjm[ip] * S[0] + ImSjm[ip] * S[1]);
191  auto const s4z = +(ReSjp[ip] * S[2] + ImSjp[ip] * S[3]);
192 
193  auto const ss = s1 + s2 + s3 + s4;
194  fs[ip][0] += fa1 * gx * ss;
195  fs[ip][1] += fa1 * gy * ss;
196  fs[ip][2] += fa1 * gr * (s1z + s2z + s3z + s4z);
197  }
198  {
199  // compute contributions to the electrical field
200  auto const s1 = -(-ReGrad_Mup[ip] * S[3] + ImGrad_Mup[ip] * S[2]);
201  auto const s2 = +(ReGrad_Mum[ip] * S[1] - ImGrad_Mum[ip] * S[0]);
202  auto const s3 = -(-ReGrad_Mum[ip] * S[1] + ImGrad_Mum[ip] * S[0]);
203  auto const s4 = +(ReGrad_Mup[ip] * S[3] - ImGrad_Mup[ip] * S[2]);
204 
205  auto const s1z = +(ReGrad_Mup[ip] * S[2] + ImGrad_Mup[ip] * S[3]);
206  auto const s2z = -(ReGrad_Mum[ip] * S[0] + ImGrad_Mum[ip] * S[1]);
207  auto const s3z = -(ReGrad_Mum[ip] * S[0] + ImGrad_Mum[ip] * S[1]);
208  auto const s4z = +(ReGrad_Mup[ip] * S[2] + ImGrad_Mup[ip] * S[3]);
209 
210  auto const ss = s1 + s2 + s3 + s4;
211  ts[ip][0] += fa1 * gx * ss;
212  ts[ip][1] += fa1 * gy * ss;
213  ts[ip][2] += fa1 * gr * (s1z + s2z + s3z + s4z);
214  }
215  }
216  ++ip;
217  }
218  }
219  }
220 
221  // Convert from the corrections to the electrical field to the corrections
222  // for the torques
223  std::size_t ip = 0;
224  for (auto const &p : particles) {
225  if (p.dipm() != 0.) {
226  ts[ip] = vector_product(p.calc_dip(), ts[ip]);
227  }
228  ++ip;
229  }
230 
231  // Multiply by the factors we have left during the loops
232 
233  auto const piarea =
234  Utils::pi() * box_geo.length_inv()[0] * box_geo.length_inv()[1];
235  for (std::size_t j = 0; j < n_local_particles; ++j) {
236  fs[j] *= piarea;
237  ts[j] *= piarea;
238  }
239 }

241 /**
242  * @brief Compute the dipolar DLC energy correction.
243  * %Algorithm implemented accordingly to @cite brodka04a.
244  */
245 static double dipolar_energy_correction(int kcut,
246  ParticleRange const &particles) {
247  auto const facux = 2. * Utils::pi() * box_geo.length_inv()[0];
248  auto const facuy = 2. * Utils::pi() * box_geo.length_inv()[1];
249 
250  double energy = 0.;
251  double sum_S[4] = {0., 0., 0., 0.};
252  for (int ix = -kcut; ix <= +kcut; ix++) {
253  for (int iy = -kcut; iy <= +kcut; iy++) {
254  if ((ix == 0 && iy == 0)) {
255  continue;
256  }
257  auto const gx = static_cast<double>(ix) * facux;
258  auto const gy = static_cast<double>(iy) * facuy;
259  auto const gr = sqrt(gx * gx + gy * gy);
260 
261  // We assume short slab direction is in the z-direction
262  auto const fa1 = 1. / (gr * (exp(gr * box_geo.length()[2]) - 1.));
263 
264  // ... Compute S+,(S+)*,S-,(S-)*, and Spj,Smj for the current g
265 
266  double S[4] = {0., 0., 0., 0.}; // S of Brodka method, or is S[4] =
267  // {Re(S+), Im(S+), Re(S-), Im(S-)}
268  for (auto const &p : particles) {
269  if (p.dipm() != 0.) {
270  auto const &pos = p.pos();
271  auto const dip = p.calc_dip();
272 
273  auto const a = gx * dip[0] + gy * dip[1];
274  auto const b = gr * dip[2];
275  auto const er = gx * pos[0] + gy * pos[1];
276  auto const ez = gr * pos[2];
277  auto const c = cos(er);
278  auto const d = sin(er);
279  auto const f = exp(ez);
280 
281  S[0] += (b * c - a * d) * f;
282  S[1] += (c * a + b * d) * f;
283  S[2] += (-b * c - a * d) / f;
284  S[3] += (c * a - b * d) / f;
285  }
286  }
287  boost::mpi::reduce(comm_cart, S, 4, sum_S, std::plus<>(), 0);
288 
289  // compute contribution to the energy
290  // s2=(ReSm*ReSp+ImSm*ImSp); s2=s1!!!
291  energy += fa1 * 2. * (sum_S[0] * sum_S[2] + sum_S[1] * sum_S[3]);
292  }
293  }
294 
295  auto const piarea =
296  Utils::pi() * box_geo.length_inv()[0] * box_geo.length_inv()[1];
297  energy *= -piarea;
298  return (this_node == 0) ? energy : 0.;
299 }

301 void DipolarLayerCorrection::add_force_corrections(
302  ParticleRange const &particles) const {
303  assert(dlc.far_cut > 0.);
304  auto const volume = box_geo.volume();
305  auto const correc = 4. * Utils::pi() / volume;
306 
307  // --- Create arrays that should contain the corrections to
308  // the forces and torques, and set them to zero.
309  std::vector<Utils::Vector3d> dip_DLC_f(particles.size());
310  std::vector<Utils::Vector3d> dip_DLC_t(particles.size());
311 
312  //---- Compute the corrections ----------------------------------
313 
314  // First the DLC correction
315  auto const kcut = static_cast<int>(std::round(dlc.far_cut));
316  dipolar_force_corrections(kcut, dip_DLC_f, dip_DLC_t, particles);
317 
318  // Now we compute the correction like Yeh and Klapp to take into account
319  // the fact that you are using a 3D PBC method which uses spherical
320  // summation instead of slab-wise summation.
321  // Slab-wise summation is the one required to apply DLC correction.
322  // This correction is often called SDC = Shape Dependent Correction.
323  // See @cite brodka04a.
324 
325  auto const box_dip = calc_slab_dipole(particles);
326 
327  // --- Transfer the computed corrections to the Forces, Energy and torques
328  // of the particles
329 
330  std::size_t ip = 0;
331  for (auto &p : particles) {
332  check_gap(p);
333 
334  if (p.dipm() != 0.) {
335  // SDC correction term is zero for the forces
336  p.force() += prefactor * dip_DLC_f[ip];
337 
338  auto const dip = p.calc_dip();
339  // SDC correction for the torques
340  auto d = Utils::Vector3d{0., 0., -correc * box_dip[2]};
341 #ifdef DP3M
342  if (epsilon != P3M_EPSILON_METALLIC) {
343  d += correc * epsilon_correction * box_dip;
344  }
345 #endif
346  p.torque() += prefactor * (dip_DLC_t[ip] + vector_product(dip, d));
347  }
348  ++ip;
349  }
350 }

352 double DipolarLayerCorrection::energy_correction(
353  ParticleRange const &particles) const {
354  assert(dlc.far_cut > 0.);
355  auto const volume = box_geo.volume();
356  auto const pref = prefactor * 2. * Utils::pi() / volume;
357 
358  // Check if particles aren't in the forbidden gap region
359  // This loop is needed, because there is no other guaranteed
360  // single pass over all particles in this function.
361  for (auto const &p : particles) {
362  check_gap(p);
363  }
364 
365  //---- Compute the corrections ----------------------------------
366 
367  // First the DLC correction
368  auto const k_cut = static_cast<int>(std::round(dlc.far_cut));
369  auto dip_DLC_energy = prefactor * dipolar_energy_correction(k_cut, particles);
370 
371  // Now we compute the correction like Yeh and Klapp to take into account
372  // the fact that you are using a 3D PBC method which uses spherical
373  // summation instead of slab-wise summation.
374  // Slab-wise summation is the one required to apply DLC correction.
375  // This correction is often called SDC = Shape Dependent Correction.
376  // See @cite brodka04a.
377 
378  auto const box_dip = calc_slab_dipole(particles);
379 
380  if (this_node == 0) {
381  dip_DLC_energy += pref * Utils::sqr(box_dip[2]);
382 #ifdef DP3M
383  if (epsilon != P3M_EPSILON_METALLIC) {
384  dip_DLC_energy -= pref * epsilon_correction * box_dip.norm2();
385  }
386 #endif
387  return dip_DLC_energy;
388  }
389  return 0.;
390 }

404 /** Compute the cut-off in the DLC dipolar part to get a certain accuracy.
405  * We assume particles to have all the same value of the dipolar momentum
406  * modulus, which is taken as the largest value of mu inside the system.
407  * If we assume the gap has a width @c gap_size (within
408  * which there is no particles): <tt>Lz = h + gap_size</tt>
409  */
410 double DipolarLayerCorrection::tune_far_cut() const {
411  /* we take the maximum dipole in the system, to be sure that the errors
412  * in the other case will be equal or less than for this one */
413  auto const mu_max_sq = Utils::sqr(calc_mu_max());
414  auto const lx = box_geo.length()[0];
415  auto const ly = box_geo.length()[1];
416  auto const lz = box_geo.length()[2];
417 
418  if (std::abs(lx - ly) > 0.001) {
419  throw std::runtime_error("DLC tuning: box size in x direction is "
420  "different from y direction. The tuning "
421  "formula requires both to be equal.");
422  }
423 
424  auto constexpr limitkc = 200;
425  auto const piarea = Utils::pi() / (lx * ly);
426  auto const nmp = static_cast<double>(count_magnetic_particles());
427  auto const h = dlc.box_h;
428  auto far_cut = -1.;
429  for (int kc = 1; kc < limitkc; kc++) {
430  auto const gc = kc * 2. * Utils::pi() / lx;
431  auto const fa0 = sqrt(9. * exp(+2. * gc * h) * g1_DLC_dip(gc, lz - h) +
432  9. * exp(-2. * gc * h) * g1_DLC_dip(gc, lz + h) +
433  22. * g1_DLC_dip(gc, lz));
434  auto const fa1 = sqrt(0.125 * piarea) * fa0;
435  auto const fa2 = g2_DLC_dip(gc, lz);
436  auto const de = nmp * mu_max_sq / (4. * (exp(gc * lz) - 1.)) * (fa1 + fa2);
437  if (de < dlc.maxPWerror) {
438  far_cut = static_cast<double>(kc);
439  break;
440  }
441  }
442  if (far_cut <= 0.) {
443  throw std::runtime_error("DLC tuning failed: maxPWerror too small");
444  }
445  return far_cut;
446 }
