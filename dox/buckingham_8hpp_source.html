43 /**Resultant Force due to a Buckingham potential between two particles at
44  * interatomic separation r greater than or equal to discont*/
45 inline double buck_force_r(double A, double B, double C, double D, double r) {
46  return (A * B * exp(-B * r) - 6.0 * C / pow(r, 7) - 4.0 * D / pow(r, 5));
47 }
48 
49 /**Potential Energy due to a Buckingham potential between two particles at
50  * interatomic separation r greater than or equal to discont*/
51 inline double buck_energy_r(double A, double B, double C, double D,
52  double shift, double r) {
53  return (A * exp(-B * r) - C / pow(r, 6) - D / pow(r, 4) + shift);
54 }

56 /** Calculate Buckingham force factor */
57 inline double buck_pair_force_factor(IA_parameters const &ia_params,
58  double dist) {
59  if (dist < ia_params.buckingham.cut) {
60  /* case: resulting force/energy greater than discontinuity and
61  less than cutoff (true Buckingham region) */
62  double fac;
63  if (dist > ia_params.buckingham.discont) {
64  fac = buck_force_r(ia_params.buckingham.A, ia_params.buckingham.B,
65  ia_params.buckingham.C, ia_params.buckingham.D, dist) /
66  dist;
67  } else {
68  /* resulting force/energy in the linear region*/
69  fac = -ia_params.buckingham.F2 / dist;
70  }
71  return fac;
72  }
73  return 0.0;
74 }
75 
76 /** Calculate Buckingham energy */
77 inline double buck_pair_energy(IA_parameters const &ia_params, double dist) {
78  if (dist < ia_params.buckingham.cut) {
79  /* case: resulting force/energy greater than discont and
80  less than cutoff (true Buckingham region) */
81  if (dist > ia_params.buckingham.discont) {
82  return buck_energy_r(ia_params.buckingham.A, ia_params.buckingham.B,
83  ia_params.buckingham.C, ia_params.buckingham.D,
84  ia_params.buckingham.shift, dist);
85  }
86 
87  /* resulting force/energy in the linear region*/
88  return ia_params.buckingham.F1 + ia_params.buckingham.F2 * dist;
89  }
90  return 0.0;
91 }
