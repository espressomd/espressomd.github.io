320 [[nodiscard]] std::optional<CellInterval>
321 get_interval(Utils::Vector3i const &lower_corner,
322 Utils::Vector3i const &upper_corner) const {
323 auto const &lattice = get_lattice();
324 auto const &cell_min = lower_corner;
325 auto const cell_max = upper_corner - Utils::Vector3i::broadcast(1);
326 auto const lower_bc = get_block_and_cell(lattice, cell_min, true);
327 auto const upper_bc = get_block_and_cell(lattice, cell_max, true);
328 if (not lower_bc or not upper_bc) {
329 return std::nullopt;
330 }
331 assert(&(*(lower_bc->block)) == &(*(upper_bc->block)));
332 return {CellInterval(lower_bc->cell, upper_bc->cell)};
333 }

335 /**
336 * @brief Convenience function to add a field with a custom allocator.
337 *
338 * When vectorization is off, let waLBerla decide which memory allocator
339 * to use. When vectorization is on, the aligned memory allocator is
340 * required, otherwise <tt>cpu_vectorize_info["assume_aligned"]</tt> will
341 * trigger assertions. That is because for single-precision kernels the
342 * waLBerla heuristic in <tt>src/field/allocation/FieldAllocator.h</tt>
343 * will fall back to @c StdFieldAlloc, yet @c AllocateAligned is needed
344 * for intrinsics to work.
345 */
346 template <typename Field> auto add_to_storage(std::string const tag) {
347 auto const &blocks = m_lattice->get_blocks();
348 auto const n_ghost_layers = m_lattice->get_ghost_layers();
349 if constexpr (Architecture == lbmpy::Arch::CPU) {
350#ifdef ESPRESSO_BUILD_WITH_AVX_KERNELS
351#if defined(__AVX512F__)
352 constexpr uint_t alignment = 64;
353#elif defined(__AVX__)
354 constexpr uint_t alignment = 32;
355#elif defined(__SSE__)
356 constexpr uint_t alignment = 16;
357#else
358#error "Unsupported arch, check walberla src/field/allocation/FieldAllocator.h"
359#endif
360 using value_type = typename Field::value_type;
361 using Allocator = field::AllocateAligned<value_type, alignment>;
362 auto const allocator = std::make_shared<Allocator>();
363 auto const empty_set = Set<SUID>::emptySet();
364 return field::addToStorage<Field>(
365 blocks, tag, field::internal::defaultSize, FloatType{0}, field::fzyx,
366 n_ghost_layers, false, {}, empty_set, empty_set, allocator);
367#else // ESPRESSO_BUILD_WITH_AVX_KERNELS
368 return field::addToStorage<Field>(blocks, tag, FloatType{0}, field::fzyx,
369 n_ghost_layers);
370#endif // ESPRESSO_BUILD_WITH_AVX_KERNELS
371 }
372#if defined(__CUDACC__)
373 else {
374 auto field_id = gpu::addGPUFieldToStorage<GPUField>(
375 blocks, tag, Field::F_SIZE, field::fzyx, n_ghost_layers);
376 if constexpr (std::is_same_v<Field, _VectorField>) {
377 for (auto block = blocks->begin(); block != blocks->end(); ++block) {
378 auto field = block->template getData<GPUField>(field_id);
379 lbm::accessor::Vector::initialize(field, Vector3<FloatType>{0});
380 }
381 } else if constexpr (std::is_same_v<Field, _PdfField>) {
382 for (auto block = blocks->begin(); block != blocks->end(); ++block) {
383 auto field = block->template getData<GPUField>(field_id);
384 lbm::accessor::Population::initialize(
385 field, std::array<FloatType, Stencil::Size>{});
386 }
387 }
388 return field_id;
389 }
390#endif
391 }

393public:
394 LBWalberlaImpl(std::shared_ptr<LatticeWalberla> lattice, double viscosity,
395 double density)
396 : m_viscosity(FloatType_c(viscosity)), m_density(FloatType_c(density)),
397 m_kT(FloatType{0}), m_lattice(std::move(lattice)) {
398
399 auto const &blocks = m_lattice->get_blocks();
400 auto const n_ghost_layers = m_lattice->get_ghost_layers();
401 if (n_ghost_layers == 0u)
402 throw std::runtime_error("At least one ghost layer must be used");
403
404 // Initialize and register fields (must use the "underlying" types)
405 m_pdf_field_id = add_to_storage<_PdfField>("pdfs");
406 m_pdf_tmp_field_id = add_to_storage<_PdfField>("pdfs_tmp");
407 m_last_applied_force_field_id = add_to_storage<_VectorField>("force last");
408 m_force_to_be_applied_id = add_to_storage<_VectorField>("force next");
409 m_velocity_field_id = add_to_storage<_VectorField>("velocity");
410 m_vec_tmp_field_id = add_to_storage<_VectorField>("velocity_tmp");
411
412 // Initialize and register pdf field
413 auto pdf_setter =
414 InitialPDFsSetter(m_force_to_be_applied_id, m_pdf_field_id,
415 m_velocity_field_id, m_density);
416 for (auto b = blocks->begin(); b != blocks->end(); ++b) {
417 pdf_setter(&*b);
418 }
419
420 // Initialize and register flag field (fluid/boundary)
421 m_flag_field_id = field::addFlagFieldToStorage<FlagField>(
422 blocks, "flag field", n_ghost_layers);
423 // Initialize boundary sweep
424 reset_boundary_handling();
425
426 // Set up the communication and register fields
427 m_pdf_streaming_communicator =
428 std::make_shared<PDFStreamingCommunicator>(blocks);
429 m_pdf_streaming_communicator->addPackInfo(
430 std::make_shared<PackInfo<PdfField>>(m_pdf_field_id));
431 m_pdf_streaming_communicator->addPackInfo(
432 std::make_shared<PackInfo<VectorField>>(m_last_applied_force_field_id));
433
434 m_pdf_full_communicator = std::make_shared<RegularFullCommunicator>(blocks);
435 m_pdf_full_communicator->addPackInfo(
436 std::make_shared<PackInfo<PdfField>>(m_pdf_field_id));
437 m_pdf_full_communicator->addPackInfo(
438 std::make_shared<PackInfo<VectorField>>(m_last_applied_force_field_id));
439 m_pdf_full_communicator->addPackInfo(
440 std::make_shared<PackInfo<VectorField>>(m_velocity_field_id));
441
442 m_boundary_communicator =
443 std::make_shared<BoundaryFullCommunicator>(blocks);
444 m_boundary_communicator->addPackInfo(
445 std::make_shared<field::communication::PackInfo<FlagField>>(
446 m_flag_field_id));
447 auto boundary_packinfo = std::make_shared<
448 field::communication::BoundaryPackInfo<FlagField, BoundaryModel>>(
449 m_flag_field_id);
450 boundary_packinfo->setup_boundary_handle(m_lattice, m_boundary);
451 m_boundary_communicator->addPackInfo(boundary_packinfo);
452
453 // Instantiate the sweep responsible for force double buffering and
454 // external forces
455 m_reset_force = std::make_shared<ResetForce<PdfField, VectorField>>(
456 m_last_applied_force_field_id, m_force_to_be_applied_id);
457
458 // Prepare LB sweeps
459 // Note: For now, combined collide-stream sweeps cannot be used,
460 // because the collide-push variant is not supported by lbmpy.
461 // The following functors are individual in-place collide and stream steps
462 m_stream = std::make_shared<StreamSweep>(
463 m_last_applied_force_field_id, m_pdf_field_id, m_velocity_field_id);
464 }

595 void set_collision_model(
596 std::unique_ptr<LeesEdwardsPack> &&lees_edwards_pack) override {
597 assert(m_kT == 0.);
598#if defined(__CUDACC__)
599 if constexpr (Architecture == lbmpy::Arch::GPU) {
600 throw std::runtime_error("Lees-Edwards LB doesn't support GPU yet");
601 }
602#endif
603 auto const shear_direction = lees_edwards_pack->shear_direction;
604 auto const shear_plane_normal = lees_edwards_pack->shear_plane_normal;
605 auto const shear_vel = FloatType_c(lees_edwards_pack->get_shear_velocity());
606 auto const omega = shear_mode_relaxation_rate();
607 if (shear_plane_normal != 1u) {
608 throw std::domain_error(
609 "Lees-Edwards LB only supports shear_plane_normal=\"y\"");
610 }
611 auto const &lattice = get_lattice();
612 auto const n_ghost_layers = lattice.get_ghost_layers();
613 auto const blocks = lattice.get_blocks();
614 auto const agrid =
615 FloatType_c(lattice.get_grid_dimensions()[shear_plane_normal]);
616 auto obj = CollisionModelLeesEdwards(
617 m_last_applied_force_field_id, m_pdf_field_id, agrid, omega, shear_vel);

773 std::vector<Utils::Vector3d> const &forces) override {

783#if defined(__CUDACC__)
784 if constexpr (Architecture == lbmpy::Arch::GPU) {

792 assert(++(lattice.get_blocks()->begin()) == lattice.get_blocks()->end());
793 for (auto const &vec : pos) {
794#pragma unroll
795 for (std::size_t i : {0ul, 1ul, 2ul}) {
796 host_pos.emplace_back(static_cast<FloatType>(vec[i] - origin[i]));

798 }
799 for (auto const &vec : forces) {
800#pragma unroll

802 host_force.emplace_back(static_cast<FloatType>(vec[i]));
803 }

806 auto field = block.template uncheckedFastGetData<VectorField>(
