266  [[nodiscard]] std::optional<CellInterval>
267  get_interval(Utils::Vector3i const &lower_corner,
268  Utils::Vector3i const &upper_corner) const {
269  auto const &lattice = get_lattice();
270  auto const &cell_min = lower_corner;
271  auto const cell_max = upper_corner - Utils::Vector3i::broadcast(1);
272  auto const lower_bc = get_block_and_cell(lattice, cell_min, true);
273  auto const upper_bc = get_block_and_cell(lattice, cell_max, true);
274  if (not lower_bc or not upper_bc) {
275  return std::nullopt;
276  }
277  assert(&(*(lower_bc->block)) == &(*(upper_bc->block)));
278  return {CellInterval(lower_bc->cell, upper_bc->cell)};
279  }

281  /**
282  * @brief Convenience function to add a field with a custom allocator.
283  *
284  * When vectorization is off, let waLBerla decide which memory allocator
285  * to use. When vectorization is on, the aligned memory allocator is
286  * required, otherwise <tt>cpu_vectorize_info["assume_aligned"]</tt> will
287  * trigger assertions. That is because for single-precision kernels the
288  * waLBerla heuristic in <tt>src/field/allocation/FieldAllocator.h</tt>
289  * will fall back to @c StdFieldAlloc, yet @c AllocateAligned is needed
290  * for intrinsics to work.
291  */
292  template <typename Field> auto add_to_storage(std::string const tag) {
293  auto const &blocks = m_lattice->get_blocks();
294  auto const n_ghost_layers = m_lattice->get_ghost_layers();
295  if constexpr (Architecture == lbmpy::Arch::CPU) {
296 #ifdef ESPRESSO_BUILD_WITH_AVX_KERNELS
297 #if defined(__AVX512F__)
298  constexpr uint_t alignment = 64;
299 #elif defined(__AVX__)
300  constexpr uint_t alignment = 32;
301 #elif defined(__SSE__)
302  constexpr uint_t alignment = 16;
303 #else
304 #error "Unsupported arch, check walberla src/field/allocation/FieldAllocator.h"
305 #endif
306  using value_type = typename Field::value_type;
307  using Allocator = field::AllocateAligned<value_type, alignment>;
308  auto const allocator = std::make_shared<Allocator>();
309  auto const empty_set = Set<SUID>::emptySet();
310  return field::addToStorage<Field>(
311  blocks, tag, field::internal::defaultSize, FloatType{0}, field::fzyx,
312  n_ghost_layers, false, {}, empty_set, empty_set, allocator);
313 #else // ESPRESSO_BUILD_WITH_AVX_KERNELS
314  return field::addToStorage<Field>(blocks, tag, FloatType{0}, field::fzyx,
315  n_ghost_layers);
316 #endif // ESPRESSO_BUILD_WITH_AVX_KERNELS
317  }
318  }

532  void set_collision_model(
533  std::unique_ptr<LeesEdwardsPack> &&lees_edwards_pack) override {
534  assert(m_kT == 0.);
535  auto const shear_direction = lees_edwards_pack->shear_direction;
536  auto const shear_plane_normal = lees_edwards_pack->shear_plane_normal;
537  auto const shear_vel = FloatType_c(lees_edwards_pack->get_shear_velocity());
538  auto const omega = shear_mode_relaxation_rate();
539  if (shear_plane_normal != 1u) {
540  throw std::domain_error(
541  "Lees-Edwards LB only supports shear_plane_normal=\"y\"");
542  }
543  auto const &lattice = get_lattice();
544  auto const n_ghost_layers = lattice.get_ghost_layers();
545  auto const blocks = lattice.get_blocks();
546  auto const agrid =
547  FloatType_c(lattice.get_grid_dimensions()[shear_plane_normal]);
548  auto obj = CollisionModelLeesEdwards(
549  m_last_applied_force_field_id, m_pdf_field_id, agrid, omega, shear_vel);
550  m_collision_model = std::make_shared<CollisionModel>(std::move(obj));
551  m_lees_edwards_callbacks = std::move(lees_edwards_pack);
552  m_run_collide_sweep = CollideSweepVisitor{m_lees_edwards_callbacks};
553  m_lees_edwards_pdf_interpol_sweep =
554  std::make_shared<InterpolateAndShiftAtBoundary<PdfField, FloatType>>(
555  blocks, m_pdf_field_id, m_pdf_tmp_field_id, n_ghost_layers,
556  shear_direction, shear_plane_normal,
557  m_lees_edwards_callbacks->get_pos_offset);
558  m_lees_edwards_vel_interpol_sweep =
559  std::make_shared<InterpolateAndShiftAtBoundary<VectorField, FloatType>>(
560  blocks, m_velocity_field_id, m_vec_tmp_field_id, n_ghost_layers,
561  shear_direction, shear_plane_normal,
562  m_lees_edwards_callbacks->get_pos_offset,
563  m_lees_edwards_callbacks->get_shear_velocity);
564  m_lees_edwards_last_applied_force_interpol_sweep =
