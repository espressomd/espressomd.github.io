113 /**
114  * @brief Serialize particle data, possibly with reduction.
115  * The reduction can take place during the save stage, e.g. to apply
116  * a ghost shift to the particle position, or during the load stage,
117  * e.g. to transfer momentum between particles in two local cells.

119 template <class Archive>
120 static void
121 serialize_and_reduce(Archive &ar, Particle &p, unsigned int data_parts,
122  ReductionPolicy policy, SerializationDirection direction,
123  Utils::Vector3d const *ghost_shift) {
124  if (data_parts & GHOSTTRANS_PROPRTS) {
125  ar &p.id() & p.mol_id() & p.type();
126 #ifdef VIRTUAL_SITES
127  ar &p.virtual_flag();
128 #endif
129 #ifdef ROTATION
130  ar &p.rotation();
131 #ifdef ROTATIONAL_INERTIA
132  ar &p.rinertia();
133 #endif
134 #endif
135 #ifdef MASS
136  ar &p.mass();
137 #endif
138 #ifdef ELECTROSTATICS
139  ar &p.q();
140 #endif
141 #ifdef DIPOLES
142  ar &p.dipm();
143 #endif
144 #ifdef LB_ELECTROHYDRODYNAMICS
145  ar &p.mu_E();
146 #endif
147 #ifdef VIRTUAL_SITES_RELATIVE
148  ar &p.vs_relative();
149 #endif
150 #ifdef THERMOSTAT_PER_PARTICLE
151  ar &p.gamma();
152 #ifdef ROTATION
153  ar &p.gamma_rot();
154 #endif
155 #endif
156 #ifdef EXTERNAL_FORCES
157  ar &p.fixed();
158  ar &p.ext_force();
159 #ifdef ROTATION
160  ar &p.ext_torque();
161 #endif
162 #endif
163 #ifdef ENGINE
164  ar &p.swimming();
165 #endif
166  }
167  if (data_parts & GHOSTTRANS_POSITION) {
168  if (direction == SerializationDirection::SAVE and ghost_shift != nullptr) {
169  /* ok, this is not nice, but perhaps fast */
170  auto pos = p.pos() + *ghost_shift;
171  ar &pos;
172  } else {
173  ar &p.pos();
174  }
175 #ifdef ROTATION
176  ar &p.quat();
177 #endif
178 #ifdef BOND_CONSTRAINT
179  ar &p.pos_last_time_step();
180 #endif
181  }
182  if (data_parts & GHOSTTRANS_MOMENTUM) {
183  ar &p.v();
184 #ifdef ROTATION
185  ar &p.omega();
186 #endif
187  }
188  if (data_parts & GHOSTTRANS_FORCE) {
189  if (policy == ReductionPolicy::UPDATE and
190  direction == SerializationDirection::LOAD) {
191  Utils::Vector3d force;
192  ar &force;
193  p.force() += force;
194  } else {
195  ar &p.force();
196  }
197 #ifdef ROTATION
198  if (policy == ReductionPolicy::UPDATE and
199  direction == SerializationDirection::LOAD) {
200  Utils::Vector3d torque;
201  ar &torque;
202  p.torque() += torque;
203  } else {
204  ar &p.torque();
205  }
206 #endif
207  }
208 #ifdef BOND_CONSTRAINT
209  if (data_parts & GHOSTTRANS_RATTLE) {
210  if (policy == ReductionPolicy::UPDATE and
211  direction == SerializationDirection::LOAD) {
212  Utils::Vector3d correction;
213  ar &correction;
214  p.rattle_correction() += correction;
215  } else {
216  ar &p.rattle_correction();
217  }
218  }
219 #endif
220 }

222 static std::size_t calc_transmit_size(unsigned data_parts) {
223  SerializationSizeCalculator sizeof_archive;
224  Particle p{};
225  serialize_and_reduce(sizeof_archive, p, data_parts, ReductionPolicy::MOVE,
226  SerializationDirection::SAVE, nullptr);
227  return sizeof_archive.size();

230 static std::size_t calc_transmit_size(const GhostCommunication &ghost_comm,
231  unsigned int data_parts) {
232  if (data_parts & GHOSTTRANS_PARTNUM)
233  return sizeof(int) * ghost_comm.part_lists.size();
234 
235  auto const n_part = boost::accumulate(
236  ghost_comm.part_lists, 0ul,
237  [](std::size_t sum, auto part_list) { return sum + part_list->size(); });
238 
239  return n_part * calc_transmit_size(data_parts);
240 }

242 static void prepare_send_buffer(CommBuf &send_buffer,
243  const GhostCommunication &ghost_comm,
244  unsigned int data_parts) {
245  /* reallocate send buffer */
246  send_buffer.resize(calc_transmit_size(ghost_comm, data_parts));
247  send_buffer.bonds().clear();
248 
249  auto archiver = Utils::MemcpyOArchive{Utils::make_span(send_buffer)};
250 
251  /* Construct archive that pushes back to the bond buffer */
252  namespace io = boost::iostreams;
253  io::stream<io::back_insert_device<std::vector<char>>> os{
254  io::back_inserter(send_buffer.bonds())};
255  boost::archive::binary_oarchive bond_archiver{os};
256 
257  /* put in data */
258  for (auto part_list : ghost_comm.part_lists) {
259  if (data_parts & GHOSTTRANS_PARTNUM) {
260  int np = static_cast<int>(part_list->size());
261  archiver << np;
262  } else {
263  for (auto &p : *part_list) {
264  serialize_and_reduce(archiver, p, data_parts, ReductionPolicy::MOVE,
265  SerializationDirection::SAVE, &ghost_comm.shift);
266  if (data_parts & GHOSTTRANS_BONDS) {
267  bond_archiver << p.bonds();
268  }
269  }
270  }
271  }
272 
273  assert(archiver.bytes_written() == send_buffer.size());
274 }

295 static void put_recv_buffer(CommBuf &recv_buffer,
296  const GhostCommunication &ghost_comm,
297  unsigned int data_parts) {
298  /* put back data */
299  auto archiver = Utils::MemcpyIArchive{Utils::make_span(recv_buffer)};
300 
301  if (data_parts & GHOSTTRANS_PARTNUM) {
302  for (auto part_list : ghost_comm.part_lists) {
303  int np;
304  archiver >> np;
305  prepare_ghost_cell(part_list, np);
306  }
307  } else {
308  for (auto part_list : ghost_comm.part_lists) {
309  for (auto &p : *part_list) {
310  serialize_and_reduce(archiver, p, data_parts, ReductionPolicy::MOVE,
311  SerializationDirection::LOAD, nullptr);
312  }
313  }
314  if (data_parts & GHOSTTRANS_BONDS) {
315  namespace io = boost::iostreams;
316  io::stream<io::array_source> bond_stream(io::array_source{
317  recv_buffer.bonds().data(), recv_buffer.bonds().size()});
318  boost::archive::binary_iarchive bond_archiver(bond_stream);
319 
320  for (auto part_list : ghost_comm.part_lists) {
321  for (auto &p : *part_list) {
322  bond_archiver >> p.bonds();
323  }
324  }
325  }
326  }
327 
328  assert(archiver.bytes_read() == recv_buffer.size());
329 
330  recv_buffer.bonds().clear();
331 }

362 static void cell_cell_transfer(const GhostCommunication &ghost_comm,
363  unsigned int data_parts) {
364  CommBuf buffer;
365  if (!(data_parts & GHOSTTRANS_PARTNUM)) {
366  buffer.resize(calc_transmit_size(data_parts));
367  }
368  /* transfer data */
369  auto const offset = ghost_comm.part_lists.size() / 2;
370  for (std::size_t pl = 0; pl < offset; pl++) {
371  auto *src_list = ghost_comm.part_lists[pl];
372  auto *dst_list = ghost_comm.part_lists[pl + offset];
373 
374  if (data_parts & GHOSTTRANS_PARTNUM) {
375  prepare_ghost_cell(dst_list, static_cast<int>(src_list->size()));
376  } else {
377  auto &src_part = *src_list;
378  auto &dst_part = *dst_list;
379  assert(src_part.size() == dst_part.size());
380 
381  for (std::size_t i = 0; i < src_part.size(); i++) {
382  auto ar_out = Utils::MemcpyOArchive{Utils::make_span(buffer)};
383  auto ar_in = Utils::MemcpyIArchive{Utils::make_span(buffer)};
384  auto &p1 = src_part.begin()[i];
385  auto &p2 = dst_part.begin()[i];
386  serialize_and_reduce(ar_out, p1, data_parts, ReductionPolicy::UPDATE,
387  SerializationDirection::SAVE, &ghost_comm.shift);
388  serialize_and_reduce(ar_in, p2, data_parts, ReductionPolicy::UPDATE,
389  SerializationDirection::LOAD, nullptr);
390  if (data_parts & GHOSTTRANS_BONDS) {
391  p2.bonds() = p1.bonds();
392  }
393  }
394  }
395  }

425 void ghost_communicator(const GhostCommunicator &gcr, unsigned int data_parts) {
426  if (GHOSTTRANS_NONE == data_parts)
427  return;
428 
429  static CommBuf send_buffer, recv_buffer;
430 
431  auto const &comm = gcr.mpi_comm;
432 
433  for (auto it = gcr.communications.begin(); it != gcr.communications.end();
434  ++it) {
435  const GhostCommunication &ghost_comm = *it;
436  int const comm_type = ghost_comm.type & GHOST_JOBMASK;
437 
438  if (comm_type == GHOST_LOCL) {
439  cell_cell_transfer(ghost_comm, data_parts);
440  continue;
441  }
442 
443  int const prefetch = ghost_comm.type & GHOST_PREFETCH;
444  int const poststore = ghost_comm.type & GHOST_PSTSTORE;
445  int const node = ghost_comm.node;
446 
447  /* prepare send buffer if necessary */
448  if (is_send_op(comm_type, node, comm.rank())) {
449  /* ok, we send this step, prepare send buffer if not yet done */
450  if (!prefetch) {
451  prepare_send_buffer(send_buffer, ghost_comm, data_parts);
452  }
453  // Check prefetched send buffers (must also hold for buffers allocated
454  // in the previous lines.)
455  assert(send_buffer.size() == calc_transmit_size(ghost_comm, data_parts));
456  } else if (prefetch) {
457  /* we do not send this time, let's look for a prefetch */
458  auto prefetch_ghost_comm = std::find_if(
459  std::next(it), gcr.communications.end(),
460  [this_node = comm.rank()](GhostCommunication const &ghost_comm) {
461  return is_prefetchable(ghost_comm, this_node);
462  });
463 
464  if (prefetch_ghost_comm != gcr.communications.end())
465  prepare_send_buffer(send_buffer, *prefetch_ghost_comm, data_parts);
466  }
467 
468  /* recv buffer for recv and multinode operations to this node */
469  if (is_recv_op(comm_type, node, comm.rank()))
470  prepare_recv_buffer(recv_buffer, ghost_comm, data_parts);
471 
472  /* transfer data */
473  // Use two send/recvs in order to avoid having to serialize CommBuf
474  // (which consists of already serialized data).
475  switch (comm_type) {
476  case GHOST_RECV:
477  comm.recv(node, REQ_GHOST_SEND, recv_buffer.data(),
478  static_cast<int>(recv_buffer.size()));
479  comm.recv(node, REQ_GHOST_SEND, recv_buffer.bonds());
480  break;
481  case GHOST_SEND:
482  comm.send(node, REQ_GHOST_SEND, send_buffer.data(),
483  static_cast<int>(send_buffer.size()));
484  comm.send(node, REQ_GHOST_SEND, send_buffer.bonds());
485  break;
486  case GHOST_BCST:
487  if (node == comm.rank()) {
488  boost::mpi::broadcast(comm, send_buffer.data(),
489  static_cast<int>(send_buffer.size()), node);
490  boost::mpi::broadcast(comm, send_buffer.bonds(), node);
491  } else {
492  boost::mpi::broadcast(comm, recv_buffer.data(),
493  static_cast<int>(recv_buffer.size()), node);
494  boost::mpi::broadcast(comm, recv_buffer.bonds(), node);
495  }
496  break;
497  case GHOST_RDCE:
498  if (node == comm.rank())
499  boost::mpi::reduce(
500  comm, reinterpret_cast<double *>(send_buffer.data()),
501  static_cast<int>(send_buffer.size() / sizeof(double)),
502  reinterpret_cast<double *>(recv_buffer.data()), std::plus<double>{},
503  node);
504  else
505  boost::mpi::reduce(
506  comm, reinterpret_cast<double *>(send_buffer.data()),
507  static_cast<int>(send_buffer.size() / sizeof(double)),
508  std::plus<double>{}, node);
509  break;
510  }
511 
512  // recv op; write back data directly, if no PSTSTORE delay is requested.
513  if (is_recv_op(comm_type, node, comm.rank())) {
514  if (!poststore) {
515  /* forces have to be added, the rest overwritten. Exception is RDCE,
516  * where the addition is integrated into the communication. */
517  if (data_parts == GHOSTTRANS_FORCE && comm_type != GHOST_RDCE)
518  add_forces_from_recv_buffer(recv_buffer, ghost_comm);
519 #ifdef BOND_CONSTRAINT
520  else if (data_parts == GHOSTTRANS_RATTLE && comm_type != GHOST_RDCE)
521  add_rattle_correction_from_recv_buffer(recv_buffer, ghost_comm);
522 #endif
523  else
524  put_recv_buffer(recv_buffer, ghost_comm, data_parts);
525  }
526  } else if (poststore) {
527  /* send op; write back delayed data from last recv, when this was a
528  * prefetch send. */
529  /* find previous action where we recv and which has PSTSTORE set */
530  auto poststore_ghost_comm = std::find_if(
531  std::make_reverse_iterator(it), gcr.communications.rend(),
532  [this_node = comm.rank()](GhostCommunication const &ghost_comm) {
533  return is_poststorable(ghost_comm, this_node);
534  });
535 
536  if (poststore_ghost_comm != gcr.communications.rend()) {
537  assert(recv_buffer.size() ==
538  calc_transmit_size(*poststore_ghost_comm, data_parts));
539  /* as above */
540  if (data_parts == GHOSTTRANS_FORCE && comm_type != GHOST_RDCE)
541  add_forces_from_recv_buffer(recv_buffer, *poststore_ghost_comm);
542 #ifdef BOND_CONSTRAINT
543  else if (data_parts == GHOSTTRANS_RATTLE && comm_type != GHOST_RDCE)
544  add_rattle_correction_from_recv_buffer(recv_buffer,
545  *poststore_ghost_comm);
546 #endif
547  else
548  put_recv_buffer(recv_buffer, *poststore_ghost_comm, data_parts);
549  }
550  }
551  }
