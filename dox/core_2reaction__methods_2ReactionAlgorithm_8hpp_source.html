101  set_exclusion_radius_per_type(std::unordered_map<int, double> const &map) {
102  auto max_exclusion_range = exclusion_range;
103  for (auto const &item : map) {
104  auto const type = item.first;
105  auto const exclusion_radius = item.second;
106  if (exclusion_radius < 0.) {
107  throw std::domain_error("Invalid excluded_radius value for type " +
108  std::to_string(type) + ": radius " +
109  std::to_string(exclusion_radius));
110  }
111  max_exclusion_range =
112  std::max(max_exclusion_range, 2. * exclusion_radius);
113  }
114  exclusion_radius_per_type = map;
115  m_max_exclusion_range = max_exclusion_range;
116  }

144  /**
145  * @brief Carry out a generic one-way chemical reaction.
146  *
147  * Generic one way reaction of the type
148  * <tt>A+B+...+G +... --> K+...X + Z +...</tt>
149  * You need to use <tt>2A --> B</tt> instead of <tt>A+A --> B</tt> since
150  * in the latter you assume distinctness of the particles, however both
151  * ways to describe the reaction are equivalent in the thermodynamic limit
152  * (large particle numbers). Furthermore, it is crucial for the function
153  * in which order you provide the reactant and product types since particles
154  * will be replaced correspondingly! If there are less reactants than
155  * products, new product particles are created randomly in the box.
156  * Matching particles simply change the types. If there are more reactants
157  * than products, old reactant particles are deleted.
158  *
159  * @param[in,out] current_reaction The reaction to attempt.
160  * @param[in,out] E_pot_old The current potential energy.
161  */
