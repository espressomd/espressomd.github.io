53int RegularDecomposition::position_to_cell_index(
54 Utils::Vector3d const &pos) const {
55 Utils::Vector3i cpos;
56
57 for (auto i = 0u; i < 3u; i++) {
58 cpos[i] = static_cast<int>(std::floor(pos[i] * inv_cell_size[i])) + 1 -
59 cell_offset[i];
60
61 /* particles outside our box. Still take them if
62 nonperiodic boundary. We also accept the particle if we are at
63 the box boundary, and the particle is within the box. In this case
64 the particle belongs here and could otherwise potentially be dismissed
65 due to rounding errors. */
66 if (cpos[i] < 1) {
67 if ((!m_box.periodic(i) or (pos[i] >= m_box.length()[i])) and
68 m_local_box.boundary()[2u * i])
69 cpos[i] = 1;
70 else
71 return -1;
72 } else if (cpos[i] > cell_grid[i]) {
73 if ((!m_box.periodic(i) or (pos[i] < m_box.length()[i])) and
74 m_local_box.boundary()[2u * i + 1u])
75 cpos[i] = cell_grid[i];
76 else
77 return -1;
78 }
79 }
80
81 return get_linear_index(cpos, ghost_cell_grid);
82}

84void RegularDecomposition::move_if_local(
85 ParticleList &src, ParticleList &rest,
86 std::vector<ParticleChange> &modified_cells) {
87 for (auto &part : src) {
88 auto target_cell = position_to_cell(part.pos());
89
90 if (target_cell) {
91 target_cell->particles().insert(std::move(part));
92 modified_cells.emplace_back(ModifiedList{target_cell->particles()});
93 } else {
94 rest.insert(std::move(part));
95 }
96 }
97
98 src.clear();

101void RegularDecomposition::move_left_or_right(ParticleList &src,
102 ParticleList &left,
103 ParticleList &right,
104 int dir) const {
105 auto const is_open_boundary_left = m_local_box.boundary()[2 * dir] != 0;
106 auto const is_open_boundary_right = m_local_box.boundary()[2 * dir + 1] != 0;
107 auto const can_move_left = m_box.periodic(dir) or not is_open_boundary_left;
108 auto const can_move_right = m_box.periodic(dir) or not is_open_boundary_right;
109 auto const my_left = m_local_box.my_left()[dir];
110 auto const my_right = m_local_box.my_right()[dir];
111 for (auto it = src.begin(); it != src.end();) {
112 auto const pos = it->pos()[dir];
113 if (m_box.get_mi_coord(pos, my_right, dir) >= 0. and can_move_right) {
114 right.insert(std::move(*it));
115 it = src.erase(it);
116 } else if (m_box.get_mi_coord(pos, my_left, dir) < 0. and can_move_left) {
117 left.insert(std::move(*it));
118 it = src.erase(it);
119 } else {
120 ++it;
121 }
122 }
123}

125void RegularDecomposition::exchange_neighbors(
126 ParticleList &pl, std::vector<ParticleChange> &modified_cells) {
127 auto const node_neighbors = Utils::Mpi::cart_neighbors<3>(m_comm);
128 static ParticleList send_buf_l, send_buf_r, recv_buf_l, recv_buf_r;
129
130 for (int dir = 0; dir < 3; dir++) {
131 /* Single node direction, no action needed. */
132 if (Utils::Mpi::cart_get<3>(m_comm).dims[dir] == 1) {
133 continue;
134 /* In this (common) case left and right neighbors are
135 the same, and we need only one communication */
136 }
137 if (Utils::Mpi::cart_get<3>(m_comm).dims[dir] == 2) {
138 move_left_or_right(pl, send_buf_l, send_buf_l, dir);
139
140 Utils::Mpi::sendrecv(m_comm, node_neighbors[2 * dir], 0, send_buf_l,
141 node_neighbors[2 * dir], 0, recv_buf_l);
142
143 send_buf_l.clear();
144 } else {
145 using boost::mpi::request;
146 using Utils::Mpi::isendrecv;
147
148 move_left_or_right(pl, send_buf_l, send_buf_r, dir);
149
150 auto req_l = isendrecv(m_comm, node_neighbors[2 * dir], 0, send_buf_l,
151 node_neighbors[2 * dir], 0, recv_buf_l);
152 auto req_r = isendrecv(m_comm, node_neighbors[2 * dir + 1], 0, send_buf_r,
153 node_neighbors[2 * dir + 1], 0, recv_buf_r);
154
155 std::array<request, 4> reqs{{req_l[0], req_l[1], req_r[0], req_r[1]}};
156 boost::mpi::wait_all(reqs.begin(), reqs.end());
157
158 send_buf_l.clear();
159 send_buf_r.clear();
160 }
161
162 move_if_local(recv_buf_l, pl, modified_cells);
163 move_if_local(recv_buf_r, pl, modified_cells);
164 }
165}

176void RegularDecomposition::resort(bool global,
177 std::vector<ParticleChange> &diff) {
178 ParticleList displaced_parts;
179
180 for (auto &c : local_cells()) {
181 for (auto it = c->particles().begin(); it != c->particles().end();) {
182 fold_and_reset(*it, m_box);
183
184 auto target_cell = particle_to_cell(*it);
185
186 /* Particle is in place */
187 if (target_cell == c) {
188 std::advance(it, 1);
189 continue;
190 }
191
192 auto p = std::move(*it);
193 it = c->particles().erase(it);
194 diff.emplace_back(ModifiedList{c->particles()});
195
196 /* Particle is not local */
197 if (target_cell == nullptr) {
198 diff.emplace_back(RemovedParticle{p.id()});
199 displaced_parts.insert(std::move(p));
200 }
201 /* Particle belongs on this node but is in the wrong cell. */
202 else if (target_cell != c) {
203 target_cell->particles().insert(std::move(p));
204 diff.emplace_back(ModifiedList{target_cell->particles()});
205 }
206 }
207 }
208
209 if (global) {
210 auto const grid = Utils::Mpi::cart_get<3>(m_comm).dims;
211 /* Worst case we need grid - 1 rounds per direction.
212 * This correctly implies that if there is only one node,
213 * no action should be taken. */
214 int rounds_left = grid[0] + grid[1] + grid[2] - 3;
215 for (; rounds_left > 0; rounds_left--) {
216 exchange_neighbors(displaced_parts, diff);
217
218 auto left_over = boost::mpi::all_reduce(m_comm, displaced_parts.size(),
219 std::plus<std::size_t>());
220
221 if (left_over == 0) {
222 break;
223 }
224 }
225 } else {
226 exchange_neighbors(displaced_parts, diff);
227 }
228
229 if (not displaced_parts.empty()) {
230 auto sort_cell = local_cells()[0];
231
232 for (auto &part : displaced_parts) {
233 runtimeErrorMsg() << "Particle " << part.id() << " moved more "
234 << "than one local box length in one timestep";
235 sort_cell->particles().insert(std::move(part));
236
237 diff.emplace_back(ModifiedList{sort_cell->particles()});
238 }
239 }

242void RegularDecomposition::mark_cells() {
243 m_local_cells.clear();
244 m_ghost_cells.clear();
245
246 int cnt_c = 0;
247 for (int o = 0; o < ghost_cell_grid[2]; o++)
248 for (int n = 0; n < ghost_cell_grid[1]; n++)
249 for (int m = 0; m < ghost_cell_grid[0]; m++) {
250 if ((m > 0 && m < ghost_cell_grid[0] - 1 && n > 0 &&
251 n < ghost_cell_grid[1] - 1 && o > 0 && o < ghost_cell_grid[2] - 1))
252 m_local_cells.push_back(&cells.at(cnt_c++));
253 else
254 m_ghost_cells.push_back(&cells.at(cnt_c++));
255 }

279int RegularDecomposition::calc_processor_min_num_cells() const {
280 /* the minimal number of cells can be lower if there are at least two nodes
281 serving a direction,
282 since this also ensures that the cell size is at most half the box
283 length. However, if there is only one processor for a direction, there
284 have to be at least two cells for this direction. */
285 return boost::accumulate(Utils::Mpi::cart_get<3>(m_comm).dims, 1,
286 [](int n_cells, int grid) {
287 return (grid == 1) ? 2 * n_cells : n_cells;
288 });
289}

291void RegularDecomposition::create_cell_grid(double range) {
292 auto const cart_info = Utils::Mpi::cart_get<3>(m_comm);
293
294 int n_local_cells;
295 auto cell_range = Utils::Vector3d::broadcast(range);
296 auto const min_num_cells = calc_processor_min_num_cells();
297
298 if (range <= 0.) {
299 /* this is the non-interacting case */
300 auto const cells_per_dir =
301 static_cast<int>(std::ceil(std::cbrt(min_num_cells)));
302
303 cell_grid = Utils::Vector3i::broadcast(cells_per_dir);
304 n_local_cells = Utils::product(cell_grid);
305 } else {
306 /* Calculate initial cell grid */
307 auto const &local_box_l = m_local_box.length();
308 auto const volume = Utils::product(local_box_l);
309 auto const scale = std::cbrt(RegularDecomposition::max_num_cells / volume);
310
311 for (auto i = 0u; i < 3u; i++) {
312 /* this is at least 1 */
313 cell_grid[i] = static_cast<int>(std::ceil(local_box_l[i] * scale));
314 cell_range[i] = local_box_l[i] / static_cast<double>(cell_grid[i]);
315
316 if (cell_range[i] < range) {
317 /* ok, too many cells for this direction, set to minimum */
318 cell_grid[i] = static_cast<int>(std::floor(local_box_l[i] / range));
319 if (cell_grid[i] < 1) {
320 runtimeErrorMsg()
321 << "interaction range " << range << " in direction " << i
322 << " is larger than the local box size " << local_box_l[i];
323 cell_grid[i] = 1;
324 }
325 cell_range[i] = local_box_l[i] / static_cast<double>(cell_grid[i]);
326 }
327 }
328
329 /* It may be necessary to asymmetrically assign the scaling to the
330 coordinates, which the above approach will not do.
331 For a symmetric box, it gives a symmetric result. Here we correct that.
332 */
333 for (;;) {
334 n_local_cells = Utils::product(cell_grid);
335
336 /* done */
337 if (n_local_cells <= RegularDecomposition::max_num_cells)
338 break;
339
340 /* find coordinate with the smallest cell range */
341 auto min_ind = 0u;
342 auto min_size = cell_range[0];
343
344 for (auto i = 1u; i < 3u; ++i) {
345 if (cell_grid[i] > 1 and cell_range[i] < min_size) {
346 min_ind = i;
347 min_size = cell_range[i];
348 }
349 }
350
351 cell_grid[min_ind]--;
352 cell_range[min_ind] = m_local_box.length()[min_ind] / cell_grid[min_ind];
353 }
354
355 /* sanity check */
356 if (n_local_cells < min_num_cells) {
357 runtimeErrorMsg() << "number of cells " << n_local_cells
358 << " is smaller than minimum " << min_num_cells
359 << ": either interaction range is too large for "
360 << "the current skin (range=" << range << ", "
361 << "half_local_box_l=[" << local_box_l / 2. << "]) "
362 << "or min_num_cells too large";
363 }
364 }
365
366 if (n_local_cells > RegularDecomposition::max_num_cells) {
367 runtimeErrorMsg() << "no suitable cell grid found";
368 }
369
370 auto const node_pos = cart_info.coords;
371
372 /* now set all dependent variables */
373 int new_cells = 1;
374 for (auto i = 0u; i < 3u; i++) {
375 ghost_cell_grid[i] = cell_grid[i] + 2;
376 new_cells *= ghost_cell_grid[i];
377 cell_size[i] = m_local_box.length()[i] / static_cast<double>(cell_grid[i]);
378 inv_cell_size[i] = 1.0 / cell_size[i];
379 cell_offset[i] = node_pos[i] * cell_grid[i];
380 }
381
382 /* allocate cell array and cell pointer arrays */
383 cells.clear();
384 cells.resize(static_cast<unsigned int>(new_cells));
385 m_local_cells.resize(n_local_cells);
386 m_ghost_cells.resize(new_cells - n_local_cells);
387}

394void RegularDecomposition::init_cell_interactions() {
395
396 // Note: the global index for physical cells is 0-based.
397 // I.e., a global index of -1 refers to a ghost cell.
398 auto const halo = Utils::Vector3i{1, 1, 1}; // number of ghost layers
399 auto const cart_info = Utils::Mpi::cart_get<3>(m_comm);
400 // 3D index of the MPI rank in the Cartesian grid of MPI ranks
401 auto const &node_pos = cart_info.coords;
402 // size of the Cartesian grid of MPI ranks
403 auto const &node_grid = ::communicator.node_grid;
404 auto const global_halo_offset = hadamard_product(node_pos, cell_grid) - halo;
405 // MD cell index of lower halo layer on this MPI rank
406 auto const global_size = hadamard_product(node_grid, cell_grid);
407
408 // is a cell at the system boundary in the given coord
409 auto const at_boundary = [&global_size](int coord, Utils::Vector3i cell_idx) {
410 return (cell_idx[coord] == 0 or cell_idx[coord] == global_size[coord]);
411 };
412
413 // For the fully connected feature (cells that don't share at least a corner)
414 // only apply if one cell is a ghost cell (i.e. connections across the
415 // periodic boundary.
416 auto const fcb_is_inner_connection = [&global_size, this](Utils::Vector3i a,
417 Utils::Vector3i b) {
418 if (fully_connected_boundary()) {
419 auto const [fc_normal, fc_dir] = *fully_connected_boundary();
420 auto const involves_ghost_cell =
421 (a[fc_normal] == -1 or a[fc_normal] == global_size[fc_normal] or
422 b[fc_normal] == -1 or b[fc_normal] == global_size[fc_normal]);
423 if (not involves_ghost_cell) {
424 // check if cells do not share at least a corner
425 return std::abs((a - b)[fc_dir]) > 1;
426 }
427 }
428 return false;
429 };
430
431 /* Translate a node local index (relative to the origin of the local grid)
432 * to a global index. */
433 auto global_index =
434 [&](Utils::Vector3i const &local_index) -> Utils::Vector3i {
435 return (global_halo_offset + local_index);
436 };
437
438 /* Linear index in the global cell grid. */
439 auto folded_linear_index = [&](Utils::Vector3i const &global_index) {
440 auto const folded_index = (global_index + global_size) % global_size;
441
442 return get_linear_index(folded_index, global_size);
443 };
444
445 /* Translate a global index into a local one */
446 auto local_index =
447 [&](Utils::Vector3i const &global_index) -> Utils::Vector3i {
448 return (global_index - global_halo_offset);
449 };
450
451 // sanity checks
452 if (fully_connected_boundary()) {
453 auto const [fc_normal, fc_dir] = *fully_connected_boundary();
454 if (fc_normal == fc_dir) {
455 throw std::domain_error("fully_connected_boundary normal and connection "
456 "coordinates need to differ.");
457 }
458 if (node_grid[fc_dir] != 1) {
459 throw std::runtime_error(
460 "The MPI nodegrid must be 1 in the fully connected direction.");
461 }
462 }
463
464 /* We only consider local cells (e.g. not halo cells), which
465 * span the range [(1,1,1), cell_grid) in local coordinates. */
466 auto const start = global_index(Utils::Vector3i{1, 1, 1});
467 auto const end = start + cell_grid;
468
469 bool one_mpi_rank = m_comm.size() == 1;
470
471 /* loop all local cells */
472 for (int o = start[2]; o < end[2]; o++)
473 for (int n = start[1]; n < end[1]; n++)
474 for (int m = start[0]; m < end[0]; m++) {
475 /* next-nearest neighbors in every direction */
476 Utils::Vector3i lower_index = {m - 1, n - 1, o - 1};
477 Utils::Vector3i upper_index = {m + 1, n + 1, o + 1};
478
479 /* In the fully connected case, we consider all cells
480 * in the direction as neighbors, not only the nearest ones.
481 // */
482 if (fully_connected_boundary()) {
483 auto const [fc_boundary, fc_direction] = *fully_connected_boundary();
484
485 // Fully connected is only needed at the box surface
486 if (at_boundary(fc_boundary, {m, n, o})) {
487 lower_index[fc_direction] = -1;
488 upper_index[fc_direction] = global_size[fc_boundary];
489 }
490 }
491
492 /* In non-periodic directions, the halo needs not
493 * be considered. */
494 for (auto i = 0u; i < 3u; i++) {
495 if (not m_box.periodic(i)) {
496 lower_index[i] = std::max(0, lower_index[i]);
497 upper_index[i] = std::min(global_size[i] - 1, upper_index[i]);
498 }
499 }
500
501 /* Unique set of neighbors, cells are compared by their linear
502 * index in the global cell grid. */
503 auto neighbors = make_flat_set<Utils::Vector3i>(
504 [&](Utils::Vector3i const &a, Utils::Vector3i const &b) {
505 return folded_linear_index(a) < folded_linear_index(b);
506 });
507
508 /* Collect neighbors */
509 for (int p = lower_index[2]; p <= upper_index[2]; p++)
510 for (int q = lower_index[1]; q <= upper_index[1]; q++)
511 for (int r = lower_index[0]; r <= upper_index[0]; r++) {
512 if (fully_connected_boundary()) {
513 // Avoid fully connecting the boundary layer and the
514 // next INNER layer
515 if (fcb_is_inner_connection({m, n, o}, {r, q, p}))
516 continue;
517 }
518 neighbors.insert(Utils::Vector3i{r, q, p});
519 }
520
521 /* Red-black partition by global index. */
522 auto const ind1 = folded_linear_index({m, n, o});
523
524 std::vector<Cell *> red_neighbors;
525 std::vector<Cell *> black_neighbors;
526
527 /* If we are running on a single MPI rank, it is not necessary to use
528 * ghost cells. Instead of adding a ghost cell as neighbor,
529 * we directly connect to the corresponding
530 * physical cell across the periodic boundary */
531 for (auto &neighbor : neighbors) {
532 if (one_mpi_rank) {
533 for (auto coord : {0u, 1u, 2u}) {
534 if (neighbor[coord] == -1) {
535 neighbor[coord] += cell_grid[coord];
536 } else if (neighbor[coord] == cell_grid[coord]) {
537 neighbor[coord] -= cell_grid[coord];
538 }
539 }
540 }
541 auto const ind2 = folded_linear_index(neighbor);
542 /* Exclude cell itself */
543 if (ind1 == ind2)
544 continue;
545
546 auto cell = &cells.at(
547 get_linear_index(local_index(neighbor), ghost_cell_grid));
548
549 // Divide red and black neighbors
550 if (ind2 > ind1) {
551 red_neighbors.push_back(cell);
552 } else {
553 black_neighbors.push_back(cell);
554 }
555 }
556
557 // Assign neighbors to the cell
558 cells[get_linear_index(local_index({m, n, o}), ghost_cell_grid)]
559 .m_neighbors = Neighbors<Cell *>(red_neighbors, black_neighbors);
560 }
561}

564/** Revert the order of a communicator: After calling this the
565 * communicator is working in reverted order with exchanged
566 * communication types GHOST_SEND <-> GHOST_RECV.
567 */
568void revert_comm_order(GhostCommunicator &comm) {
569 /* revert order */
570 boost::reverse(comm.communications);
571
572 /* exchange SEND/RECV */
573 for (auto &c : comm.communications) {
574 if (c.type == GHOST_SEND)
575 c.type = GHOST_RECV;
576 else if (c.type == GHOST_RECV)
577 c.type = GHOST_SEND;
578 else if (c.type == GHOST_LOCL) {
579 boost::reverse(c.part_lists);
580 }
581 }
582}

584/** Of every two communication rounds, set the first receivers to prefetch and
585 * poststore
586 */
587void assign_prefetches(GhostCommunicator &comm) {
588 for (auto it = comm.communications.begin(); it != comm.communications.end();
589 it += 2) {
590 auto next = std::next(it);
591 if (it->type == GHOST_RECV && next->type == GHOST_SEND) {
592 it->type |= GHOST_PREFETCH | GHOST_PSTSTORE;
593 next->type |= GHOST_PREFETCH | GHOST_PSTSTORE;
594 }
595 }

600GhostCommunicator RegularDecomposition::prepare_comm() {
601 // When running on a single MPI rank, the ghost cells are not needed,
602 // as the corresponding physical cell is available on the same MPI rank.
603 // The cell neighborships are set up such that cells across the periodic
604 // boundaries are connected as neighbors directly.
605 if (m_comm.size() == 1)
606 return {};
607
608 int dir, lr, i, cnt, n_comm_cells[3];
609 Utils::Vector3i lc{}, hc{}, done{};
610
611 auto const comm_info = Utils::Mpi::cart_get<3>(m_comm);
612 auto const node_neighbors = Utils::Mpi::cart_neighbors<3>(m_comm);
613
614 /* calculate number of communications */
615 std::size_t num = 0;
616 for (dir = 0; dir < 3; dir++) {
617 for (lr = 0; lr < 2; lr++) {
618 /* No communication for border of non periodic direction */
619 if (comm_info.dims[dir] == 1)
620 num++;
621 else
622 num += 2;
623 }
624 }
625
626 /* prepare communicator */
627 auto ghost_comm = GhostCommunicator{m_comm, num};
628
629 /* number of cells to communicate in a direction */
630 n_comm_cells[0] = cell_grid[1] * cell_grid[2];
631 n_comm_cells[1] = cell_grid[2] * ghost_cell_grid[0];
632 n_comm_cells[2] = ghost_cell_grid[0] * ghost_cell_grid[1];
633
634 cnt = 0;
635 /* direction loop: x, y, z */
636 for (dir = 0; dir < 3; dir++) {
637 lc[(dir + 1) % 3] = 1 - done[(dir + 1) % 3];
638 lc[(dir + 2) % 3] = 1 - done[(dir + 2) % 3];
639 hc[(dir + 1) % 3] = cell_grid[(dir + 1) % 3] + done[(dir + 1) % 3];
640 hc[(dir + 2) % 3] = cell_grid[(dir + 2) % 3] + done[(dir + 2) % 3];
641 /* lr loop: left right */
642 /* here we could in principle build in a one sided ghost
643 communication, simply by taking the lr loop only over one
644 value */
645 for (lr = 0; lr < 2; lr++) {
646 if (comm_info.dims[dir] == 1) {
647 /* just copy cells on a single node */
648 ghost_comm.communications[cnt].type = GHOST_LOCL;
649 ghost_comm.communications[cnt].node = m_comm.rank();
650
651 /* Buffer has to contain Send and Recv cells -> factor 2 */
652 ghost_comm.communications[cnt].part_lists.resize(2 * n_comm_cells[dir]);
653
654 /* fill send ghost_comm cells */
655 lc[dir] = hc[dir] = 1 + lr * (cell_grid[dir] - 1);
656
657 fill_comm_cell_lists(ghost_comm.communications[cnt].part_lists.data(),
658 lc, hc);
659
660 /* fill recv ghost_comm cells */
661 lc[dir] = hc[dir] = 0 + (1 - lr) * (cell_grid[dir] + 1);
662
663 /* place receive cells after send cells */
664 fill_comm_cell_lists(
665 &ghost_comm.communications[cnt].part_lists[n_comm_cells[dir]], lc,
666 hc);
667
668 cnt++;
669 } else {
670 /* i: send/recv loop */
671 for (i = 0; i < 2; i++) {
672 if ((comm_info.coords[dir] + i) % 2 == 0) {
673 ghost_comm.communications[cnt].type = GHOST_SEND;
674 ghost_comm.communications[cnt].node = node_neighbors[2 * dir + lr];
675 ghost_comm.communications[cnt].part_lists.resize(n_comm_cells[dir]);
676 /* prepare folding of ghost positions */
677
678 lc[dir] = hc[dir] = 1 + lr * (cell_grid[dir] - 1);
679
680 fill_comm_cell_lists(
681 ghost_comm.communications[cnt].part_lists.data(), lc, hc);
682 cnt++;
683 }
684 if ((comm_info.coords[dir] + (1 - i)) % 2 == 0) {
685 ghost_comm.communications[cnt].type = GHOST_RECV;
686 ghost_comm.communications[cnt].node =
687 node_neighbors[2 * dir + (1 - lr)];
688 ghost_comm.communications[cnt].part_lists.resize(n_comm_cells[dir]);
689
690 lc[dir] = hc[dir] = (1 - lr) * (cell_grid[dir] + 1);
691
692 fill_comm_cell_lists(
693 ghost_comm.communications[cnt].part_lists.data(), lc, hc);
694 cnt++;
695 }
696 }
697 }
698 done[dir] = 1;
699 }
700 }
701
702 return ghost_comm;
703}
