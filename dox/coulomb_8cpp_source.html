303 /** @brief Compute the net charge rescaled by the smallest non-zero charge. */
304 static auto calc_charge_excess_ratio(std::vector<double> const &charges) {
305  using namespace boost::accumulators;
306  using KahanSum = accumulator_set<double, features<tag::sum_kahan>>;
307 
308  KahanSum q_sum;
309  auto q_min = std::numeric_limits<double>::infinity();
310 
311  for (auto const q : charges) {
312  if (q != 0.) {
313  q_sum(q);
314  q_min = std::min(q_min, std::abs(q));
315  }
316  }
317 
318  return std::abs(sum_kahan(q_sum)) / q_min;
319 }

321 void check_charge_neutrality(double relative_tolerance) {
322  // collect non-zero particle charges from all nodes
323  auto const &local_particles = cell_structure.local_particles();
324  std::vector<double> local_charges;
325  for (auto const &p : local_particles) {
326  local_charges.push_back(p.q());
327  }
328  std::vector<std::vector<double>> node_charges;
329  boost::mpi::gather(comm_cart, local_charges, node_charges, 0);
330 
331  // run Kahan sum on charges
332  auto excess_ratio = 0.;
333  if (this_node == 0) {
334  auto charges = std::move(local_charges);
335  for (auto it = ++node_charges.begin(); it != node_charges.end(); ++it) {
336  charges.insert(charges.end(), it->begin(), it->end());
337  }
338  excess_ratio = calc_charge_excess_ratio(charges);
339  }
340  boost::mpi::broadcast(comm_cart, excess_ratio, 0);
341 
342  if (excess_ratio >= relative_tolerance) {
343  std::ostringstream serializer;
344  serializer << std::scientific << std::setprecision(4);
345  serializer << excess_ratio;
346  throw std::runtime_error(
347  "The system is not charge neutral. Please neutralize the system "
348  "before adding a new actor by adding the corresponding counterions "
349  "to the system. Alternatively you can turn off the electroneutrality "
350  "check by supplying check_neutrality=False when creating the actor. "
351  "In this case you may be simulating a non-neutral system which will "
352  "affect physical observables like e.g. the pressure, the chemical "
353  "potentials of charged species or potential energies of the system. "
354  "Since simulations of non charge neutral systems are special please "
355  "make sure you know what you are doing. The relative charge excess "
356  "was " +
357  serializer.str());
358  }
359 }
