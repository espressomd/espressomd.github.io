268/** @brief Compute the net charge rescaled by the smallest non-zero charge. */
269static auto calc_charge_excess_ratio(std::vector<double> const &charges) {
270 using namespace boost::accumulators;
271 using KahanSum = accumulator_set<double, features<tag::sum_kahan>>;
272
273 KahanSum q_sum;
274 auto q_min = std::numeric_limits<double>::infinity();
275
276 for (auto const q : charges) {
277 if (q != 0.) {
278 q_sum(q);
279 q_min = std::min(q_min, std::abs(q));
280 }
281 }
282
283 return std::abs(sum_kahan(q_sum)) / q_min;
284}

286void check_charge_neutrality(System::System const &system,
287 double relative_tolerance) {
288 // collect non-zero particle charges from all nodes
289 std::vector<double> local_charges;
290 for (auto const &p : system.cell_structure->local_particles()) {
291 local_charges.push_back(p.q());
292 }
293 std::vector<std::vector<double>> node_charges;
294 boost::mpi::gather(comm_cart, local_charges, node_charges, 0);
295
296 // run Kahan sum on charges
297 auto excess_ratio = 0.;
298 if (this_node == 0) {
299 auto charges = std::move(local_charges);
300 for (auto it = ++node_charges.begin(); it != node_charges.end(); ++it) {
301 charges.insert(charges.end(), it->begin(), it->end());
302 }
303 excess_ratio = calc_charge_excess_ratio(charges);
304 }
305 boost::mpi::broadcast(comm_cart, excess_ratio, 0);
306
307 if (excess_ratio >= relative_tolerance) {
308 std::ostringstream serializer;
309 serializer << std::scientific << std::setprecision(4);
310 serializer << excess_ratio;
311 throw std::runtime_error(
312 "The system is not charge neutral. Please neutralize the system "
313 "before adding a new actor by adding the corresponding counterions "
314 "to the system. Alternatively you can turn off the electroneutrality "
315 "check by supplying check_neutrality=False when creating the actor. "
316 "In this case you may be simulating a non-neutral system which will "
317 "affect physical observables like e.g. the pressure, the chemical "
318 "potentials of charged species or potential energies of the system. "
319 "Since simulations of non charge neutral systems are special please "
320 "make sure you know what you are doing. The relative charge excess "
321 "was " +
322 serializer.str());
323 }
324}
