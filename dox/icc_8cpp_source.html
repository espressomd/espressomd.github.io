59 /** Calculate the electrostatic forces between source charges (= real charges)
60  * and wall charges. For each electrostatic method, the proper functions
61  * for short- and long-range parts are called. Long-range parts are calculated
62  * directly, short-range parts need helper functions according to the particle
63  * data organisation. This is a modified version of \ref force_calc.
64  */
65 static void force_calc_icc(
66  CellStructure &cell_structure, ParticleRange const &particles,
67  ParticleRange const &ghost_particles,
68  Coulomb::ShortRangeForceKernel::result_type const &coulomb_kernel,
69  Coulomb::ShortRangeForceCorrectionsKernel::result_type const &elc_kernel) {
70  // reset forces
71  for (auto &p : particles) {
72  p.force() = {};
73  }
74  for (auto &p : ghost_particles) {
75  p.force() = {};
76  }
77 
78  // calc ICC forces
79  cell_structure.non_bonded_loop(
80  [coulomb_kernel_ptr = coulomb_kernel.get_ptr(),
81  elc_kernel_ptr = elc_kernel.get_ptr()](Particle &p1, Particle &p2,
82  Distance const &d) {
83  auto const q1q2 = p1.q() * p2.q();
84  if (q1q2 != 0.) {
85  auto force = (*coulomb_kernel_ptr)(q1q2, d.vec21, std::sqrt(d.dist2));
86  p1.force() += force;
87  p2.force() -= force;
88 #ifdef P3M
89  if (elc_kernel_ptr) {
90  (*elc_kernel_ptr)(p1, p2, q1q2);
91  }
92 #endif // P3M
93  }
94  });
95 
96  Coulomb::calc_long_range_force(particles);
97 }

99 void ICCStar::iteration(CellStructure &cell_structure,
100  ParticleRange const &particles,
101  ParticleRange const &ghost_particles) {
102 
103  try {
104  sanity_check();
105  } catch (std::runtime_error const &err) {
106  runtimeErrorMsg() << err.what();
107  return;
108  }
109 
110  auto const prefactor =
111  boost::apply_visitor(GetCoulombPrefactor{}, *electrostatics_actor);
112  auto const pref = 1. / (prefactor * 2. * Utils::pi());
113  auto const kernel = Coulomb::pair_force_kernel();
114  auto const elc_kernel = Coulomb::pair_force_elc_kernel();
115  icc_cfg.citeration = 0;
116 
117  auto global_max_rel_diff = 0.;
118 
119  for (int j = 0; j < icc_cfg.max_iterations; j++) {
120  auto charge_density_max = 0.;
121 
122  // calculate electrostatic forces (SR+LR) excluding self-interactions
123  force_calc_icc(cell_structure, particles, ghost_particles, kernel,
124  elc_kernel);
125  cell_structure.ghosts_reduce_forces();
126 
127  auto max_rel_diff = 0.;
128 
129  for (auto &p : particles) {
130  auto const pid = p.id();
131  if (pid >= icc_cfg.first_id and pid < icc_cfg.n_icc + icc_cfg.first_id) {
132  auto const id = p.id() - icc_cfg.first_id;
133  /* the dielectric-related prefactor: */
134  auto const eps_in = icc_cfg.epsilons[id];
135  auto const eps_out = icc_cfg.eps_out;
136  auto const del_eps = (eps_in - eps_out) / (eps_in + eps_out);
137  /* calculate the electric field at the certain position */
138  auto const local_e_field = p.force() / p.q() + icc_cfg.ext_field;
139 
140  if (local_e_field.norm2() == 0.) {
141  runtimeErrorMsg()
142  << "ICC found zero electric field on a charge. This must "
143  "never happen";
144  }
145 
146  auto const charge_density_old = p.q() / icc_cfg.areas[id];
147 
148  charge_density_max =
149  std::max(charge_density_max, std::abs(charge_density_old));
150 
151  auto const charge_density_update =
152  del_eps * pref * (local_e_field * icc_cfg.normals[id]) +
153  2. * icc_cfg.eps_out / (icc_cfg.eps_out + icc_cfg.epsilons[id]) *
154  icc_cfg.sigmas[id];
155  /* relative variation: never use an estimator which can be negative
156  * here */
157  auto const charge_density_new =
158  (1. - icc_cfg.relaxation) * charge_density_old +
159  (icc_cfg.relaxation) * charge_density_update;
160 
161  /* Take the largest error to check for convergence */
162  auto const relative_difference =
163  std::abs((charge_density_new - charge_density_old) /
164  (charge_density_max +
165  std::abs(charge_density_new + charge_density_old)));
166 
167  max_rel_diff = std::max(max_rel_diff, relative_difference);
168 
169  p.q() = charge_density_new * icc_cfg.areas[id];
170 
171  /* check if the charge now is more than 1e6, to determine if ICC still
172  * leads to reasonable results. This is kind of an arbitrary measure
173  * but does a good job of spotting divergence! */
174  if (std::abs(p.q()) > 1e6) {
175  runtimeErrorMsg()
176  << "Particle with id " << p.id() << " has a charge (q=" << p.q()
177  << ") that is too large for the ICC algorithm";
178 
179  max_rel_diff = std::numeric_limits<double>::infinity();
180  break;
181  }
182  }
183  }
184 
185  /* Update charges on ghosts. */
186  cell_structure.ghosts_update(Cells::DATA_PART_PROPERTIES);
187 
188  icc_cfg.citeration++;
189 
190  boost::mpi::all_reduce(comm_cart, max_rel_diff, global_max_rel_diff,
191  boost::mpi::maximum<double>());
192 
193  if (global_max_rel_diff < icc_cfg.convergence)
194  break;
195  }
196 
197  if (global_max_rel_diff > icc_cfg.convergence) {
198  runtimeErrorMsg()
199  << "ICC failed to converge in the given number of maximal steps.";
200  }
201 
202  on_particle_charge_change();
203 }

205 void icc_data::sanity_checks() const {
206  if (convergence <= 0.)
207  throw std::domain_error("Parameter 'convergence' must be > 0");
208  if (relaxation < 0. or relaxation > 2.)
209  throw std::domain_error("Parameter 'relaxation' must be >= 0 and <= 2");
210  if (max_iterations <= 0)
211  throw std::domain_error("Parameter 'max_iterations' must be > 0");
212  if (first_id < 0)
213  throw std::domain_error("Parameter 'first_id' must be >= 0");
214  if (eps_out <= 0.)
215  throw std::domain_error("Parameter 'eps_out' must be > 0");
216 
217  assert(n_icc >= 1);
218  assert(areas.size() == n_icc);
219  assert(epsilons.size() == n_icc);
220  assert(sigmas.size() == n_icc);
221  assert(normals.size() == n_icc);
222 }

235  template <typename T>

250 #endif // P3M

255  throw std::runtime_error("ICC does not work with ReactionField.");
256  }
