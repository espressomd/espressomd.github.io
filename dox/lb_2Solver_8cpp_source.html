155 std::optional<Utils::Vector3d>
156 Solver::get_interpolated_velocity(Utils::Vector3d const &pos) const {
157  /* calculate fluid velocity at particle's position
158  this is done by linear interpolation
159  (Eq. (11) Ahlrichs and Duenweg, JCP 111(17):8225 (1999)) */
160  return std::visit(
161  [&](auto &ptr) {
162  auto const agrid = ptr->get_agrid();
163  auto const &box_geo = *System::get_system().box_geo;
164  auto const lb_pos = box_geo.folded_position(pos) / agrid;
165  return ptr->get_velocity_at_pos(lb_pos, false);
166  },
167  *impl->solver);
168 }

182 Utils::Vector3d
183 Solver::get_coupling_interpolated_velocity(Utils::Vector3d const &pos) const {
184  return std::visit(
185  [&](auto &ptr) {
186  auto const agrid = ptr->get_agrid();
187  auto const res = ptr->get_velocity_at_pos(pos / agrid, true);
188  assert(res);
189  return *res * (ptr->get_agrid() / ptr->get_tau());
190  },
191  *impl->solver);
192 }
