66 Variant LBFluid::do_call_method(std::string const &name,
67  VariantMap const &params) {
68  if (name == "activate") {
69  context()->parallel_try_catch([this]() {
70  ::System::get_system().lb.set<::LB::LBWalberla>(m_instance, m_lb_params);
71  });
72  m_is_active = true;
73  return {};
74  }
75  if (name == "deactivate") {
76  if (m_is_active) {
77  ::System::get_system().lb.reset();
78  m_is_active = false;
79  }
80  return {};
81  }
82  if (name == "add_force_at_pos") {
83  auto const &box_geo = *::System::get_system().box_geo;
84  auto const pos = get_value<Utils::Vector3d>(params, "pos");
85  auto const f = get_value<Utils::Vector3d>(params, "force");
86  auto const folded_pos = box_geo.folded_position(pos);
87  m_instance->add_force_at_pos(folded_pos * m_conv_dist, f * m_conv_force);
88  return {};
89  }
90  if (name == "get_interpolated_velocity") {
91  auto const pos = get_value<Utils::Vector3d>(params, "pos");
92  return get_interpolated_velocity(pos);
93  }
94  if (name == "get_pressure_tensor") {
95  return get_average_pressure_tensor();
96  }
97  if (name == "load_checkpoint") {
98  auto const path = get_value<std::string>(params, "path");
99  auto const mode = get_value<int>(params, "mode");
100  load_checkpoint(path, mode);
101  return {};
102  }
103  if (name == "save_checkpoint") {
104  auto const path = get_value<std::string>(params, "path");
105  auto const mode = get_value<int>(params, "mode");
106  save_checkpoint(path, mode);
107  return {};
108  }
109  if (name == "clear_boundaries") {
110  m_instance->clear_boundaries();
111  ::System::get_system().on_lb_boundary_conditions_change();
112  return {};
113  }
114  if (name == "add_boundary_from_shape") {
115  m_instance->update_boundary_from_shape(
116  get_value<std::vector<int>>(params, "raster"),
117  get_value<std::vector<double>>(params, "values"));
118  return {};
119  }
120  if (name == "get_lattice_speed") {
121  return 1. / m_conv_speed;
122  }
123 
124  return Base::do_call_method(name, params);
125 }

127 void LBFluid::do_construct(VariantMap const &params) {
128  m_lattice = get_value<std::shared_ptr<LatticeWalberla>>(params, "lattice");
129  m_vtk_writers =
130  get_value_or<decltype(m_vtk_writers)>(params, "vtk_writers", {});
131  auto const tau = get_value<double>(params, "tau");
132  auto const agrid = get_value<double>(m_lattice->get_parameter("agrid"));
133  auto const visc = get_value<double>(params, "kinematic_viscosity");
134  auto const dens = get_value<double>(params, "density");
135  auto const kT = get_value<double>(params, "kT");
136  auto const ext_f = get_value<Utils::Vector3d>(params, "ext_force_density");
137  auto const single_precision = get_value<bool>(params, "single_precision");
138  m_lb_params = std::make_shared<::LB::LBWalberlaParams>(agrid, tau);
139  m_is_active = false;
140  m_seed = get_value<int>(params, "seed");
141  context()->parallel_try_catch([&]() {
142  if (tau <= 0.) {
143  throw std::domain_error("Parameter 'tau' must be > 0");
144  }
145  m_conv_dist = 1. / agrid;
146  m_conv_visc = Utils::int_pow<1>(tau) / Utils::int_pow<2>(agrid);
147  m_conv_energy = Utils::int_pow<2>(tau) / Utils::int_pow<2>(agrid);
148  m_conv_dens = Utils::int_pow<3>(agrid);
149  m_conv_speed = Utils::int_pow<1>(tau) / Utils::int_pow<1>(agrid);
150  m_conv_press = Utils::int_pow<2>(tau) * Utils::int_pow<1>(agrid);
151  m_conv_force = Utils::int_pow<2>(tau) / Utils::int_pow<1>(agrid);
152  m_conv_force_dens = Utils::int_pow<2>(tau) * Utils::int_pow<2>(agrid);
153  auto const lb_lattice = m_lattice->lattice();
154  auto const lb_visc = m_conv_visc * visc;
155  auto const lb_dens = m_conv_dens * dens;
156  auto const lb_kT = m_conv_energy * kT;
157  auto const lb_ext_f = m_conv_force_dens * ext_f;
158  if (m_seed < 0) {
159  throw std::domain_error("Parameter 'seed' must be >= 0");
160  }
161  if (lb_kT < 0.) {
162  throw std::domain_error("Parameter 'kT' must be >= 0");
163  }
164  if (lb_dens <= 0.) {
165  throw std::domain_error("Parameter 'density' must be > 0");
166  }
167  if (lb_visc < 0.) {
168  throw std::domain_error("Parameter 'kinematic_viscosity' must be >= 0");
169  }
170  m_instance =
171  new_lb_walberla(lb_lattice, lb_visc, lb_dens, single_precision);
172  auto const &system = ::System::get_system();
173  if (auto le_protocol = system.lees_edwards->get_protocol()) {
174  if (lb_kT != 0.) {
175  throw std::runtime_error(
176  "Lees-Edwards LB doesn't support thermalization");
177  }
178  auto const &le_bc = system.box_geo->lees_edwards_bc();
179  auto lees_edwards_object = std::make_unique<LeesEdwardsPack>(
180  le_bc.shear_direction, le_bc.shear_plane_normal,
181  [this, le_protocol, &system]() {
182  return get_pos_offset(system.get_sim_time(), *le_protocol) /
183  m_lb_params->get_agrid();
184  },
185  [this, le_protocol, &system]() {
186  return get_shear_velocity(system.get_sim_time(), *le_protocol) *
187  (m_lb_params->get_tau() / m_lb_params->get_agrid());
188  });
189  m_instance->set_collision_model(std::move(lees_edwards_object));
190  } else {
191  m_instance->set_collision_model(lb_kT, m_seed);
192  }
193  m_instance->ghost_communication(); // synchronize ghost layers
194  m_instance->set_external_force(lb_ext_f);
195  for (auto &vtk : m_vtk_writers) {
196  vtk->attach_to_lattice(m_instance, get_latice_to_md_units_conversion());
197  }
198  });
199 }

201 std::vector<Variant> LBFluid::get_average_pressure_tensor() const {
202  auto const local = m_instance->get_pressure_tensor() / m_conv_press;
203  auto const tensor_flat = mpi_reduce_sum(context()->get_comm(), local);
204  auto tensor = Utils::Matrix<double, 3, 3>{};
205  std::copy(tensor_flat.begin(), tensor_flat.end(), tensor.m_data.begin());
206  return std::vector<Variant>{tensor.row<0>().as_vector(),
207  tensor.row<1>().as_vector(),
208  tensor.row<2>().as_vector()};
209 }

211 Variant LBFluid::get_interpolated_velocity(Utils::Vector3d const &pos) const {
212  auto const &box_geo = *::System::get_system().box_geo;
213  auto const lb_pos = box_geo.folded_position(pos) * m_conv_dist;

219 void LBFluid::load_checkpoint(std::string const &filename, int mode) {
220  auto &lb_obj = *m_instance;
221 
222  auto const read_metadata = [&lb_obj](CheckpointFile &cpfile) {
223  auto const expected_grid_size = lb_obj.get_lattice().get_grid_dimensions();
224  auto const expected_pop_size = lb_obj.stencil_size();
225  Utils::Vector3i read_grid_size;
226  std::size_t read_pop_size;
227  cpfile.read(read_grid_size);
228  cpfile.read(read_pop_size);
229  if (read_grid_size != expected_grid_size) {
230  std::stringstream message;
231  message << "grid dimensions mismatch, "
232  << "read [" << read_grid_size << "], "
233  << "expected [" << expected_grid_size << "].";
234  throw std::runtime_error(message.str());
235  }
236  if (read_pop_size != expected_pop_size) {
237  throw std::runtime_error("population size mismatch, read " +
238  std::to_string(read_pop_size) + ", expected " +
239  std::to_string(expected_pop_size) + ".");
240  }
241  };
242 
243  auto const read_data = [&lb_obj](CheckpointFile &cpfile) {
244  auto const grid_size = lb_obj.get_lattice().get_grid_dimensions();
245  auto const i_max = grid_size[0];
246  auto const j_max = grid_size[1];
247  auto const k_max = grid_size[2];
248  LBWalberlaNodeState cpnode;
249  cpnode.populations.resize(lb_obj.stencil_size());
250  for (int i = 0; i < i_max; i++) {
251  for (int j = 0; j < j_max; j++) {
252  for (int k = 0; k < k_max; k++) {
253  auto const ind = Utils::Vector3i{{i, j, k}};
254  cpfile.read(cpnode.populations);
255  cpfile.read(cpnode.last_applied_force);
256  cpfile.read(cpnode.is_boundary);
257  if (cpnode.is_boundary) {
258  cpfile.read(cpnode.slip_velocity);
259  }
260  lb_obj.set_node_population(ind, cpnode.populations);
261  lb_obj.set_node_last_applied_force(ind, cpnode.last_applied_force);
262  if (cpnode.is_boundary) {
263  lb_obj.set_node_velocity_at_boundary(ind, cpnode.slip_velocity);
264  }
265  }
266  }
267  }
268  };
269 
270  auto const on_success = [&lb_obj]() {
271  lb_obj.ghost_communication();
272  lb_obj.reallocate_ubb_field();
273  };
274 
275  load_checkpoint_common(*context(), "LB", filename, mode, read_metadata,
276  read_data, on_success);
277 }

279 void LBFluid::save_checkpoint(std::string const &filename, int mode) {
280  auto &lb_obj = *m_instance;
281 
282  auto const write_metadata = [&lb_obj,
283  mode](std::shared_ptr<CheckpointFile> cpfile_ptr,
284  Context const &context) {
285  auto const grid_size = lb_obj.get_lattice().get_grid_dimensions();
286  auto const pop_size = lb_obj.stencil_size();
287  if (context.is_head_node()) {
288  cpfile_ptr->write(grid_size);
289  cpfile_ptr->write(pop_size);
290  unit_test_handle(mode);
291  }
292  };
293 
294  auto const on_failure = [](std::shared_ptr<CheckpointFile> const &,
295  Context const &context) {
296  if (context.is_head_node()) {
297  auto failure = true;
298  boost::mpi::broadcast(context.get_comm(), failure, 0);
299  }
300  };
301 
302  auto const write_data = [&lb_obj,
303  mode](std::shared_ptr<CheckpointFile> cpfile_ptr,
304  Context const &context) {
305  auto const get_node_checkpoint =
306  [&](Utils::Vector3i const &ind) -> std::optional<LBWalberlaNodeState> {
307  auto const pop = lb_obj.get_node_population(ind);
308  auto const laf = lb_obj.get_node_last_applied_force(ind);
309  auto const lbb = lb_obj.get_node_is_boundary(ind);
310  auto const vbb = lb_obj.get_node_velocity_at_boundary(ind);
311  if (pop and laf and lbb and ((*lbb) ? vbb.has_value() : true)) {
312  LBWalberlaNodeState cpnode;
313  cpnode.populations = *pop;
314  cpnode.last_applied_force = *laf;
315  cpnode.is_boundary = *lbb;
316  if (*lbb) {
317  cpnode.slip_velocity = *vbb;
318  }
319  return {cpnode};
320  }
321  return std::nullopt;
322  };
323 
324  auto failure = false;
325  auto const &comm = context.get_comm();
326  auto const is_head_node = context.is_head_node();
327  auto const unit_test_mode = (mode != static_cast<int>(CptMode::ascii)) and
328  (mode != static_cast<int>(CptMode::binary));
329  auto const grid_size = lb_obj.get_lattice().get_grid_dimensions();
330  auto const i_max = grid_size[0];
331  auto const j_max = grid_size[1];
332  auto const k_max = grid_size[2];
333  LBWalberlaNodeState cpnode;
334  for (int i = 0; i < i_max; i++) {
335  for (int j = 0; j < j_max; j++) {
336  for (int k = 0; k < k_max; k++) {
337  auto const ind = Utils::Vector3i{{i, j, k}};
338  auto const result = get_node_checkpoint(ind);
339  if (!unit_test_mode) {
340  assert(1 == boost::mpi::all_reduce(comm, static_cast<int>(!!result),
341  std::plus<>()) &&
342  "Incorrect number of return values");
343  }
344  if (is_head_node) {
345  if (result) {
346  cpnode = *result;
347  } else {
348  comm.recv(boost::mpi::any_source, 42, cpnode);
349  }
350  auto &cpfile = *cpfile_ptr;
351  cpfile.write(cpnode.populations);
352  cpfile.write(cpnode.last_applied_force);
353  cpfile.write(cpnode.is_boundary);
354  if (cpnode.is_boundary) {
355  cpfile.write(cpnode.slip_velocity);
356  }
357  boost::mpi::broadcast(comm, failure, 0);
358  } else {
359  if (result) {
360  comm.send(0, 42, *result);
361  }
362  boost::mpi::broadcast(comm, failure, 0);
363  if (failure) {
364  return;
365  }
366  }
367  }
368  }
369  }
370  };
371 
372  save_checkpoint_common(*context(), "LB", filename, mode, write_metadata,
373  write_data, on_failure);
374 }
