34BMHTF_Parameters::BMHTF_Parameters(double a, double b, double c, double d,
35 double sig, double cutoff)
36 : A{a}, B{b}, C{c}, D{d}, sig{sig}, cut{cutoff} {
37 if (a < 0.) {
38 throw std::domain_error("BMHTF parameter 'a' has to be >= 0");
39 }
40 if (c < 0.) {
41 throw std::domain_error("BMHTF parameter 'c' has to be >= 0");
42 }
43 if (d < 0.) {
44 throw std::domain_error("BMHTF parameter 'd' has to be >= 0");
45 }
46 if (cutoff < 0.) {
47 throw std::domain_error("BMHTF parameter 'cutoff' has to be >= 0");
48 }
49 computed_shift = C / Utils::int_pow<6>(cut) + D / Utils::int_pow<8>(cut) -
50 A * std::exp(B * (sig - cut));
51}
