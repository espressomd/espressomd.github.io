37 /**
38  * @brief Vector pointing from the real particle to the virtual site.
39  *
40  * @return Relative distance.
41  */
42 static auto connection_vector(Particle const &p_ref, Particle const &p_vs) {
43  // Calculate the quaternion defining the orientation of the vector connecting
44  // the virtual site and the real particle
45  // This is obtained, by multiplying the quaternion representing the director
46  // of the real particle with the quaternion of the virtual particle, which
47  // specifies the relative orientation.
48  auto const director = Utils::convert_quaternion_to_director(
49  p_ref.quat() * p_vs.vs_relative().rel_orientation)
50  .normalize();
51 
52  return p_vs.vs_relative().distance * director;
53 }

55 /**
56  * @brief Velocity of the virtual site
57  * @param p_ref Reference particle for the virtual site.
58  * @param p_vs Virtual site.
59  * @return Velocity of the virtual site.
60  */
61 static Utils::Vector3d velocity(Particle const &p_ref, Particle const &p_vs) {
62  auto const d = connection_vector(p_ref, p_vs);
63 
64  // Get omega of real particle in space-fixed frame
65  auto const omega_space_frame =
66  convert_vector_body_to_space(p_ref, p_ref.omega());
67  // Obtain velocity from v = v_real particle + omega_real_particle * director
68  return vector_product(omega_space_frame, d) + p_ref.v();
69 }

71 /**
72  * @brief Get real particle tracked by a virtual site.
73  *
74  * @param p Virtual site.
75  * @return Pointer to real particle.
76  */
77 static Particle *get_reference_particle(Particle const &p) {
78  if (!p.is_virtual()) {
79  return nullptr;
80  }
81  auto const &vs_rel = p.vs_relative();
82  auto p_ref_ptr = cell_structure.get_local_particle(vs_rel.to_particle_id);
83  if (!p_ref_ptr) {
84  runtimeErrorMsg() << "No real particle with id " << vs_rel.to_particle_id
85  << " for virtual site with id " << p.id();
86  }
87  return p_ref_ptr;
88 }

90 /**
91  * @brief Constraint force to hold the particle at its prescribed position.
92  *
93  * @param p_ref Reference particle.
94  * @param p_vs Virtual site.
95  * @return Constraint force.
96  */
97 static auto constraint_stress(Particle const &p_ref, Particle const &p_vs) {
98  /* The constraint force is minus the force on the particle, make it force
99  * free. The counter force is translated by the connection vector to the
100  * real particle, hence the virial stress is */
101  return tensor_product(-p_vs.force(), connection_vector(p_ref, p_vs));
102 }

104 void VirtualSitesRelative::update() const {
105  cell_structure.ghosts_update(Cells::DATA_PART_POSITION |
106  Cells::DATA_PART_MOMENTUM);
107 
108  auto const particles = cell_structure.local_particles();
109  for (auto &p : particles) {
110  auto const *p_ref_ptr = get_reference_particle(p);
111  if (!p_ref_ptr)
112  continue;
113 
114  auto const &p_ref = *p_ref_ptr;
115  auto new_pos = p_ref.pos() + connection_vector(p_ref, p);
116  /* The shift has to respect periodic boundaries: if the reference
117  * particles is not in the same image box, we potentially avoid shifting
118  * to the other side of the box. */
119  auto shift = box_geo.get_mi_vector(new_pos, p.pos());
120  p.pos() += shift;
121  Utils::Vector3i image_shift{};
122  fold_position(shift, image_shift, box_geo);
123  p.image_box() = p_ref.image_box() - image_shift;
124 
125  p.v() = velocity(p_ref, p);
126 
127  if (box_geo.type() == BoxType::LEES_EDWARDS) {
128  auto const &lebc = box_geo.lees_edwards_bc();
129  auto const shear_dir = lebc.shear_direction;
130  auto const shear_normal = lebc.shear_plane_normal;
131  auto const le_vel = lebc.shear_velocity;
132  Utils::Vector3i n_shifts{};
133  fold_position(new_pos, n_shifts, box_geo);
134  p.v()[shear_dir] -= n_shifts[shear_normal] * le_vel;
135  }
136 
137  if (have_quaternions())
138  p.quat() = p_ref.quat() * p.vs_relative().quat;
139  }
140 
141  if (cell_structure.check_resort_required(particles, skin)) {
142  cell_structure.set_resort_particles(Cells::RESORT_LOCAL);
143  }
144 }

146 // Distribute forces that have accumulated on virtual particles to the
147 // associated real particles
148 void VirtualSitesRelative::back_transfer_forces_and_torques() const {
149  cell_structure.ghosts_reduce_forces();
150 
151  init_forces_ghosts(cell_structure.ghost_particles());
152 
153  // Iterate over all the particles in the local cells
154  for (auto &p : cell_structure.local_particles()) {
155  auto *p_ref_ptr = get_reference_particle(p);
156  if (!p_ref_ptr)
157  continue;
158 
159  // Add forces and torques
160  auto &p_ref = *p_ref_ptr;
161  p_ref.force() += p.force();
162  p_ref.torque() +=
163  vector_product(connection_vector(p_ref, p), p.force()) + p.torque();
164  }
165 }

167 // Rigid body contribution to scalar pressure and pressure tensor
168 Utils::Matrix<double, 3, 3> VirtualSitesRelative::pressure_tensor() const {
169  Utils::Matrix<double, 3, 3> pressure_tensor = {};
170 
171  for (auto const &p : cell_structure.local_particles()) {
172  auto const *p_ref_ptr = get_reference_particle(p);
173  if (p_ref_ptr) {
174  pressure_tensor += constraint_stress(*p_ref_ptr, p);
175  }
176  }
177 
178  return pressure_tensor;
179 }
