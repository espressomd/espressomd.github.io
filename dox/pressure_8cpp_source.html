51std::shared_ptr<Observable_stat> System::calculate_pressure() {
52
53 auto obs_pressure_ptr = std::make_shared<Observable_stat>(
54 9ul, static_cast<std::size_t>(bonded_ias->get_next_key()),
55 nonbonded_ias->get_max_seen_particle_type());
56
57 if (long_range_interactions_sanity_checks()) {
58 return obs_pressure_ptr;
59 }
60
61 auto &obs_pressure = *obs_pressure_ptr;
62
63 on_observable_calc();
64
65 auto const volume = box_geo->volume();
66
67 // Kinetic virial — reduction over local particles
68 auto const kinetic = reduce_over_local_particles<Utils::Matrix<double, 3, 3>>(
69 *cell_structure,
70 [](Utils::Matrix<double, 3, 3> &acc, Particle const &p) {
71 if (!p.is_virtual())
72 acc += Utils::tensor_product(p.v(), p.mass() * p.v());
73 },
74 [](auto &a, auto const &b) { a += b; });
75 std::ranges::copy(Utils::flatten(kinetic), obs_pressure.kinetic_lin.begin());
76
77 auto const coulomb_force_kernel = coulomb.pair_force_kernel();
78 auto const coulomb_pressure_kernel = coulomb.pair_pressure_kernel();
79
80 VerletCriterion<> const verlet_criterion{*this,
81 cell_structure->get_verlet_skin(),
82 get_interaction_range(),
83 coulomb.cutoff(),
84 dipoles.cutoff(),
85 inactive_cutoff};
86 update_cabana_state(*cell_structure, verlet_criterion,
87 get_interaction_range(), propagation->integ_switch);
88
89 PressureBinLayout layout{
90 static_cast<std::size_t>(bonded_ias->get_next_key()),
91 std::size_t(nonbonded_ias->get_max_seen_particle_type() + 1)};
92
93 using exec = Kokkos::DefaultExecutionSpace;
94 Kokkos::View<double **, Kokkos::LayoutRight> local_pressure(
95 "local_pressure", exec().concurrency(), layout.total * 9);
96
97 auto const &unique_particles = cell_structure->get_unique_particles();
98 auto const n_particles = static_cast<int>(unique_particles.size());
99 Kokkos::View<int *> mol_id_view("mol_id", n_particles);
100 auto mol_id_host = Kokkos::create_mirror_view(mol_id_view);
101 for (int i = 0; i < n_particles; ++i)
102 mol_id_host(i) = unique_particles[i]->mol_id();
103 Kokkos::deep_copy(mol_id_view, mol_id_host);
104
105 PressureKernel pair_p_kernel{*bonded_ias,
106 *nonbonded_ias,
107 coulomb,
108 get_ptr(coulomb_force_kernel),
109 get_ptr(coulomb_pressure_kernel),
110 *box_geo,
111#ifdef ESPRESSO_DPD
112 thermostat->dpd.get(),
113#endif
114 cell_structure->get_unique_particles(),
115 local_pressure,
116 layout,
117 cell_structure->get_aosoa(),
118 mol_id_view,
119 maximal_cutoff(),
120 thermostat->thermo_switch};
121
122 auto &bs = cell_structure->bond_state();
123 BondsPressureKernelData bonds_p_data{*bonded_ias, *box_geo, local_pressure,
124 layout, cell_structure->get_aosoa()};
125 PairBondsPressureKernel pair_bp_kernel{
126 bonds_p_data, bs.pair_list, bs.pair_ids, get_ptr(coulomb_force_kernel)};
127 AngleBondsPressureKernel angle_bp_kernel{bonds_p_data, bs.angle_list,
128 bs.angle_ids};
129 DihedralBondsPressureKernel dih_bp_kernel{bonds_p_data, bs.dihedral_list,
130 bs.dihedral_ids};
131
132 cabana_short_range(pair_bp_kernel, angle_bp_kernel, dih_bp_kernel,
133 pair_p_kernel, *cell_structure, get_interaction_range(),
134 bonded_ias->maximal_cutoff(), verlet_criterion,
135 propagation->integ_switch);
136
137 reduce_cabana_pressure(local_pressure, layout, obs_pressure, *bonded_ias,
138 nonbonded_ias->get_max_seen_particle_type() + 1);
139
140#ifdef ESPRESSO_ELECTROSTATICS
141 /* calculate k-space part of electrostatic interaction. */
142 auto const coulomb_pressure = coulomb.calc_pressure_long_range();
143 std::ranges::copy(coulomb_pressure, obs_pressure.coulomb.begin() + 9u);
144#endif
145#ifdef ESPRESSO_DIPOLES
146 /* calculate k-space part of magnetostatic interaction. */
147 dipoles.calc_pressure_long_range();
148#endif
149
150#ifdef ESPRESSO_VIRTUAL_SITES_RELATIVE
151 if (!obs_pressure.virtual_sites.empty()) {
152 auto const vs_pressure = vs_relative_pressure_tensor(*cell_structure);
153 std::ranges::copy(Utils::flatten(vs_pressure),
154 obs_pressure.virtual_sites.begin());
155 }
156#endif
157
158#ifdef ESPRESSO_BOND_CONSTRAINT
159 if (propagation->is_inertial() and bonded_ias->get_n_rigid_bonds() >= 1) {
160 // rigid_bond_virial was accumulated bond-by-bond inside
161 // correct_position_shake() during the last integration step, so it is
162 // already correct regardless of how many rigid bonds a particle
163 // participates in; only the deferred 1/dt^2 factor is applied here.
164 auto const sq_dt = Utils::sqr(get_time_step());
165 auto const &rigid_bond_virial = bonded_ias->rigid_bond_virial;
166 for (std::size_t bond_id = 0; bond_id < rigid_bond_virial.size();
167 ++bond_id) {
168 auto const stress = rigid_bond_virial[bond_id] / sq_dt;
169 auto dest = obs_pressure.bonded_contribution(static_cast<int>(bond_id));
170 for (std::size_t k = 0; k < 9u; ++k)
171 dest[k] += stress[k];
172 }
173 }
174#endif // ESPRESSO_BOND_CONSTRAINT
175
176 obs_pressure.rescale(volume);
177
178 obs_pressure.mpi_reduce();
179 return obs_pressure_ptr;
180}
181} // namespace System
