37  std::vector<Cell *> black_neighbors;
38 
39  /* distribute force calculation work */
40  for (int n = 0; n < m_comm.size(); n++) {
41  if (m_comm.rank() == n) {
42  continue;
43  }

54 GhostCommunicator AtomDecomposition::prepare_comm() {
55  /* no need for comm for only 1 node */
56  if (m_comm.size() == 1) {
57  return GhostCommunicator{m_comm, 0};
58  }
59 
60  auto ghost_comm =
61  GhostCommunicator{m_comm, static_cast<std::size_t>(m_comm.size())};
62  /* every node has its dedicated comm step */
63  for (int n = 0; n < m_comm.size(); n++) {
64  ghost_comm.communications[n].part_lists.resize(1);
65  ghost_comm.communications[n].part_lists[0] = &(cells.at(n).particles());
66  ghost_comm.communications[n].node = n;
67  }
68 
69  return ghost_comm;
70 }
71 void AtomDecomposition::configure_comms() {
72  m_exchange_ghosts_comm = prepare_comm();
73  m_collect_ghost_force_comm = prepare_comm();
74 
75  if (m_comm.size() > 1) {
76  for (int n = 0; n < m_comm.size(); n++) {
77  /* use the prefetched send buffers. Node 0 transmits first and never
78  * prefetches. */
79  if (m_comm.rank() == 0 || m_comm.rank() != n) {
80  m_exchange_ghosts_comm.communications[n].type = GHOST_BCST;
81  } else {
82  m_exchange_ghosts_comm.communications[n].type =
83  GHOST_BCST | GHOST_PREFETCH;
84  }
85  m_collect_ghost_force_comm.communications[n].type = GHOST_RDCE;
86  }
87  /* first round: all nodes except the first one prefetch their send data */
88  if (m_comm.rank() != 0) {
89  m_exchange_ghosts_comm.communications[0].type |= GHOST_PREFETCH;
90  }
91  }

102 void AtomDecomposition::resort(bool global_flag,
103  std::vector<ParticleChange> &diff) {
104  for (auto &p : local().particles()) {
105  m_box.fold_position(p.pos(), p.image_box());
106 
107  p.pos_at_last_verlet_update() = p.pos();
108  }
109 
110  /* Local updates are a NoOp for this decomposition. */
111  if (not global_flag) {
112  return;
113  }
114 
115  /* Sort displaced particles by the node they belong to. */
116  std::vector<std::vector<Particle>> send_buf(m_comm.size());
117  for (auto it = local().particles().begin();
118  it != local().particles().end();) {
119  auto const target_node = id_to_rank(it->id());
120  if (target_node != m_comm.rank()) {
121  diff.emplace_back(RemovedParticle{it->id()});
122  send_buf.at(target_node).emplace_back(std::move(*it));
123  it = local().particles().erase(it);
124  } else {
125  ++it;
126  }
127  }
128 
129  /* Exchange particles */
130  std::vector<std::vector<Particle>> recv_buf(m_comm.size());
131  boost::mpi::all_to_all(m_comm, send_buf, recv_buf);
132 
133  diff.emplace_back(ModifiedList{local().particles()});
134 
135  /* Add new particles belonging to this node */
136  for (auto &parts : recv_buf) {
137  for (auto &p : parts) {
138  local().particles().insert(std::move(p));
139  }
140  }
141 }
