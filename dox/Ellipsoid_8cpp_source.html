31void Ellipsoid::calculate_dist(const Utils::Vector3d &pos, double &dist,
32 Utils::Vector3d &vec) const {
33
34 /* get particle position in reference frame of ellipsoid */
35 Utils::Vector3d const ppos_e = pos - m_center;
36
37 /* set appropriate initial point for Newton's method */
38 double l = 0.;
39 int distance_prefactor = -1;
40 if (not inside_ellipsoid(ppos_e)) {
41 l = *std::ranges::max_element(m_semiaxes) * ppos_e.norm();
42 distance_prefactor = 1;
43 }
44
45 /* find root via Newton's method */
46 double eps = 10.;
47 int step = 0;
48 while ((eps >= 1e-12) and (step < 100)) {
49 auto const l0 = l;
50 l -= newton_term(ppos_e, l0);
51 eps = std::abs(l - l0);
52 step++;
53 }
54
55 /* calculate dist and vec */
56 for (unsigned int i = 0; i < 3; i++) {
57 auto const semi_sq = Utils::sqr(m_semiaxes[i]);
58 vec[i] = (ppos_e[i] - semi_sq * ppos_e[i] / (l + semi_sq));
59 }
60
61 dist = distance_prefactor * m_direction * vec.norm();
62}

64bool Ellipsoid::inside_ellipsoid(const Utils::Vector3d &ppos) const {
65 return Utils::hadamard_division(ppos, m_semiaxes).norm2() <= 1;
66}

68double Ellipsoid::newton_term(const Utils::Vector3d &ppos,
69 const double &l) const {
70 Utils::Vector3d axpos, lax, lax2;
71 for (unsigned int i = 0; i < 3; i++) {
72 axpos[i] = Utils::sqr(m_semiaxes[i]) * Utils::sqr(ppos[i]);
73 lax[i] = l + Utils::sqr(m_semiaxes[i]);
74 lax2[i] = Utils::sqr(lax[i]);
75 }
76
77 return (axpos[0] * lax2[1] * lax2[2] + axpos[1] * lax2[2] * lax2[0] +
78 axpos[2] * lax2[0] * lax2[1] - lax2[0] * lax2[1] * lax2[2]) /
79 (2 * (axpos[0] * lax[1] * lax2[2] + axpos[1] * lax[2] * lax2[0] +
80 axpos[2] * lax[0] * lax2[1] + axpos[0] * lax2[1] * lax[2] +
81 axpos[1] * lax2[2] * lax[0] + axpos[2] * lax2[0] * lax[1] -
82 lax[0] * lax2[1] * lax2[2] - lax2[0] * lax[1] * lax2[2] -
83 lax2[0] * lax2[1] * lax[2]));
84}
