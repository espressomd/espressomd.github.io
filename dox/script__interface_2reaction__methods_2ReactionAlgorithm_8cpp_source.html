105 static auto get_real_particle(boost::mpi::communicator const &comm, int p_id) {
106  assert(p_id >= 0);
107  auto const &system = ::System::get_system();
108  auto &cell_structure = *system.cell_structure;
109  auto ptr = cell_structure.get_local_particle(p_id);
110  if (ptr != nullptr and ptr->is_ghost()) {
111  ptr = nullptr;
112  }
113  assert(boost::mpi::all_reduce(comm, static_cast<int>(ptr != nullptr),
114  std::plus<>()) == 1);
115  return ptr;
116 }

118 Variant ReactionAlgorithm::do_call_method(std::string const &name,
119  VariantMap const &params) {
120  if (name == "calculate_factorial_expression") {
121  if (context()->is_head_node()) {
122  auto &bookkeeping = RE()->get_old_system_state();
123  auto &old_particle_numbers = bookkeeping.old_particle_numbers;
124  auto &reaction = *m_reactions[bookkeeping.reaction_id]->get_reaction();
125  return calculate_factorial_expression(reaction, old_particle_numbers);
126  }
127  return {};
128  }
129  if (name == "get_random_reaction_index") {
130  return RE()->i_random(static_cast<int>(RE()->reactions.size()));
131  }
132  if (name == "potential_energy") {
133  return RE()->calculate_potential_energy();
134  }
135  if (name == "create_new_trial_state") {
136  auto const reaction_id = get_value<int>(params, "reaction_id");
137  Variant result{};
138  context()->parallel_try_catch([&]() {
139  auto const optional = RE()->create_new_trial_state(reaction_id);
140  if (optional) {
141  result = *optional;
142  }
143  });
144  return result;
145  }
146  if (name == "make_reaction_mc_move_attempt") {
147  auto const bf = get_value<double>(params, "bf");
148  auto const E_pot_old = get_value<double>(params, "E_pot_old");
149  auto const E_pot_new = get_value<double>(params, "E_pot_new");
150  auto const reaction_id = get_value<int>(params, "reaction_id");
151  Variant result;
152  context()->parallel_try_catch([&]() {
153  result = RE()->make_reaction_mc_move_attempt(reaction_id, bf, E_pot_old,
154  E_pot_new);
155  });
156  return result;
157  }
158  if (name == "setup_bookkeeping_of_empty_pids") {
159  RE()->setup_bookkeeping_of_empty_pids();
160  } else if (name == "remove_constraint") {
161  RE()->remove_constraint();
162  } else if (name == "set_cylindrical_constraint_in_z_direction") {
163  context()->parallel_try_catch([&]() {
164  RE()->set_cyl_constraint(get_value<double>(params, "center_x"),
165  get_value<double>(params, "center_y"),
166  get_value<double>(params, "radius"));
167  });
168  } else if (name == "set_wall_constraints_in_z_direction") {
169  context()->parallel_try_catch([&]() {
170  RE()->set_slab_constraint(get_value<double>(params, "slab_start_z"),
171  get_value<double>(params, "slab_end_z"));
172  });
173  } else if (name == "get_wall_constraints_in_z_direction") {
174  if (context()->is_head_node()) {
175  return RE()->get_slab_constraint_parameters();
176  }
177  return {};
178  } else if (name == "set_volume") {
179  context()->parallel_try_catch(
180  [&]() { RE()->set_volume(get_value<double>(params, "volume")); });
181  } else if (name == "get_volume") {
182  return RE()->get_volume();
183  } else if (name == "get_acceptance_rate_reaction") {
184  auto const index = get_value<int>(params, "reaction_id");
185  context()->parallel_try_catch([&]() {
186  if (index < 0 or index >= static_cast<int>(m_reactions.size())) {
187  throw std::out_of_range("No reaction with id " + std::to_string(index));
188  }
189  });
190  return m_reactions[index]->get_reaction()->get_acceptance_rate();
191  } else if (name == "set_non_interacting_type") {
192  auto const type = get_value<int>(params, "type");
193  context()->parallel_try_catch([&]() {
194  if (type < 0) {
195  throw std::domain_error("Invalid type: " + std::to_string(type));
196  }
197  });
198  RE()->non_interacting_type = type;
199  ::init_type_map(type);
200  } else if (name == "get_non_interacting_type") {
201  return RE()->non_interacting_type;
202  } else if (name == "displacement_mc_move_for_particles_of_type") {
203  auto const type = get_value<int>(params, "type_mc");
204  auto const n_particles =
205  get_value_or<int>(params, "particle_number_to_be_changed", 1);
206  auto result = false;
207  context()->parallel_try_catch([&]() {
208  result = RE()->make_displacement_mc_move_attempt(type, n_particles);
209  });
210  return result;
211  } else if (name == "delete_particle") {
212  context()->parallel_try_catch(
213  [&]() { RE()->delete_particle(get_value<int>(params, "p_id")); });
214  } else if (name == "delete_reaction") {
215  auto const reaction_id = get_value<int>(params, "reaction_id");
216  auto const index = get_reaction_index(reaction_id);
217  // delete forward and backward reactions
218  delete_reaction(index + 1);
219  delete_reaction(index + 0);
220  } else if (name == "add_reaction") {
221  auto const reaction =
222  get_value<std::shared_ptr<SingleReaction>>(params, "reaction");
223  m_reactions.push_back(reaction);
224  RE()->add_reaction(reaction->get_reaction());
225  } else if (name == "change_reaction_constant") {
226  auto const gamma = get_value<double>(params, "gamma");
227  auto const reaction_id = get_value<int>(params, "reaction_id");
228  context()->parallel_try_catch([&]() {
229  if (reaction_id % 2 == 1) {
230  throw std::invalid_argument("Only forward reactions can be selected");
231  }
232  if (gamma <= 0.) {
233  throw std::domain_error("gamma needs to be a strictly positive value");
234  }
235  });
236  auto const index = get_reaction_index(reaction_id);
237  m_reactions[index + 0]->get_reaction()->gamma = gamma;
238  m_reactions[index + 1]->get_reaction()->gamma = 1. / gamma;
239  } else if (name == "set_charge_of_type") {
240  auto const type = get_value<int>(params, "type");
241  auto const charge = get_value<double>(params, "charge");
242  RE()->charges_of_types[type] = charge;
243  } else if (context()->is_head_node()) {
244  throw std::runtime_error("unknown method '" + name + "()'");
245  }
246  return {};
247 }
