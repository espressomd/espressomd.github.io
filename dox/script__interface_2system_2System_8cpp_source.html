162 void System::do_construct(VariantMap const &params) {
163  /* When reloading the system state from a checkpoint file,
164  * the order of global variables instantiation matters.
165  * The @c box_l must be set before any other global variable.
166  * All these globals re-initialize the cell system, and we
167  * cannot use the default-constructed @c box_geo when e.g.
168  * long-range interactions exist in the system, otherwise
169  * runtime errors about the local geometry being smaller
170  * than the interaction range would be raised.
171  */
172  m_instance = ::System::System::create();
173  ::System::set_system(m_instance);
174 
175  // domain decomposition can only be set after box_l is set
176  m_instance->set_cell_structure_topology(CellStructureType::NSQUARE);
177  do_set_parameter("box_l", params.at("box_l"));
178  m_instance->set_cell_structure_topology(CellStructureType::REGULAR);
179 
180  m_instance->lb.bind_system(m_instance);
181  m_instance->ek.bind_system(m_instance);
182 
183  for (auto const &key : get_parameter_insertion_order()) {
184  if (key != "box_l" and params.count(key) != 0ul) {
185  do_set_parameter(key, params.at(key));
186  }
187  }
188 }

190 static void rotate_system(CellStructure &cell_structure, double phi,
191  double theta, double alpha) {
192  auto const particles = cell_structure.local_particles();
193 
194  // Calculate center of mass
195  Utils::Vector3d local_com{};
196  double local_mass = 0.0;
197 
198  for (auto const &p : particles) {
199  if (not p.is_virtual()) {
200  local_com += p.mass() * p.pos();
201  local_mass += p.mass();
202  }
203  }
204 
205  auto const total_mass =
206  boost::mpi::all_reduce(comm_cart, local_mass, std::plus<>());
207  auto const com =
208  boost::mpi::all_reduce(comm_cart, local_com, std::plus<>()) / total_mass;
209 
210  // Rotation axis in Cartesian coordinates
211  Utils::Vector3d axis;
212  axis[0] = std::sin(theta) * std::cos(phi);
213  axis[1] = std::sin(theta) * std::sin(phi);
214  axis[2] = std::cos(theta);
215 
216  // Rotate particle coordinates
217  for (auto &p : particles) {
218  // Move the center of mass of the system to the origin
219  p.pos() = com + Utils::vec_rotate(axis, alpha, p.pos() - com);
220 #ifdef ROTATION
221  local_rotate_particle(p, axis, alpha);
222 #endif
223  }
224 
225  cell_structure.set_resort_particles(Cells::RESORT_GLOBAL);
226  cell_structure.update_ghosts_and_resort_particle(Cells::DATA_PART_POSITION |
227  Cells::DATA_PART_PROPERTIES);
228 }

230 /** Rescale all particle positions in direction @p dir by a factor @p scale.
231  * @param cell_structure cell structure
232  * @param dir direction to scale (0/1/2 = x/y/z, 3 = x+y+z isotropically)
233  * @param scale factor by which to rescale (>1: stretch, <1: contract)
234  */
235 static void rescale_particles(CellStructure &cell_structure, int dir,
236  double scale) {
237  for (auto &p : cell_structure.local_particles()) {
238  if (dir < 3)
239  p.pos()[dir] *= scale;
240  else {
241  p.pos() *= scale;
242  }
243  }
244 }

246 Variant System::do_call_method(std::string const &name,
247  VariantMap const &parameters) {
248  if (name == "is_system_created") {
249  return system_created;
250  }
251  if (name == "lock_system_creation") {
252  system_created = true;
253  return {};
254  }
255  if (name == "rescale_boxl") {
256  auto &box_geo = *m_instance->box_geo;
257  auto const coord = get_value<int>(parameters, "coord");
258  auto const length = get_value<double>(parameters, "length");
259  auto const scale = (coord == 3) ? length * box_geo.length_inv()[0]
260  : length * box_geo.length_inv()[coord];
261  context()->parallel_try_catch([&]() {
262  if (length <= 0.) {
263  throw std::domain_error("Parameter 'd_new' must be > 0");
264  }
265  });
266  auto new_value = Utils::Vector3d{};
267  if (coord == 3) {
268  new_value = Utils::Vector3d::broadcast(length);
269  } else {
270  new_value = box_geo.length();
271  new_value[coord] = length;
272  }
273  // when shrinking, rescale the particles first
274  if (scale <= 1.) {
275  rescale_particles(*m_instance->cell_structure, coord, scale);
276  m_instance->on_particle_change();
277  }
278  m_instance->box_geo->set_length(new_value);
279  m_instance->on_boxl_change();
280  if (scale > 1.) {
281  rescale_particles(*m_instance->cell_structure, coord, scale);
282  m_instance->on_particle_change();
283  }
284  return {};
285  }
286  if (name == "setup_type_map") {
287  auto const types = get_value<std::vector<int>>(parameters, "type_list");
288  for (auto const type : types) {
289  ::init_type_map(type);
290  }
291  return {};
292  }
293  if (name == "number_of_particles") {
294  auto const type = get_value<int>(parameters, "type");
295  return ::number_of_particles_with_type(type);
296  }
297  if (name == "velocity_difference") {
298  auto const pos1 = get_value<Utils::Vector3d>(parameters, "pos1");
299  auto const pos2 = get_value<Utils::Vector3d>(parameters, "pos2");
300  auto const v1 = get_value<Utils::Vector3d>(parameters, "v1");
301  auto const v2 = get_value<Utils::Vector3d>(parameters, "v2");
302  return m_instance->box_geo->velocity_difference(pos2, pos1, v2, v1);
303  }
304  if (name == "distance_vec") {
305  auto const pos1 = get_value<Utils::Vector3d>(parameters, "pos1");
306  auto const pos2 = get_value<Utils::Vector3d>(parameters, "pos2");
307  return m_instance->box_geo->get_mi_vector(pos2, pos1);
308  }
309  if (name == "rotate_system") {
310  rotate_system(*m_instance->cell_structure,
311  get_value<double>(parameters, "phi"),
312  get_value<double>(parameters, "theta"),
313  get_value<double>(parameters, "alpha"));
314  m_instance->on_particle_change();
315  m_instance->update_dependent_particles();
316  return {};
317  }
318  if (name == "get_propagation_modes_enum") {
319  return make_unordered_map_of_variants(propagation_flags_map());
320  }
321  if (name == "session_shutdown") {
322  if (m_instance) {
323  if (&::System::get_system() == m_instance.get()) {
324  ::System::reset_system();
325  }
326  assert(m_instance.use_count() == 1l);
327  m_instance.reset();
328  }
329  return {};
330  }
331  return {};
332 }
