55 public:
56  Variant do_call_method(std::string const &method_name,
57  VariantMap const &params) override {
58  if (method_name == "add_vtk_writer") {
59  auto vtk = get_value<std::shared_ptr<VTKHandle>>(params, "vtk");
60  auto const needle = find_vtk(vtk);
61  ObjectHandle::context()->parallel_try_catch([&]() {
62  if (needle != m_vtk_writers.end()) {
63  throw std::runtime_error(
64  "VTK object is already attached to this lattice");
65  }
66  vtk->attach_to_lattice(m_instance, get_latice_to_md_units_conversion());
67  m_vtk_writers.emplace_back(vtk);
68  });
69  return {};
70  }
71  if (method_name == "remove_vtk_writer") {
72  auto const vtk = get_value<std::shared_ptr<VTKHandle>>(params, "vtk");
73  auto const needle = find_vtk(vtk);
74  ObjectHandle::context()->parallel_try_catch([&]() {
75  if (needle == m_vtk_writers.end()) {
76  throw std::runtime_error(
77  "VTK object is not attached to this lattice");
78  }
79  vtk->detach_from_lattice();
80  });
81  m_vtk_writers.erase(needle);
82  return {};
83  }
84  if (method_name == "clear_vtk_writers") {
85  for (auto const &vtk : m_vtk_writers) {
86  vtk->detach_from_lattice();
87  }
88  m_vtk_writers.clear();
89  }
90  return {};
91  }
92 };
