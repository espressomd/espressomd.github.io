41 void ParallelExceptionHandler::handle_impl(std::exception const *error) const {
42  auto const head_node = 0;
43  auto const this_node = m_comm.rank();
44 
45  enum : unsigned char {
46  NO_RANK_FAILED = 0u,
47  SOME_RANK_FAILED = 1u,
48  THIS_RANK_SUCCESS = 0u,
49  THIS_RANK_FAILED = 1u,
50  MAIN_RANK_FAILED = 2u,
51  };
52  auto const this_fail_flag =
53  ((error)
54  ? ((this_node == head_node) ? MAIN_RANK_FAILED : THIS_RANK_FAILED)
55  : THIS_RANK_SUCCESS);
56  auto const fail_flag = boost::mpi::all_reduce(
57  m_comm, static_cast<unsigned char>(this_fail_flag), std::bit_or<>());
58  auto const main_rank_failed = fail_flag & MAIN_RANK_FAILED;
59  auto const some_rank_failed = fail_flag & SOME_RANK_FAILED;
60 
61  if (main_rank_failed) {
62  flush_runtime_errors_local();
63  if (this_node == head_node) {
64  throw;
65  }
66  throw Exception("");
67  }
68 
69  if (some_rank_failed) {
70  flush_runtime_errors_local();
71  std::vector<std::string> messages;
72  std::string this_message{(error) ? error->what() : ""};
73  boost::mpi::gather(m_comm, this_message, messages, head_node);
74  if (this_node == head_node) {
75  std::string error_message{"an error occurred on one or more MPI ranks:"};
76  for (std::size_t i = 0; i < messages.size(); ++i) {
77  error_message += "\n rank " + std::to_string(i) + ": " + messages[i];
78  }
79  throw std::runtime_error(error_message.c_str());
80  }
81  throw Exception("");
82  }
83 }
