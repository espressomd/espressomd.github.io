65 /** Structure to hold P3M parameters and some dependent variables. */
66 struct P3MParameters {
67  /** tuning or production? */
68  bool tuning;
69  /** Ewald splitting parameter (0<alpha<1), rescaled to
70  * @p alpha_L = @p alpha * @p box_l. */
71  double alpha_L;
72  /** cutoff radius for real space electrostatics (>0), rescaled to
73  * @p r_cut_iL = @p r_cut * @p box_l_i. */
74  double r_cut_iL;
75  /** number of mesh points per coordinate direction (>0). */
76  Utils::Vector3i mesh = {};
77  /** offset of the first mesh point (lower left corner) from the
78  * coordinate origin ([0,1[). */
79  Utils::Vector3d mesh_off;
80  /** charge assignment order ([0,7]). */
81  int cao;
82  /** accuracy of the actual parameter set. */
83  double accuracy;
84 
85  /** epsilon of the "surrounding dielectric". */
86  double epsilon;
87  /** cutoff for charge assignment. */
88  Utils::Vector3d cao_cut;
89  /** mesh constant. */
90  Utils::Vector3d a;
91  /** inverse mesh constant. */
92  Utils::Vector3d ai;
93  /** unscaled @ref P3MParameters::alpha_L "alpha_L" for use with fast
94  * inline functions only */
95  double alpha;
96  /** unscaled @ref P3MParameters::r_cut_iL "r_cut_iL" for use with fast
97  * inline functions only */
98  double r_cut;
99  /** number of points unto which a single charge is interpolated, i.e.
100  * @ref P3MParameters::cao "cao" cubed */
101  int cao3;

103  P3MParameters(bool tuning, double epsilon, double r_cut,
104  Utils::Vector3i const &mesh, Utils::Vector3d const &mesh_off,
105  int cao, double alpha, double accuracy)
106  : tuning{tuning}, alpha_L{0.}, r_cut_iL{0.}, mesh{mesh},
107  mesh_off{mesh_off}, cao{cao}, accuracy{accuracy}, epsilon{epsilon},
108  cao_cut{}, a{}, ai{}, alpha{alpha}, r_cut{r_cut}, cao3{-1} {
109 
110  auto constexpr value_to_tune = -1.;
111 
112  if (epsilon < 0.) {
113  throw std::domain_error("Parameter 'epsilon' must be >= 0");
114  }
115 
116  if (accuracy <= 0.) {
117  throw std::domain_error("Parameter 'accuracy' must be > 0");
118  }
119 
120  if (r_cut <= 0.) {
121  if (tuning and r_cut == value_to_tune) {
122  this->r_cut = 0.;
123  } else {
124  throw std::domain_error("Parameter 'r_cut' must be > 0");
125  }
126  }
127 
128  if (alpha <= 0.) {
129  if (tuning and alpha == value_to_tune) {
130  this->alpha = 0.;
131  } else {
132  throw std::domain_error("Parameter 'alpha' must be > 0");
133  }
134  }
135 
136  if (not(mesh >= Utils::Vector3i::broadcast(1) or
137  ((mesh[0] >= 1) and
138  (mesh == Utils::Vector3i{{mesh[0], -1, -1}}))) and
139  not(tuning and mesh == Utils::Vector3i::broadcast(-1))) {
140  throw std::domain_error("Parameter 'mesh' must be > 0");
141  }
142 
143  if (not(mesh_off >= Utils::Vector3d::broadcast(0.) and
144  mesh_off <= Utils::Vector3d::broadcast(1.))) {
145  if (mesh_off == Utils::Vector3d::broadcast(-1.)) {
146  this->mesh_off = Utils::Vector3d::broadcast(P3M_MESHOFF);
147  } else {
148  throw std::domain_error("Parameter 'mesh_off' must be >= 0 and <= 1");
149  }
150  }
151 
152  if ((cao < 1 or cao > 7) and not(tuning and cao == -1)) {
153  throw std::domain_error("Parameter 'cao' must be >= 1 and <= 7");
154  }
155 
156  if (not tuning and (Utils::Vector3i::broadcast(cao) > mesh)) {
157  throw std::domain_error("Parameter 'cao' cannot be larger than 'mesh'");
158  }
159  }
160 
161  /**
162  * @brief Recalculate quantities derived from the mesh and box length:
163  * @ref P3MParameters::a "a",
164  * @ref P3MParameters::ai "ai" and
165  * @ref P3MParameters::cao_cut "cao_cut".
166  */
167  void recalc_a_ai_cao_cut(Utils::Vector3d const &box_l) {
168  ai = Utils::hadamard_division(mesh, box_l);
169  a = Utils::hadamard_division(Utils::Vector3d::broadcast(1.), ai);
170  cao_cut = (static_cast<double>(cao) / 2.) * a;
171  }
172 };

234 /** Calculate indices that shift @ref P3MParameters::mesh "mesh" by `mesh/2`.
235  * For each mesh size @f$ n @f$ in @c mesh_size, create a sequence of integer
236  * values @f$ \left( 0, \ldots, \lfloor n/2 \rfloor, -\lfloor n/2 \rfloor,
237  * \ldots, -1\right) @f$ if @c zero_out_midpoint is false, otherwise
238  * @f$ \left( 0, \ldots, \lfloor n/2 - 1 \rfloor, 0, -\lfloor n/2 \rfloor,
239  * \ldots, -1\right) @f$.
240  */
241 std::array<std::vector<int>, 3> inline calc_meshift(
242  Utils::Vector3i const &mesh_size, bool zero_out_midpoint = false) {
243  std::array<std::vector<int>, 3> ret{};
244 
245  for (std::size_t i = 0; i < 3; i++) {
246  ret[i] = std::vector<int>(mesh_size[i]);
247 
248  for (int j = 1; j <= mesh_size[i] / 2; j++) {
249  ret[i][j] = j;
250  ret[i][mesh_size[i] - j] = -j;
251  }
252  if (zero_out_midpoint)
253  ret[i][mesh_size[i] / 2] = 0;
254  }
255 
256  return ret;
257 }
