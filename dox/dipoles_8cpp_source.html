116  actor->dipole_assign(m_particles);

155 struct LongRangeEnergy : public boost::static_visitor<double> {
156  ParticleRange const &m_particles;
157  explicit LongRangeEnergy(ParticleRange const &particles)
158  : m_particles(particles) {}
159 
160 #ifdef DP3M
161  double operator()(std::shared_ptr<DipolarP3M> const &actor) const {
162  actor->dipole_assign(m_particles);
163  return actor->kernel(false, true, m_particles);
164  }
165 #endif // DP3M
166  double
167  operator()(std::shared_ptr<DipolarLayerCorrection> const &actor) const {
168  auto energy = boost::apply_visitor(*this, actor->base_solver);
169  return energy + actor->energy_correction(m_particles);
170  }
171  double operator()(std::shared_ptr<DipolarDirectSum> const &actor) const {
172  return actor->kernel(false, true, m_particles);
173  }
174  double
175  operator()(std::shared_ptr<DipolarDirectSumWithReplica> const &actor) const {
176  return actor->kernel(false, true, m_particles);
177  }
