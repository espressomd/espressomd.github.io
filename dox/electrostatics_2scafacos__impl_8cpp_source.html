87 void CoulombScafacosImpl::tune_r_cut() {
88  auto constexpr convergence_threshold = 1e-3;
89 
90  auto const min_box_l = *boost::min_element(box_geo.length());
91  auto const min_local_box_l = *boost::min_element(local_geo.length());
92 
93  /* The bisection code breaks down when r_min < 1 for several methods
94  * (e.g. p2nfft, p3m, ewald) if the mesh size is not fixed (ScaFaCoS
95  * either hangs or allocates too much memory) */
96  auto r_min = 1.0;
97  auto r_max = std::min(min_local_box_l, min_box_l / 2.0) - skin;
98  assert(r_max >= r_min);
99  auto t_min = 0.0;
100  auto t_max = std::numeric_limits<double>::max();
101  auto r_opt = -1.0;
102 
103  /* Run bisection */
104  while (std::fabs(r_min - r_max) > convergence_threshold) {
105  r_opt = (r_max + r_min) / 2.;
106  auto const dr = 0.5 * (r_max - r_min);
107  auto const t_mid = time_r_cut(r_min + dr);
108  t_min = time_r_cut(r_min);
109  t_max = time_r_cut(r_max);
110 
111  if (t_min <= 0.0 or t_max <= 0.0) {
112  break;
113  }
114 
115  if (t_mid > t_min) {
116  r_max = r_min += dr;
117  } else {
118  r_min += dr;
119  }
120  }
121  assert(r_opt >= 0.);
122  set_r_cut(r_opt);
123 }
