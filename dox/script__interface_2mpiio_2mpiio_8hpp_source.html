41class MPIIOScript : public ObjectHandle {
42 std::weak_ptr<System::System> m_system;
43 std::shared_ptr<Mpiio::write_buffers> m_buffers;
44
45public:
46 void do_construct(VariantMap const &parameters) override {
47 m_system = get_value<std::shared_ptr<System::System>>(parameters, "system");
48 m_buffers = std::make_shared<Mpiio::write_buffers>();
49 }
50
51 Variant do_call_method(std::string const &name,
52 VariantMap const &parameters) override {
53
54 auto prefix = get_value<std::filesystem::path>(parameters.at("prefix"));
55 auto pos = get_value<bool>(parameters.at("pos"));
56 auto vel = get_value<bool>(parameters.at("vel"));
57 auto typ = get_value<bool>(parameters.at("typ"));
58 auto bnd = get_value<bool>(parameters.at("bond"));
59
60 auto const fields = ((pos) ? Mpiio::MPIIO_OUT_POS : Mpiio::MPIIO_OUT_NON) |
61 ((vel) ? Mpiio::MPIIO_OUT_VEL : Mpiio::MPIIO_OUT_NON) |
62 ((typ) ? Mpiio::MPIIO_OUT_TYP : Mpiio::MPIIO_OUT_NON) |
63 ((bnd) ? Mpiio::MPIIO_OUT_BND : Mpiio::MPIIO_OUT_NON);
64
65 if (name == "write") {
66 auto const system_si = m_system.lock();
67 auto &system = system_si->get_system();
68 auto &cell_structure = *system.cell_structure;
69 auto &bonded_ias = *system.bonded_ias;
70 Mpiio::mpi_mpiio_common_write(prefix.string(), fields, bonded_ias,
71 cell_structure.local_particles(),
72 *m_buffers);
73 } else if (name == "read") {
74 auto const system_si = m_system.lock();
75 auto &system = system_si->get_system();
76 auto &cell_structure = *system.cell_structure;
77 Mpiio::mpi_mpiio_common_read(prefix.string(), fields, cell_structure);
78 }
79
80 return {};
81 }
82};
