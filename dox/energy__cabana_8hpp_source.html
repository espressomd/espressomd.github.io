40struct EnergyBinLayout {
41 std::size_t n_bonded; // initialized by bonded_ias->get_next_key()
42 std::size_t n_types; // max_seen_particle_type
43 std::size_t off_bonded = 0; // [0, n_bonded)
44 std::size_t off_nb_inter;
45 std::size_t off_nb_intra;
46 std::size_t off_coulomb;
47 std::size_t off_dipolar;
48 std::size_t total;
49
50 EnergyBinLayout(std::size_t n_bonded_, std::size_t n_types_)
51 : n_bonded(n_bonded_), n_types(n_types_) {
52 auto const n_nb = n_types * (n_types + 1) / 2;
53 off_nb_inter = off_bonded + n_bonded;
54 off_nb_intra = off_nb_inter + n_nb;
55 off_coulomb = off_nb_intra + n_nb;
56 off_dipolar = off_coulomb + 1;
57 total = off_dipolar + 1;
58 }
59
60 KOKKOS_INLINE_FUNCTION
61 std::size_t nb_inter_idx(int t1, int t2) const {
62 return off_nb_inter +
63 Utils::lower_triangular(std::max(t1, t2), std::min(t1, t2));
64 }
65
66 KOKKOS_INLINE_FUNCTION
67 std::size_t nb_intra_idx(int t1, int t2) const {
68 return off_nb_intra +
69 Utils::lower_triangular(std::max(t1, t2), std::min(t1, t2));
70 }
71
72 KOKKOS_INLINE_FUNCTION std::size_t dipolar_idx() const { return off_dipolar; }
73 KOKKOS_INLINE_FUNCTION std::size_t coulomb_idx() const { return off_coulomb; }
74 KOKKOS_INLINE_FUNCTION std::size_t bonded_idx(int b) const {
75 return off_bonded + b;
76 }
77};

96 Coulomb::Solver const &coulomb_,
97 Coulomb::ShortRangeEnergyKernel::kernel_type const *coulomb_u_kernel_,

104 : bonded_ias(bonded_ias_), nonbonded_ias(nonbonded_ias_),
105 coulomb(coulomb_), coulomb_u_kernel(coulomb_u_kernel_),
106 dipoles_u_kernel(dipoles_u_kernel_), box_geo(box_geo_),
107 unique_particles(unique_particles_), local_energy(local_energy_),
108 layout(layout_), aosoa(aosoa_), mol_id_view(std::move(mol_id_view_)),
109 system_max_cutoff_sq(system_max_cutoff_ * system_max_cutoff_) {}
110
111 KOKKOS_INLINE_FUNCTION
112 void operator()(std::size_t i, std::size_t j) const {
113 auto const d = box_geo.get_mi_vector(
114 aosoa.position(i, 0), aosoa.position(i, 1), aosoa.position(i, 2),
115 aosoa.position(j, 0), aosoa.position(j, 1), aosoa.position(j, 2));
116 auto const dist_sq = d.norm2();
117 if (dist_sq > system_max_cutoff_sq)
118 return;
119 auto const dist = std::sqrt(dist_sq);
120
121 auto const t1 = aosoa.type(i);
122 auto const t2 = aosoa.type(j);
123 auto const &ia_params = nonbonded_ias.get_ia_param(t1, t2);
124
125 // Determine which data needs to be loaded based on active algorithms
126#if defined(ESPRESSO_EXCLUSIONS) or defined(ESPRESSO_THOLE)
127 bool need_particle_pointers = false;
128#ifdef ESPRESSO_EXCLUSIONS
129 need_particle_pointers |= aosoa.has_exclusion(i) or aosoa.has_exclusion(j);
130#endif
131#ifdef ESPRESSO_THOLE
132 need_particle_pointers |=
133 thole_active(ia_params, coulomb_u_kernel != nullptr);
134#endif
135
136 Particle const *p1_ptr = nullptr;
137 Particle const *p2_ptr = nullptr;
138 if (need_particle_pointers) {
139 p1_ptr = unique_particles.at(i);
140 p2_ptr = unique_particles.at(j);
141 }
142#endif
143
144 auto const tid = omp_get_thread_num();
145 double e_nb = 0.0;
146
147 if (dist <= ia_params.max_cut) {
148#ifdef ESPRESSO_EXCLUSIONS
149 bool skip = false;
150 if (aosoa.has_exclusion(i) or aosoa.has_exclusion(j))
151 skip = not do_nonbonded(*p1_ptr, *p2_ptr);
152 if (not skip)
153#endif
154 {
155 e_nb += calc_central_radial_energy(ia_params, dist);
156
157 // Only call Thole energy kernel if active
158#ifdef ESPRESSO_THOLE
159 if (thole_active(ia_params, coulomb_u_kernel != nullptr)) {
160 e_nb += thole_pair_energy(*p1_ptr, *p2_ptr, ia_params, d, dist,
161 bonded_ias, coulomb, coulomb_u_kernel);
162 }
163#endif
164 // Only call Gay-Berne energy kernel if active
165#ifdef ESPRESSO_GAY_BERNE
166 if (gay_berne_active(dist, ia_params)) {
167 auto const dir1 = aosoa.get_vector_at(aosoa.director, i);
168 auto const dir2 = aosoa.get_vector_at(aosoa.director, j);
169 e_nb += gb_pair_energy(dir1, dir2, ia_params, d, dist);
170 }
171#endif
172 }
173 }
174 // pick inter vs intra bin like Observable_stat::non_bonded_contribution
175 // does
176 auto const bin = (mol_id_view(i) == mol_id_view(j))
177 ? layout.nb_intra_idx(t1, t2)
178 : layout.nb_inter_idx(t1, t2);
179 local_energy(tid, bin) += e_nb;
180
181#ifdef ESPRESSO_ELECTROSTATICS
182 if (coulomb_u_kernel != nullptr) {
183 auto const q1 = aosoa.charge(i), q2 = aosoa.charge(j);
184 if (q1 != 0. and q2 != 0.) {
185 auto const pos1 = aosoa.get_vector_at(aosoa.position, i);
186 auto const pos2 = aosoa.get_vector_at(aosoa.position, j);
187 double const e_c = (*coulomb_u_kernel)(pos1, pos2, q1 * q2, d, dist);
188 local_energy(tid, layout.coulomb_idx()) += e_c;
189 }
190 }
191#endif
192
193#ifdef ESPRESSO_DIPOLES
194 if (dipoles_u_kernel != nullptr) {
195 if (aosoa.dipm(i) != 0. and aosoa.dipm(j) != 0.) {
196 auto const dir1 = aosoa.get_vector_at(aosoa.director, i);
197 auto const dir2 = aosoa.get_vector_at(aosoa.director, j);
198 double const e_d = (*dipoles_u_kernel)(
199 aosoa.dipm(i) * dir1, aosoa.dipm(j) * dir2, d, dist, dist_sq);
200 local_energy(tid, layout.dipolar_idx()) += e_d;
201 }
202 }
203#endif
204 }
205};

207static void reduce_cabana_energy(
208 Kokkos::View<double **, Kokkos::LayoutRight> const &local_energy,
209 EnergyBinLayout const &layout, Observable_stat &obs,
210 BondedInteractionsMap const &bonded_ias, int n_types) {
211 auto const nthreads = local_energy.extent(0);
212 auto host =
213 Kokkos::create_mirror_view_and_copy(Kokkos::HostSpace{}, local_energy);
214
215 auto sum_bin = [&](std::size_t bin) {
216 double s = 0.;
217 for (std::size_t t = 0; t < nthreads; ++t)
218 s += host(t, bin);
219 return s;
220 };
221
222 for (int b = 0; b < int(layout.n_bonded); ++b)
223 obs.bonded_contribution(b)[0] += sum_bin(layout.bonded_idx(b));
224
225 for (int t1 = 0; t1 < n_types; ++t1)
226 for (int t2 = 0; t2 <= t1; ++t2)
227 obs.non_bonded_inter_contribution(t1, t2)[0] +=
228 sum_bin(layout.nb_inter_idx(t1, t2));
229
230 for (int t1 = 0; t1 < n_types; ++t1)
231 for (int t2 = 0; t2 <= t1; ++t2)
232 obs.non_bonded_intra_contribution(t1, t2)[0] +=
233 sum_bin(layout.nb_intra_idx(t1, t2));
234
235 obs.coulomb[0] += sum_bin(layout.coulomb_idx());
236 obs.dipolar[0] += sum_bin(layout.dipolar_idx());
237}
