165void System::set_cell_structure_topology(CellStructureType topology) {
166 if (topology == CellStructureType::REGULAR) {
167 if (cell_structure->decomposition_type() == CellStructureType::REGULAR) {
168 // get fully connected info from exising regular decomposition
169 auto &old_regular_decomposition =
170 dynamic_cast<RegularDecomposition const &>(
171 std::as_const(*cell_structure).decomposition());
172 cell_structure->set_regular_decomposition(
173 get_interaction_range(),
174 old_regular_decomposition.fully_connected_boundary());
175 } else { // prev. decomposition is not a regular decomposition
176 cell_structure->set_regular_decomposition(get_interaction_range(), {});
177 }
178 } else if (topology == CellStructureType::NSQUARE) {
179 cell_structure->set_atom_decomposition();
180 } else {
181 assert(topology == CellStructureType::HYBRID);
182 /* Get current HybridDecomposition to extract n_square_types */
183 auto &old_hybrid_decomposition = dynamic_cast<HybridDecomposition const &>(
184 std::as_const(*cell_structure).decomposition());
185 cell_structure->set_hybrid_decomposition(
186 old_hybrid_decomposition.get_cutoff_regular(),
187 old_hybrid_decomposition.get_n_square_types());
188 }
189}

213void System::veto_boxl_change(bool skip_particle_checks) const {
214 if (not skip_particle_checks) {
215 auto const n_part = boost::mpi::all_reduce(
216 ::comm_cart, cell_structure->local_particles().size(), std::plus<>());
217 if (n_part > 0ul) {
218 throw std::runtime_error(
219 "Cannot reset the box length when particles are present");
220 }
221 }
222 constraints->veto_boxl_change();
223 lb.veto_boxl_change();
224 ek.veto_boxl_change();

430double System::maximal_cutoff() const {
431 auto max_cut = inactive_cutoff;
432 max_cut = std::max(max_cut, get_min_global_cut());
433 max_cut = std::max(max_cut, coulomb.cutoff());
434 max_cut = std::max(max_cut, dipoles.cutoff());
435 if (::communicator.size > 1) {
436 // If there is just one node, the bonded cutoff can be omitted
437 // because bond partners are always on the local node.
438 max_cut = std::max(max_cut, bonded_ias->maximal_cutoff());
439 }
440 max_cut = std::max(max_cut, nonbonded_ias->maximal_cutoff());
441
442#ifdef ESPRESSO_COLLISION_DETECTION
443 max_cut = std::max(max_cut, collision_detection->cutoff());
444#endif
445 return max_cut;
446}

463double System::get_interaction_range() const {
464 auto const max_cut = maximal_cutoff();
465 auto const verlet_skin = cell_structure->get_verlet_skin();
466 /* Consider skin only if there are actually interactions */
467 return (max_cut > 0.) ? max_cut + verlet_skin : inactive_cutoff;

475void System::on_integration_start() {
476 // sanity checks
477 integrator_sanity_checks();
478 long_range_interactions_sanity_checks();
479 lb.sanity_checks();
480 ek.sanity_checks();
481
482#ifdef ESPRESSO_NPT
483 if (propagation->integ_switch == INTEG_METHOD_NPT_ISO_AND ||
484 propagation->integ_switch == INTEG_METHOD_NPT_ISO_MTK) {
485 npt_ensemble_init(propagation->recalc_forces);
486 }
487#endif
488
489 /* Prepare the thermostat */
490 if (reinit_thermo) {
491 thermostat->recalc_prefactors(time_step);
492 reinit_thermo = false;
493 propagation->recalc_forces = true;
494 }
495
496 invalidate_fetch_cache();
497#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
498 cell_structure->clear_local_properties();
499#endif
500
501#ifdef ESPRESSO_ADDITIONAL_CHECKS
502 if (!Utils::Mpi::all_compare(::comm_cart, cell_structure->use_verlet_list)) {
503 runtimeErrorMsg() << "Nodes disagree about use of verlet lists.";
504 }
505#ifdef ESPRESSO_ELECTROSTATICS
506 {
507 auto const &actor = coulomb.impl->solver;
508 if (not Utils::Mpi::all_compare(::comm_cart, static_cast<bool>(actor)) or
509 (actor and not Utils::Mpi::all_compare(::comm_cart, (*actor).index())))
510 runtimeErrorMsg() << "Nodes disagree about Coulomb long-range method";
511 }
512#endif
513#ifdef ESPRESSO_DIPOLES
514 {
515 auto const &actor = dipoles.impl->solver;
516 if (not Utils::Mpi::all_compare(::comm_cart, static_cast<bool>(actor)) or
517 (actor and not Utils::Mpi::all_compare(::comm_cart, (*actor).index())))
518 runtimeErrorMsg() << "Nodes disagree about dipolar long-range method";
519 }
520#endif
521#endif /* ESPRESSO_ADDITIONAL_CHECKS */
522
523 on_observable_calc();
524}

526/**
527 * @brief Returns the ghost flags required for running pair
528 * kernels for the global state, e.g. the force calculation.
529 * @return Required data parts;
530 */
531unsigned System::get_global_ghost_flags() const {
532 /* Position and Properties are always requested. */
533 unsigned data_parts = Cells::DATA_PART_POSITION | Cells::DATA_PART_PROPERTIES;
534
535 if (lb.is_solver_set())
536 data_parts |= Cells::DATA_PART_MOMENTUM;
537
538 if (thermostat->thermo_switch & THERMO_DPD)
539 data_parts |= Cells::DATA_PART_MOMENTUM;
540
541 if (thermostat->thermo_switch & THERMO_BOND) {
542 data_parts |= Cells::DATA_PART_MOMENTUM;
543 data_parts |= Cells::DATA_PART_BONDS;
544 }
545
546#ifdef ESPRESSO_COLLISION_DETECTION
547 if (not collision_detection->is_off()) {
548 data_parts |= Cells::DATA_PART_BONDS;
549 }
550#endif
551
552 return data_parts;
553}
