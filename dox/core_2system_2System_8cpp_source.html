132 void System::set_cell_structure_topology(CellStructureType topology) {
133  if (topology == CellStructureType::REGULAR) {
134  cell_structure->set_regular_decomposition(get_interaction_range());
135  } else if (topology == CellStructureType::NSQUARE) {
136  cell_structure->set_atom_decomposition();
137  } else {
138  assert(topology == CellStructureType::HYBRID);
139  /* Get current HybridDecomposition to extract n_square_types */
140  auto &old_hybrid_decomposition = dynamic_cast<HybridDecomposition const &>(
141  std::as_const(*cell_structure).decomposition());
142  cell_structure->set_hybrid_decomposition(
143  old_hybrid_decomposition.get_cutoff_regular(),
144  old_hybrid_decomposition.get_n_square_types());
145  }
146 }

339 double System::maximal_cutoff() const {
340  auto max_cut = INACTIVE_CUTOFF;
341  max_cut = std::max(max_cut, get_min_global_cut());
342 #ifdef ELECTROSTATICS
343  max_cut = std::max(max_cut, coulomb.cutoff());
344 #endif
345 #ifdef DIPOLES
346  max_cut = std::max(max_cut, dipoles.cutoff());
347 #endif
348  if (::communicator.size > 1) {
349  // If there is just one node, the bonded cutoff can be omitted
350  // because bond partners are always on the local node.
351  max_cut = std::max(max_cut, maximal_cutoff_bonded());
352  }
353  max_cut = std::max(max_cut, nonbonded_ias->maximal_cutoff());
354 
355  max_cut = std::max(max_cut, collision_detection_cutoff());
356  return max_cut;
357 }

374 double System::get_interaction_range() const {
375  auto const max_cut = maximal_cutoff();

377  /* Consider skin only if there are actually interactions */

386 void System::on_integration_start() {
387  // sanity checks
388  integrator_sanity_checks();
389 #ifdef NPT
390  integrator_npt_sanity_checks();
391 #endif
392  long_range_interactions_sanity_checks();
393  lb.sanity_checks();
394  ek.sanity_checks();
395 
396  /* Prepare the thermostat */
397  if (reinit_thermo) {
398  thermostat->recalc_prefactors(time_step);
399  reinit_thermo = false;
400  propagation->recalc_forces = true;
401  }
402 
403 #ifdef NPT
404  if (propagation->integ_switch == INTEG_METHOD_NPT_ISO) {
405  npt_ensemble_init(box_geo->length(), propagation->recalc_forces);
406  }
407 #endif
408 
409  invalidate_fetch_cache();
410 
411 #ifdef ADDITIONAL_CHECKS
412  if (!Utils::Mpi::all_compare(::comm_cart, cell_structure->use_verlet_list)) {
413  runtimeErrorMsg() << "Nodes disagree about use of verlet lists.";
414  }
415 #ifdef ELECTROSTATICS
416  {
417  auto const &actor = coulomb.impl->solver;
418  if (not Utils::Mpi::all_compare(::comm_cart, static_cast<bool>(actor)) or
419  (actor and not Utils::Mpi::all_compare(::comm_cart, (*actor).index())))
420  runtimeErrorMsg() << "Nodes disagree about Coulomb long-range method";
421  }
422 #endif
423 #ifdef DIPOLES
424  {
425  auto const &actor = dipoles.impl->solver;
426  if (not Utils::Mpi::all_compare(::comm_cart, static_cast<bool>(actor)) or
427  (actor and not Utils::Mpi::all_compare(::comm_cart, (*actor).index())))
428  runtimeErrorMsg() << "Nodes disagree about dipolar long-range method";
429  }
430 #endif
431 #endif /* ADDITIONAL_CHECKS */
432 
433  on_observable_calc();
434 }

436 /**
437  * @brief Returns the ghost flags required for running pair
438  * kernels for the global state, e.g. the force calculation.
439  * @return Required data parts;
440  */
441 unsigned System::get_global_ghost_flags() const {
442  /* Position and Properties are always requested. */
443  unsigned data_parts = Cells::DATA_PART_POSITION | Cells::DATA_PART_PROPERTIES;
444 
445  if (lb.is_solver_set())
446  data_parts |= Cells::DATA_PART_MOMENTUM;
447 
448  if (thermostat->thermo_switch & THERMO_DPD)
449  data_parts |= Cells::DATA_PART_MOMENTUM;
450 
451  if (thermostat->thermo_switch & THERMO_BOND) {
452  data_parts |= Cells::DATA_PART_MOMENTUM;
453  data_parts |= Cells::DATA_PART_BONDS;
454  }
455 
456 #ifdef COLLISION_DETECTION
457  if (::collision_params.mode != CollisionModeType::OFF) {
458  data_parts |= Cells::DATA_PART_BONDS;
459  }
460 #endif
461 
462  return data_parts;
463 }
