49 void add_md_force(LB::Solver &lb, Utils::Vector3d const &pos,
50  Utils::Vector3d const &force, double time_step) {
51  /* transform momentum transfer to lattice units
52  (eq. (12) @cite ahlrichs99a) */
53  auto const delta_j = (time_step / lb.get_lattice_speed()) * force;
54  lb.add_force_density(pos, delta_j);
55 }

72 Utils::Vector3d lb_drag_force(LB::Solver const &lb, double lb_gamma,
73  Particle const &p,
74  Utils::Vector3d const &shifted_pos,
75  Utils::Vector3d const &vel_offset) {
76  /* calculate fluid velocity at particle's position
77  this is done by linear interpolation (eq. (11) @cite ahlrichs99a) */
78  auto const v_fluid = lb.get_coupling_interpolated_velocity(shifted_pos);
79  auto const v_drift = v_fluid + vel_offset;
80  auto const gamma = lb_handle_particle_anisotropy(p, lb_gamma);
81 
82  /* calculate viscous force (eq. (9) @cite ahlrichs99a) */
83  return Utils::hadamard_product(gamma, v_drift - p.v());
84 }

116 /**
117  * @brief Return a vector of positions shifted by +,- box length in each
118  * coordinate
119  */
120 std::vector<Utils::Vector3d> positions_in_halo(Utils::Vector3d const &pos,
121  BoxGeometry const &box_geo,
122  LocalBox const &local_box,
123  double agrid) {
124  auto const halo = 0.5 * agrid;
125  auto const halo_vec = Utils::Vector3d::broadcast(halo);
126  auto const fully_inside_lower = local_box.my_left() + 2. * halo_vec;
127  auto const fully_inside_upper = local_box.my_right() - 2. * halo_vec;
128  if (in_box(pos, fully_inside_lower, fully_inside_upper)) {
129  return {pos};
130  }
131  auto const halo_lower_corner = local_box.my_left() - halo_vec;
132  auto const halo_upper_corner = local_box.my_right() + halo_vec;
133 
134  std::vector<Utils::Vector3d> res;
135  for (int i : {-1, 0, 1}) {
136  for (int j : {-1, 0, 1}) {
137  for (int k : {-1, 0, 1}) {
138  Utils::Vector3d shift{{double(i), double(j), double(k)}};
139  Utils::Vector3d pos_shifted =
140  pos + Utils::hadamard_product(box_geo.length(), shift);
141 
142  if (box_geo.type() == BoxType::LEES_EDWARDS) {
143  auto le = box_geo.lees_edwards_bc();
144  auto normal_shift = (pos_shifted - pos)[le.shear_plane_normal];
145  if (normal_shift > std::numeric_limits<double>::epsilon())
146  pos_shifted[le.shear_direction] += le.pos_offset;
147  if (normal_shift < -std::numeric_limits<double>::epsilon())
148  pos_shifted[le.shear_direction] -= le.pos_offset;
149  }
150 
151  if (in_box(pos_shifted, halo_lower_corner, halo_upper_corner)) {
152  res.push_back(pos_shifted);
153  }
154  }
155  }
156  }
157  return res;
158 }

162 Utils::Vector3d ParticleCoupling::get_noise_term(Particle const &p) const {
163  if (not m_thermalized) {
164  return Utils::Vector3d{};
165  }
166  using std::sqrt;
167  using Utils::sqrt;
168  auto const gamma = lb_handle_particle_anisotropy(p, m_thermostat.gamma);
169  auto const noise = Random::noise_uniform<RNGSalt::PARTICLES>(
170  m_thermostat.rng_counter(), m_thermostat.rng_seed(), p.id());
171  return m_noise_pref_wo_gamma * Utils::hadamard_product(sqrt(gamma), noise);
172 }

174 void ParticleCoupling::kernel(Particle &p) {
175  auto const agrid = m_lb.get_agrid();
176 
177  // Calculate coupling force
178  Utils::Vector3d force_on_particle = {};
179  auto const halo_pos = positions_in_halo(m_box_geo.folded_position(p.pos()),
180  m_box_geo, m_local_box, agrid);
181 
182 #ifdef ENGINE
183  if (not p.swimming().is_engine_force_on_fluid)
184 #endif
185  for (auto const &pos : halo_pos) {
186  if (in_local_halo(m_local_box, pos, agrid)) {
187  auto const vel_offset = lb_drift_velocity_offset(p);
188  auto const drag_force =
189  lb_drag_force(m_lb, m_thermostat.gamma, p, pos, vel_offset);
190  auto const random_force = get_noise_term(p);
191  force_on_particle = drag_force + random_force;
192  break;
193  }
194  }
195 
196  auto force_on_fluid = -force_on_particle;
197 #ifdef ENGINE
198  if (p.swimming().is_engine_force_on_fluid) {
199  force_on_fluid = p.calc_director() * p.swimming().f_swim;
200  }
201 #endif
202 
203  // couple positions including shifts by one box length to add
204  // forces to ghost layers
205  for (auto const &pos : halo_pos) {
206  if (in_local_domain(m_local_box, pos)) {
207  /* Particle is in our LB volume, so this node
208  * is responsible to adding its force */
209  p.force() += force_on_particle;
210  }
211  add_md_force(m_lb, pos, force_on_fluid, m_time_step);
212  }
213 }

231 void System::System::lb_couple_particles(double time_step) {
232 #ifdef CALIPER
233  CALI_CXX_MARK_FUNCTION;
234 #endif
235  assert(thermostat->lb != nullptr);
236  if (thermostat->lb->couple_to_md) {
237  if (not lb.is_solver_set()) {
238  runtimeErrorMsg() << "The LB thermostat requires a LB fluid";
239  return;
240  }
241  auto const real_particles = cell_structure->local_particles();
242  auto const ghost_particles = cell_structure->ghost_particles();
243  LB::ParticleCoupling coupling{*thermostat->lb, lb, *box_geo, *local_geo,
244  time_step};
245  LB::CouplingBookkeeping bookkeeping{*cell_structure};
246  for (auto const *particle_range : {&real_particles, &ghost_particles}) {
247  for (auto &p : *particle_range) {
248  if (not LB::is_tracer(p) and bookkeeping.should_be_coupled(p)) {
249 #if defined(THERMOSTAT_PER_PARTICLE) and defined(PARTICLE_ANISOTROPY)
250  LB::lb_coupling_sanity_checks(p);
251 #endif
252  coupling.kernel(p);
253  }
254  }
255  }
256  }
