71 void set_preferred_kspace_decomposition(Utils::Vector3i const &node_grid) {
72 auto const global_box = heffte::box3d<>(
73 {0, 0, 0}, to_array(m_global_mesh - Utils::Vector3i::broadcast(1)),
74 to_array(m_memory_layout));
75 auto const n_procs = Utils::product(node_grid);
76 auto best_grid = node_grid;
77 for (auto i : {0u, 1u, 2u}) {
78 if (m_global_mesh[i] % (2 * n_procs) == 0) {
79 best_grid = {n_procs, 1, 1};
80 break;
81 }
82 }
83 auto all_boxes = heffte::split_world(
84 global_box, {best_grid[0], best_grid[1], best_grid[2]});
85 out_box = std::make_shared<Box>(all_boxes[comm.rank()]);
86 init_fft();
87 }

89 void init_fft() {
90 // at this stage we can manually adjust some HeFFTe options
91 heffte::plan_options options = heffte::default_options<backend_tag>();
92
93 // use strided 1-D FFT operations
94 // some backends work just as well when the entries of the data are not
95 // contiguous then there is no need to reorder the data in the intermediate
96 // stages which saves time
97 options.use_reorder = false;
98
99 // use point-to-point communications
100 // collaborative all-to-all and individual point-to-point communications are
101 // two alternatives one may be better than the other depending on the
102 // version of MPI, the hardware interconnect, and the problem size
103 options.algorithm = heffte::reshape_algorithm::p2p_plined;
104
105 // in the intermediate steps, the data can be shapes as either 2-D slabs or
106 // 1-D pencils for sufficiently large problem, it is expected that the
107 // pencil decomposition is better but for smaller problems, the slabs may
108 // perform better (depending on hardware and backend)
109 options.use_pencils = true;
110 fft3d = heffte::fft3d<backend_tag>(*in_box, *out_box, comm, options);
111 m_buffer.resize(fft3d.size_workspace());
112 }
113
114 Utils::Vector3i ks_local_ld_index() const {
115 return Utils::Vector3i(out_box->low);
116 }
117 Utils::Vector3i ks_local_ur_index() const {
118 return Utils::Vector3i(out_box->high) + Utils::Vector3i::broadcast(1);
119 }
120 Utils::Vector3i ks_local_size() const {
121 return ks_local_ur_index() - ks_local_ld_index();
122 }
123 template <typename In, typename Out> void forward(In in, Out out) {
124 fft3d.forward(in, out, m_buffer.data());
125 }
126 template <typename T1, typename T2> auto backward(T1 &in, T2 &out) {
127 return fft3d.backward(in, out, m_buffer.data());
128 }
129 auto const &get_memory_layout() const { return m_memory_layout; }
130};
