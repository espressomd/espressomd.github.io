95 /**
96  * @brief Calculate cached sin/cos values for one direction.
97  *
98  * @tparam dir Index of the dimension to consider (e.g. 0 for x ...).
99  *
100  * @param particles Particle to calculate values for
101  * @param n_freq Number of frequencies to calculate per particle
102  * @param u Inverse box length
103  * @return Calculated values.
104  */
105 template <std::size_t dir>
106 static std::vector<SCCache> calc_sc_cache(ParticleRange const &particles,
107  std::size_t n_freq, double u) {
108  auto constexpr c_2pi = 2. * Utils::pi();
109  auto const n_part = particles.size();
110  std::vector<SCCache> ret(n_freq * n_part);
111 
112  for (std::size_t freq = 1; freq <= n_freq; freq++) {
113  auto const pref = c_2pi * u * static_cast<double>(freq);
114 
115  std::size_t o = (freq - 1) * n_part;
116  for (auto const &p : particles) {
117  auto const arg = pref * p.pos()[dir];
118  ret[o++] = {sin(arg), cos(arg)};
119  }
120  }
121 
122  return ret;
123 }

127  assert(far_cut >= 0.);
128  auto const n_freq_x =

196 /** Calculate the dipole force.
197  * See @cite yeh99a.
198  */
199 void ElectrostaticLayerCorrection::add_dipole_force(
200  ParticleRange const &particles) const {
201  constexpr std::size_t size = 3;
202  auto const pref = prefactor * 4. * Utils::pi() / box_geo.volume();
203 
204  /* for non-neutral systems, this shift gives the background contribution
205  * (rsp. for this shift, the DM of the background is zero) */
206  auto const shift = box_geo.length_half()[2];
207 
208  // collect moments
209 
210  gblcblk[0] = 0.; // sum q_i (z_i - L/2)
211  gblcblk[1] = 0.; // sum q_i z_i
212  gblcblk[2] = 0.; // sum q_i
213 
214  for (auto const &p : particles) {
215  check_gap(p);
216  auto const q = p.q();
217  auto const z = p.pos()[2];
218 
219  gblcblk[0] += q * (z - shift);
220  gblcblk[1] += q * z;
221  gblcblk[2] += q;
222 
223  if (elc.dielectric_contrast_on) {
224  if (z < elc.space_layer) {
225  gblcblk[0] += elc.delta_mid_bot * q * (-z - shift);
226  gblcblk[2] += elc.delta_mid_bot * q;
227  }
228  if (z > (elc.box_h - elc.space_layer)) {
229  gblcblk[0] += elc.delta_mid_top * q * (2. * elc.box_h - z - shift);
230  gblcblk[2] += elc.delta_mid_top * q;
231  }
232  }
233  }
234 
235  gblcblk[0] *= pref;
236  gblcblk[1] *= pref / elc.box_h * box_geo.length()[2];
237  gblcblk[2] *= pref;
238 
239  distribute(size);
240 
241  // Yeh + Berkowitz dipole term @cite yeh99a
242  auto field_tot = gblcblk[0];
243 
244  // Constand potential contribution
245  if (elc.const_pot) {
246  auto const field_induced = gblcblk[1];
247  auto const field_applied = elc.pot_diff / elc.box_h;
248  field_tot -= field_applied + field_induced;
249  }
250 
251  for (auto &p : particles) {
252  p.force()[2] -= field_tot * p.q();
253 
254  if (!elc.neutralize) {
255  // SUBTRACT the forces of the P3M homogeneous neutralizing background
256  p.force()[2] += gblcblk[2] * p.q() * (p.pos()[2] - shift);
257  }
258  }
259 }

261 /** Calculate the dipole energy.
262  * See @cite yeh99a.
263  */
264 double ElectrostaticLayerCorrection::dipole_energy(
265  ParticleRange const &particles) const {
266  constexpr std::size_t size = 7;
267  auto const pref = prefactor * 2. * Utils::pi() / box_geo.volume();
268  auto const lz = box_geo.length()[2];
269  /* for nonneutral systems, this shift gives the background contribution
270  (rsp. for this shift, the DM of the background is zero) */
271  auto const shift = box_geo.length_half()[2];
272 
273  // collect moments
274 
275  gblcblk[0] = 0.; // sum q_i primary box
276  gblcblk[1] = 0.; // sum q_i boundary layers
277  gblcblk[2] = 0.; // sum q_i (z_i - L/2) primary box
278  gblcblk[3] = 0.; // sum q_i (z_i - L/2) boundary layers
279  gblcblk[4] = 0.; // sum q_i (z_i - L/2)^2 primary box
280  gblcblk[5] = 0.; // sum q_i (z_i - L/2)^2 boundary layers
281  gblcblk[6] = 0.; // sum q_i z_i primary box
282 
283  for (auto const &p : particles) {
284  check_gap(p);
285  auto const q = p.q();
286  auto const z = p.pos()[2];
287 
288  gblcblk[0] += q;
289  gblcblk[2] += q * (z - shift);
290  gblcblk[4] += q * (Utils::sqr(z - shift));
291  gblcblk[6] += q * z;
292 
293  if (elc.dielectric_contrast_on) {
294  if (z < elc.space_layer) {
295  gblcblk[1] += elc.delta_mid_bot * q;
296  gblcblk[3] += elc.delta_mid_bot * q * (-z - shift);
297  gblcblk[5] += elc.delta_mid_bot * q * (Utils::sqr(-z - shift));
298  }
299  if (z > (elc.box_h - elc.space_layer)) {
300  gblcblk[1] += elc.delta_mid_top * q;
301  gblcblk[3] += elc.delta_mid_top * q * (2. * elc.box_h - z - shift);
302  gblcblk[5] +=
303  elc.delta_mid_top * q * (Utils::sqr(2. * elc.box_h - z - shift));
304  }
305  }
306  }
307 
308  distribute(size);
309 
310  // Yeh + Berkowitz term @cite yeh99a
311  auto energy = 2. * pref * (Utils::sqr(gblcblk[2]) + gblcblk[2] * gblcblk[3]);
312 
313  if (!elc.neutralize) {
314  // SUBTRACT the energy of the P3M homogeneous neutralizing background
315  energy += 2. * pref *
316  (-gblcblk[0] * gblcblk[4] -
317  (.25 - .5 / 3.) * Utils::sqr(gblcblk[0] * lz));
318  }
319 
320  if (elc.dielectric_contrast_on) {
321  if (elc.const_pot) {
322  // zero potential difference contribution
323  energy += pref / elc.box_h * lz * Utils::sqr(gblcblk[6]);
324  // external potential shift contribution
325  energy -= 2. * elc.pot_diff / elc.box_h * gblcblk[6];
326  }
327 
328  /* counter the P3M homogeneous background contribution to the
329  boundaries. We never need that, since a homogeneous background
330  spanning the artificial boundary layers is aphysical. */
331  energy +=
332  pref * (-(gblcblk[1] * gblcblk[4] + gblcblk[0] * gblcblk[5]) -
333  (1. - 2. / 3.) * gblcblk[0] * gblcblk[1] * Utils::sqr(lz));
334  }
335 
336  return this_node == 0 ? energy : 0.;
337 }

341 static auto image_sum_b(double q, double z, double d) {
342  auto const shift = box_geo.length_half()[2];
343  auto const lz = box_geo.length()[2];
344  return q / (1. - d) * (z - 2. * d * lz / (1. - d)) - q * shift / (1. - d);
345 }
346 
347 static auto image_sum_t(double q, double z, double d) {
348  auto const shift = box_geo.length_half()[2];
349  auto const lz = box_geo.length()[2];
350  return q / (1. - d) * (z + 2. * d * lz / (1. - d)) - q * shift / (1. - d);
351 }

353 double
354 ElectrostaticLayerCorrection::z_energy(ParticleRange const &particles) const {
355  constexpr std::size_t size = 4;
356  auto const xy_area_inv = box_geo.length_inv()[0] * box_geo.length_inv()[1];
357  auto const pref = prefactor * 2. * Utils::pi() * xy_area_inv;
358  auto const delta = elc.delta_mid_top * elc.delta_mid_bot;
359  auto const fac_delta_mid_bot = elc.delta_mid_bot / (1. - delta);
360  auto const fac_delta_mid_top = elc.delta_mid_top / (1. - delta);
361  auto const fac_delta = delta / (1. - delta);
362 
363  /* for non-neutral systems, this shift gives the background contribution
364  * (rsp. for this shift, the DM of the background is zero) */
365  double const shift = box_geo.length_half()[2];
366 
367  if (elc.dielectric_contrast_on) {
368  if (elc.const_pot) {
369  clear_vec(gblcblk, size);
370  for (auto const &p : particles) {
371  auto const z = p.pos()[2];
372  auto const q = p.q();
373  gblcblk[0] += q;
374  gblcblk[1] += q * (z - shift);
375  if (z < elc.space_layer) {
376  gblcblk[2] -= elc.delta_mid_bot * q;
377  gblcblk[3] -= elc.delta_mid_bot * q * (-z - shift);
378  }
379  if (z > (elc.box_h - elc.space_layer)) {
380  gblcblk[2] += elc.delta_mid_top * q;
381  gblcblk[3] += elc.delta_mid_top * q * (2. * elc.box_h - z - shift);
382  }
383  }
384  } else {
385  // metallic boundaries
386  clear_vec(gblcblk, size);
387  auto const h = elc.box_h;
388  for (auto const &p : particles) {
389  auto const z = p.pos()[2];
390  auto const q = p.q();
391  gblcblk[0] += q;
392  gblcblk[1] += q * (z - shift);
393  if (elc.dielectric_contrast_on) {
394  if (z < elc.space_layer) {
395  gblcblk[2] += fac_delta * (elc.delta_mid_bot + 1.) * q;
396  gblcblk[3] += q * (image_sum_b(elc.delta_mid_bot * delta,
397  -(2. * h + z), delta) +
398  image_sum_b(delta, -(2. * h - z), delta));
399  } else {
400  gblcblk[2] += fac_delta_mid_bot * (1. + elc.delta_mid_top) * q;
401  gblcblk[3] += q * (image_sum_b(elc.delta_mid_bot, -z, delta) +
402  image_sum_b(delta, -(2. * h - z), delta));
403  }
404  if (z > (h - elc.space_layer)) {
405  // note the minus sign here which is required due to |z_i-z_j|
406  gblcblk[2] -= fac_delta * (elc.delta_mid_top + 1.) * q;
407  gblcblk[3] -=
408  q * (image_sum_t(elc.delta_mid_top * delta, 4. * h - z, delta) +
409  image_sum_t(delta, 2. * h + z, delta));
410  } else {
411  // note the minus sign here which is required due to |z_i-z_j|
412  gblcblk[2] -= fac_delta_mid_top * (1. + elc.delta_mid_bot) * q;
413  gblcblk[3] -=
414  q * (image_sum_t(elc.delta_mid_top, 2. * h - z, delta) +
415  image_sum_t(delta, 2. * h + z, delta));
416  }
417  }
418  }
419  }
420  }
421  distribute(size);
422 
423  auto const energy = gblcblk[1] * gblcblk[2] - gblcblk[0] * gblcblk[3];
424  return (this_node == 0) ? -pref * energy : 0.;
425 }

427 void ElectrostaticLayerCorrection::add_z_force(
428  ParticleRange const &particles) const {
429  constexpr std::size_t size = 1;
430  auto const xy_area_inv = box_geo.length_inv()[0] * box_geo.length_inv()[1];
431  auto const pref = prefactor * 2. * Utils::pi() * xy_area_inv;
432  auto const delta = elc.delta_mid_top * elc.delta_mid_bot;
433  auto const fac_delta_mid_bot = elc.delta_mid_bot / (1. - delta);
434  auto const fac_delta_mid_top = elc.delta_mid_top / (1. - delta);
435  auto const fac_delta = delta / (1. - delta);
436 
437  if (elc.dielectric_contrast_on) {
438  if (elc.const_pot) {
439  clear_vec(gblcblk, size);
440  /* just counter the 2 pi |z| contribution stemming from P3M */
441  for (auto const &p : particles) {
442  auto const z = p.pos()[2];
443  auto const q = p.q();
444  if (z < elc.space_layer)
445  gblcblk[0] -= elc.delta_mid_bot * q;
446  if (z > (elc.box_h - elc.space_layer))
447  gblcblk[0] += elc.delta_mid_top * q;
448  }
449  } else {
450  clear_vec(gblcblk, size);
451  for (auto const &p : particles) {
452  auto const z = p.pos()[2];
453  auto const q = p.q();
454  if (z < elc.space_layer) {
455  gblcblk[0] += fac_delta * (elc.delta_mid_bot + 1.) * q;
456  } else {
457  gblcblk[0] += fac_delta_mid_bot * (elc.delta_mid_top + 1.) * q;
458  }
459  if (z > (elc.box_h - elc.space_layer)) {
460  // note the minus sign here which is required due to |z_i-z_j|
461  gblcblk[0] -= fac_delta * (elc.delta_mid_top + 1.) * q;
462  } else {
463  // note the minus sign here which is required due to |z_i-z_j|
464  gblcblk[0] -= fac_delta_mid_top * (elc.delta_mid_bot + 1.) * q;
465  }
466  }
467  }
468 
469  gblcblk[0] *= pref;
470 
471  distribute(size);
472 
473  for (auto &p : particles) {
474  p.force()[2] += gblcblk[0] * p.q();
475  }
476  }

483 /** \name q=0 or p=0 per frequency code */
484 /**@{*/
485 template <PoQ axis>
486 void setup_PoQ(elc_data const &elc, double prefactor, std::size_t index,
487  double omega, ParticleRange const &particles) {
488  assert(index >= 1);
489  constexpr std::size_t size = 4;
490  auto const xy_area_inv = box_geo.length_inv()[0] * box_geo.length_inv()[1];
491  auto const pref_di = prefactor * 4. * Utils::pi() * xy_area_inv;
492  auto const pref = -pref_di / expm1(omega * box_geo.length()[2]);
493  double lclimgebot[4], lclimgetop[4], lclimge[4];
494  double fac_delta_mid_bot = 1., fac_delta_mid_top = 1., fac_delta = 1.;
495 
496  if (elc.dielectric_contrast_on) {
497  auto const delta = elc.delta_mid_top * elc.delta_mid_bot;
498  auto const fac_elc = 1. / (1. - delta * exp(-omega * 2. * elc.box_h));
499  fac_delta_mid_bot = elc.delta_mid_bot * fac_elc;
500  fac_delta_mid_top = elc.delta_mid_top * fac_elc;
501  fac_delta = fac_delta_mid_bot * elc.delta_mid_top;
502  }
503 
504  clear_vec(lclimge, size);
505  clear_vec(gblcblk, size);
506  auto const &sc_cache = (axis == PoQ::P) ? scxcache : scycache;
507 
508  std::size_t ic = 0;
509  auto const o = (index - 1) * particles.size();
510  for (auto const &p : particles) {
511  auto const z = p.pos()[2];
512  auto const q = p.q();
513  auto e = exp(omega * z);
514 
515  partblk[size * ic + POQESM] = q * sc_cache[o + ic].s / e;
516  partblk[size * ic + POQESP] = q * sc_cache[o + ic].s * e;
517  partblk[size * ic + POQECM] = q * sc_cache[o + ic].c / e;
518  partblk[size * ic + POQECP] = q * sc_cache[o + ic].c * e;
519 
520  add_vec(gblcblk, gblcblk, block(partblk.data(), ic, size), size);
521 
522  if (elc.dielectric_contrast_on) {
523  if (z < elc.space_layer) { // handle the lower case first
524  // negative sign is okay here as the image is located at -z
525 
526  e = exp(-omega * z);
527 
528  auto const scale = q * elc.delta_mid_bot;
529 
530  lclimgebot[POQESM] = sc_cache[o + ic].s / e;
531  lclimgebot[POQESP] = sc_cache[o + ic].s * e;
532  lclimgebot[POQECM] = sc_cache[o + ic].c / e;
533  lclimgebot[POQECP] = sc_cache[o + ic].c * e;
534 
535  addscale_vec(gblcblk, scale, lclimgebot, gblcblk, size);
536 
537  e = (exp(omega * (-z - 2. * elc.box_h)) * elc.delta_mid_bot +
538  exp(omega * (+z - 2. * elc.box_h))) *
539  fac_delta;
540  } else {
541  e = (exp(-omega * z) +
542  exp(omega * (z - 2. * elc.box_h)) * elc.delta_mid_top) *
543  fac_delta_mid_bot;
544  }
545 
546  lclimge[POQESP] += q * sc_cache[o + ic].s * e;
547  lclimge[POQECP] += q * sc_cache[o + ic].c * e;
548 
549  if (z > (elc.box_h - elc.space_layer)) { // handle the upper case now

625 static void setup_PQ(elc_data const &elc, double prefactor, std::size_t index_p,
626  std::size_t index_q, double omega,

628  assert(index_p >= 1);
629  assert(index_q >= 1);
630  constexpr std::size_t size = 8;
631  auto const xy_area_inv = box_geo.length_inv()[0] * box_geo.length_inv()[1];
632  auto const pref_di = prefactor * 8 * Utils::pi() * xy_area_inv;
633  auto const pref = -pref_di / expm1(omega * box_geo.length()[2]);
634  double lclimgebot[8], lclimgetop[8], lclimge[8];
635  double fac_delta_mid_bot = 1, fac_delta_mid_top = 1, fac_delta = 1;
636  if (elc.dielectric_contrast_on) {
637  auto const delta = elc.delta_mid_top * elc.delta_mid_bot;
638  auto const fac_elc = 1. / (1. - delta * exp(-omega * 2. * elc.box_h));
639  fac_delta_mid_bot = elc.delta_mid_bot * fac_elc;
640  fac_delta_mid_top = elc.delta_mid_top * fac_elc;
641  fac_delta = fac_delta_mid_bot * elc.delta_mid_top;

812 void ElectrostaticLayerCorrection::add_force(
813  ParticleRange const &particles) const {
814  auto constexpr c_2pi = 2. * Utils::pi();
815  auto const n_freqs = prepare_sc_cache(particles, elc.far_cut);
816  auto const n_scxcache = std::get<0>(n_freqs);
817  auto const n_scycache = std::get<1>(n_freqs);
818  partblk.resize(particles.size() * 8);
819 
820  add_dipole_force(particles);
821  add_z_force(particles);
822 
823  /* the second condition is just for the case of numerical accident */
824  for (std::size_t p = 1;
825  box_geo.length_inv()[0] * static_cast<double>(p - 1) < elc.far_cut &&
826  p <= n_scxcache;
827  p++) {
828  auto const omega = c_2pi * box_geo.length_inv()[0] * static_cast<double>(p);
829  setup_PoQ<PoQ::P>(elc, prefactor, p, omega, particles);
830  distribute(4);
831  add_PoQ_force<PoQ::P>(particles);
832  }
833 
834  for (std::size_t q = 1;
835  box_geo.length_inv()[1] * static_cast<double>(q - 1) < elc.far_cut &&
836  q <= n_scycache;
837  q++) {
838  auto const omega = c_2pi * box_geo.length_inv()[1] * static_cast<double>(q);
839  setup_PoQ<PoQ::Q>(elc, prefactor, q, omega, particles);
840  distribute(4);
841  add_PoQ_force<PoQ::Q>(particles);
842  }
843 
844  for (std::size_t p = 1;
845  box_geo.length_inv()[0] * static_cast<double>(p - 1) < elc.far_cut &&
846  p <= n_scxcache;
847  p++) {
848  for (std::size_t q = 1;
849  Utils::sqr(box_geo.length_inv()[0] * static_cast<double>(p - 1)) +
850  Utils::sqr(box_geo.length_inv()[1] *
851  static_cast<double>(q - 1)) <
852  elc.far_cut2 &&
853  q <= n_scycache;
854  q++) {
855  auto const omega =
856  c_2pi *
857  sqrt(Utils::sqr(box_geo.length_inv()[0] * static_cast<double>(p)) +
858  Utils::sqr(box_geo.length_inv()[1] * static_cast<double>(q)));
859  setup_PQ(elc, prefactor, p, q, omega, particles);
860  distribute(8);
861  add_PQ_force(p, q, omega, particles);
862  }
863  }

922 double ElectrostaticLayerCorrection::tune_far_cut() const {
923  // Largest reasonable cutoff for far formula
924  auto constexpr maximal_far_cut = 50.;
925  auto const box_l_x_inv = box_geo.length_inv()[0];
926  auto const box_l_y_inv = box_geo.length_inv()[1];
927  auto const min_inv_boxl = std::min(box_l_x_inv, box_l_y_inv);
928  auto const box_l_z = box_geo.length()[2];
929  // adjust lz according to dielectric layer method
930  auto const lz =
931  (elc.dielectric_contrast_on) ? elc.box_h + elc.space_layer : box_l_z;
932 
933  auto tuned_far_cut = min_inv_boxl;
934  double err;
935  do {
936  auto const pref = 2. * Utils::pi() * tuned_far_cut;
937  auto const sum = pref + 2. * (box_l_x_inv + box_l_y_inv);
938  auto const den = -expm1(-pref * lz);
939  auto const num1 = exp(pref * (elc.box_h - lz));
940  auto const num2 = exp(-pref * (elc.box_h + lz));
941 
942  err = 0.5 / den *
943  (num1 * (sum + 1. / (lz - elc.box_h)) / (lz - elc.box_h) +
944  num2 * (sum + 1. / (lz + elc.box_h)) / (lz + elc.box_h));
945 
946  tuned_far_cut += min_inv_boxl;
947  } while (err > elc.maxPWerror and tuned_far_cut < maximal_far_cut);
948  if (tuned_far_cut >= maximal_far_cut) {
949  throw std::runtime_error("ELC tuning failed: maxPWerror too small");
950  }
951  return tuned_far_cut - min_inv_boxl;
952 }
