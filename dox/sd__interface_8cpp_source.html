77void register_integrator(StokesianDynamicsParameters const &obj) {
78 auto const &box_geo = *System::get_system().box_geo;
79 if (box_geo.periodic(0) or box_geo.periodic(1) or box_geo.periodic(2)) {
80 throw std::runtime_error(

86/** Update translational and rotational velocities of all particles. */
87template <typename ParticleIterable>
88void sd_update_locally(ParticleIterable const &parts) {
89 std::size_t i = 0;
90
91 // Even though on the head node, the v_sd vector is larger than
92 // the (local) parts vector, this should still work. Because the local
93 // particles correspond to the first 6*n entries in the head node's v_sd
94 // (which holds the velocities of ALL particles).
95
96 for (auto &p : parts) {
97 // Copy velocities
98 p.v()[0] = v_sd[6 * i + 0];
99 p.v()[1] = v_sd[6 * i + 1];
100 p.v()[2] = v_sd[6 * i + 2];
101
102 p.omega()[0] = v_sd[6 * i + 3];
103 p.omega()[1] = v_sd[6 * i + 4];
104 p.omega()[2] = v_sd[6 * i + 5];
105
106 ++i;
107 }
108}

110StokesianDynamicsParameters::StokesianDynamicsParameters(
111 double viscosity, std::unordered_map<int, double> radii, int flags)
112 : viscosity{viscosity}, radii{radii}, flags{flags} {
113 if (viscosity < 0.) {
114 throw std::domain_error("Viscosity has an invalid value: " +
115 std::to_string(viscosity));
116 }
117 /* Check that radii are positive */
118 for (auto const &[p_type, radius] : radii) {
119 if (radius < 0.) {
120 throw std::domain_error(
121 "Particle radius for type " + std::to_string(p_type) +
122 " has an invalid value: " + std::to_string(radius));
123 }
124 }
125}

127void propagate_vel_pos_sd(ParticleRangeStokesian const &particles,
128 StokesianThermostat const &stokesian,
129 double const time_step, double const kT) {
130
131 static std::vector<SD_particle_data> parts_buffer{};
132
133 parts_buffer.clear();
134 std::ranges::transform(particles, std::back_inserter(parts_buffer),
135 [](auto const &p) { return SD_particle_data(p); });
136 Utils::Mpi::gather_buffer(parts_buffer, ::comm_cart, 0);
137
138 /* Buffer that holds local particle data, and all particles on the head
139 * node used for sending particle data to head node. */
140 if (::comm_cart.rank() == 0) {
141 std::size_t n_part = parts_buffer.size();
142
143 static std::vector<double> x_host{};
144 static std::vector<double> f_host{};
145 static std::vector<double> a_host{};
146
147 x_host.resize(6 * n_part);
148 f_host.resize(6 * n_part);
149 a_host.resize(n_part);
150
151 std::size_t i = 0;
152 for (auto const &p : parts_buffer) {
153 x_host[6 * i + 0] = p.pos[0];
154 x_host[6 * i + 1] = p.pos[1];
155 x_host[6 * i + 2] = p.pos[2];
156 // Actual orientation is not needed, just need default.
157 x_host[6 * i + 3] = 1;
158 x_host[6 * i + 4] = 0;
159 x_host[6 * i + 5] = 0;
160
161 f_host[6 * i + 0] = p.ext_force.f[0];
162 f_host[6 * i + 1] = p.ext_force.f[1];
163 f_host[6 * i + 2] = p.ext_force.f[2];
164
165 f_host[6 * i + 3] = p.ext_force.torque[0];
166 f_host[6 * i + 4] = p.ext_force.torque[1];
167 f_host[6 * i + 5] = p.ext_force.torque[2];
168
169 a_host[i] = params.radii.at(p.type);
170
171 ++i;
172 }
173
174 v_sd = sd_cpu(x_host, f_host, a_host, n_part, params.viscosity,
175 std::sqrt(kT / time_step),
176 static_cast<std::size_t>(stokesian.rng_counter()),
177 static_cast<std::size_t>(stokesian.rng_seed()), params.flags);
178 } else { // if (this_node == 0)
179 v_sd.resize(particles.size() * 6);
180 } // if (this_node == 0) {...} else
181
182 Utils::Mpi::scatter_buffer(
183 v_sd.data(), static_cast<int>(particles.size() * 6), ::comm_cart, 0);
184 sd_update_locally(particles);
185}
