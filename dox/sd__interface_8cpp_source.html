72/** Update translational and rotational velocities of all particles. */
73template <typename ParticleIterable>
74static void sd_update_locally(ParticleIterable const &parts,
75 std::vector<double> const &v_sd) {
76 std::size_t i = 0;
77
78 // Even though on the head node, the v_sd vector is larger than
79 // the (local) parts vector, this should still work. Because the local
80 // particles correspond to the first 6*n entries in the head node's v_sd
81 // (which holds the velocities of ALL particles).
82
83 for (auto &p : parts) {
84 // Copy velocities
85 p.v()[0] = v_sd[6 * i + 0];
86 p.v()[1] = v_sd[6 * i + 1];
87 p.v()[2] = v_sd[6 * i + 2];
88
89 p.omega()[0] = v_sd[6 * i + 3];
90 p.omega()[1] = v_sd[6 * i + 4];
91 p.omega()[2] = v_sd[6 * i + 5];
92
93 ++i;
94 }
95}

97StokesianDynamics::StokesianDynamics(double viscosity,
98 std::unordered_map<int, double> radii,
99 int flags)
100 : viscosity{viscosity}, radii{radii}, flags{flags} {
101 if (viscosity < 0.) {
102 throw std::domain_error("Viscosity has an invalid value: " +
103 std::to_string(viscosity));
104 }
105 /* Check that radii are positive */
106 for (auto const &[p_type, radius] : radii) {
107 if (radius < 0.) {
108 throw std::domain_error(
109 "Particle radius for type " + std::to_string(p_type) +
110 " has an invalid value: " + std::to_string(radius));
111 }
112 }
113}

115void StokesianDynamics::propagate_vel_pos(
116 ParticleRangeStokesian const &particles,
117 StokesianThermostat const &stokesian, double const time_step,
118 double const kT) const {
119
120 std::vector<SD_particle_data> parts_buffer{};
121 parts_buffer.reserve(particles.size());
122
123 for (auto const &p : particles) {
124 parts_buffer.emplace_back(p);
125 }
126 Utils::Mpi::gather_buffer(parts_buffer, ::comm_cart, 0);
127
128 /* Buffer that holds local particle data, and all particles on the head
129 * node used for sending particle data to head node. */
130 if (::comm_cart.rank() == 0) {
131 std::size_t n_part = parts_buffer.size();
132
133 x_host.resize(6 * n_part);
134 f_host.resize(6 * n_part);
135 a_host.resize(n_part);
136
137 std::size_t i = 0;
138 for (auto const &p : parts_buffer) {
139 x_host[6 * i + 0] = p.pos[0];
140 x_host[6 * i + 1] = p.pos[1];
141 x_host[6 * i + 2] = p.pos[2];
142 // Actual orientation is not needed, just need default.
143 x_host[6 * i + 3] = 1;
144 x_host[6 * i + 4] = 0;
145 x_host[6 * i + 5] = 0;
146
147 f_host[6 * i + 0] = p.ext_force.f[0];
148 f_host[6 * i + 1] = p.ext_force.f[1];
149 f_host[6 * i + 2] = p.ext_force.f[2];
150
151 f_host[6 * i + 3] = p.ext_force.torque[0];
152 f_host[6 * i + 4] = p.ext_force.torque[1];
153 f_host[6 * i + 5] = p.ext_force.torque[2];
154
155 a_host[i] = radii.at(p.type);
156
157 ++i;
158 }
159
160 v_sd = sd_cpu(x_host, f_host, a_host, n_part, viscosity,
161 std::sqrt(kT / time_step),
162 static_cast<std::size_t>(stokesian.rng_counter()),
163 static_cast<std::size_t>(stokesian.rng_seed()), flags);
164 } else { // if (this_node == 0)
165 v_sd.resize(particles.size() * 6);
166 } // if (this_node == 0) {...} else
167
168 Utils::Mpi::scatter_buffer(
169 v_sd.data(), static_cast<int>(particles.size() * 6), ::comm_cart, 0);
170 sd_update_locally(particles, v_sd);
171}
