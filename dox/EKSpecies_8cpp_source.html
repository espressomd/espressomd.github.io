51Variant EKSpecies::do_call_method(std::string const &method,
52 VariantMap const &parameters) {
53 if (method == "update_flux_boundary_from_shape") {
54 auto values = get_value<std::vector<double>>(parameters, "values");
55 std::ranges::for_each(values, [this](double &v) { v *= m_conv_flux; });
56
57 m_instance->update_flux_boundary_from_shape(
58 get_value<std::vector<int>>(parameters, "raster"), values);
59 return {};
60 }
61 if (method == "update_density_boundary_from_shape") {
62 auto values = get_value<std::vector<double>>(parameters, "values");
63 std::ranges::for_each(values, [this](double &v) { v *= m_conv_density; });
64 m_instance->update_density_boundary_from_shape(
65 get_value<std::vector<int>>(parameters, "raster"), values);
66 return {};
67 }
68 if (method == "clear_flux_boundaries") {
69 m_instance->clear_flux_boundaries();
70 return {};
71 }
72 if (method == "clear_density_boundaries") {
73 m_instance->clear_density_boundaries();
74 return {};
75 }
76 if (method == "save_checkpoint") {
77 auto const path = get_value<std::filesystem::path>(parameters, "path");
78 auto const mode = get_value<int>(parameters, "mode");
79 save_checkpoint(path, mode);
80 return {};
81 }
82 if (method == "load_checkpoint") {
83 auto const path = get_value<std::filesystem::path>(parameters, "path");
84 auto const mode = get_value<int>(parameters, "mode");
85 load_checkpoint(path, mode);
86 return {};
87 }
88 return Base::do_call_method(method, parameters);
89}

91void EKSpeciesCPU::make_instance(VariantMap const &params) {
92 auto const diffusion = get_value<double>(params, "diffusion");
93 auto const ext_efield = get_value<Utils::Vector3d>(params, "ext_efield");
94 auto const density = get_value<double>(params, "density");
95 auto const kT = get_value<double>(params, "kT");
96 auto const ek_diffusion = diffusion * m_conv_diffusion;
97 auto const ek_ext_efield = ext_efield * m_conv_ext_efield;
98 auto const ek_density = density * m_conv_density;
99 auto const ek_kT = kT * m_conv_energy;
100 m_instance = ::walberla::new_ek_walberla_cpu(
101 m_lattice->lattice(), ek_diffusion, ek_kT,
102 get_value<double>(params, "valency"), ek_ext_efield, ek_density,
103 get_value<bool>(params, "advection"),
104 get_value<bool>(params, "friction_coupling"),
105 get_value<bool>(params, "single_precision"),
106 get_value_or<bool>(params, "thermalized", false),
107 static_cast<uint>(get_value_or<int>(params, "seed", 0)));
108 m_instance->ghost_communication();
109}

133void EKSpecies::do_construct(VariantMap const &params) {
134 m_lattice = get_value<std::shared_ptr<LatticeWalberla>>(params, "lattice");
135 m_vtk_writers =
136 get_value_or<decltype(m_vtk_writers)>(params, "vtk_writers", {});
137 auto const agrid = get_value<double>(m_lattice->get_parameter("agrid"));
138 auto const density = get_value<double>(params, "density");
139 auto const kT = get_value<double>(params, "kT");
140 auto const tau = m_tau = get_value<double>(params, "tau");
141 auto const seed = get_value_or<int>(params, "seed", 0);
142 context()->parallel_try_catch([&]() {
143 if (get_value<bool>(params, "thermalized") and
144 not params.contains("seed")) {
145 throw std::invalid_argument(
146 "Parameter 'seed' is required for thermalized EKSpecies");
147 }
148 if (seed < 0) {
149 throw std::domain_error("Parameter 'seed' must be >= 0");
150 }
151 if (tau <= 0.) {
152 throw std::domain_error("Parameter 'tau' must be > 0");
153 }
154 if (kT < 0.) {
155 throw std::domain_error("Parameter 'kT' must be >= 0");
156 }
157 if (density < 0.) {
158 throw std::domain_error("Parameter 'density' must be >= 0");
159 }
160 m_conv_energy = Utils::int_pow<2>(tau) / Utils::int_pow<2>(agrid);
161 m_conv_diffusion = tau / Utils::int_pow<2>(agrid);
162 m_conv_ext_efield = Utils::int_pow<2>(tau) / agrid;
163 m_conv_density = Utils::int_pow<3>(agrid);
164 m_conv_flux = tau * Utils::int_pow<2>(agrid);
165 m_density = density * m_conv_density;
166 make_instance(params);
167 for (auto &vtk : m_vtk_writers) {
168 vtk->attach_to_lattice(m_instance, get_lattice_to_md_units_conversion());
169 }
170 });
171}

173void EKSpecies::load_checkpoint(std::filesystem::path const &path, int mode) {
174 auto &ek_obj = *m_instance;
175
176 auto const read_metadata = [&ek_obj](CheckpointFile &cpfile) {
177 auto const expected_grid_size = ek_obj.get_lattice().get_grid_dimensions();
178 Utils::Vector3i read_grid_size;
179 cpfile.read(read_grid_size);
180 if (read_grid_size != expected_grid_size) {
181 std::stringstream message;
182 message << "grid dimensions mismatch, read [" << read_grid_size << "], "
183 << "expected [" << expected_grid_size << "].";
184 throw std::runtime_error(message.str());
185 }
186 };
187
188 auto const read_data = [&ek_obj](CheckpointFile &cpfile) {
189 auto const grid_size = ek_obj.get_lattice().get_grid_dimensions();
190 auto const i_max = grid_size[0];
191 auto const j_max = grid_size[1];
192 auto const k_max = grid_size[2];
193 EKWalberlaNodeState cpnode;
194 for (int i = 0; i < i_max; i++) {
195 for (int j = 0; j < j_max; j++) {
196 for (int k = 0; k < k_max; k++) {
197 auto const ind = Utils::Vector3i{{i, j, k}};
198 cpfile.read(cpnode.density);
199 cpfile.read(cpnode.is_boundary_density);
200 if (cpnode.is_boundary_density) {
201 cpfile.read(cpnode.density_boundary);
202 }
203 cpfile.read(cpnode.is_boundary_flux);
204 if (cpnode.is_boundary_flux) {
205 cpfile.read(cpnode.flux_boundary);
206 }
207 ek_obj.set_node_density(ind, cpnode.density);
208 if (cpnode.is_boundary_density) {
209 ek_obj.set_node_density_boundary(ind, cpnode.density_boundary);
210 }
211 if (cpnode.is_boundary_flux) {
212 ek_obj.set_node_flux_boundary(ind, cpnode.flux_boundary);
213 }
214 }
215 }
216 }
217 };
218
219 auto const on_success = [&ek_obj]() { ek_obj.ghost_communication(); };
220
221 load_checkpoint_common(*context(), "EK", path, mode, read_metadata, read_data,
222 on_success);
223}

225void EKSpecies::save_checkpoint(std::filesystem::path const &path, int mode) {
226 auto &ek_obj = *m_instance;
227
228 auto const write_metadata = [&ek_obj,
229 mode](std::shared_ptr<CheckpointFile> cpfile_ptr,
230 Context const &context) {
231 auto const grid_size = ek_obj.get_lattice().get_grid_dimensions();
232 if (context.is_head_node()) {
233 cpfile_ptr->write(grid_size);
234 unit_test_handle(mode);
235 }
236 };
237
238 auto const on_failure = [](std::shared_ptr<CheckpointFile> const &,
239 Context const &context) {
240 if (context.is_head_node()) {
241 auto failure = true;
242 boost::mpi::broadcast(context.get_comm(), failure, 0);
243 }
244 };
245
246 auto const write_data = [&ek_obj,
247 mode](std::shared_ptr<CheckpointFile> cpfile_ptr,
248 Context const &context) {
249 auto const get_node_checkpoint =
250 [&](Utils::Vector3i const &ind) -> std::optional<EKWalberlaNodeState> {
251 auto const density = ek_obj.get_node_density(ind);
252 auto const is_b_d = ek_obj.get_node_is_density_boundary(ind);
253 auto const dens_b = ek_obj.get_node_density_at_boundary(ind);
254 auto const is_b_f = ek_obj.get_node_is_flux_boundary(ind);
255 auto const flux_b = ek_obj.get_node_flux_at_boundary(ind);
256 if (density and is_b_d and is_b_f and
257 ((*is_b_d) ? dens_b.has_value() : true) and
258 ((*is_b_f) ? flux_b.has_value() : true)) {
259 EKWalberlaNodeState cpnode;
260 cpnode.density = *density;
261 cpnode.is_boundary_density = *is_b_d;
262 if (*is_b_d) {
263 cpnode.density_boundary = *dens_b;
264 }
265 cpnode.is_boundary_flux = *is_b_f;
266 if (*is_b_f) {
267 cpnode.flux_boundary = *flux_b;
268 }
269 return cpnode;
270 }
271 return std::nullopt;
272 };
273
274 auto failure = false;
275 auto const &comm = context.get_comm();
276 auto const is_head_node = context.is_head_node();
277 auto const unit_test_mode = (mode != static_cast<int>(CptMode::ascii)) and
278 (mode != static_cast<int>(CptMode::binary));
279 auto const grid_size = ek_obj.get_lattice().get_grid_dimensions();
280 auto const i_max = grid_size[0];
281 auto const j_max = grid_size[1];
282 auto const k_max = grid_size[2];
283 EKWalberlaNodeState cpnode;
284 for (int i = 0; i < i_max; i++) {
285 for (int j = 0; j < j_max; j++) {
286 for (int k = 0; k < k_max; k++) {
287 auto const ind = Utils::Vector3i{{i, j, k}};
288 auto const result = get_node_checkpoint(ind);
289 if (!unit_test_mode) {
290 assert(1 == boost::mpi::all_reduce(comm, static_cast<int>(!!result),
291 std::plus<>()) &&
292 "Incorrect number of return values");
293 }
294 if (is_head_node) {
295 if (result) {
296 cpnode = *result;
297 } else {
298 comm.recv(boost::mpi::any_source, 42, cpnode);
299 }
300 auto &cpfile = *cpfile_ptr;
301 cpfile.write(cpnode.density);
302 cpfile.write(cpnode.is_boundary_density);
303 if (cpnode.is_boundary_density) {
304 cpfile.write(cpnode.density_boundary);
305 }
306 cpfile.write(cpnode.is_boundary_flux);
307 if (cpnode.is_boundary_flux) {
308 cpfile.write(cpnode.flux_boundary);
309 }
310 boost::mpi::broadcast(comm, failure, 0);
311 } else {
312 if (result) {
313 comm.send(0, 42, *result);
314 }
315 boost::mpi::broadcast(comm, failure, 0);
316 if (failure) {
317 return;
318 }
319 }
320 }
321 }
322 }
323 };
324
325 save_checkpoint_common(*context(), "EK", path, mode, write_metadata,
326 write_data, on_failure);
327}
