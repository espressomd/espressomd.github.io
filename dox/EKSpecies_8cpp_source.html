49 Variant EKSpecies::do_call_method(std::string const &method,
50  VariantMap const &parameters) {
51  if (method == "update_flux_boundary_from_shape") {
52  auto values = get_value<std::vector<double>>(parameters, "values");
53  std::transform(values.begin(), values.end(), values.begin(),
54  [this](double v) { return v * m_conv_flux; });
55 
56  m_instance->update_flux_boundary_from_shape(
57  get_value<std::vector<int>>(parameters, "raster"), values);
58  return {};
59  }
60  if (method == "update_density_boundary_from_shape") {
61  auto values = get_value<std::vector<double>>(parameters, "values");
62  std::transform(values.begin(), values.end(), values.begin(),
63  [this](double v) { return v * m_conv_density; });
64  m_instance->update_density_boundary_from_shape(
65  get_value<std::vector<int>>(parameters, "raster"), values);
66  return {};
67  }
68  if (method == "clear_flux_boundaries") {
69  m_instance->clear_flux_boundaries();
70  return {};
71  }
72  if (method == "clear_density_boundaries") {
73  m_instance->clear_density_boundaries();
74  return {};
75  }
76  if (method == "save_checkpoint") {
77  auto const path = get_value<std::string>(parameters, "path");
78  auto const mode = get_value<int>(parameters, "mode");
79  save_checkpoint(path, mode);
80  return {};
81  }
82  if (method == "load_checkpoint") {
83  auto const path = get_value<std::string>(parameters, "path");
84  auto const mode = get_value<int>(parameters, "mode");
85  load_checkpoint(path, mode);
86  return {};
87  }
88  return Base::do_call_method(method, parameters);
89 }

91 void EKSpecies::do_construct(VariantMap const &args) {
92  m_lattice = get_value<std::shared_ptr<LatticeWalberla>>(args, "lattice");
93  m_vtk_writers =
94  get_value_or<decltype(m_vtk_writers)>(args, "vtk_writers", {});
95  auto const single_precision = get_value<bool>(args, "single_precision");
96  auto const agrid = get_value<double>(m_lattice->get_parameter("agrid"));
97  auto const diffusion = get_value<double>(args, "diffusion");
98  auto const ext_efield = get_value<Utils::Vector3d>(args, "ext_efield");
99  auto const density = get_value<double>(args, "density");
100  auto const kT = get_value<double>(args, "kT");
101  auto const tau = m_tau = get_value<double>(args, "tau");
102  context()->parallel_try_catch([&]() {
103  if (tau <= 0.) {
104  throw std::domain_error("Parameter 'tau' must be > 0");
105  }
106  if (kT < 0.) {
107  throw std::domain_error("Parameter 'kT' must be >= 0");
108  }
109  if (density < 0.) {
110  throw std::domain_error("Parameter 'density' must be >= 0");
111  }
112  m_conv_energy = Utils::int_pow<2>(tau) / Utils::int_pow<2>(agrid);
113  m_conv_diffusion = tau / Utils::int_pow<2>(agrid);
114  m_conv_ext_efield = Utils::int_pow<2>(tau) / agrid;
115  m_conv_density = Utils::int_pow<3>(agrid);
116  m_conv_flux = tau * Utils::int_pow<2>(agrid);
117  auto const ek_diffusion = diffusion * m_conv_diffusion;
118  auto const ek_ext_efield = ext_efield * m_conv_ext_efield;
119  auto const ek_density = m_density = density * m_conv_density;
120  auto const ek_kT = kT * m_conv_energy;
121  m_instance = ::walberla::new_ek_walberla(
122  m_lattice->lattice(), ek_diffusion, ek_kT,
123  get_value<double>(args, "valency"), ek_ext_efield, ek_density,
124  get_value<bool>(args, "advection"),
125  get_value<bool>(args, "friction_coupling"), single_precision);
126  for (auto &vtk : m_vtk_writers) {
127  vtk->attach_to_lattice(m_instance, get_latice_to_md_units_conversion());
128  }
129  });
130 }

132 void EKSpecies::load_checkpoint(std::string const &filename, int mode) {
133  auto &ek_obj = *m_instance;
134 
135  auto const read_metadata = [&ek_obj](CheckpointFile &cpfile) {
136  auto const expected_grid_size = ek_obj.get_lattice().get_grid_dimensions();
137  Utils::Vector3i read_grid_size;
138  cpfile.read(read_grid_size);
139  if (read_grid_size != expected_grid_size) {
140  std::stringstream message;
141  message << "grid dimensions mismatch, "
142  << "read [" << read_grid_size << "], "
143  << "expected [" << expected_grid_size << "].";
144  throw std::runtime_error(message.str());
145  }
146  };
147 
148  auto const read_data = [&ek_obj](CheckpointFile &cpfile) {
149  auto const grid_size = ek_obj.get_lattice().get_grid_dimensions();
150  auto const i_max = grid_size[0];
151  auto const j_max = grid_size[1];
152  auto const k_max = grid_size[2];
153  EKWalberlaNodeState cpnode;
154  for (int i = 0; i < i_max; i++) {
155  for (int j = 0; j < j_max; j++) {
156  for (int k = 0; k < k_max; k++) {
157  auto const ind = Utils::Vector3i{{i, j, k}};
158  cpfile.read(cpnode.density);
159  cpfile.read(cpnode.is_boundary_density);
160  if (cpnode.is_boundary_density) {
161  cpfile.read(cpnode.density_boundary);
162  }
163  cpfile.read(cpnode.is_boundary_flux);
164  if (cpnode.is_boundary_flux) {
165  cpfile.read(cpnode.flux_boundary);
166  }
167  ek_obj.set_node_density(ind, cpnode.density);
168  if (cpnode.is_boundary_density) {
169  ek_obj.set_node_density_boundary(ind, cpnode.density_boundary);
170  }
171  if (cpnode.is_boundary_flux) {
172  ek_obj.set_node_flux_boundary(ind, cpnode.flux_boundary);
173  }
174  }
175  }
176  }
177  };
178 
179  auto const on_success = [&ek_obj]() { ek_obj.ghost_communication(); };
180 
181  load_checkpoint_common(*context(), "EK", filename, mode, read_metadata,
182  read_data, on_success);
183 }

185 void EKSpecies::save_checkpoint(std::string const &filename, int mode) {
186  auto &ek_obj = *m_instance;
187 
188  auto const write_metadata = [&ek_obj,
189  mode](std::shared_ptr<CheckpointFile> cpfile_ptr,
190  Context const &context) {
191  auto const grid_size = ek_obj.get_lattice().get_grid_dimensions();
192  if (context.is_head_node()) {
193  cpfile_ptr->write(grid_size);
194  unit_test_handle(mode);
195  }
196  };
197 
198  auto const on_failure = [](std::shared_ptr<CheckpointFile> const &,
199  Context const &context) {
200  if (context.is_head_node()) {
201  auto failure = true;
202  boost::mpi::broadcast(context.get_comm(), failure, 0);
203  }
204  };
205 
206  auto const write_data = [&ek_obj,
207  mode](std::shared_ptr<CheckpointFile> cpfile_ptr,
208  Context const &context) {
209  auto const get_node_checkpoint =
210  [&](Utils::Vector3i const &ind) -> std::optional<EKWalberlaNodeState> {
211  auto const density = ek_obj.get_node_density(ind);
212  auto const is_b_d = ek_obj.get_node_is_density_boundary(ind);
213  auto const dens_b = ek_obj.get_node_density_at_boundary(ind);
214  auto const is_b_f = ek_obj.get_node_is_flux_boundary(ind);
215  auto const flux_b = ek_obj.get_node_flux_at_boundary(ind);
216  if (density and is_b_d and is_b_f and
217  ((*is_b_d) ? dens_b.has_value() : true) and
218  ((*is_b_f) ? flux_b.has_value() : true)) {
219  EKWalberlaNodeState cpnode;
220  cpnode.density = *density;
221  cpnode.is_boundary_density = *is_b_d;
222  if (*is_b_d) {
223  cpnode.density_boundary = *dens_b;
224  }
225  cpnode.is_boundary_flux = *is_b_f;
226  if (*is_b_f) {
227  cpnode.flux_boundary = *flux_b;
228  }
229  return cpnode;
230  }
231  return std::nullopt;
232  };
233 
234  auto failure = false;
235  auto const &comm = context.get_comm();
236  auto const is_head_node = context.is_head_node();
237  auto const unit_test_mode = (mode != static_cast<int>(CptMode::ascii)) and
238  (mode != static_cast<int>(CptMode::binary));
239  auto const grid_size = ek_obj.get_lattice().get_grid_dimensions();
240  auto const i_max = grid_size[0];
241  auto const j_max = grid_size[1];
242  auto const k_max = grid_size[2];
243  EKWalberlaNodeState cpnode;
244  for (int i = 0; i < i_max; i++) {
245  for (int j = 0; j < j_max; j++) {
246  for (int k = 0; k < k_max; k++) {
247  auto const ind = Utils::Vector3i{{i, j, k}};
248  auto const result = get_node_checkpoint(ind);
249  if (!unit_test_mode) {
250  assert(1 == boost::mpi::all_reduce(comm, static_cast<int>(!!result),
251  std::plus<>()) &&
252  "Incorrect number of return values");
253  }
254  if (is_head_node) {
255  if (result) {
256  cpnode = *result;
257  } else {
258  comm.recv(boost::mpi::any_source, 42, cpnode);
259  }
260  auto &cpfile = *cpfile_ptr;
261  cpfile.write(cpnode.density);
262  cpfile.write(cpnode.is_boundary_density);
263  if (cpnode.is_boundary_density) {
264  cpfile.write(cpnode.density_boundary);
265  }
266  cpfile.write(cpnode.is_boundary_flux);
267  if (cpnode.is_boundary_flux) {
268  cpfile.write(cpnode.flux_boundary);
269  }
270  boost::mpi::broadcast(comm, failure, 0);
271  } else {
272  if (result) {
273  comm.send(0, 42, *result);
274  }
275  boost::mpi::broadcast(comm, failure, 0);
276  if (failure) {
277  return;
278  }
279  }
280  }
281  }
282  }
283  };
284 
285  save_checkpoint_common(*context(), "EK", filename, mode, write_metadata,
286  write_data, on_failure);
287 }
