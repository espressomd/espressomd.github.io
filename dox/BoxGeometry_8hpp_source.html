34 namespace detail {
35 /**
36  * @brief Get the minimum-image distance between two coordinates.
37  * @param a Coordinate of the terminal point.
38  * @param b Coordinate of the initial point.
39  * @param box_length Box length.
40  * @param box_length_inv Inverse box length
41  * @param box_length_half Half box length
42  * @param periodic Box periodicity.
43  * @return Shortest distance from @p b to @p a across periodic images,
44  * i.e. <tt>a - b</tt>. Can be negative.
45  */
46 template <typename T>
47 T get_mi_coord(T a, T b, T box_length, T box_length_inv, T box_length_half,
48  bool periodic) {
49  auto const dx = a - b;
50 
51  if (periodic && (std::abs(dx) > box_length_half)) {
52  return dx - std::round(dx * box_length_inv) * box_length;
53  }
54 
55  return dx;
56 }
57 
58 /**
59  * @brief Get the minimum-image distance between two coordinates.
60  * @param a Coordinate of the terminal point.
61  * @param b Coordinate of the initial point.
62  * @param box_length Box length.
63  * @param periodic Box periodicity.
64  * @return Shortest distance from @p b to @p a across periodic images,
65  * i.e. <tt>a - b</tt>. Can be negative.
66  */
67 template <typename T> T get_mi_coord(T a, T b, T box_length, bool periodic) {
68  return get_mi_coord(a, b, box_length, 1. / box_length, 0.5 * box_length,
69  periodic);
70 }

176  /**
177  * @brief Get the minimum-image vector between two coordinates.
178  *
179  * @tparam T Floating point type.
180  *
181  * @param a Coordinate of the terminal point.
182  * @param b Coordinate of the initial point.
183  * @return Vector from @p b to @p a that minimizes the distance across
184  * periodic images, i.e. <tt>a - b</tt>.
185  */
186  template <typename T>
187  Utils::Vector<T, 3> get_mi_vector(const Utils::Vector<T, 3> &a,
188  const Utils::Vector<T, 3> &b) const {
189  if (type() == BoxType::LEES_EDWARDS) {
190  return lees_edwards_bc().distance(a - b, length(), length_half(),
191  length_inv(), m_periodic);
192  }
193  assert(type() == BoxType::CUBOID);
194  return {get_mi_coord(a[0], b[0], 0), get_mi_coord(a[1], b[1], 1),
195  get_mi_coord(a[2], b[2], 2)};
196  }

214  /** Calculate the velocity difference including the Lees-Edwards velocity */
215  Utils::Vector3d velocity_difference(Utils::Vector3d const &x,
216  Utils::Vector3d const &y,
217  Utils::Vector3d const &u,
218  Utils::Vector3d const &v) const {
219  auto ret = u - v;
220  if (type() == BoxType::LEES_EDWARDS) {
221  auto const &le = m_lees_edwards_bc;
222  auto const dy = x[le.shear_plane_normal] - y[le.shear_plane_normal];
223  if (fabs(dy) > 0.5 * length_half()[le.shear_plane_normal]) {
224  ret[le.shear_direction] -= Utils::sgn(dy) * le.shear_velocity;
225  }
226  }
227  return ret;
228  }

231 /** @brief Fold a coordinate to primary simulation box.
232  * @param pos coordinate to fold
233  * @param image_box image box offset
234  * @param length box length
235  */
236 inline std::pair<double, int> fold_coordinate(double pos, int image_box,
237  double const &length) {
238  std::tie(pos, image_box) = Algorithm::periodic_fold(pos, image_box, length);
239 
240  if ((image_box == std::numeric_limits<int>::min()) ||
241  (image_box == std::numeric_limits<int>::max())) {
242  throw std::runtime_error(
243  "Overflow in the image box count while folding a particle coordinate "
244  "into the primary simulation box. Maybe a particle experienced a "
245  "huge force.");
246  }
247 
248  return {pos, image_box};
249 }

251 /** @brief Fold particle coordinates to primary simulation box.
252  * Lees-Edwards offset is ignored.
253  * @param[in,out] pos coordinate to fold
254  * @param[in,out] image_box image box offset
255  * @param[in] box box parameters (side lengths, periodicity)
256  */
257 inline void fold_position(Utils::Vector3d &pos, Utils::Vector3i &image_box,
258  const BoxGeometry &box) {
259  for (int i = 0; i < 3; i++) {
260  if (box.periodic(i)) {
261  std::tie(pos[i], image_box[i]) =
262  fold_coordinate(pos[i], image_box[i], box.length()[i]);
263  }
264  }
265 }

267 /** @brief Fold particle coordinates to primary simulation box.

272 inline Utils::Vector3d folded_position(const Utils::Vector3d &p,
273  const BoxGeometry &box) {
274  Utils::Vector3d p_folded;
275  for (int i = 0; i < 3; i++) {
276  if (box.periodic(i)) {
277  p_folded[i] = Algorithm::periodic_fold(p[i], box.length()[i]);
278  } else {
279  p_folded[i] = p[i];
280  }
281  }
282 
283  return p_folded;
284 }
