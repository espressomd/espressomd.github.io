71template <typename FloatType, Arch Architecture>
72struct DipolarP3MImpl : public DipolarP3M {

74
75 /** @brief Coulomb P3M meshes and FFT algorithm. */

80 template <typename... Args>
81 DipolarP3MImpl(
82 std::unique_ptr<p3m_data_struct_dipoles<FloatType>> &&dp3m_handle,
83 Args &&...args)
84 : DipolarP3M(*dp3m_handle, std::forward<Args>(args)...),
85 dp3m_impl{std::move(dp3m_handle)}, dp3m{*dp3m_impl} {}
86
87 void init() override {
88 if constexpr (Architecture == Arch::CPU) {
89 init_cpu_kernels();
90 }
91 }
92 void tune() override;
93 void count_magnetic_particles() override;
94
95 [[nodiscard]] bool is_gpu() const noexcept override {

99 void on_activation() override {

101 tune();
102 }
103
104 double long_range_energy(ParticleRange const &particles) override {
105 return long_range_kernel(false, true, particles);
106 }
107
108 void add_long_range_forces(ParticleRange const &particles) override {
109 if constexpr (Architecture == Arch::CPU) {
110 long_range_kernel(true, false, particles);

113
114 void dipole_assign(ParticleRange const &particles) override;
115
116protected:
117 /** Compute the k-space part of forces and energies. */
118 double long_range_kernel(bool force_flag, bool energy_flag,
119 ParticleRange const &particles);
120 double calc_average_self_energy_k_space() const override;
121 void calc_energy_correction() override;
122 void calc_influence_function_force() override;
123 void calc_influence_function_energy() override;
124 void init_cpu_kernels();
125 void scaleby_box_l() override {
126 DipolarP3M::scaleby_box_l();
127 dp3m.energy_correction = 0.;
128 }
129};

131template <typename FloatType, Arch Architecture,
132 template <typename> class FFTBackendImpl, class... Args>
133std::shared_ptr<DipolarP3M> new_dp3m_handle(P3MParameters &&p3m,
134 Args &&...args) {
135 auto obj = std::make_shared<DipolarP3MImpl<FloatType, Architecture>>(
136 std::make_unique<p3m_data_struct_dipoles<FloatType>>(std::move(p3m)),
137 std::forward<Args>(args)...);
138 obj->dp3m.template make_fft_instance<FFTBackendImpl<FloatType>>(true);
139 return obj;
140}
