93static auto fields_list_to_bitfield(std::vector<std::string> const &fields) {
94 unsigned int bitfield = H5MD_OUT_NONE;
95 for (auto const &field_name : fields) {
96 if (not fields_map.contains(field_name)) {
97 throw std::invalid_argument("Unknown field '" + field_name + "'");
98 }
99 bitfield |= fields_map.at(field_name);
100 }
101 return bitfield;
102}

104static void backup_file(std::filesystem::path const &from,
105 std::filesystem::path const &to) {
106 /*
107 * If the file itself *and* a backup file exists, something must
108 * have gone wrong.
109 */
110 auto constexpr option_fail_if_exists = std::filesystem::copy_options::none;
111 try {
112 std::filesystem::copy_file(from, to, option_fail_if_exists);
113 } catch (std::filesystem::filesystem_error const &) {
114 throw left_backupfile();
115 }
116}

118template <typename extent_type>
119static void extend_dataset(HighFive::DataSet &dataset,
120 extent_type const &change_extent) {
121 auto extents = dataset.getSpace().getDimensions();
122 auto const rank = extents.size();
123 /* Extend the dataset for another timestep */
124 for (auto i = 0u; i < rank; i++) {
125 extents[i] += change_extent[i];
126 }
127 dataset.resize(extents);
128}

130template <typename value_type, typename extent_type>
131static void write_dataset(value_type const &data, HighFive::DataSet &dataset,
132 extent_type const &offset, extent_type const &count) {
133 auto xfer_props = HighFive::DataTransferProps{};
134 xfer_props.add(HighFive::UseCollectiveIO{});
135 /* write the data to the dataset. */
136 dataset.select(offset, count).write(data, xfer_props);
137}

142 extent_type const &offset, extent_type const &count) {
143 extend_dataset(dataset, change_extent);

160/* Initialize the file-related variables after parameters have been set. */
161void File::init_file() {
162 auto const file_exists = std::filesystem::exists(m_file_path);
163 auto const backup_file_exists = std::filesystem::exists(m_backup_path);
164 /* Perform a barrier synchronization. Otherwise one process might already
165 * create the file while another still checks for its existence. */
166 m_comm.barrier();
167 if (file_exists) {
168 if (m_h5md_specification.is_compliant(m_file_path)) {
169 /*
170 * If the file exists and has a valid H5MD structure, let's create a
171 * backup of it. This has the advantage, that the new file can
172 * just be deleted if the simulation crashes at some point and we
173 * still have a valid trajectory backed up, from which we can restart.
174 */
175 if (m_comm.rank() == 0)
176 backup_file(m_file_path, m_backup_path);
177 load_file();
178 } else {
179 throw incompatible_h5mdfile();
180 }
181 } else {
182 if (backup_file_exists)
183 throw left_backupfile();
184 create_file();
185 }
186}

200 std::stringstream ss(ds.group);

202 std::string current_path = "/";

215 if (rank == 3ul) {

250void File::create_datasets() {
251 auto &datasets = *m_datasets;
252 for (auto const &ds : m_h5md_specification.get_datasets()) {
253 if (ds.is_link)
254 continue;
255 auto dims = create_dims(ds.rank, ds.data_dim);
256 auto maxdims = create_maxdims(ds.rank, ds.data_dim, H5S_UNLIMITED);
257 auto dataspace = HighFive::DataSpace(dims, maxdims);
258 auto const chunk_size = static_cast<hsize_t>(m_chunk_size);
259 auto const chunk = create_chunk_dims(ds.rank, ds.data_dim, chunk_size);
260 HighFive::DataSetCreateProps props;
261 props.add(HighFive::Chunking(chunk));
262 auto path = ds.path();
263 if (ds.type == H5T_NATIVE_INT) {
264 datasets.emplace(path,
265 m_h5md_file->createDataSet<int>(path, dataspace, props));
266 } else if (ds.type == H5T_NATIVE_DOUBLE) {
267 datasets.emplace(
268 path, m_h5md_file->createDataSet<double>(path, dataspace, props));
269 }
270 }

335void File::create_hard_links() {
336 std::string path_step = "/particles/atoms/id/step";
337 std::string path_time = "/particles/atoms/id/time";
338 for (auto &ds : m_h5md_specification.get_datasets()) {
339 if (ds.is_link) {
340 char const *from = nullptr;
341 if (ds.name == "step") {
342 from = path_step.c_str();
343 } else if (ds.name == "time") {
344 from = path_time.c_str();
345 }
346 assert(from != nullptr);
347 if (H5Lcreate_hard(m_h5md_file->getId(), from, m_h5md_file->getId(),
348 ds.path().c_str(), H5P_DEFAULT, H5P_DEFAULT) < 0) {
349 throw std::runtime_error("Error creating hard link for " + ds.path());
350 }
351 }
352 }

377template <std::size_t rank> struct slice_info {};
378
379template <> struct slice_info<3> {
380 static auto extent(hsize_t n_part_diff) {
381 return Vector3s{1ul, n_part_diff, 0ul};
382 }
383 static constexpr auto count(std::size_t local_n_part) {
384 return Vector3s{1ul, local_n_part, 3ul};
385 }
386 static auto offset(hsize_t n_time_steps, hsize_t prefix) {
387 return Vector3s{n_time_steps, prefix, 0ul};
388 }
389 template <typename T>
390 static boost::multi_array<T, 3> reshape(std::vector<T> const &v1d,
391 Vector3s const &count) {
392 if (v1d.empty()) {
393 boost::multi_array<T, 3> data(boost::extents[0][0][0]);
394 return data;
395 }
396 auto const rows = count[1];
397 auto const cols = count[2];
398
399 boost::multi_array<T, 3> data(
400 boost::extents[1][static_cast<long>(rows)][static_cast<long>(cols)]);
401
402 for (std::size_t i = 0; i < rows; ++i) {
403 for (std::size_t j = 0; j < cols; ++j) {
404 data[0][i][j] = v1d[cols * i + j];
405 }
406 }
407
408 return data;
409 }
410};
411
412template <> struct slice_info<2> {
413 static auto extent(hsize_t n_part_diff) { return Vector2s{1ul, n_part_diff}; }
414 static constexpr auto count(std::size_t local_n) {
415 return Vector2s{1ul, local_n};
416 }
417 static auto offset(hsize_t n_time_steps, hsize_t prefix) {
418 return Vector2s{n_time_steps, prefix};
419 }
420 template <typename T>
421 static boost::multi_array<T, 2> reshape(std::vector<T> const &v1d,
422 Vector2s const &count) {
423 if (v1d.empty()) {
424 boost::multi_array<T, 2> data(boost::extents[0][0]);
425 return data;
426 }
427 auto const cols = count[1];
428
429 boost::multi_array<T, 2> data(boost::extents[1][static_cast<long>(cols)]);
430
431 for (std::size_t i = 0; i < cols; ++i) {
432 data[0][i] = v1d[i];
433 }
434
435 return data;
436 }
437};

455template <typename Functor> class ParticleDataSerializer {
456 using RetVal = std::decay_t<std::invoke_result_t<Functor, Particle const &>>;
457 Functor m_getter;
458
459 template <typename T>
460 requires std::is_arithmetic_v<T>
461 void serialize(auto &buffer, T const &value) const {
462 buffer.emplace_back(value);
463 }
464
465 template <typename T, std::size_t N>
466 void serialize(auto &buffer, Utils::Vector<T, N> const &value) const {
467 buffer.insert(buffer.end(), value.cbegin(), value.cend());
468 }
469
470public:
471 explicit ParticleDataSerializer(Functor lambda) : m_getter{lambda} {}
472
473 auto operator()(ParticleRange const &particles) const {
474 auto constexpr value_dim = get_buffer_traits<RetVal>::dim;
475 std::vector<typename get_buffer_traits<RetVal>::type> buffer{};
476 buffer.reserve(particles.size() * value_dim);
477 for (auto const &p : particles) {
478 serialize(buffer, m_getter(p));
479 }
480 return buffer;
481 }
482};
483
484template <typename Functor> auto make_serializer(Functor lambda) {
485 return ParticleDataSerializer<Functor>{lambda};
486}
487template <typename RetVal>
488auto make_serializer(RetVal (Particle::*getter)() const) {
489 auto kernel = [getter](Particle const &p) -> RetVal { return (p.*getter)(); };
490 return ParticleDataSerializer<decltype(kernel)>{std::move(kernel)};
491}

495template <std::size_t dim, typename Serializer>
496void write_td_particle_property(hsize_t prefix, hsize_t n_part_global,
497 ParticleRange const &particles,
498 HighFive::DataSet &dataset,
499 Serializer serializer) {
500 auto const n_part_local = static_cast<hsize_t>(particles.size());
501 auto const old_extents = dataset.getSpace().getDimensions();
502 auto const extent_n_part =
503 std::max(n_part_global, static_cast<hsize_t>(old_extents[1])) -
504 old_extents[1];
505 extend_dataset(dataset, detail::slice_info<dim>::extent(extent_n_part));
506 auto const count = detail::slice_info<dim>::count(n_part_local);
507 auto const offset = detail::slice_info<dim>::offset(old_extents[0], prefix);
508 HighFive::DataType dtype = dataset.getDataType();
509 auto buffer = serializer(particles);
510 write_dataset(detail::slice_info<dim>::reshape(buffer, count), dataset,
511 offset, count);
512}

559void File::write(const ParticleRange &particles, double time, int step,
560 BoxGeometry const &box_geo) {
561 auto &datasets = *m_datasets;
562 if (m_fields & H5MD_OUT_BOX_L) {
563 write_box(box_geo, datasets.at("/particles/atoms/box/edges/value"));
564 }
565 auto const &lebc = box_geo.lees_edwards_bc();
566 if (m_fields & H5MD_OUT_LE_OFF) {
567 write_le_off(lebc,
568 datasets.at("/particles/atoms/lees_edwards/offset/value"));
569 }
570 if (m_fields & H5MD_OUT_LE_DIR) {
571 write_le_dir(lebc,
572 datasets.at("/particles/atoms/lees_edwards/direction/value"));
573 }
574 if (m_fields & H5MD_OUT_LE_NORMAL) {
575 write_le_normal(lebc,
576 datasets.at("/particles/atoms/lees_edwards/normal/value"));
577 }
578
579 // calculate particle count and offset
580 static_assert(sizeof(hsize_t) == 8ul);
581 auto const n_part_local = static_cast<hsize_t>(particles.size());
582 hsize_t prefix{0ul};
583 BOOST_MPI_CHECK_RESULT(
584 MPI_Exscan, (&n_part_local, &prefix, 1, MPI_UINT64_T, MPI_SUM, m_comm));
585 auto const n_part_global =
586 boost::mpi::all_reduce(m_comm, n_part_local, std::plus<hsize_t>());
587
588 write_td_particle_property<2>(prefix, n_part_global, particles,
589 datasets.at("/particles/atoms/id/value"),
590 detail::make_serializer(&Particle::id));
591
592 {
593 HighFive::DataSet &dataset = datasets.at("/particles/atoms/id/value");
594 auto const extents = dataset.getSpace().getDimensions();
595 write_dataset(std::vector<double>{time},
596 datasets.at("/particles/atoms/id/time"), Vector1s{1},
597 Vector1s{extents[0]}, Vector1s{1});
598 write_dataset(std::vector<int>{step},
599 datasets.at("/particles/atoms/id/step"), Vector1s{1},
600 Vector1s{extents[0]}, Vector1s{1});
601 }
602
603 if (m_fields & H5MD_OUT_TYPE) {
604 write_td_particle_property<2>(prefix, n_part_global, particles,
605 datasets.at("/particles/atoms/species/value"),
606 detail::make_serializer(&Particle::type));
607 }
608 if (m_fields & H5MD_OUT_MASS) {
609 write_td_particle_property<2>(prefix, n_part_global, particles,
610 datasets.at("/particles/atoms/mass/value"),
611 detail::make_serializer(&Particle::mass));
612 }
613 if (m_fields & H5MD_OUT_POS) {
614 write_td_particle_property<3>(
615 prefix, n_part_global, particles,
616 datasets.at("/particles/atoms/position/value"),
617 detail::make_serializer([&](Particle const &p) {
618 return box_geo.folded_position(p.pos());
619 }));

650void File::write_connectivity(const ParticleRange &particles) {
651 MultiArray3i bond(boost::extents[0][0][0]);
652 for (auto const &p : particles) {
653 auto nbonds_local = static_cast<decltype(bond)::index>(bond.shape()[1]);
654 for (auto const b : p.bonds()) {
655 auto const &partner_ids = b.partner_ids();
656 if (partner_ids.size() == 1u) {
657 bond.resize(boost::extents[1][nbonds_local + 1][2]);
658 bond[0][nbonds_local][0] = p.id();
659 bond[0][nbonds_local][1] = partner_ids[0];
660 nbonds_local++;
661 }
662 }
663 }
664
665 auto const n_bonds_local = static_cast<int>(bond.shape()[1]);
666 auto &datasets = *m_datasets;
667 int prefix_bonds = 0;
668 BOOST_MPI_CHECK_RESULT(
669 MPI_Exscan, (&n_bonds_local, &prefix_bonds, 1, MPI_INT, MPI_SUM, m_comm));
670 auto const n_bonds_total =
671 boost::mpi::all_reduce(m_comm, n_bonds_local, std::plus<int>());
672 auto const extents =
673 datasets.at("/connectivity/atoms/value").getSpace().getDimensions();
674 Vector3s offset_bonds = {extents[0], static_cast<std::size_t>(prefix_bonds),
675 0};
676 Vector3s count_bonds = {1, static_cast<std::size_t>(n_bonds_local), 2};
677 auto const n_bond_diff = std::max(static_cast<hsize_t>(n_bonds_total),
678 static_cast<hsize_t>(extents[1])) -
679 extents[1];
680 Vector3s change_extent_bonds = {1, static_cast<std::size_t>(n_bond_diff), 0};
681 write_dataset(bond, datasets.at("/connectivity/atoms/value"),
682 change_extent_bonds, offset_bonds, count_bonds);
683}
