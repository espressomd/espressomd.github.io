41 if (name == "version_major") {

43 }
44 if (name == "version_minor") {

55 version_tuple.emplace_back(std::stoi(x));
56 }
57 return version_tuple;

60 return std::string{GIT_BRANCH};

64 }
65 if (name == "git_state") {
