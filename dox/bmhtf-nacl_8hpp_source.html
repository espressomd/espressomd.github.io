44 /** Calculate BMHTF force factor */
45 inline double BMHTF_pair_force_factor(IA_parameters const &ia_params,
46  double dist) {
47  if (dist < ia_params.bmhtf.cut) {
48  auto const dist8 = Utils::int_pow<8>(dist);
49  auto const dist10 = Utils::int_pow<10>(dist);
50  return ia_params.bmhtf.A * ia_params.bmhtf.B *
51  exp(ia_params.bmhtf.B * (ia_params.bmhtf.sig - dist)) / dist -
52  6 * ia_params.bmhtf.C / dist8 - 8 * ia_params.bmhtf.D / dist10;
53  }
54  return 0.0;
55 }
56 
57 /** Calculate BMHTF potential energy */
58 inline double BMHTF_pair_energy(IA_parameters const &ia_params, double dist) {
59  if (dist < ia_params.bmhtf.cut) {
60  auto const dist6 = Utils::int_pow<6>(dist);
61  auto const dist8 = Utils::int_pow<8>(dist);
62  return ia_params.bmhtf.A *
63  exp(ia_params.bmhtf.B * (ia_params.bmhtf.sig - dist)) -
64  ia_params.bmhtf.C / dist6 - ia_params.bmhtf.D / dist8 +
65  ia_params.bmhtf.computed_shift;
66  }
67  return 0.0;
68 }
