95 ESPRESSO_ATTR_ALWAYS_INLINE KOKKOS_INLINE_FUNCTION void
96 operator()(std::size_t i, std::size_t j) const {
97
98 auto const thread_id = omp_get_thread_num();
99
100 auto const &ia_params =
101 nonbonded_ias.get_ia_param(aosoa.type(i), aosoa.type(j));
102
103 ParticleForce pf{};
104 auto const pos1 = aosoa.get_vector_at(aosoa.position, i);
105 auto const pos2 = aosoa.get_vector_at(aosoa.position, j);
106#if defined(ESPRESSO_GAY_BERNE) or defined(ESPRESSO_DIPOLES)
107 auto const dir1 = aosoa.get_vector_at(aosoa.director, i);
108 auto const dir2 = aosoa.get_vector_at(aosoa.director, j);
109#endif
110
111#ifdef ESPRESSO_NPT
112 Utils::Vector3d virial{};
113 auto *const virial_handle = global_virial ? &virial : nullptr;
114#endif
115 auto const d = box_geo.get_mi_vector(pos1, pos2);
116 auto const dist = d.norm();
117
118#if defined(ESPRESSO_EXCLUSIONS) or defined(ESPRESSO_THOLE)
119 auto const &p1 = *unique_particles.at(i);
120 auto const &p2 = *unique_particles.at(j);
121#endif
122
123 /***********************************************/
124 /* non-bonded pair potentials */
125 /***********************************************/
126
127 if (dist < ia_params.max_cut) {
128#ifdef ESPRESSO_EXCLUSIONS
129 if (do_nonbonded(p1, p2)) {
130#endif
131 pf += calc_central_radial_force(ia_params, d, dist);
132#ifdef ESPRESSO_THOLE
133 pf.f += thole_pair_force(p1, p2, ia_params, d, dist, bonded_ias,
134 coulomb_kernel);
135#endif
136#ifdef ESPRESSO_GAY_BERNE
137 pf += gb_pair_force(dir1, dir2, ia_params, d, dist);
138#endif
139#ifdef ESPRESSO_EXCLUSIONS
140 }
141#endif
142 }
143
144 /*********************************************************************/
145 /* everything before this contributes to the virial pressure in NpT, */
146 /* but nothing afterwards, since the contribution to pressure from */
147 /* electrostatic is calculated by energy */
148 /*********************************************************************/
149#ifdef ESPRESSO_NPT
150 if (virial_handle) {
151 *virial_handle += hadamard_product(pf.f, d);
152 }
153#endif // ESPRESSO_NPT
154
155 /***********************************************/
156 /* thermostat */
157 /***********************************************/
158
159 /* The inter dpd force should not be part of the virial */
160#ifdef ESPRESSO_DPD
161 if (thermostat.thermo_switch & THERMO_DPD) {
162 auto const dist2 = dist * dist;
163 auto const vel1 = aosoa.get_vector_at(aosoa.velocity, i);
164 auto const vel2 = aosoa.get_vector_at(aosoa.velocity, j);
165 auto const force =
166 dpd_pair_force(pos1, vel1, aosoa.id(i), pos2, vel2, aosoa.id(j),
167 *thermostat.dpd, box_geo, ia_params, d, dist, dist2);
168 pf += force;
169 }
170#endif // ESPRESSO_DPD
171
172#ifdef ESPRESSO_ELECTROSTATICS
173 auto const q1q2 = aosoa.charge(i) * aosoa.charge(j);
174 ParticleForce p1f_asym{};
175 ParticleForce p2f_asym{};
176 // real-space electrostatic charge-charge interaction
177 if (q1q2 != 0. and coulomb_kernel != nullptr) {
178 pf.f += (*coulomb_kernel)(q1q2, d, dist);
179 if (elc_kernel) {
180 (*elc_kernel)(pos1, pos2, p1f_asym, p2f_asym, q1q2);
181 }
182#ifdef ESPRESSO_NPT
183 if (virial_handle) {
184 (*virial_handle)[0] += (*coulomb_u_kernel)(pos1, pos2, q1q2, d, dist);
185 }
186#endif // ESPRESSO_NPT
187 }
188#endif // ESPRESSO_ELECTROSTATICS
189#ifdef ESPRESSO_DIPOLES
190 // real-space magnetic dipole-dipole interaction
191 if (dipoles_kernel) {
192 auto const d1d2 = aosoa.dipm(i) * aosoa.dipm(j);
193 if (d1d2 != 0.) {
194 pf += (*dipoles_kernel)(d1d2, aosoa.dipm(i) * dir1,
195 aosoa.dipm(j) * dir2, d, dist, dist * dist);
196 }
197 }
198#endif // ESPRESSO_DIPOLES
199
200 auto opf = calc_opposing_force(pf, d);
201#ifdef ESPRESSO_ELECTROSTATICS
202 pf += p1f_asym;
203 opf += p2f_asym;
204#endif // ESPRESSO_ELECTROSTATICS
205
206 local_force(i, thread_id, 0) += pf.f[0];
207 local_force(i, thread_id, 1) += pf.f[1];
208 local_force(i, thread_id, 2) += pf.f[2];
209#ifdef ESPRESSO_ROTATION
210 local_torque(i, thread_id, 0) += pf.torque[0];
211 local_torque(i, thread_id, 1) += pf.torque[1];
212 local_torque(i, thread_id, 2) += pf.torque[2];
213#endif
214
215 local_force(j, thread_id, 0) += opf.f[0];
216 local_force(j, thread_id, 1) += opf.f[1];
217 local_force(j, thread_id, 2) += opf.f[2];
218#ifdef ESPRESSO_ROTATION
219 local_torque(j, thread_id, 0) += opf.torque[0];
220 local_torque(j, thread_id, 1) += opf.torque[1];
221 local_torque(j, thread_id, 2) += opf.torque[2];
222#endif
223#ifdef ESPRESSO_NPT
224 if (virial_handle) {
225 local_virial(thread_id, 0) += virial[0];
226 local_virial(thread_id, 1) += virial[1];
227 local_virial(thread_id, 2) += virial[2];
228 }
229#endif
230 }
231};
