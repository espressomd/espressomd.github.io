40void CoulombP3MGPU::add_long_range_forces(ParticleRange const &particles) {
41#ifdef NPT
42 if (get_system().propagation->integ_switch == INTEG_METHOD_NPT_ISO) {
43 auto const energy = long_range_energy(particles);
44 npt_add_virial_contribution(energy);
45 }
46#else
47 static_cast<void>(particles);
48#endif
49 if (this_node == 0) {
50 auto &gpu = get_system().gpu;
51 p3m_gpu_add_farfield_force(*m_gpu_data, gpu, prefactor, gpu.n_particles());
52 }
53}

55void CoulombP3MGPU::init() {
56 auto &system = get_system();
57 if (has_actor_of_type<ElectrostaticLayerCorrection>(
58 system.coulomb.impl->solver)) {
59 init_cpu_kernels();
60 }
61 p3m_gpu_init(m_gpu_data, p3m.params.cao, p3m.params.mesh, p3m.params.alpha,
62 system.box_geo->length(), system.gpu.n_particles());
63}
