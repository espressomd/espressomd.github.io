97 [[nodiscard]] std::optional<double>
98 get_node_potential(Utils::Vector3i const &node,
99 bool consider_ghosts = false) override {
100 auto bc = get_block_and_cell(get_lattice(), node, consider_ghosts);
101
102 if (!bc || (get_potential_field_id() == 0))
103 return std::nullopt;
104
105 auto const potential_field = bc->block->template getData<PotentialField>(
106 domain_decomposition::BlockDataID(get_potential_field_id()));
107 return {double_c(
108 walberla::ek::accessor::Scalar::get(potential_field, bc->cell))};
109 }

111 [[nodiscard]] std::vector<double>
112 get_slice_potential(Utils::Vector3i const &lower_corner,
113 Utils::Vector3i const &upper_corner) const override {
114 std::vector<double> out;
115 uint_t values_size = 0;
116 auto const &lattice = get_lattice();
117 if (auto const ci = get_interval(lattice, lower_corner, upper_corner)) {
118 out = std::vector<double>(ci->numCells());
119 for (auto &block : *lattice.get_blocks()) {
120 auto const block_offset = lattice.get_block_corner(block, true);
121 if (auto const bci = get_block_interval(
122 lattice, lower_corner, upper_corner, block_offset, block)) {
123 auto const potential_field = block.template getData<PotentialField>(
124 domain_decomposition::BlockDataID(get_potential_field_id()));
125 auto const values = ek::accessor::Scalar::get(potential_field, *bci);
126 assert(values.size() == bci->numCells());
127 values_size += bci->numCells();
128 auto kernel = [&values, &out](unsigned const block_index,
129 unsigned const local_index,
130 Utils::Vector3i const &node) {
131 out[local_index] = double_c(values[block_index]);
132 };
133
134 copy_block_buffer(*bci, *ci, block_offset, lower_corner, kernel);
135 }
136 }
137 assert(values_size == ci->numCells());
138 }
139 return out;
140 }

163 template <typename OutputType = float>
164 class PotentialVTKWriter
165 : public VTKWriter<PotentialFieldCpu, 1u, OutputType> {
166 public:
167 using Base = VTKWriter<PotentialFieldCpu, 1u, OutputType>;
168 using Base::Base;
169 using Base::evaluate;
170
171 protected:
172 OutputType evaluate(cell_idx_t const x, cell_idx_t const y,
173 cell_idx_t const z, cell_idx_t const) override {
174 WALBERLA_ASSERT_NOT_NULLPTR(this->m_field);
175 auto const potential =
176 walberla::ek::accessor::Scalar::get(this->m_field, {x, y, z});
177 return numeric_cast<OutputType>(this->m_conversion * potential);
178 }
179 };

181 void register_vtk_field_writers(walberla::vtk::VTKOutput &vtk_obj,
182 LatticeModel::units_map const &units,
183 int flag_observables) override {
184 auto const allocate_cpu_field_if_empty =
185 [&]<typename Field>(auto const &blocks, std::string name,
186 std::optional<BlockDataID> &cpu_field) {
187 if (not cpu_field) {
188 cpu_field = field::addToStorage<Field>(
189 blocks, name, FloatType{0}, field::fzyx,
190 get_lattice().get_ghost_layers(), m_host_field_allocator);
191 }
192 };
193 if (flag_observables & static_cast<int>(EKPoissonOutputVTK::potential)) {
194 auto const unit_conversion = FloatType_c(units.at("potential"));
195 auto const &blocks = get_lattice().get_blocks();
196 allocate_cpu_field_if_empty.template operator()<PotentialFieldCpu>(
197 blocks, "potential_cpu", m_potential_cpu_field_id);
198 vtk_obj.addBeforeFunction(
199 gpu::fieldCpyFunctor<PotentialFieldCpu, PotentialField>(
200 blocks, *m_potential_cpu_field_id,
201 domain_decomposition::BlockDataID(get_potential_field_id())));
202 vtk_obj.addCellDataWriter(make_shared<PotentialVTKWriter<float>>(
203 *m_potential_cpu_field_id, "potential", unit_conversion));
204 }
205 }
206
207 void integrate_vtk_writers() override {
208 for (auto const &it : m_vtk_auto) {
209 auto &vtk_handle = it.second;
210 if (vtk_handle->enabled) {
211 vtk::writeFiles(vtk_handle->ptr)();
212 vtk_handle->execution_count++;
213 }
214 }
215 }
