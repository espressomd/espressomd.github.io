50/** Return a random uniform 3D vector with the Philox thermostat.
51 * Random numbers depend on
52 * 1. dpd_rng_counter (initialized by seed) which is increased on integration
53 * 2. Salt (decorrelates different counters)
54 * 3. Two particle IDs (order-independent, decorrelates particles, gets rid of
55 * seed-per-node)
56 */
57static Utils::Vector3d dpd_noise(DPDThermostat const &dpd, int pid1, int pid2) {
58 const double pref = (pid1 < pid2) ? 1.0 : -1.0;
59 return pref * Random::noise_uniform<RNGSalt::SALT_DPD>(
60 dpd.rng_counter(), dpd.rng_seed(),
61 (pid1 < pid2) ? pid2 : pid1, (pid1 < pid2) ? pid1 : pid2);
62}

64void dpd_init(double kT, double time_step) {
65 auto &nonbonded_ias = *System::get_system().nonbonded_ias;
66 auto const max_type = nonbonded_ias.get_max_seen_particle_type();
67 for (int type_a = 0; type_a <= max_type; type_a++) {
68 for (int type_b = type_a; type_b <= max_type; type_b++) {
69 auto &ia_params = nonbonded_ias.get_ia_param(type_a, type_b);
70
71 ia_params.dpd.radial.pref =
72 sqrt(24.0 * kT * ia_params.dpd.radial.gamma / time_step);
73 ia_params.dpd.trans.pref =
74 sqrt(24.0 * kT * ia_params.dpd.trans.gamma / time_step);
75 }
76 }
77}

79static double weight(int type, double r_cut, double k, double r) {
80 if (type == 0) {
81 return 1.;
82 }
83 return 1. - pow((r / r_cut), k);
84}

86Utils::Vector3d dpd_pair_force(DPDParameters const &params,
87 Utils::Vector3d const &v, double dist,
88 Utils::Vector3d const &noise) {
89 if (dist < params.cutoff) {
90 auto const omega = weight(params.wf, params.cutoff, params.k, dist);
91 auto const omega2 = Utils::sqr(omega);
92
93 auto const f_d = params.gamma * omega2 * v;
94 auto const f_r = params.pref * omega * noise;
95
96 return f_r - f_d;
97 }
98
99 return {};
100}

102Utils::Vector3d
103dpd_pair_force(Utils::Vector3d const &p1_position,
104 Utils::Vector3d const &p1_velocity, int const &p1_id,
105 Utils::Vector3d const &p2_position,
106 Utils::Vector3d const &p2_velocity, int const &p2_id,
107 DPDThermostat const &dpd, BoxGeometry const &box_geo,
108 IA_parameters const &ia_params, Utils::Vector3d const &d,
109 double dist, double dist2) {
110 if (ia_params.dpd.radial.cutoff <= 0.0 && ia_params.dpd.trans.cutoff <= 0.0) {
111 return {};
112 }
113
114 auto const v21 = box_geo.velocity_difference(p1_position, p2_position,
115 p1_velocity, p2_velocity);
116 auto const noise_vec =
117 (ia_params.dpd.radial.pref > 0.0 || ia_params.dpd.trans.pref > 0.0)
118 ? dpd_noise(dpd, p1_id, p2_id)
119 : Utils::Vector3d{};
120
121 auto const f_r = dpd_pair_force(ia_params.dpd.radial, v21, dist, noise_vec);
122 auto const f_t = dpd_pair_force(ia_params.dpd.trans, v21, dist, noise_vec);
123
124 /* Projection operator to radial direction */
125 auto const P = tensor_product(d / dist2, d);
126 /* This is equivalent to P * f_r + (1 - P) * f_t, but with
127 * doing only one matrix-vector multiplication */
128 auto const force = P * (f_r - f_t) + f_t;
129 return force;
130}

132static auto dpd_viscous_stress_local() {
133 auto &system = System::get_system();
134 auto const &box_geo = *system.box_geo;
135 auto const &nonbonded_ias = *system.nonbonded_ias;
136 auto &cell_structure = *system.cell_structure;
137 system.on_observable_calc();
138
139 Utils::Matrix<double, 3, 3> stress{};
140 cell_structure.non_bonded_loop([&stress, &box_geo, &nonbonded_ias](
141 Particle const &p1, Particle const &p2,
142 Distance const &d) {
143 auto const v21 =
144 box_geo.velocity_difference(p1.pos(), p2.pos(), p1.v(), p2.v());
145
146 auto const &ia_params = nonbonded_ias.get_ia_param(p1.type(), p2.type());
147 auto const dist = std::sqrt(d.dist2);
148
149 auto const f_r = dpd_pair_force(ia_params.dpd.radial, v21, dist, {});
150 auto const f_t = dpd_pair_force(ia_params.dpd.trans, v21, dist, {});
151
152 /* Projection operator to radial direction */
153 auto const P = tensor_product(d.vec21 / d.dist2, d.vec21);
154 /* This is equivalent to P * f_r + (1 - P) * f_t, but with
155 * doing only one matrix-vector multiplication */
156 auto const f = P * (f_r - f_t) + f_t;
157
158 stress += tensor_product(d.vec21, f);
159 });
160
161 return stress;
162}

164Utils::Vector9d dpd_pressure_local() {
165 auto const local_stress = dpd_viscous_stress_local();
166 return -Utils::flatten(local_stress);
167}

169/**
170 * @brief Viscous stress tensor of the DPD interaction.
171 *
172 * This calculates the total viscous stress contribution of the
173 * DPD interaction. It contains only the dissipative contributions
174 * of the interaction without noise. It's calculated as the
175 * sum over all pair virials as
176 * \f[
177 * \sigma^{\nu\mu} = V^{-1}\sum_i \sum_{j < i} r_{i,j}^{\nu} (- \gamma_{i,j}
178 * v_{i,j})^{\mu} \f] where \f$\gamma_{i,j}\f$ is the (in general tensor valued)
179 * DPD friction coefficient for particles i and j, \f$v_{i,j}\f$, \f$r_{i,j}\f$
180 * are their relative velocity and distance and \f$V\f$ is the box volume.
181 *
182 * @return Stress tensor contribution.
183 */
184Utils::Vector9d dpd_stress(boost::mpi::communicator const &comm) {
185 auto const &box_geo = *System::get_system().box_geo;
186 auto const local_stress = dpd_viscous_stress_local();
187 std::remove_const_t<decltype(local_stress)> global_stress{};
188
189 boost::mpi::reduce(comm, local_stress, global_stress, std::plus<>(), 0);
190
191 return Utils::flatten(global_stress) / box_geo.volume();
192}
