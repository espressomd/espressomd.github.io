85static void pack_particles(ParticleRange const &particles,
86 GpuParticleData::GpuParticle *buffer) {
87 auto const &box = *System::get_system().box_geo;
88 std::size_t i = 0u;
89 for (auto const &p : particles) {
90 buffer[i].p = static_cast<Utils::Vector3f>(box.folded_position(p.pos()));
91#ifdef ESPRESSO_DIPOLES
92 buffer[i].dip = static_cast<Utils::Vector3f>(p.calc_dip());
93#endif
94#ifdef ESPRESSO_ELECTROSTATICS
95 buffer[i].q = static_cast<float>(p.q());
96#endif
97 buffer[i].identity = p.id();
98 i++;
99 }
100}

102void GpuParticleData::gather_particle_data(
103 ParticleRange const &particles,
104 pinned_vector<GpuParticle> &particle_data_host, int this_node) {
105 auto const n_part = particles.size();
106
107 if (this_node > 0) {
108 static std::vector<GpuParticle> buffer;
109 buffer.resize(n_part);
110 /* pack local parts into buffer */
111 pack_particles(particles, buffer.data());
112
113 Utils::Mpi::gather_buffer(buffer, comm_cart);
114 } else {
115 particle_data_host.resize(n_part);
116
117 /* Pack own particles */
118 pack_particles(particles, particle_data_host.data());
119
120 Utils::Mpi::gather_buffer(particle_data_host, comm_cart);
121 }
122}

124/**
125 * @brief Add a flat force (and torque) array to a range of particles.
126 *
127 * @param particles The particles the forces (and torques) should be added to
128 * @param forces The forces as flat array of size 3 * particles.size()
129 * @param torques The torques as flat array of size 3 * particles.size(),
130 * this is only touched if ROTATION is active.
131 */
132static void add_forces_and_torques(ParticleRange const &particles,
133 std::span<const float> forces,
134 std::span<const float> torques) {
135 std::size_t i = 0ul;
136 for (auto &p : particles) {
137 for (std::size_t j = 0ul; j < 3ul; j++) {
138 p.force()[j] += static_cast<double>(forces[3ul * i + j]);
139#ifdef ESPRESSO_ROTATION
140 p.torque()[j] += static_cast<double>(torques[3ul * i + j]);
141#endif
142 }
143 i++;
144 }
145}
146#ifdef ESPRESSO_DIPOLE_FIELD_TRACKING
147static void add_dip_fld(ParticleRange const &particles,
148 std::span<const float> dip_fld) {
149 std::size_t i = 0ul;
150 for (auto &p : particles) {
151 for (std::size_t j = 0ul; j < 3ul; j++) {
152 p.dip_fld()[j] += static_cast<double>(dip_fld[3ul * i + j]);
153 }
154 i++;
155 }
156}

159/**
160 * @brief Distribute forces to the worker nodes, and add them to the particles.
161 *
162 * @param particles The particles for which the forces (and torques) should
163 * be added to.
164 * @param host_forces The forces as flat array of size 3 * particles.size(),
165 * only relevant on the head node.
166 * @param host_torques The torques as flat array of size 3 * particles.size(),
167 * this is only touched if ROTATION is active. Only
168 * relevant on the head node.
169 */
170void GpuParticleData::particles_scatter_forces(
171 ParticleRange const &particles, std::span<float> host_forces,
172 std::span<float> host_torques) const {
173
174 auto const size = 3ul * particles.size();
175 auto const n_elements = static_cast<int>(size);
176
177 if (::this_node > 0) {
178 static std::vector<float> buffer_forces;
179 static std::vector<float> buffer_torques;
180
181 buffer_forces.resize(size);
182 Utils::Mpi::scatter_buffer(buffer_forces.data(), n_elements, ::comm_cart);
183#ifdef ESPRESSO_ROTATION
184 buffer_torques.resize(size);
185 Utils::Mpi::scatter_buffer(buffer_torques.data(), n_elements, ::comm_cart);
186#endif
187 add_forces_and_torques(particles, buffer_forces, buffer_torques);
188 } else {
189 Utils::Mpi::scatter_buffer(host_forces.data(), n_elements, ::comm_cart);
190#ifdef ESPRESSO_ROTATION
191 Utils::Mpi::scatter_buffer(host_torques.data(), n_elements, ::comm_cart);
192#endif
193 add_forces_and_torques(particles, host_forces, host_torques);
194 }
195}
