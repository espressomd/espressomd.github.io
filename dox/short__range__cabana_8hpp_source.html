39template <class KokkosRangePolicy = Kokkos::RangePolicy<>>
40ESPRESSO_ATTR_ALWAYS_INLINE inline void
41kokkos_parallel_range_for(auto const &name, auto start, auto end,
42 auto const &kernel) {
43 if (Kokkos::num_threads() > 1) {
44 KokkosRangePolicy policy(start, end);
45 Kokkos::parallel_for(name, policy, kernel);
46 } else {
47 for (auto p_index = start; p_index < end; ++p_index) {
48 kernel(p_index);
49 }
50 }
51}

53ESPRESSO_ATTR_ALWAYS_INLINE inline void
54commit_particle(Particle const &p, auto const index,
55 CellStructure::AoSoA_pack &aosoa) {
56 aosoa.id(index) = p.id();
57#ifdef ESPRESSO_ELECTROSTATICS
58 aosoa.charge(index) = p.q();
59#endif
60 aosoa.type(index) = p.type();
61 aosoa.set_vector_at(aosoa.position, index, p.pos());
62#ifdef ESPRESSO_DPD
63 aosoa.set_vector_at(aosoa.velocity, index, p.v());
64#endif
65#if defined(ESPRESSO_GAY_BERNE) or defined(ESPRESSO_DIPOLES)
66 aosoa.set_vector_at(aosoa.director, index,
67 Utils::convert_quaternion_to_director(p.quat()));
68#endif
69#ifdef ESPRESSO_DIPOLES
70 aosoa.dipm(index) = p.dipm();
71#endif
72}

74ESPRESSO_ATTR_ALWAYS_INLINE inline void
75link_cell_kokkos(std::span<Cell *const> cells, BoxGeometry const &box_geo,
76 auto const &verlet_criterion,
77 Kokkos::View<int *> const &id_to_index, int const max_id,
78 auto const &intra_operator, auto const &inter_operator) {
79
80 auto const distance_function = detail::MinimalImageDistance{box_geo};
81
82 // implementation detail: max_id refers to the max local particle id,
83 // but ghost particles from other ranks may have larger particle ids;
84 // -1 is used as a sentinel value for particle ids from other threads
85
86 auto intra_kernel = [&cells, &distance_function, &verlet_criterion,
87 &id_to_index, &intra_operator, max_id](const int i) {
88 auto &local_particles = cells[i]->particles();
89 for (auto it = local_particles.begin(); it != local_particles.end(); ++it) {
90 auto const &p1 = *it;
91 if (p1.id() <= max_id) {
92 auto const ii = id_to_index(p1.id());
93 if (ii >= 0) {
94 // pairs in this cell
95 for (auto jt = std::next(it); jt != local_particles.end(); ++jt) {
96 if ((*jt).id() <= max_id) {
97 if (verlet_criterion(p1, *jt, distance_function(p1, *jt))) {
98 auto const jj = id_to_index((*jt).id());
99 if (jj >= 0) {
100 intra_operator(ii, jj);
101 }
102 }
103 }
104 }
105 }
106 }
107 }
108 };
109
110 auto inter_kernel = [&cells, &distance_function, &verlet_criterion,
111 &id_to_index, &inter_operator, max_id](const int i) {
112 auto &local_particles = cells[i]->particles();
113 for (auto it = local_particles.begin(); it != local_particles.end(); ++it) {
114 auto const &p1 = *it;
115 if (p1.id() <= max_id) {
116 auto const ii = id_to_index(p1.id());
117 if (ii >= 0) {
118 // pairs with neighboring cells
119 for (auto &neighbor : cells[i]->neighbors().red()) {
120 for (auto const &p2 : neighbor->particles()) {
121 if (p2.id() <= max_id) {
122 if (verlet_criterion(p1, p2, distance_function(p1, p2))) {
123 auto const jj = id_to_index(p2.id());
124 if (jj >= 0) {
125 inter_operator(ii, jj);
126 }
127 }
128 }
129 }
130 }
131 }
132 }
133 }
134 };
135
136 Kokkos::parallel_for("inter", cells.size(), intra_kernel);
137 Kokkos::fence();
138
139 Kokkos::parallel_for("intra", cells.size(), inter_kernel);
140 Kokkos::fence();
141}

143ESPRESSO_ATTR_ALWAYS_INLINE inline void
144update_cabana_state(CellStructure &cell_structure, auto const &verlet_criterion,
145 double const pair_cutoff, auto const integ_switch) {
146 using execution_space = Kokkos::DefaultExecutionSpace;
147 using policy_type = Kokkos::RangePolicy<execution_space>;
148 auto const rebuild = cell_structure.prepare_verlet_list_cabana(pair_cutoff);
149 auto const &unique_particles = cell_structure.get_unique_particles();
150 auto const n_part = unique_particles.size();
151 auto const max_id = cell_structure.get_cached_max_local_particle_id();
152 auto &aosoa = cell_structure.get_aosoa();
153
154 if (rebuild) {
155 auto &id_to_index = cell_structure.get_id_to_index();
156
157 // ===================================================
158 // Fill particle storage
159 // ===================================================
160 kokkos_parallel_range_for<policy_type>(
161 "AoSoA write", std::size_t{0}, n_part,
162 [&unique_particles, &aosoa, &id_to_index](int const index) {
163 auto const &p = *unique_particles.at(index);
164 commit_particle(p, index, aosoa);
165 id_to_index(p.id()) = index;
166 });
167 Kokkos::fence();
168
169 // ===================================================
170 // Get Verlet pairs and fill Verlet list
171 // ===================================================
172 bool rebuild_vl = (integ_switch != INTEG_METHOD_STEEPEST_DESCENT and
173 cell_structure.use_verlet_list);
174 cell_structure.rebuild_verlet_list_cabana(
175 [&](std::span<Cell *const> cells, BoxGeometry const &box,
176 CellStructure::ListType &verlet_list) {
177 link_cell_kokkos(
178 std::move(cells), box, verlet_criterion, id_to_index, max_id,
179 [&](const int i, const int j) {
180 // intra cell loop
181 verlet_list.addNeighborLB(i, j);
182 },
183 [&](const int i, const int j) {
184 // inter cell loop
185 verlet_list.addNeighbor(i, j);
186 });
187 if (verlet_list.hasOverflow()) {
188 cell_structure.use_verlet_list = false;
189 runtimeWarningMsg()
190 << "Verlet list overflow detected: neighbor count exceeded "
191 "max_counts. Falling back to the link cell algorithm.";
192 }
193 },
194 rebuild_vl);
195 } else {
196 // ===================================================
197 // Fill particle storage
198 // ===================================================
199 kokkos_parallel_range_for<policy_type>(
200 "AoSoA write", std::size_t{0}, n_part,
201 [&unique_particles, &aosoa](int const index) {
202 auto const &p = *unique_particles.at(index);
203 commit_particle(p, index, aosoa);
204 });
205 Kokkos::fence();
206 }
207}

226void cabana_short_range(auto const &bond_kernel, auto const &forces_kernel,
227 CellStructure &cell_structure, double pair_cutoff,
228 double bond_cutoff, auto const &verlet_criterion,
229 auto const integ_switch) {
230 using execution_space = Kokkos::DefaultExecutionSpace;
231 assert(cell_structure.get_resort_particles() == Cells::RESORT_NONE);
232
233 if (bond_cutoff >= 0.) {
234 cell_structure.bond_loop(bond_kernel);
235 }
236
237 // Cabana short range loop
238 if (pair_cutoff > 0.) {
239 if (integ_switch != INTEG_METHOD_STEEPEST_DESCENT and
240 cell_structure.use_verlet_list) {
241 auto const &verlet_list = cell_structure.get_verlet_list_cabana();
242 Kokkos::RangePolicy<execution_space> policy(
243 std::size_t{0}, cell_structure.get_unique_particles().size());
244 Cabana::neighbor_parallel_for(policy, forces_kernel, verlet_list,
245 Cabana::FirstNeighborsTag(),
246 Cabana::SerialOpTag());
247 } else {
248 cell_structure.cell_list_loop(
249 [&](std::span<Cell *const> cells, BoxGeometry const &box) {
250 link_cell_kokkos(
251 std::move(cells), box, verlet_criterion,
252 cell_structure.get_id_to_index(),
253 cell_structure.get_cached_max_local_particle_id(),
254 [&](const int i, const int j) {
255 // intra cell loop
256 forces_kernel(i, j);
257 },
258 [&](const int i, const int j) {
259 // inter cell loop
260 forces_kernel(i, j);
261 });
262 });
263 }
264 Kokkos::fence();
265 }
266}
