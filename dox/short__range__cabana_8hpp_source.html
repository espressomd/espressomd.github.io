39ESPRESSO_ATTR_ALWAYS_INLINE inline void
40commit_particle(Particle const &p, int const index,
41 CellStructure::AoSoA_pack &aosoa) {
42 aosoa.id(index) = p.id();
43#ifdef ELECTROSTATICS
44 aosoa.charge(index) = p.q();
45#endif
46 aosoa.type(index) = p.type();
47 auto const &pos = p.pos();
48 aosoa.position(index, 0) = pos[0];
49 aosoa.position(index, 1) = pos[1];
50 aosoa.position(index, 2) = pos[2];
51}

53ESPRESSO_ATTR_ALWAYS_INLINE inline void construct_verlet_list(
54 std::span<Cell *const> cells, BoxGeometry const &box_geo,
55 CellStructure::ListType &verlet_list, auto const &verlet_criterion,
56 Kokkos::View<int *> const &id_to_index, int const max_id) {
57
58 auto const distance_function = detail::MinimalImageDistance{box_geo};
59
60 // implementation detail: max_id refers to the max local particle id,
61 // but ghost particles from other ranks may have larger particle ids;
62 // -1 is used as a sentinel value for particle ids from other threads
63
64 auto intra_kernel = [&cells, &distance_function, &verlet_criterion,
65 &id_to_index, &verlet_list, max_id](const int i) {
66 auto &local_particles = cells[i]->particles();
67 for (auto it = local_particles.begin(); it != local_particles.end(); ++it) {
68 auto const &p1 = *it;
69 if (p1.id() <= max_id) {
70 auto const ii = id_to_index(p1.id());
71 if (ii >= 0) {
72 // pairs in this cell
73 for (auto jt = std::next(it); jt != local_particles.end(); ++jt) {
74 if ((*jt).id() <= max_id) {
75 if (verlet_criterion(p1, *jt, distance_function(p1, *jt))) {
76 auto const jj = id_to_index((*jt).id());
77 if (jj >= 0) {
78 verlet_list.addNeighborLB(ii, jj);
79 }
80 }
81 }
82 }
83 }
84 }
85 }
86 };
87
88 auto inter_kernel = [&cells, &distance_function, &verlet_criterion,
89 &id_to_index, &verlet_list, max_id](const int i) {
90 auto &local_particles = cells[i]->particles();
91 for (auto it = local_particles.begin(); it != local_particles.end(); ++it) {
92 auto const &p1 = *it;
93 if (p1.id() <= max_id) {
94 auto const ii = id_to_index(p1.id());
95 if (ii >= 0) {
96 // pairs with neighboring cells
97 for (auto &neighbor : cells[i]->neighbors().red()) {
98 for (auto const &p2 : neighbor->particles()) {
99 if (p2.id() <= max_id) {
100 if (verlet_criterion(p1, p2, distance_function(p1, p2))) {
101 auto const jj = id_to_index(p2.id());
102 if (jj >= 0) {
103 verlet_list.addNeighbor(ii, jj);
104 }
105 }
106 }
107 }
108 }
109 }
110 }
111 }
112 };
113
114 Kokkos::parallel_for("inter", cells.size(), intra_kernel);
115 Kokkos::fence();
116
117 Kokkos::parallel_for("intra", cells.size(), inter_kernel);
118 Kokkos::fence();
119}

121ESPRESSO_ATTR_ALWAYS_INLINE inline void
122update_cabana_state(CellStructure &cell_structure, auto const &verlet_criterion,
123 double const pair_cutoff) {
124 using execution_space = Kokkos::DefaultExecutionSpace;
125 auto const num_threads = execution_space().concurrency();
126 auto const rebuild =
127 cell_structure.prepare_verlet_list_cabana(num_threads, pair_cutoff);
128 auto const &unique_particles = cell_structure.get_unique_particles();
129 auto const max_id = cell_structure.get_cached_max_local_particle_id();
130 auto &aosoa = cell_structure.get_aosoa();
131
132 // ===================================================
133 // Fill particle storage
134 // ===================================================
135 Kokkos::View<int *> id_to_index(
136 Kokkos::ViewAllocateWithoutInitializing("id_to_index"), max_id + 1);
137 Kokkos::deep_copy(id_to_index, -1);
138
139 using policy_type = Kokkos::RangePolicy<execution_space>;
140 Kokkos::parallel_for(
141 "AoSoA write", policy_type(0, unique_particles.size()),
142 [&unique_particles, &aosoa, &id_to_index](int const index) {
143 auto const &p = *unique_particles.at(index);
144 commit_particle(p, index, aosoa);
145 id_to_index(p.id()) = index;
146 });
147 Kokkos::fence();
148
149 // ===================================================
150 // Get Verlet pairs and fill Verlet list
151 // ===================================================
152 if (rebuild) {
153 cell_structure.rebuild_verlet_list_cabana(
154 [&](std::span<Cell *const> cells, BoxGeometry const &box,
155 CellStructure::ListType &verlet_list) {
156 construct_verlet_list(std::move(cells), box, verlet_list,
157 verlet_criterion, id_to_index, max_id);
158 });
159 }
160}

162void cabana_short_range(auto const &bond_kernel, auto const &forces_kernel,
163 CellStructure &cell_structure, double pair_cutoff,
164 double bond_cutoff) {
165 using execution_space = Kokkos::DefaultExecutionSpace;
166 assert(cell_structure.get_resort_particles() == Cells::RESORT_NONE);
167
168 if (bond_cutoff >= 0.) {
169 cell_structure.bond_loop(bond_kernel);
170 }
171
172 // Cabana short range loop
173 if (pair_cutoff > 0.) {
174 auto const &verlet_list = cell_structure.get_verlet_list_cabana();
175 Kokkos::RangePolicy<execution_space> policy(
176 0, cell_structure.get_unique_particles().size());
177 Cabana::neighbor_parallel_for(policy, forces_kernel, verlet_list,
178 Cabana::FirstNeighborsTag(),
179 Cabana::SerialOpTag());
180 Kokkos::fence();
181 }
182}
