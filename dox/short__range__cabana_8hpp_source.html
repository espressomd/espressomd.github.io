39template <class KokkosRangePolicy = Kokkos::RangePolicy<>>
40ESPRESSO_ATTR_ALWAYS_INLINE inline void
41kokkos_parallel_range_for(auto const &name, auto start, auto end,
42 auto const &kernel) {
43 if (Kokkos::num_threads() > 1) {
44 KokkosRangePolicy policy(start, end);
45 Kokkos::parallel_for(name, policy, kernel);
46 } else {
47 for (auto p_index = start; p_index < end; ++p_index) {
48 kernel(p_index);
49 }
50 }
51}

53ESPRESSO_ATTR_ALWAYS_INLINE inline void
54commit_particle(Particle const &p, auto const index,
55 CellStructure::AoSoA_pack &aosoa) {
56 aosoa.id(index) = p.id();
57#ifdef ESPRESSO_ELECTROSTATICS
58 aosoa.charge(index) = p.q();
59#endif
60 aosoa.type(index) = p.type();
61 auto const &pos = p.pos();
62 aosoa.position(index, 0) = pos[0];
63 aosoa.position(index, 1) = pos[1];
64 aosoa.position(index, 2) = pos[2];
65}

67ESPRESSO_ATTR_ALWAYS_INLINE inline void
68link_cell_kokkos(std::span<Cell *const> cells, BoxGeometry const &box_geo,
69 auto const &verlet_criterion,
70 Kokkos::View<int *> const &id_to_index, int const max_id,
71 auto const &intra_operator, auto const &inter_operator) {
72
73 auto const distance_function = detail::MinimalImageDistance{box_geo};
74
75 // implementation detail: max_id refers to the max local particle id,
76 // but ghost particles from other ranks may have larger particle ids;
77 // -1 is used as a sentinel value for particle ids from other threads
78
79 auto intra_kernel = [&cells, &distance_function, &verlet_criterion,
80 &id_to_index, &intra_operator, max_id](const int i) {
81 auto &local_particles = cells[i]->particles();
82 for (auto it = local_particles.begin(); it != local_particles.end(); ++it) {
83 auto const &p1 = *it;
84 if (p1.id() <= max_id) {
85 auto const ii = id_to_index(p1.id());
86 if (ii >= 0) {
87 // pairs in this cell
88 for (auto jt = std::next(it); jt != local_particles.end(); ++jt) {
89 if ((*jt).id() <= max_id) {
90 if (verlet_criterion(p1, *jt, distance_function(p1, *jt))) {
91 auto const jj = id_to_index((*jt).id());
92 if (jj >= 0) {
93 intra_operator(ii, jj);
94 }
95 }
96 }
97 }
98 }
99 }
100 }
101 };
102
103 auto inter_kernel = [&cells, &distance_function, &verlet_criterion,
104 &id_to_index, &inter_operator, max_id](const int i) {
105 auto &local_particles = cells[i]->particles();
106 for (auto it = local_particles.begin(); it != local_particles.end(); ++it) {
107 auto const &p1 = *it;
108 if (p1.id() <= max_id) {
109 auto const ii = id_to_index(p1.id());
110 if (ii >= 0) {
111 // pairs with neighboring cells
112 for (auto &neighbor : cells[i]->neighbors().red()) {
113 for (auto const &p2 : neighbor->particles()) {
114 if (p2.id() <= max_id) {
115 if (verlet_criterion(p1, p2, distance_function(p1, p2))) {
116 auto const jj = id_to_index(p2.id());
117 if (jj >= 0) {
118 inter_operator(ii, jj);
119 }
120 }
121 }
122 }
123 }
124 }
125 }
126 }
127 };
128
129 Kokkos::parallel_for("inter", cells.size(), intra_kernel);
130 Kokkos::fence();
131
132 Kokkos::parallel_for("intra", cells.size(), inter_kernel);
133 Kokkos::fence();
134}

136ESPRESSO_ATTR_ALWAYS_INLINE inline void
137update_cabana_state(CellStructure &cell_structure, auto const &verlet_criterion,
138 double const pair_cutoff, auto const integ_switch) {
139 using execution_space = Kokkos::DefaultExecutionSpace;
140 using policy_type = Kokkos::RangePolicy<execution_space>;
141 auto const rebuild = cell_structure.prepare_verlet_list_cabana(pair_cutoff);
142 auto const &unique_particles = cell_structure.get_unique_particles();
143 auto const n_part = unique_particles.size();
144 auto const max_id = cell_structure.get_cached_max_local_particle_id();
145 auto &aosoa = cell_structure.get_aosoa();
146
147 if (rebuild) {
148 auto &id_to_index = cell_structure.get_id_to_index();
149
150 // ===================================================
151 // Fill particle storage
152 // ===================================================
153 kokkos_parallel_range_for<policy_type>(
154 "AoSoA write", std::size_t{0}, n_part,
155 [&unique_particles, &aosoa, &id_to_index](int const index) {
156 auto const &p = *unique_particles.at(index);
157 commit_particle(p, index, aosoa);
158 id_to_index(p.id()) = index;
159 });
160 Kokkos::fence();
161
162 if (integ_switch != INTEG_METHOD_STEEPEST_DESCENT) {
163 // ===================================================
164 // Get Verlet pairs and fill Verlet list
165 // ===================================================
166 cell_structure.rebuild_verlet_list_cabana(
167 [&](std::span<Cell *const> cells, BoxGeometry const &box,
168 CellStructure::ListType &verlet_list) {
169 link_cell_kokkos(
170 std::move(cells), box, verlet_criterion, id_to_index, max_id,
171 [&](const int i, const int j) {
172 // intra cell loop
173 verlet_list.addNeighborLB(i, j);
174 },
175 [&](const int i, const int j) {
176 // inter cell loop
177 verlet_list.addNeighbor(i, j);
178 });
179 });
180 }
181 } else {
182 // ===================================================
183 // Fill particle storage
184 // ===================================================
185 kokkos_parallel_range_for<policy_type>(
186 "AoSoA write", std::size_t{0}, n_part,
187 [&unique_particles, &aosoa](int const index) {
188 auto const &p = *unique_particles.at(index);
189 commit_particle(p, index, aosoa);
190 });
191 Kokkos::fence();
192 }
193}

212void cabana_short_range(auto const &bond_kernel, auto const &forces_kernel,
213 CellStructure &cell_structure, double pair_cutoff,
214 double bond_cutoff, auto const &verlet_criterion,
215 auto const integ_switch) {
216 using execution_space = Kokkos::DefaultExecutionSpace;
217 assert(cell_structure.get_resort_particles() == Cells::RESORT_NONE);
218
219 if (bond_cutoff >= 0.) {
220 cell_structure.bond_loop(bond_kernel);
221 }
222
223 // Cabana short range loop
224 if (pair_cutoff > 0.) {
225 if (integ_switch != INTEG_METHOD_STEEPEST_DESCENT and
226 cell_structure.use_verlet_list) {
227 auto const &verlet_list = cell_structure.get_verlet_list_cabana();
228 Kokkos::RangePolicy<execution_space> policy(
229 std::size_t{0}, cell_structure.get_unique_particles().size());
230 Cabana::neighbor_parallel_for(policy, forces_kernel, verlet_list,
231 Cabana::FirstNeighborsTag(),
232 Cabana::SerialOpTag());
233 } else {
234 cell_structure.cell_list_loop(
235 [&](std::span<Cell *const> cells, BoxGeometry const &box) {
236 link_cell_kokkos(
237 std::move(cells), box, verlet_criterion,
238 cell_structure.get_id_to_index(),
239 cell_structure.get_cached_max_local_particle_id(),
240 [&](const int i, const int j) {
241 // intra cell loop
242 forces_kernel(i, j);
243 },
244 [&](const int i, const int j) {
245 // inter cell loop
246 forces_kernel(i, j);
247 });
248 });
249 }
250 Kokkos::fence();
251 }
252}
