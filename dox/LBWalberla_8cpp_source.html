80 void LBWalberla::veto_kT(double kT) const {
81  auto const energy_conversion =
82  Utils::int_pow<2>(lb_params->get_agrid() / lb_params->get_tau());
83  auto const lb_kT = lb_fluid->get_kT() * energy_conversion;
84  if (not ::Thermostat::are_kT_equal(lb_kT, kT)) {
85  throw std::runtime_error("Temperature change not supported by LB");

89 void LBWalberla::sanity_checks(System::System const &system) const {
90  auto const agrid = lb_params->get_agrid();
91  auto [lb_left, lb_right] = lb_fluid->get_lattice().get_local_domain();
92  lb_left *= agrid;
93  lb_right *= agrid;
94  auto const &md_left = system.local_geo->my_left();
95  auto const &md_right = system.local_geo->my_right();
96  walberla_agrid_sanity_checks("LB", md_left, md_right, lb_left, lb_right,
97  agrid);
98  // LB time step and MD time step must agree
99  walberla_tau_sanity_checks("LB", lb_params->get_tau(),
100  system.get_time_step());
101 }
