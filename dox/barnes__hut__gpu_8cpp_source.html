59 int DipolarBarnesHutGpu::initialize_data_structure() {
60  auto &system = EspressoSystemInterface::Instance();
61  auto const n_part = static_cast<int>(system.npart_gpu());
62  auto const error_code = call_kernel(allocBHmemCopy, n_part, &m_bh_data);
63 
64  if (error_code == ES_OK) {
65  fill_bh_data(system.rGpuBegin(), system.dipGpuBegin(), &m_bh_data);
66  initBHgpu(m_bh_data.blocks);
67  buildBoxBH(m_bh_data.blocks);
68  buildTreeBH(m_bh_data.blocks);
69  summarizeBH(m_bh_data.blocks);
70  sortBH(m_bh_data.blocks);
71  }
72 
73  return error_code;
74 }
