40 /** Calculate hat force factor */
41 inline double hat_pair_force_factor(IA_parameters const &ia_params,
42  double dist) {
43  if (dist > 0. && dist < ia_params.hat.r) {
44  return ia_params.hat.Fmax * (1.0 - dist / ia_params.hat.r) / dist;
45  }
46  return 0.0;
47 }
48 
49 /** Calculate hat energy */
50 inline double hat_pair_energy(IA_parameters const &ia_params, double dist) {
51  if (dist < ia_params.hat.r) {
52  auto const r = ia_params.hat.r;
53  return ia_params.hat.Fmax * (dist - r) * ((dist + r) / (2.0 * r) - 1.0);
54  }
55  return 0.0;
56 }
