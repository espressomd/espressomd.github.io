22/** @file
23 * P3M algorithm for long-range Coulomb interaction.
24 *
25 * We use a P3M (Particle-Particle Particle-Mesh) method based on the
26 * Ewald summation. Details of the used method can be found in
27 * @cite hockney88a and @cite deserno98a @cite deserno98b.
28 *
29 * Further reading: @cite ewald21a, @cite hockney88a, @cite deserno98a,
30 * @cite deserno98b, @cite deserno00e, @cite deserno00b, @cite cerda08d.
31 *
32 * Implementation in p3m.cpp.
33 */

81 template <typename... Args>
82 CoulombP3MImpl(
83 std::unique_ptr<p3m_data_struct_coulomb<FloatType>> &&p3m_handle,
84 Args &&...args)
85 : CoulombP3M(*p3m_handle, std::forward<Args>(args)...),
86 p3m_impl{std::move(p3m_handle)}, p3m{*p3m_impl} {}
87
88 void init() override {
89 if constexpr (Architecture == Arch::CPU) {
90 init_cpu_kernels();
91 }
92#ifdef CUDA
93 if constexpr (Architecture == Arch::GPU) {
94 init_gpu_kernels();
95 }
96#endif
97 }
98 void tune() override;
99 void count_charged_particles() override;
100 void count_charged_particles_elc(int n, double sum_q2,
101 double square_sum_q) override {
102 p3m.sum_qpart = n;
103 p3m.sum_q2 = sum_q2;
104 p3m.square_sum_q = square_sum_q;
105 }
106
107 [[nodiscard]] bool is_gpu() const noexcept override {
108 return Architecture == Arch::GPU;
109 }
110
111 void on_activation() override {
112#ifdef CUDA
113 if constexpr (Architecture == Arch::GPU) {
114 request_gpu();
115 }
116#endif
117 sanity_checks();
118 tune();
119#ifdef CUDA
120 if constexpr (Architecture == Arch::GPU) {
121 if (is_tuned()) {
122 init_cpu_kernels();
123 }
124 }
125#endif
126 }
127
128 double long_range_energy(ParticleRange const &particles) override {
129 return long_range_kernel(false, true, particles);
130 }
131
132 void add_long_range_forces(ParticleRange const &particles) override {
133 if constexpr (Architecture == Arch::CPU) {
134 long_range_kernel(true, false, particles);
135 }
136#ifdef CUDA
137 if constexpr (Architecture == Arch::GPU) {
138 add_long_range_forces_gpu(particles);
139 }
140#endif
141 }
142
143 Utils::Vector9d long_range_pressure(ParticleRange const &particles) override;
144
145 void charge_assign(ParticleRange const &particles) override;
146 void assign_charge(double q, Utils::Vector3d const &real_pos,
147 bool skip_cache) override;
148 void prepare_fft_mesh(bool reset_weights) override {
149 if (reset_weights) {
150 p3m.inter_weights.reset(p3m.params.cao);
151 }
152 for (int i = 0; i < p3m.local_mesh.size; i++) {
153 p3m.mesh.rs_scalar[i] = FloatType(0);
154 }
155 }
156
157protected:
158 /** Compute the k-space part of forces and energies. */
159 double long_range_kernel(bool force_flag, bool energy_flag,
160 ParticleRange const &particles);
161 void calc_influence_function_force() override;
162 void calc_influence_function_energy() override;
163 void init_cpu_kernels();
164#ifdef CUDA
165 void init_gpu_kernels();
166 void add_long_range_forces_gpu(ParticleRange const &particles);
167 std::shared_ptr<P3MGpuParams> m_gpu_data = nullptr;
168 void request_gpu() const;
169#endif
170};

175 Args &&...args) {
176 auto obj = std::make_shared<CoulombP3MImpl<FloatType, Architecture>>(
177 std::make_unique<p3m_data_struct_coulomb<FloatType>>(std::move(p3m)),
