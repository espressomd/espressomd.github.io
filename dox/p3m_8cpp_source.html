125 /** Aliasing sum used by @ref p3m_k_space_error. */
126 static void p3m_tune_aliasing_sums(int nx, int ny, int nz,
127  Utils::Vector3i const &mesh,
128  Utils::Vector3d const &mesh_i, int cao,
129  double alpha_L_i, double *alias1,
130  double *alias2) {
131  using Utils::sinc;
132 
133  auto const factor1 = Utils::sqr(Utils::pi() * alpha_L_i);
134 
135  *alias1 = *alias2 = 0.0;
136  for (int mx = -P3M_BRILLOUIN; mx <= P3M_BRILLOUIN; mx++) {
137  auto const nmx = nx + mx * mesh[0];
138  auto const fnmx = mesh_i[0] * nmx;
139  for (int my = -P3M_BRILLOUIN; my <= P3M_BRILLOUIN; my++) {
140  auto const nmy = ny + my * mesh[1];
141  auto const fnmy = mesh_i[1] * nmy;
142  for (int mz = -P3M_BRILLOUIN; mz <= P3M_BRILLOUIN; mz++) {
143  auto const nmz = nz + mz * mesh[2];
144  auto const fnmz = mesh_i[2] * nmz;
145 
146  auto const nm2 = Utils::sqr(nmx) + Utils::sqr(nmy) + Utils::sqr(nmz);
147  auto const ex = exp(-factor1 * nm2);
148  auto const ex2 = Utils::sqr(ex);
149 
150  auto const U2 = pow(sinc(fnmx) * sinc(fnmy) * sinc(fnmz), 2.0 * cao);
151 
152  *alias1 += ex2 / nm2;
153  *alias2 += U2 * ex * (nx * nmx + ny * nmy + nz * nmz) / nm2;
154  }
155  }
156  }
157 }

159 /** Calculate the real space contribution to the rms error in the force (as
160  * described by Kolafa and Perram).
161  * \param pref Prefactor of Coulomb interaction.
162  * \param r_cut_iL rescaled real space cutoff for p3m method.
163  * \param n_c_part number of charged particles in the system.
164  * \param sum_q2 sum of square of charges in the system
165  * \param alpha_L rescaled Ewald splitting parameter.
166  * \return real space error
167  */
168 static double p3m_real_space_error(double pref, double r_cut_iL, int n_c_part,
169  double sum_q2, double alpha_L) {
170  return (2. * pref * sum_q2 * exp(-Utils::sqr(r_cut_iL * alpha_L))) /
171  sqrt(static_cast<double>(n_c_part) * r_cut_iL * box_geo.length()[0] *
172  box_geo.volume());
173 }

175 /** Calculate the analytic expression of the error estimate for the
176  * P3M method in @cite hockney88a (eq. 8-23 p. 275) in
177  * order to obtain the rms error in the force for a system of N
178  * randomly distributed particles in a cubic box (k-space part).
179  * \param pref Prefactor of Coulomb interaction.
180  * \param mesh number of mesh points in one direction.
181  * \param cao charge assignment order.
182  * \param n_c_part number of charged particles in the system.
183  * \param sum_q2 sum of square of charges in the system
184  * \param alpha_L rescaled Ewald splitting parameter.
185  * \return reciprocal (k) space error
186  */
187 static double p3m_k_space_error(double pref, Utils::Vector3i const &mesh,
188  int cao, int n_c_part, double sum_q2,
189  double alpha_L) {
190  auto const mesh_i =
191  Utils::hadamard_division(Utils::Vector3d::broadcast(1.), mesh);
192  auto const alpha_L_i = 1. / alpha_L;
193  auto he_q = 0.;
194 
195  for (int nx = -mesh[0] / 2; nx < mesh[0] / 2; nx++) {
196  auto const ctan_x = p3m_analytic_cotangent_sum(nx, mesh_i[0], cao);
197  for (int ny = -mesh[1] / 2; ny < mesh[1] / 2; ny++) {
198  auto const ctan_y =
199  ctan_x * p3m_analytic_cotangent_sum(ny, mesh_i[1], cao);
200  for (int nz = -mesh[2] / 2; nz < mesh[2] / 2; nz++) {
201  if ((nx != 0) || (ny != 0) || (nz != 0)) {
202  auto const n2 = Utils::sqr(nx) + Utils::sqr(ny) + Utils::sqr(nz);
203  auto const cs =
204  p3m_analytic_cotangent_sum(nz, mesh_i[2], cao) * ctan_y;
205  double alias1, alias2;
206  p3m_tune_aliasing_sums(nx, ny, nz, mesh, mesh_i, cao, alpha_L_i,
207  &alias1, &alias2);
208 
209  auto const d = alias1 - Utils::sqr(alias2 / cs) / n2;
210  /* at high precision, d can become negative due to extinction;
211  also, don't take values that have no significant digits left*/
212  if (d > 0 && (fabs(d / alias1) > ROUND_ERROR_PREC))
213  he_q += d;
214  }
215  }
216  }
217  }
218  return 2. * pref * sum_q2 * sqrt(he_q / static_cast<double>(n_c_part)) /
219  (box_geo.length()[1] * box_geo.length()[2]);
220 }

236 void CoulombP3M::init() {
237  assert(p3m.params.mesh >= Utils::Vector3i::broadcast(1));
238  assert(p3m.params.cao >= 1 and p3m.params.cao <= 7);
239  assert(p3m.params.alpha > 0.);
240 
241  p3m.params.cao3 = Utils::int_pow<3>(p3m.params.cao);
242  p3m.params.recalc_a_ai_cao_cut(box_geo.length());
243 
244  sanity_checks();
245 
246  double elc_layer = 0.;
247  if (auto elc_actor = get_actor_by_type<ElectrostaticLayerCorrection>(
248  electrostatics_actor)) {
249  elc_layer = elc_actor->elc.space_layer;
250  }
251 
252  p3m.local_mesh.calc_local_ca_mesh(p3m.params, local_geo, skin, elc_layer);
253  p3m.sm.resize(comm_cart, p3m.local_mesh);
254 
255  int ca_mesh_size =
256  fft_init(p3m.local_mesh.dim, p3m.local_mesh.margin, p3m.params.mesh,
257  p3m.params.mesh_off, p3m.ks_pnum, p3m.fft, node_grid, comm_cart);
258  p3m.rs_mesh.resize(ca_mesh_size);
259 
260  for (auto &e : p3m.E_mesh) {
261  e.resize(ca_mesh_size);
262  }
263 
264  p3m.calc_differential_operator();
265 
266  /* fix box length dependent constants */
267  scaleby_box_l();
268 
269  count_charged_particles();
270 }

338 template <std::size_t cao> struct AssignForces {
339  void operator()(p3m_data_struct &p3m, double force_prefac,
340  ParticleRange const &particles) const {
341  using Utils::make_const_span;
342  using Utils::Span;
343  using Utils::Vector;
344 
345  assert(cao == p3m.inter_weights.cao());
346 
347  /* charged particle counter */
348  auto p_index = std::size_t{0ul};
349 
350  for (auto &p : particles) {
351  if (p.q() != 0.0) {
352  auto const pref = p.q() * force_prefac;
353  auto const w = p3m.inter_weights.load<cao>(p_index);
354 
355  Utils::Vector3d force{};
356  p3m_interpolate(p3m.local_mesh, w, [&force, &p3m](int ind, double w) {
357  force += w * Utils::Vector3d{p3m.E_mesh[0][ind], p3m.E_mesh[1][ind],
358  p3m.E_mesh[2][ind]};
359  });
360 
361  p.force() -= pref * force;
362  ++p_index;
363  }
364  }
365  }
366 };

384 /** @details Calculate the long range electrostatics part of the pressure
385  * tensor. This is part \f$\Pi_{\textrm{dir}, \alpha, \beta}\f$ eq. (2.6)
386  * in @cite essmann95a. The part \f$\Pi_{\textrm{corr}, \alpha, \beta}\f$
387  * eq. (2.8) is not present here since M is the empty set in our simulations.
388  */
389 Utils::Vector9d CoulombP3M::p3m_calc_kspace_pressure_tensor() {
390  using namespace detail::FFT_indexing;
391 
392  Utils::Vector9d node_k_space_pressure_tensor{};
393 
394  if (p3m.sum_q2 > 0.) {
395  p3m.sm.gather_grid(p3m.rs_mesh.data(), comm_cart, p3m.local_mesh.dim);
396  fft_perform_forw(p3m.rs_mesh.data(), p3m.fft, comm_cart);
397 
398  auto diagonal = 0.;
399  int ind = 0;
400  int j[3];
401  auto const half_alpha_inv_sq = Utils::sqr(1. / 2. / p3m.params.alpha);
402  for (j[0] = 0; j[0] < p3m.fft.plan[3].new_mesh[RX]; j[0]++) {
403  for (j[1] = 0; j[1] < p3m.fft.plan[3].new_mesh[RY]; j[1]++) {
404  for (j[2] = 0; j[2] < p3m.fft.plan[3].new_mesh[RZ]; j[2]++) {
405  auto const kx = 2. * Utils::pi() *
406  p3m.d_op[RX][j[KX] + p3m.fft.plan[3].start[KX]] *
407  box_geo.length_inv()[RX];
408  auto const ky = 2. * Utils::pi() *
409  p3m.d_op[RY][j[KY] + p3m.fft.plan[3].start[KY]] *
410  box_geo.length_inv()[RY];
411  auto const kz = 2. * Utils::pi() *
412  p3m.d_op[RZ][j[KZ] + p3m.fft.plan[3].start[KZ]] *
413  box_geo.length_inv()[RZ];
414  auto const sqk = Utils::sqr(kx) + Utils::sqr(ky) + Utils::sqr(kz);
415 
416  if (sqk != 0.) {
417  auto const node_k_space_energy =
418  p3m.g_energy[ind] * (Utils::sqr(p3m.rs_mesh[2 * ind]) +
419  Utils::sqr(p3m.rs_mesh[2 * ind + 1]));
420  auto const vterm = -2. * (1. / sqk + half_alpha_inv_sq);
421  auto const pref = node_k_space_energy * vterm;
422  node_k_space_pressure_tensor[0] += pref * kx * kx; /* sigma_xx */
423  node_k_space_pressure_tensor[1] += pref * kx * ky; /* sigma_xy */
424  node_k_space_pressure_tensor[2] += pref * kx * kz; /* sigma_xz */
425  node_k_space_pressure_tensor[3] += pref * ky * kx; /* sigma_yx */
426  node_k_space_pressure_tensor[4] += pref * ky * ky; /* sigma_yy */
427  node_k_space_pressure_tensor[5] += pref * ky * kz; /* sigma_yz */
428  node_k_space_pressure_tensor[6] += pref * kz * kx; /* sigma_zx */
429  node_k_space_pressure_tensor[7] += pref * kz * ky; /* sigma_zy */
430  node_k_space_pressure_tensor[8] += pref * kz * kz; /* sigma_zz */
431  diagonal += node_k_space_energy;
432  }
433  ind++;
434  }
435  }
436  }
437  node_k_space_pressure_tensor[0] += diagonal;
438  node_k_space_pressure_tensor[4] += diagonal;
439  node_k_space_pressure_tensor[8] += diagonal;
440  }
441 
442  return node_k_space_pressure_tensor * prefactor / (2. * box_geo.volume());
443 }

445 double CoulombP3M::long_range_kernel(bool force_flag, bool energy_flag,
446  ParticleRange const &particles) {
447  /* Gather information for FFT grid inside the nodes domain (inner local mesh)
448  * and perform forward 3D FFT (Charge Assignment Mesh). */
449  p3m.sm.gather_grid(p3m.rs_mesh.data(), comm_cart, p3m.local_mesh.dim);
450  fft_perform_forw(p3m.rs_mesh.data(), p3m.fft, comm_cart);
451 
452  // Note: after these calls, the grids are in the order yzx and not xyz
453  // anymore!!!
454  /* The dipole moment is only needed if we don't have metallic boundaries. */
455  auto const box_dipole = (p3m.params.epsilon != P3M_EPSILON_METALLIC)
456  ? boost::make_optional(calc_dipole_moment(
457  comm_cart, particles, box_geo))
458  : boost::none;
459  auto const volume = box_geo.volume();
460  auto const pref = 4. * Utils::pi() / volume / (2. * p3m.params.epsilon + 1.);
461 
462  /* === k-space force calculation === */
463  if (force_flag) {
464  /* sqrt(-1)*k differentiation */
465  int j[3];
466  int ind = 0;
467  for (j[0] = 0; j[0] < p3m.fft.plan[3].new_mesh[0]; j[0]++) {
468  for (j[1] = 0; j[1] < p3m.fft.plan[3].new_mesh[1]; j[1]++) {
469  for (j[2] = 0; j[2] < p3m.fft.plan[3].new_mesh[2]; j[2]++) {
470  auto const rho_hat = std::complex<double>(p3m.rs_mesh[2 * ind + 0],
471  p3m.rs_mesh[2 * ind + 1]);
472  auto const phi_hat = p3m.g_force[ind] * rho_hat;
473 
474  for (int d = 0; d < 3; d++) {
475  /* direction in r-space: */
476  int d_rs = (d + p3m.ks_pnum) % 3;
477  /* directions */
478  auto const k = 2. * Utils::pi() *
479  p3m.d_op[d_rs][j[d] + p3m.fft.plan[3].start[d]] *
480  box_geo.length_inv()[d_rs];
481 
482  /* i*k*(Re+i*Im) = - Im*k + i*Re*k (i=sqrt(-1)) */
483  p3m.E_mesh[d_rs][2 * ind + 0] = -k * phi_hat.imag();
484  p3m.E_mesh[d_rs][2 * ind + 1] = +k * phi_hat.real();
485  }
486 
487  ind++;
488  }
489  }
490  }
491 
492  /* Back FFT force component mesh */
493  auto const check_complex = !p3m.params.tuning;
494  for (int d = 0; d < 3; d++) {
495  fft_perform_back(p3m.E_mesh[d].data(), check_complex, p3m.fft, comm_cart);
496  }
497 
498  /* redistribute force component mesh */
499  std::array<double *, 3> E_fields = {
500  {p3m.E_mesh[0].data(), p3m.E_mesh[1].data(), p3m.E_mesh[2].data()}};
501  p3m.sm.spread_grid(Utils::make_span(E_fields), comm_cart,
502  p3m.local_mesh.dim);
503 
504  auto const force_prefac = prefactor / volume;
505  Utils::integral_parameter<AssignForces, 1, 7>(p3m.params.cao, p3m,
506  force_prefac, particles);
507 
508  // add dipole forces
509  if (p3m.params.epsilon != P3M_EPSILON_METALLIC) {
510  auto const dm = prefactor * pref * box_dipole.value();
511  for (auto &p : particles) {
512  p.force() -= p.q() * dm;
513  }
514  }
515  }
516 
517  /* === k-space energy calculation === */
518  if (energy_flag) {
519  auto node_energy = 0.;
520  for (int i = 0; i < p3m.fft.plan[3].new_size; i++) {
521  // Use the energy optimized influence function for energy!
522  node_energy += p3m.g_energy[i] * (Utils::sqr(p3m.rs_mesh[2 * i]) +
523  Utils::sqr(p3m.rs_mesh[2 * i + 1]));
524  }
525  node_energy /= 2. * volume;
526 
527  auto energy = 0.;
528  boost::mpi::reduce(comm_cart, node_energy, energy, std::plus<>(), 0);
529  if (this_node == 0) {
530  /* self energy correction */
531  energy -= p3m.sum_q2 * p3m.params.alpha * Utils::sqrt_pi_i();
532  /* net charge correction */
533  energy -= p3m.square_sum_q * Utils::pi() /
534  (2. * volume * Utils::sqr(p3m.params.alpha));
535  /* dipole correction */
536  if (p3m.params.epsilon != P3M_EPSILON_METALLIC) {
537  energy += pref * box_dipole.value().norm2();
538  }
539  }
540  return prefactor * energy;
541  }
542 
543  return 0.;
544 }

584  std::tuple<double, double, double, double>
585  calculate_accuracy(Utils::Vector3i const &mesh, int cao,
586  double r_cut_iL) const override {
587 
588  double alpha_L, rs_err, ks_err;
589 
590  /* calc maximal real space error for setting */
591  rs_err = p3m_real_space_error(m_prefactor, r_cut_iL, p3m.sum_qpart,
592  p3m.sum_q2, 0.);
593 
594  if (Utils::sqrt_2() * rs_err > p3m.params.accuracy) {
595  /* assume rs_err = ks_err -> rs_err = accuracy/sqrt(2.0) -> alpha_L */
596  alpha_L =
597  sqrt(log(Utils::sqrt_2() * rs_err / p3m.params.accuracy)) / r_cut_iL;
598  } else {
599  /* even alpha=0 is ok, however, we cannot choose it since it kills the
600  k-space error formula.
601  Anyways, this very likely NOT the optimal solution */
602  alpha_L = 0.1;
603  }
604 
605  /* calculate real-space and k-space error for this alpha_L */
606  rs_err = p3m_real_space_error(m_prefactor, r_cut_iL, p3m.sum_qpart,
607  p3m.sum_q2, alpha_L);
608 #ifdef CUDA
609  if (has_actor_of_type<CoulombP3MGPU>(electrostatics_actor)) {
610  ks_err = p3mgpu_k_space_error(m_prefactor, mesh, cao, p3m.sum_qpart,
611  p3m.sum_q2, alpha_L);
612  } else
613 #endif
614  ks_err = p3m_k_space_error(m_prefactor, mesh, cao, p3m.sum_qpart,
615  p3m.sum_q2, alpha_L);
616 
617  return {Utils::Vector2d{rs_err, ks_err}.norm(), rs_err, ks_err, alpha_L};
618  }

620  void determine_mesh_limits() override {
621  auto const mesh_density =
622  static_cast<double>(p3m.params.mesh[0]) * box_geo.length_inv()[0];
623 
624  if (p3m.params.mesh == Utils::Vector3i::broadcast(-1)) {
625  /* avoid using more than 1 GB of FFT arrays */
626  auto const normalized_box_dim = std::cbrt(box_geo.volume());
627  auto const max_npart_per_dim = 512.;
628  /* simple heuristic to limit the tried meshes if the accuracy cannot
629  be obtained with smaller meshes, but normally not all these
630  meshes have to be tested */
631  auto const min_npart_per_dim = std::min(
632  max_npart_per_dim, std::cbrt(static_cast<double>(p3m.sum_qpart)));
633  m_mesh_density_min = min_npart_per_dim / normalized_box_dim;
634  m_mesh_density_max = max_npart_per_dim / normalized_box_dim;
635  m_tune_mesh = true;
636  } else {
637  m_mesh_density_min = m_mesh_density_max = mesh_density;
638  assert(p3m.params.mesh[0] >= 1);
639  if (p3m.params.mesh[1] == -1 and p3m.params.mesh[2] == -1) {
640  // determine the two missing values by rescaling by the box length
641  for (int i : {1, 2}) {
642  p3m.params.mesh[i] =
643  static_cast<int>(std::round(mesh_density * box_geo.length()[i]));
644  // make the mesh even in all directions
645  p3m.params.mesh[i] += p3m.params.mesh[i] % 2;
646  }
647  }
648  m_logger->report_fixed_mesh(p3m.params.mesh);
649  }
650  }
651 
652  TuningAlgorithm::Parameters get_time() override {
653  auto tuned_params = TuningAlgorithm::Parameters{};
654  auto time_best = time_sentinel;
655  for (auto mesh_density = m_mesh_density_min;
656  mesh_density <= m_mesh_density_max; mesh_density += 0.1) {
657  auto trial_params = TuningAlgorithm::Parameters{};
658  if (m_tune_mesh) {
659  for (int i : {0, 1, 2}) {
660  trial_params.mesh[i] =
661  static_cast<int>(std::round(box_geo.length()[i] * mesh_density));
662  // make the mesh even in all directions
663  trial_params.mesh[i] += trial_params.mesh[i] % 2;
664  }
665  } else {
666  trial_params.mesh = p3m.params.mesh;
667  }
668  trial_params.cao = cao_best;
669 
670  auto const trial_time =
671  get_m_time(trial_params.mesh, trial_params.cao, trial_params.r_cut_iL,
672  trial_params.alpha_L, trial_params.accuracy);
673 
674  /* this mesh does not work at all */
675  if (trial_time < 0.)
676  continue;
677 
678  /* the optimum r_cut for this mesh is the upper limit for higher meshes,
679  everything else is slower */
680  if (has_actor_of_type<CoulombP3M>(electrostatics_actor)) {
681  m_r_cut_iL_max = trial_params.r_cut_iL;
682  }
683 
684  if (trial_time < time_best) {
685  /* new optimum */
686  reset_n_trials();
687  tuned_params = trial_params;
688  time_best = tuned_params.time = trial_time;
689  } else if (trial_time > time_best + time_granularity or
690  get_n_trials() > max_n_consecutive_trials) {
691  /* no hope of further optimisation */
692  break;
693  }
694  }
695  return tuned_params;
696  }

699 void CoulombP3M::tune() {
700  if (p3m.params.alpha_L == 0. and p3m.params.alpha != 0.) {
701  p3m.params.alpha_L = p3m.params.alpha * box_geo.length()[0];
702  }

710  "CoulombP3M: no charged particles in the system");

717  parameters.determine_r_cut_limits();
