139template <typename FloatType, Arch Architecture>
140void CoulombP3MImpl<FloatType, Architecture>::count_charged_particles() {
141 struct Res {
142 std::size_t local_n = std::size_t{0u};
143 double local_q = 0.0;
144 double local_q2 = 0.0;
145 };
146 Reduction::AddPartialResultKernel<Res> kernel = [](Res &acc, auto const &p) {
147 if (p.q() != 0.0) {
148 acc.local_n++;
149 acc.local_q2 += Utils::sqr(p.q());
150 acc.local_q += p.q();
151 }
152 };
153
154 Reduction::ReductionOp<Res> reduce = [](Res &a, Res const &b) {
155 a.local_n += b.local_n;
156 a.local_q += b.local_q;
157 a.local_q2 += b.local_q2;
158 };
159 auto res = reduce_over_local_particles(*(get_system().cell_structure), kernel,
160 reduce);
161
162 boost::mpi::all_reduce(comm_cart, res.local_n, p3m.sum_qpart, std::plus<>());
163 boost::mpi::all_reduce(comm_cart, res.local_q2, p3m.sum_q2, std::plus<>());
164 boost::mpi::all_reduce(comm_cart, res.local_q, p3m.square_sum_q,
165 std::plus<>());
166 p3m.square_sum_q = Utils::sqr(p3m.square_sum_q);
167}

195/** Aliasing sum used by @ref p3m_k_space_error. */
196static auto p3m_tune_aliasing_sums(Utils::Vector3i const &shift,
197 Utils::Vector3i const &mesh,
198 Utils::Vector3d const &mesh_i, int cao,
199 double alpha_L_i) {
200
201 auto constexpr mesh_start = Utils::Vector3i::broadcast(-P3M_BRILLOUIN);
202 auto constexpr mesh_stop = Utils::Vector3i::broadcast(P3M_BRILLOUIN + 1);
203 auto constexpr exp_min = -708.4; // for IEEE-compatible double
204 auto const factor1 = Utils::sqr(std::numbers::pi * alpha_L_i);
205 auto alias1 = 0.;
206 auto alias2 = 0.;
207
208 Utils::Vector3i indices{};
209 Utils::Vector3i nm{};
210 Utils::Vector3d fnm{};
211 for_each_3d(
212 mesh_start, mesh_stop, indices,
213 [&]() {
214 auto const norm_sq = nm.norm2();
215 auto const exponent = -factor1 * norm_sq;
216 auto const exp_limit = (exp_min + std::log(norm_sq)) / 2.;
217 auto const ex = (exponent < exp_limit) ? 0. : std::exp(exponent);
218 auto const energy = std::pow(Utils::product(fnm), 2 * cao);
219 alias1 += Utils::sqr(ex) / norm_sq;
220 alias2 += energy * ex * (shift * nm) / norm_sq;
221 },
222 [&](unsigned dim, int n) {
223 nm[dim] = shift[dim] + n * mesh[dim];
224 fnm[dim] = math::sinc(nm[dim] * mesh_i[dim]);
225 });
226
227 return std::make_pair(alias1, alias2);
228}

230/** Calculate the real space contribution to the rms error in the force (as
231 * described by Kolafa and Perram).
232 * \param pref Prefactor of Coulomb interaction.
233 * \param r_cut_iL rescaled real space cutoff for p3m method.
234 * \param n_c_part number of charged particles in the system.
235 * \param sum_q2 sum of square of charges in the system
236 * \param alpha_L rescaled Ewald splitting parameter.
237 * \param box_l box dimensions.
238 * \return real space error
239 */
240static double p3m_real_space_error(double pref, double r_cut_iL, int n_c_part,
241 double sum_q2, double alpha_L,
242 Utils::Vector3d const &box_l) {
243 auto const volume = Utils::product(box_l);
244 return (2. * pref * sum_q2 * exp(-Utils::sqr(r_cut_iL * alpha_L))) /
245 sqrt(static_cast<double>(n_c_part) * r_cut_iL * box_l[0] * volume);
246}
247
248/** Calculate the analytic expression of the error estimate for the
249 * P3M method in @cite hockney88a (eq. 8-23 p. 275) in
250 * order to obtain the rms error in the force for a system of N
251 * randomly distributed particles in a cubic box (k-space part).
252 * \param pref Prefactor of Coulomb interaction.
253 * \param mesh number of mesh points in one direction.
254 * \param cao charge assignment order.
255 * \param n_c_part number of charged particles in the system.
256 * \param sum_q2 sum of square of charges in the system
257 * \param alpha_L rescaled Ewald splitting parameter.
258 * \param box_l box dimensions.
259 * \return reciprocal (k) space error
260 */
261static double p3m_k_space_error(double pref, Utils::Vector3i const &mesh,
262 int cao, int n_c_part, double sum_q2,
263 double alpha_L, Utils::Vector3d const &box_l) {
264
265 auto const cotangent_sum = math::get_analytic_cotangent_sum_kernel(cao);
266 auto const mesh_i = 1. / Utils::Vector3d(mesh);
267 auto const alpha_L_i = 1. / alpha_L;
268 auto const mesh_stop = mesh / 2;
269 auto const mesh_start = -mesh_stop;
270 auto indices = Utils::Vector3i{};
271 auto values = Utils::Vector3d{};
272 auto he_q = 0.;
273
274 for_each_3d(
275 mesh_start, mesh_stop, indices,
276 [&]() {
277 if ((indices[0] != 0) or (indices[1] != 0) or (indices[2] != 0)) {
278 auto const n2 = indices.norm2();
279 auto const cs = Utils::product(values);
280 auto const [alias1, alias2] =
281 p3m_tune_aliasing_sums(indices, mesh, mesh_i, cao, alpha_L_i);
282 auto const d = alias1 - Utils::sqr(alias2 / cs) / n2;
283 /* at high precision, d can become negative due to extinction;
284 also, don't take values that have no significant digits left*/
285 if (d > 0. and std::fabs(d / alias1) > round_error_prec) {
286 he_q += d;
287 }
288 }
289 },
290 [&values, &mesh_i, cotangent_sum](unsigned dim, int n) {
291 values[dim] = cotangent_sum(n, mesh_i[dim]);
292 });
293
294 return 2. * pref * sum_q2 * sqrt(he_q / static_cast<double>(n_c_part)) /
295 (box_l[1] * box_l[2]);
296}

298template <typename FloatType, Arch Architecture>
299void CoulombP3MImpl<FloatType, Architecture>::init_cpu_kernels() {
300 assert(p3m.params.mesh >= Utils::Vector3i::broadcast(1));
301 assert(p3m.params.cao >= p3m_min_cao and p3m.params.cao <= p3m_max_cao);
302 assert(p3m.params.alpha > 0.);
303
304 auto const &system = get_system();
305 auto const &box_geo = *system.box_geo;
306 auto const &local_geo = *system.local_geo;
307 auto const skin = system.cell_structure->get_verlet_skin();
308
309 p3m.params.cao3 = Utils::int_pow<3>(p3m.params.cao);
310 p3m.params.recalc_a_ai_cao_cut(box_geo.length());
311
312 sanity_checks();
313
314 auto const &solver = system.coulomb.impl->solver;
315 double elc_layer = 0.;
316 if (auto actor = get_actor_by_type<ElectrostaticLayerCorrection>(solver)) {
317 elc_layer = actor->elc.space_layer;
318 }
319
320 p3m.local_mesh.calc_local_ca_mesh(p3m.params, local_geo, skin, elc_layer);
321 p3m.fft = std::make_shared<P3MFFT<FloatType>>(
322 comm_cart, p3m.params.mesh, p3m.local_mesh.ld_no_halo,
323 p3m.local_mesh.ur_no_halo, get_memory_layout());
324 p3m.fft->set_preferred_kspace_decomposition(::communicator.node_grid);
325 auto const rs_array_size = get_size_from_shape(p3m.local_mesh.dim);
326 auto const rs_array_size_no_halo =
327 get_size_from_shape(p3m.local_mesh.dim_no_halo);
328 auto const fft_mesh_size = get_size_from_shape(p3m.fft->ks_local_size());
329 p3m.rs_charge_density.resize(rs_array_size);
330 p3m.ks_charge_density.resize(fft_mesh_size);
331 for (auto d : {0u, 1u, 2u}) {
332 p3m.rs_E_fields[d].resize(rs_array_size);
333 }
334 p3m.ks_E_fields_storage.resize(3u * fft_mesh_size);
335 p3m.rs_E_fields_no_halo.resize(3u * rs_array_size_no_halo);
336 p3m.calc_differential_operator();
337
338 /* fix box length dependent constants */
339 scaleby_box_l();
340
341 count_charged_particles();
342}

359 auto const weights = p3m_calculate_interpolation_weights<cao, memory_order>(
360 real_pos, p3m.params.ai, p3m.local_mesh);
361 inter_weights.store(weights);
362 this->operator()(p3m, q, weights);

402 }

438template <int cao> struct AssignForces {
439 void operator()(auto &p3m, auto force_prefac,
440#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
441 CellStructure &cell_structure
442#else
443 auto const &p_q_force_range
444#endif
445 ) const {
446
447 assert(cao == p3m.inter_weights.cao());
448
449 auto const kernel = [&p3m](auto pref, auto &p_force, std::size_t p_index) {
450 auto const weights = p3m.inter_weights.template load<cao>(p_index);
451
452 Utils::Vector3d force{};
453 p3m_interpolate(p3m.local_mesh, weights,
454 [&force, &p3m](int ind, double w) {
455 force[0u] += w * double(p3m.rs_E_fields[0u][ind]);
456 force[1u] += w * double(p3m.rs_E_fields[1u][ind]);
457 force[2u] += w * double(p3m.rs_E_fields[2u][ind]);
458 });
459
460#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
461 auto const thread_id = omp_get_thread_num();
462 p_force(p_index, thread_id, 0) -= pref * force[0];
463 p_force(p_index, thread_id, 1) -= pref * force[1];
464 p_force(p_index, thread_id, 2) -= pref * force[2];
465#else
466 p_force -= pref * force;
467#endif
468 };
469
470#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
471 auto const n_part = p3m.inter_weights.size();
472 auto const &aosoa = cell_structure.get_aosoa();
473 auto &local_force = cell_structure.get_local_force();
474 kokkos_parallel_range_for(
475 "AssignForces", std::size_t{0u}, n_part, [&](std::size_t p_index) {
476 if (auto const pref = aosoa.charge(p_index) * force_prefac) {
477 kernel(pref, local_force, p_index);
478 }
479 });
480#else // ESPRESSO_SHARED_MEMORY_PARALLELISM
481 /* charged particle counter */
482 std::size_t p_index{0ul};
483
484 for (auto zipped : p_q_force_range) {
485 auto p_q = boost::get<0>(zipped);
486 if (p_q != 0.) {
487 auto &p_force = boost::get<1>(zipped);
488 kernel(p_q * force_prefac, p_force, p_index);
489 ++p_index;
490 }
491 }
492#endif // ESPRESSO_SHARED_MEMORY_PARALLELISM
493 }
494};

498static auto calc_dipole_moment(boost::mpi::communicator const &comm,
499 auto const &cs, auto const &box_geo) {
500 auto const local_dip = reduce_over_local_particles<Utils::Vector3d>(
501 cs,
502 [&box_geo](Utils::Vector3d &acc, Particle const &p) {
503 acc += p.q() * box_geo.unfolded_position(p.pos(), p.image_box());
504 },
505 [](Utils::Vector3d &a, Utils::Vector3d const &b) { a = a + b; });
506 return boost::mpi::all_reduce(comm, local_dip, std::plus<>());
507}
508#else // ESPRESSO_SHARED_MEMORY_PARALLELISM
509static auto calc_dipole_moment(boost::mpi::communicator const &comm,
510 auto const &p_q_unfolded_pos_range) {
511 auto const local_dip =
512 boost::accumulate(p_q_unfolded_pos_range, Utils::Vector3d{},
513 [](Utils::Vector3d const &dip, auto const &q_pos) {
514 auto const p_q = boost::get<0>(q_pos);
515 auto const &p_unfolded_pos = boost::get<1>(q_pos);
516 return dip + p_q * p_unfolded_pos;
517 });
518 return boost::mpi::all_reduce(comm, local_dip, std::plus<>());
519}

522template <typename FloatType, Arch Architecture>
523void CoulombP3MImpl<FloatType, Architecture>::kernel_ks_charge_density() {
524 // halo communication of real space charge density
525 p3m.halo_comm.gather_grid(comm_cart, p3m.rs_charge_density.data(),
526 p3m.local_mesh.dim);
527
528 // get real space charge density without ghost layers
529 auto charge_density_no_halos =
530 extract_block<Utils::MemoryOrder::ROW_MAJOR,
531 Utils::MemoryOrder::COLUMN_MAJOR>(
532 p3m.rs_charge_density, p3m.local_mesh.dim, p3m.local_mesh.n_halo_ld,
533 p3m.local_mesh.dim - p3m.local_mesh.n_halo_ur);
534
535 // Set up the FFT using the Heffte library.
536 // This is in global mesh coordinates without any ghost layers
537 // The memory layout has to be specified, so the parts of
538 // the mesh held by each MPI rank are assembled correctly.
539 p3m.fft->forward(charge_density_no_halos.data(),
540 p3m.ks_charge_density.data());
541}

543template <typename FloatType, Arch Architecture>
544void CoulombP3MImpl<FloatType, Architecture>::kernel_rs_electric_field() {
545 auto const mesh_start = p3m.fft->ks_local_ld_index();
546 auto const mesh_stop = mesh_start + p3m.fft->ks_local_size();
547 auto const &box_geo = *get_system().box_geo;
548
549 // hold electric field in k-space
550 std::array<std::span<std::complex<FloatType>>, 3> ks_E_fields;
551 auto const fft_mesh_length = get_size_from_shape(mesh_stop - mesh_start);
552 for (auto d : {0u, 1u, 2u}) {
553 auto const offset = d * fft_mesh_length;
554 auto const begin = p3m.ks_E_fields_storage.begin() + offset;
555 ks_E_fields[d] = {begin, fft_mesh_length};
556 }
557
558 // i*k differentiation
559 auto const wavevector =
560 Utils::Vector3<FloatType>((2. * std::numbers::pi) * box_geo.length_inv());
561
562 // compute electric field, Eq. (3.49) @cite deserno00b
563 for_each_3d_lin<Utils::MemoryOrder::COLUMN_MAJOR>(
564 mesh_start, mesh_stop,
565 [&](const Utils::Vector3i &indices, int local_index) {
566#ifdef ESPRESSO_ADDITIONAL_CHECKS
567 assert(local_index ==
568 Utils::get_linear_index<Utils::MemoryOrder::COLUMN_MAJOR>(
569 indices - mesh_start, p3m.fft->ks_local_size()));
570#endif
571 auto const phi_hat = multiply_complex_by_real(
572 p3m.ks_charge_density[local_index], p3m.g_force[local_index]);
573
574 for (auto d : {0u, 1u, 2u}) {
575 // wave vector of the current mesh point
576 auto const k = FloatType(p3m.d_op[d][indices[d]]) * wavevector[d];
577 // electric field in k-space
578 ks_E_fields[d][local_index] =
579 multiply_complex_by_imaginary(phi_hat, k);
580 }
581 });
582
583 // back-transform the k-space electric field to real space
584 auto const size = p3m.local_mesh.ur_no_halo - p3m.local_mesh.ld_no_halo;
585 auto const rs_mesh_size_no_halo = Utils::product(size);
586 for (auto d : {0u, 1u, 2u}) {
587 auto k_space = ks_E_fields[d].data();
588 auto real_space = p3m.rs_E_fields_no_halo.data() + d * rs_mesh_size_no_halo;
589 p3m.fft->backward(k_space, real_space);
590
591 // add zeros around the E-field in real space to make room for ghost layers
592 auto const offset = d * rs_mesh_size_no_halo;
593 auto const begin = p3m.rs_E_fields_no_halo.begin() + offset;
594 auto f = std::span<std::complex<FloatType>>(begin, rs_mesh_size_no_halo);
595 p3m.rs_E_fields[d] =
596 pad_with_zeros_discard_imag<Utils::MemoryOrder::COLUMN_MAJOR,
597 Utils::MemoryOrder::ROW_MAJOR>(
598 std::span<std::complex<FloatType>>(f), p3m.local_mesh.dim_no_halo,
599 p3m.local_mesh.n_halo_ld, p3m.local_mesh.n_halo_ur);
600 }
601
602 // ghost communicate the boundary layers of the E-field in real space
603 auto field_pointers = std::vector<FloatType *>{p3m.rs_E_fields[0u].data(),
604 p3m.rs_E_fields[1u].data(),
605 p3m.rs_E_fields[2u].data()};
606 p3m.halo_comm.spread_grid(comm_cart, field_pointers, p3m.local_mesh.dim);
607}

609/** @details Calculate the long range electrostatics part of the pressure
610 * tensor. This is part \f$\Pi_{\textrm{dir}, \alpha, \beta}\f$ eq. (2.6)
611 * in @cite essmann95a. The part \f$\Pi_{\textrm{corr}, \alpha, \beta}\f$
612 * eq. (2.8) is not present here since M is the empty set in our simulations.
613 */
614template <typename FloatType, Arch Architecture>
615Utils::Vector9d CoulombP3MImpl<FloatType, Architecture>::long_range_pressure(
616 ParticleRange const &particles) {
617 auto const &box_geo = *get_system().box_geo;
618 Utils::Vector9d node_k_space_pressure_tensor{};
619
620 if (p3m.sum_q2 > 0.) {
621 charge_assign(particles);
622 kernel_ks_charge_density();
623
624 auto constexpr mesh_start = Utils::Vector3i::broadcast(0);
625 auto const &mesh_stop = p3m.fft->ks_local_size();
626 auto const &k_size = p3m.fft->ks_local_size();
627 auto const half_alpha_inv_sq = Utils::sqr(1. / 2. / p3m.params.alpha);
628 auto const wavevector = (2. * std::numbers::pi) * box_geo.length_inv();
629 auto indices = Utils::Vector3i{};
630 auto diagonal = 0.;
631
632 for_each_3d(mesh_start, mesh_stop, indices, [&]() {
633 auto const global_index = indices + p3m.fft->ks_local_ld_index();
634 auto const kx = p3m.d_op[0u][global_index[0u]] * wavevector[0u];
635 auto const ky = p3m.d_op[1u][global_index[1u]] * wavevector[1u];
636 auto const kz = p3m.d_op[2u][global_index[2u]] * wavevector[2u];
637 auto const norm_sq = Utils::sqr(kx) + Utils::sqr(ky) + Utils::sqr(kz);
638
639 if (norm_sq != 0.) {
640 auto const index = Utils::get_linear_index(
641 indices, k_size, Utils::MemoryOrder::COLUMN_MAJOR);
642 auto const node_k_space_energy = static_cast<double>(
643 p3m.g_energy[index] * complex_norm2(p3m.ks_charge_density[index]));
644 auto const vterm = -2. * (1. / norm_sq + half_alpha_inv_sq);
645 auto const pref = node_k_space_energy * vterm;
646 node_k_space_pressure_tensor[0u] += pref * kx * kx; /* sigma_xx */
647 node_k_space_pressure_tensor[1u] += pref * kx * ky; /* sigma_xy */
648 node_k_space_pressure_tensor[2u] += pref * kx * kz; /* sigma_xz */
649 node_k_space_pressure_tensor[4u] += pref * ky * ky; /* sigma_yy */
650 node_k_space_pressure_tensor[5u] += pref * ky * kz; /* sigma_yz */
651 node_k_space_pressure_tensor[8u] += pref * kz * kz; /* sigma_zz */
652 diagonal += node_k_space_energy;
653 }
654 });
655
656 node_k_space_pressure_tensor[0u] += diagonal;
657 node_k_space_pressure_tensor[4u] += diagonal;
658 node_k_space_pressure_tensor[8u] += diagonal;
659 node_k_space_pressure_tensor[3u] = node_k_space_pressure_tensor[1u];
660 node_k_space_pressure_tensor[6u] = node_k_space_pressure_tensor[2u];
661 node_k_space_pressure_tensor[7u] = node_k_space_pressure_tensor[5u];
662 }
663
664 return node_k_space_pressure_tensor * prefactor / (2. * box_geo.volume());
665}

667template <typename FloatType, Arch Architecture>
668double CoulombP3MImpl<FloatType, Architecture>::long_range_kernel(
669 bool force_flag, bool energy_flag, ParticleRange const &particles) {
670
671 auto const &system = get_system();
672 auto const &box_geo = *system.box_geo;
673#ifdef ESPRESSO_NPT
674 auto const npt_flag = force_flag and system.has_npt_enabled();
675#else
676 auto constexpr npt_flag = false;
677#endif
678 if (p3m.sum_qpart == 0u) {
679 return 0.;
680 }
681
682 if (not has_actor_of_type<ElectrostaticLayerCorrection>(
683 system.coulomb.impl->solver)) {
684 charge_assign(particles);
685 }
686
687 kernel_ks_charge_density();
688
689#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
690 auto &cell_structure = *system.cell_structure;
691 auto const &local_force = cell_structure.get_local_force();
692 auto const &aosoa = cell_structure.get_aosoa();
693#else
694 auto p_q_range = ParticlePropertyRange::charge_range(particles);
695 auto p_force_range = ParticlePropertyRange::force_range(particles);
696 auto p_unfolded_pos_range =
697 ParticlePropertyRange::unfolded_pos_range(particles, box_geo);
698#endif // ESPRESSO_SHARED_MEMORY_PARALLELISM
699
700 // The dipole moment is only needed if we don't have metallic boundaries
701 auto const box_dipole = (p3m.params.epsilon != P3M_EPSILON_METALLIC)
702 ? std::make_optional(calc_dipole_moment(
703#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
704 comm_cart, cell_structure, box_geo))
705#else
706 comm_cart,
707 boost::combine(p_q_range,
708 p_unfolded_pos_range)))
709#endif
710 : std::nullopt;
711 auto const volume = box_geo.volume();
712 auto const pref =
713 4. * std::numbers::pi / volume / (2. * p3m.params.epsilon + 1.);
714 auto energy = 0.;
715
716 /* === k-space force calculation === */
717 if (force_flag) {
718 kernel_rs_electric_field();
719
720 // assign particle forces
721 auto const force_prefac = prefactor / volume;
722#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
723 auto &particle_data = cell_structure;
724#else
725 auto const particle_data = boost::combine(p_q_range, p_force_range);
726#endif
727 Utils::integral_parameter<int, AssignForces, p3m_min_cao, p3m_max_cao>(
728 p3m.params.cao, p3m, force_prefac, particle_data);
729
730 // add dipole forces
731 // Eq. (3.19) @cite deserno00b
732 if (box_dipole) {
733 auto const dm = prefactor * pref * box_dipole.value();
734#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
735 auto const n_part = cell_structure.count_local_particles();
736 kokkos_parallel_range_for(
737 "AssignForcesBoxDipole", std::size_t{0u}, n_part,
738 [&aosoa, &local_force, dm](auto p_index) {
739 auto const thread_id = omp_get_thread_num();
740 auto const q = aosoa.charge(p_index);
741 local_force(p_index, thread_id, 0) -= q * dm[0];
742 local_force(p_index, thread_id, 1) -= q * dm[1];
743 local_force(p_index, thread_id, 2) -= q * dm[2];
744 });
745#else // ESPRESSO_SHARED_MEMORY_PARALLELISM
746 for (auto zipped : boost::combine(p_q_range, p_force_range)) {
747 auto p_q = boost::get<0>(zipped);
748 auto &p_force = boost::get<1>(zipped);
749 p_force -= p_q * dm;
750 }
751#endif // ESPRESSO_SHARED_MEMORY_PARALLELISM
752 }
753 }
754
755 /* === k-space energy calculation === */
756 if (energy_flag or npt_flag) {
757 auto constexpr mesh_start = Utils::Vector3i::broadcast(0);
758 auto const &mesh_stop = p3m.fft->ks_local_size();
759 auto indices = Utils::Vector3i{};
760 auto node_energy = 0.;
761 for_each_3d(mesh_start, mesh_stop, indices, [&]() {
762 auto const index = Utils::get_linear_index(
763 indices, mesh_stop, Utils::MemoryOrder::COLUMN_MAJOR);
764 // Use the energy optimized influence function for energy!
765 // Eq. (3.40) @cite deserno00b
766 node_energy += static_cast<double>(
767 p3m.g_energy[index] * complex_norm2(p3m.ks_charge_density[index]));
768 });
769 node_energy /= 2. * volume;
770
771 // add up energy contributions from all mpi ranks
772 boost::mpi::reduce(comm_cart, node_energy, energy, std::plus<>(), 0);
773 if (this_node == 0) {
774 /* self energy correction */
775 // Eq. (3.8) @cite deserno00b
776 energy -= p3m.sum_q2 * p3m.params.alpha * std::numbers::inv_sqrtpi;
777 /* net charge correction */
778 // Eq. (3.11) @cite deserno00b
779 energy -= p3m.square_sum_q * std::numbers::pi /
780 (2. * volume * Utils::sqr(p3m.params.alpha));
781 /* dipole correction */
782 // Eq. (3.9) @cite deserno00b
783 if (box_dipole) {
784 energy += pref * box_dipole.value().norm2();
785 }
786 }
787 energy *= prefactor;
788#ifdef ESPRESSO_NPT
789 if (npt_flag) {
790 get_system().npt_add_virial_contribution(energy);
791 }
792#endif
793 if (not energy_flag) {
794 energy = 0.;
795 }
796 }
797
798 return energy;
799}

879 std::tuple<double, double, double, double>
880 calculate_accuracy(Utils::Vector3i const &mesh, int cao,
881 double r_cut_iL) const override {
882
883 auto const &box_geo = *m_system.box_geo;
884 double alpha_L, rs_err, ks_err;
885
886 /* calc maximal real space error for setting */
887 rs_err = p3m_real_space_error(m_prefactor, r_cut_iL, p3m.sum_qpart,
888 p3m.sum_q2, 0., box_geo.length());
889
890 if (std::numbers::sqrt2 * rs_err > p3m.params.accuracy) {
891 /* assume rs_err = ks_err -> rs_err = accuracy/sqrt(2.0) -> alpha_L */
892 alpha_L = sqrt(log(std::numbers::sqrt2 * rs_err / p3m.params.accuracy)) /
893 r_cut_iL;
894 } else {
895 /* even alpha=0 is ok, however, we cannot choose it since it kills the
896 k-space error formula.
897 Anyways, this very likely NOT the optimal solution */
898 alpha_L = 0.1;
899 }
900
901 /* calculate real-space and k-space error for this alpha_L */
902 rs_err = p3m_real_space_error(m_prefactor, r_cut_iL, p3m.sum_qpart,
903 p3m.sum_q2, alpha_L, box_geo.length());
904#ifdef ESPRESSO_CUDA
905 if constexpr (Architecture == Arch::GPU) {
906 if (this_node == 0) {
907 ks_err =
908 p3m_k_space_error_gpu(m_prefactor, mesh.data(), cao, p3m.sum_qpart,
909 p3m.sum_q2, alpha_L, box_geo.length().data());
910 }
911 boost::mpi::broadcast(comm_cart, ks_err, 0);
912 } else
913#endif
914 ks_err = p3m_k_space_error(m_prefactor, mesh, cao, p3m.sum_qpart,
915 p3m.sum_q2, alpha_L, box_geo.length());
916
917 return {Utils::Vector2d{rs_err, ks_err}.norm(), rs_err, ks_err, alpha_L};
918 }

920 void determine_mesh_limits() override {
921 auto const &box_geo = *m_system.box_geo;
922 auto const mesh_density =
923 static_cast<double>(p3m.params.mesh[0]) * box_geo.length_inv()[0];
924
925 if (p3m.params.mesh == Utils::Vector3i::broadcast(-1)) {
926 /* avoid using more than 1 GB of FFT arrays */
927 auto const normalized_box_dim = std::cbrt(box_geo.volume());
928 auto const max_npart_per_dim = 512.;
929 /* simple heuristic to limit the tried meshes if the accuracy cannot
930 be obtained with smaller meshes, but normally not all these
931 meshes have to be tested */
932 auto const min_npart_per_dim = std::min(
933 max_npart_per_dim, std::cbrt(static_cast<double>(p3m.sum_qpart)));
934 m_mesh_density_min = min_npart_per_dim / normalized_box_dim;
935 m_mesh_density_max = max_npart_per_dim / normalized_box_dim;
936 if (m_tune_limits.first or m_tune_limits.second) {
937 auto const &box_l = box_geo.length();
938 auto const dim = std::max({box_l[0], box_l[1], box_l[2]});
939 if (m_tune_limits.first) {
940 m_mesh_density_min = static_cast<double>(*m_tune_limits.first) / dim;
941 }
942 if (m_tune_limits.second) {
943 m_mesh_density_max = static_cast<double>(*m_tune_limits.second) / dim;
944 }
945 }
946 m_tune_mesh = true;
947 } else {
948 m_mesh_density_min = m_mesh_density_max = mesh_density;
949 assert(p3m.params.mesh[0] >= 1);
950 if (p3m.params.mesh[1] == -1 and p3m.params.mesh[2] == -1) {
951 // determine the two missing values by rescaling by the box length
952 for (auto i : {1u, 2u}) {
953 p3m.params.mesh[i] =
954 static_cast<int>(std::round(mesh_density * box_geo.length()[i]));
955 // make the mesh even in all directions
956 p3m.params.mesh[i] += p3m.params.mesh[i] % 2;
957 }
958 }
959 m_logger->report_fixed_mesh(p3m.params.mesh);
960 }
961 }
962
963 TuningAlgorithm::Parameters get_time() override {
964 auto const &box_geo = *m_system.box_geo;
965 auto const &solver = m_system.coulomb.impl->solver;
966 auto tuned_params = TuningAlgorithm::Parameters{};
967 auto time_best = time_sentinel;
968 auto mesh_density = m_mesh_density_min;
969 auto current_mesh = p3m.params.mesh;
970 if (m_tune_mesh) {
971 for (auto i : {0u, 1u, 2u}) {
972 current_mesh[i] =
973 static_cast<int>(std::round(box_geo.length()[i] * mesh_density));
974 // make the mesh even in all directions
975 current_mesh[i] += current_mesh[i] % 2;
976 }
977 }
978
979 while (mesh_density <= m_mesh_density_max) {
980 auto trial_params = TuningAlgorithm::Parameters{};
981 trial_params.mesh = current_mesh;
982 trial_params.cao = cao_best;
983 trial_params.cao = cao_best;
984
985 auto const trial_time =
986 get_m_time(trial_params.mesh, trial_params.cao, trial_params.r_cut_iL,
987 trial_params.alpha_L, trial_params.accuracy);
988
989 if (trial_time >= 0.) {
990 /* the optimum r_cut for this mesh is the upper limit for higher meshes,
991 everything else is slower */
992 if (has_actor_of_type<CoulombP3M>(solver)) {
993 m_r_cut_iL_max = trial_params.r_cut_iL;
994 }
995
996 if (trial_time < time_best) {
997 /* new optimum */
998 reset_n_trials();
999 tuned_params = trial_params;
1000 time_best = tuned_params.time = trial_time;
1001 } else if (trial_time > time_best + time_granularity or
1002 get_n_trials() > max_n_consecutive_trials) {
1003 /* no hope of further optimisation */
1004 break;
1005 }
1006 }
1007 if (m_tune_mesh) {
1008 current_mesh += Utils::Vector3i::broadcast(2);
1009 mesh_density = current_mesh[0] / box_geo.length()[0];
1010 } else {
1011 break;
1012 }
1013 }
1014 return tuned_params;
1015 }

1018template <typename FloatType, Arch Architecture>
1019void CoulombP3MImpl<FloatType, Architecture>::tune() {
1020 auto &system = get_system();
1021 auto const &box_geo = *system.box_geo;
1022 if (p3m.params.alpha_L == 0. and p3m.params.alpha != 0.) {
1023 p3m.params.alpha_L = p3m.params.alpha * box_geo.length()[0];
1024 }
1025 if (p3m.params.r_cut_iL == 0. and p3m.params.r_cut != 0.) {
1026 p3m.params.r_cut_iL = p3m.params.r_cut * box_geo.length_inv()[0];
1027 }
1028 if (not is_tuned()) {
1029 count_charged_particles();
1030 if (p3m.sum_qpart == 0) {
1031 throw std::runtime_error(
1032 "CoulombP3M: no charged particles in the system");
1033 }
1034 try {
1035 CoulombTuningAlgorithm<FloatType, Architecture> parameters(
1036 system, p3m, prefactor, tune_timings, tune_limits);
1037 parameters.setup_logger(tune_verbose);
1038 // parameter ranges
1039 parameters.determine_mesh_limits();
1040 parameters.determine_r_cut_limits();
1041 parameters.determine_cao_limits(7);
1042 // run tuning algorithm
1043 parameters.tune();
1044 m_is_tuned = true;
1045 system.on_coulomb_change();
1046 } catch (...) {
1047 p3m.params.tuning = false;
1048 throw;
1049 }
1050 }
1051 init();
1052}

1054void CoulombP3M::sanity_checks_boxl() const {
1055 auto const &system = get_system();
1056 auto const &box_geo = *system.box_geo;
1057 auto const &local_geo = *system.local_geo;
1058 for (auto i = 0u; i < 3u; i++) {
1059 /* check k-space cutoff */
1060 if (p3m_params.cao_cut[i] >= box_geo.length_half()[i]) {
1061 std::stringstream msg;
1062 msg << "P3M_init: k-space cutoff " << p3m_params.cao_cut[i]
1063 << " is larger than half of box dimension " << box_geo.length()[i];
1064 throw std::runtime_error(msg.str());
1065 }
