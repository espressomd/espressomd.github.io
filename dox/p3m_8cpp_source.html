87void CoulombP3M::count_charged_particles() {
88 auto local_n = 0;
89 auto local_q2 = 0.0;
90 auto local_q = 0.0;
91
92 for (auto const &p : get_system().cell_structure->local_particles()) {
93 if (p.q() != 0.0) {
94 local_n++;
95 local_q2 += Utils::sqr(p.q());
96 local_q += p.q();
97 }
98 }
99
100 boost::mpi::all_reduce(comm_cart, local_n, p3m.sum_qpart, std::plus<>());
101 boost::mpi::all_reduce(comm_cart, local_q2, p3m.sum_q2, std::plus<>());
102 boost::mpi::all_reduce(comm_cart, local_q, p3m.square_sum_q, std::plus<>());
103 p3m.square_sum_q = Utils::sqr(p3m.square_sum_q);
104}

106/** Calculate the optimal influence function of @cite hockney88a.
107 * (optimised for force calculations)
108 *
109 * Each node calculates only the values for its domain in k-space
110 * (see fft.plan[3].mesh and fft.plan[3].start).
111 *
112 * See also: @cite hockney88a eq. 8-22 (p. 275). Note the somewhat
113 * different convention for the prefactors, which is described in
114 * @cite deserno98a @cite deserno98b.
115 */
116void CoulombP3M::calc_influence_function_force() {
117 auto const start = Utils::Vector3i{p3m.fft.plan[3].start};
118 auto const size = Utils::Vector3i{p3m.fft.plan[3].new_mesh};
119
120 p3m.g_force = grid_influence_function<1>(p3m.params, start, start + size,
121 get_system().box_geo->length_inv());
122}

135/** Aliasing sum used by @ref p3m_k_space_error. */
136static auto p3m_tune_aliasing_sums(Utils::Vector3i const &shift,
137 Utils::Vector3i const &mesh,
138 Utils::Vector3d const &mesh_i, int cao,
139 double alpha_L_i) {
140
141 auto constexpr m_start = Utils::Vector3i::broadcast(-P3M_BRILLOUIN);
142 auto constexpr m_stop = Utils::Vector3i::broadcast(P3M_BRILLOUIN + 1);
143 auto const factor1 = Utils::sqr(std::numbers::pi * alpha_L_i);
144 auto alias1 = 0.;
145 auto alias2 = 0.;
146
147 Utils::Vector3i indices{};
148 Utils::Vector3i nm{};
149 Utils::Vector3d fnm{};
150 for_each_3d(
151 m_start, m_stop, indices,
152 [&]() {
153 auto const norm_sq = nm.norm2();
154 auto const ex = exp(-factor1 * norm_sq);
155 auto const energy = std::pow(Utils::product(fnm), 2 * cao);
156 alias1 += Utils::sqr(ex) / norm_sq;
157 alias2 += energy * ex * (shift * nm) / norm_sq;
158 },
159 [&](unsigned dim, int n) {
160 nm[dim] = shift[dim] + n * mesh[dim];
161 fnm[dim] = math::sinc(nm[dim] * mesh_i[dim]);
162 });
163
164 return std::make_pair(alias1, alias2);
165}

167/** Calculate the real space contribution to the rms error in the force (as
168 * described by Kolafa and Perram).
169 * \param pref Prefactor of Coulomb interaction.
170 * \param r_cut_iL rescaled real space cutoff for p3m method.
171 * \param n_c_part number of charged particles in the system.
172 * \param sum_q2 sum of square of charges in the system
173 * \param alpha_L rescaled Ewald splitting parameter.
174 * \param box_l box dimensions.
175 * \return real space error
176 */
177static double p3m_real_space_error(double pref, double r_cut_iL, int n_c_part,
178 double sum_q2, double alpha_L,
179 Utils::Vector3d const &box_l) {
180 auto const volume = Utils::product(box_l);
181 return (2. * pref * sum_q2 * exp(-Utils::sqr(r_cut_iL * alpha_L))) /
182 sqrt(static_cast<double>(n_c_part) * r_cut_iL * box_l[0] * volume);
183}
184
185/** Calculate the analytic expression of the error estimate for the
186 * P3M method in @cite hockney88a (eq. 8-23 p. 275) in
187 * order to obtain the rms error in the force for a system of N
188 * randomly distributed particles in a cubic box (k-space part).
189 * \param pref Prefactor of Coulomb interaction.
190 * \param mesh number of mesh points in one direction.
191 * \param cao charge assignment order.
192 * \param n_c_part number of charged particles in the system.
193 * \param sum_q2 sum of square of charges in the system
194 * \param alpha_L rescaled Ewald splitting parameter.
195 * \param box_l box dimensions.
196 * \return reciprocal (k) space error
197 */
198static double p3m_k_space_error(double pref, Utils::Vector3i const &mesh,
199 int cao, int n_c_part, double sum_q2,
200 double alpha_L, Utils::Vector3d const &box_l) {
201
202 auto const cotangent_sum = math::get_analytic_cotangent_sum_kernel(cao);
203 auto const mesh_i = 1. / Utils::Vector3d(mesh);
204 auto const alpha_L_i = 1. / alpha_L;
205 auto const m_stop = mesh / 2;
206 auto const m_start = -m_stop;
207 auto indices = Utils::Vector3i{};
208 auto values = Utils::Vector3d{};
209 auto he_q = 0.;
210
211 for_each_3d(
212 m_start, m_stop, indices,
213 [&]() {
214 if ((indices[0] != 0) or (indices[1] != 0) or (indices[2] != 0)) {
215 auto const n2 = indices.norm2();
216 auto const cs = Utils::product(values);
217 auto const [alias1, alias2] =
218 p3m_tune_aliasing_sums(indices, mesh, mesh_i, cao, alpha_L_i);
219 auto const d = alias1 - Utils::sqr(alias2 / cs) / n2;
220 /* at high precision, d can become negative due to extinction;
221 also, don't take values that have no significant digits left*/
222 if (d > 0. and std::fabs(d / alias1) > ROUND_ERROR_PREC) {
223 he_q += d;
224 }
225 }
226 },
227 [&values, &mesh_i, cotangent_sum](unsigned dim, int n) {
228 values[dim] = cotangent_sum(n, mesh_i[dim]);
229 });
230
231 return 2. * pref * sum_q2 * sqrt(he_q / static_cast<double>(n_c_part)) /
232 (box_l[1] * box_l[2]);
233}

235void CoulombP3M::init() {
236 assert(p3m.params.mesh >= Utils::Vector3i::broadcast(1));
237 assert(p3m.params.cao >= 1 and p3m.params.cao <= 7);
238 assert(p3m.params.alpha > 0.);
239
240 auto const &system = get_system();
241 auto const &box_geo = *system.box_geo;
242 auto const &local_geo = *system.local_geo;
243 auto const skin = system.cell_structure->get_verlet_skin();
244
245 p3m.params.cao3 = Utils::int_pow<3>(p3m.params.cao);
246 p3m.params.recalc_a_ai_cao_cut(box_geo.length());
247
248 sanity_checks();
249
250 auto const &solver = system.coulomb.impl->solver;
251 double elc_layer = 0.;
252 if (auto actor = get_actor_by_type<ElectrostaticLayerCorrection>(solver)) {
253 elc_layer = actor->elc.space_layer;
254 }
255
256 p3m.local_mesh.calc_local_ca_mesh(p3m.params, local_geo, skin, elc_layer);
257 p3m.sm.resize(comm_cart, p3m.local_mesh);
258
259 int ca_mesh_size = fft_init(p3m.local_mesh.dim, p3m.local_mesh.margin,
260 p3m.params.mesh, p3m.params.mesh_off, p3m.ks_pnum,
261 p3m.fft, ::communicator.node_grid, comm_cart);
262 p3m.rs_mesh.resize(ca_mesh_size);
263
264 for (auto &e : p3m.E_mesh) {
265 e.resize(ca_mesh_size);
266 }
267
268 p3m.calc_differential_operator();
269
270 /* fix box length dependent constants */
271 scaleby_box_l();
272
273 count_charged_particles();
274}

353template <int cao> struct AssignForces {
354 template <typename combined_ranges>
355 void operator()(p3m_data_struct &p3m, double force_prefac,
356 combined_ranges const &p_q_force_range) const {
357
358 assert(cao == p3m.inter_weights.cao());
359
360 /* charged particle counter */
361 auto p_index = std::size_t{0ul};
362
363 for (auto zipped : p_q_force_range) {
364 auto p_q = boost::get<0>(zipped);
365 auto &p_force = boost::get<1>(zipped);
366 if (p_q != 0.0) {
367 auto const pref = p_q * force_prefac;
368 auto const w = p3m.inter_weights.load<cao>(p_index);
369
370 Utils::Vector3d force{};
371 p3m_interpolate(p3m.local_mesh, w, [&force, &p3m](int ind, double w) {
372 force += w * Utils::Vector3d{p3m.E_mesh[0][ind], p3m.E_mesh[1][ind],
373 p3m.E_mesh[2][ind]};
374 });
375
376 p_force -= pref * force;
377 ++p_index;
378 }
379 }
380 }
381};

383template <typename combined_ranges>
384static auto calc_dipole_moment(boost::mpi::communicator const &comm,
385 combined_ranges const &p_q_unfolded_pos_range) {
386 auto const local_dip =
387 boost::accumulate(p_q_unfolded_pos_range, Utils::Vector3d{},
388 [](Utils::Vector3d const &dip, auto const &q_pos) {
389 auto const p_q = boost::get<0>(q_pos);
390 auto const &p_unfolded_pos = boost::get<1>(q_pos);
391 return dip + p_q * p_unfolded_pos;
392 });
393 return boost::mpi::all_reduce(comm, local_dip, std::plus<>());
394}

396/** @details Calculate the long range electrostatics part of the pressure
397 * tensor. This is part \f$\Pi_{\textrm{dir}, \alpha, \beta}\f$ eq. (2.6)
398 * in @cite essmann95a. The part \f$\Pi_{\textrm{corr}, \alpha, \beta}\f$
399 * eq. (2.8) is not present here since M is the empty set in our simulations.
400 */
401Utils::Vector9d
402CoulombP3M::long_range_pressure(ParticleRange const &particles) {
403 using namespace detail::FFT_indexing;
404
405 auto const &box_geo = *get_system().box_geo;
406
407 Utils::Vector9d node_k_space_pressure_tensor{};
408
409 if (p3m.sum_q2 > 0.) {
410 charge_assign(particles);
411 p3m.sm.gather_grid(p3m.rs_mesh.data(), comm_cart, p3m.local_mesh.dim);
412 fft_perform_forw(p3m.rs_mesh.data(), p3m.fft, comm_cart);
413
414 auto constexpr m_start = Utils::Vector3i::broadcast(0);
415 auto const half_alpha_inv_sq = Utils::sqr(1. / 2. / p3m.params.alpha);
416 auto const m_stop = std::span(p3m.fft.plan[3].new_mesh);
417 auto const offset = std::span(p3m.fft.plan[3].start);
418 auto const wavevector = (2. * std::numbers::pi) * box_geo.length_inv();
419 auto indices = Utils::Vector3i{};
420 auto index = std::size_t(0u);
421 auto diagonal = 0.;
422
423 for_each_3d(m_start, m_stop, indices, [&]() {
424 auto const kx = p3m.d_op[RX][indices[KX] + offset[KX]] * wavevector[RX];
425 auto const ky = p3m.d_op[RY][indices[KY] + offset[KY]] * wavevector[RY];
426 auto const kz = p3m.d_op[RZ][indices[KZ] + offset[KZ]] * wavevector[RZ];
427 auto const norm_sq = Utils::sqr(kx) + Utils::sqr(ky) + Utils::sqr(kz);
428
429 if (norm_sq != 0.) {
430 auto const node_k_space_energy =
431 p3m.g_energy[index] * (Utils::sqr(p3m.rs_mesh[2u * index + 0u]) +
432 Utils::sqr(p3m.rs_mesh[2u * index + 1u]));
433 auto const vterm = -2. * (1. / norm_sq + half_alpha_inv_sq);
434 auto const pref = node_k_space_energy * vterm;
435 node_k_space_pressure_tensor[0u] += pref * kx * kx; /* sigma_xx */
436 node_k_space_pressure_tensor[1u] += pref * kx * ky; /* sigma_xy */
437 node_k_space_pressure_tensor[2u] += pref * kx * kz; /* sigma_xz */
438 node_k_space_pressure_tensor[4u] += pref * ky * ky; /* sigma_yy */
439 node_k_space_pressure_tensor[5u] += pref * ky * kz; /* sigma_yz */
440 node_k_space_pressure_tensor[8u] += pref * kz * kz; /* sigma_zz */
441 diagonal += node_k_space_energy;
442 }
443 ++index;
444 });
445
446 node_k_space_pressure_tensor[0u] += diagonal;
447 node_k_space_pressure_tensor[4u] += diagonal;
448 node_k_space_pressure_tensor[8u] += diagonal;
449 node_k_space_pressure_tensor[3u] = node_k_space_pressure_tensor[1u];
450 node_k_space_pressure_tensor[6u] = node_k_space_pressure_tensor[2u];
451 node_k_space_pressure_tensor[7u] = node_k_space_pressure_tensor[5u];
452 }
453
454 return node_k_space_pressure_tensor * prefactor / (2. * box_geo.volume());
455}

457double CoulombP3M::long_range_kernel(bool force_flag, bool energy_flag,
458 ParticleRange const &particles) {
459 auto const &system = get_system();
460 auto const &box_geo = *system.box_geo;
461#ifdef NPT
462 auto const npt_flag =
463 force_flag and (system.propagation->integ_switch == INTEG_METHOD_NPT_ISO);
464#else
465 auto constexpr npt_flag = false;
466#endif
467
468 if (p3m.sum_q2 > 0.) {
469 if (not has_actor_of_type<ElectrostaticLayerCorrection>(
470 system.coulomb.impl->solver)) {
471 charge_assign(particles);
472 }
473 /* Gather information for FFT grid inside the nodes domain (inner local
474 * mesh) and perform forward 3D FFT (Charge Assignment Mesh). */
475 p3m.sm.gather_grid(p3m.rs_mesh.data(), comm_cart, p3m.local_mesh.dim);
476 fft_perform_forw(p3m.rs_mesh.data(), p3m.fft, comm_cart);
477 }
478
479 auto p_q_range = ParticlePropertyRange::charge_range(particles);
480 auto p_force_range = ParticlePropertyRange::force_range(particles);
481 auto p_unfolded_pos_range =
482 ParticlePropertyRange::unfolded_pos_range(particles, box_geo);
483
484 // Note: after these calls, the grids are in the order yzx and not xyz
485 // anymore!!!
486 /* The dipole moment is only needed if we don't have metallic boundaries. */
487 auto const box_dipole =
488 (p3m.params.epsilon != P3M_EPSILON_METALLIC)
489 ? std::make_optional(calc_dipole_moment(
490 comm_cart, boost::combine(p_q_range, p_unfolded_pos_range)))
491 : std::nullopt;
492 auto const volume = box_geo.volume();
493 auto const pref =
494 4. * std::numbers::pi / volume / (2. * p3m.params.epsilon + 1.);
495
496 /* === k-space force calculation === */
497 if (force_flag) {
498 /* i*k differentiation */
499 auto constexpr m_start = Utils::Vector3i::broadcast(0);
500 auto const m_stop = std::span(p3m.fft.plan[3].new_mesh);
501 auto const offset = std::span(p3m.fft.plan[3].start);
502 auto const wavevector = (2. * std::numbers::pi) * box_geo.length_inv();
503 auto indices = Utils::Vector3i{};
504 auto index = std::size_t(0u);
505
506 /* compute electric field */
507 // Eq. (3.49) @cite deserno00b
508 for_each_3d(m_start, m_stop, indices, [&]() {
509 auto const rho_hat = std::complex<double>(p3m.rs_mesh[2u * index + 0u],
510 p3m.rs_mesh[2u * index + 1u]);
511 auto const phi_hat = p3m.g_force[index] * rho_hat;
512
513 for (int d = 0; d < 3; d++) {
514 /* direction in r-space: */
515 int d_rs = (d + p3m.ks_pnum) % 3;
516 /* directions */
517 auto const k =
518 p3m.d_op[d_rs][indices[d] + offset[d]] * wavevector[d_rs];
519
520 /* i*k*(Re+i*Im) = - Im*k + i*Re*k (i=sqrt(-1)) */
521 p3m.E_mesh[d_rs][2u * index + 0u] = -k * phi_hat.imag();
522 p3m.E_mesh[d_rs][2u * index + 1u] = +k * phi_hat.real();
523 }
524
525 ++index;
526 });
527
528 /* Back FFT force component mesh */
529 auto const check_complex = !p3m.params.tuning and check_complex_residuals;
530 for (int d = 0; d < 3; d++) {
531 fft_perform_back(p3m.E_mesh[d].data(), check_complex, p3m.fft, comm_cart);
532 }
533
534 /* redistribute force component mesh */
535 std::array<double *, 3> E_fields = {
536 {p3m.E_mesh[0].data(), p3m.E_mesh[1].data(), p3m.E_mesh[2].data()}};
537 p3m.sm.spread_grid(E_fields, comm_cart, p3m.local_mesh.dim);
538
539 auto const force_prefac = prefactor / volume;
540 Utils::integral_parameter<int, AssignForces, 1, 7>(
541 p3m.params.cao, p3m, force_prefac,
542 boost::combine(p_q_range, p_force_range));
543
544 // add dipole forces
545 // Eq. (3.19) @cite deserno00b
546 if (box_dipole) {
547 auto const dm = prefactor * pref * box_dipole.value();
548 for (auto zipped : boost::combine(p_q_range, p_force_range)) {
549 auto p_q = boost::get<0>(zipped);
550 auto &p_force = boost::get<1>(zipped);
551 p_force -= p_q * dm;
552 }
553 }
554 }
555
556 /* === k-space energy calculation === */
557 if (energy_flag or npt_flag) {
558 auto node_energy = 0.;
559 for (int i = 0; i < p3m.fft.plan[3].new_size; i++) {
560 // Use the energy optimized influence function for energy!
561 // Eq. (3.40) @cite deserno00b
562 node_energy += p3m.g_energy[i] * (Utils::sqr(p3m.rs_mesh[2 * i]) +
563 Utils::sqr(p3m.rs_mesh[2 * i + 1]));
564 }
565 node_energy /= 2. * volume;
566
567 auto energy = 0.;
568 boost::mpi::reduce(comm_cart, node_energy, energy, std::plus<>(), 0);
569 if (this_node == 0) {
570 /* self energy correction */
571 // Eq. (3.8) @cite deserno00b
572 energy -= p3m.sum_q2 * p3m.params.alpha * std::numbers::inv_sqrtpi;
573 /* net charge correction */
574 // Eq. (3.11) @cite deserno00b
575 energy -= p3m.square_sum_q * std::numbers::pi /
576 (2. * volume * Utils::sqr(p3m.params.alpha));
577 /* dipole correction */
578 // Eq. (3.9) @cite deserno00b
579 if (box_dipole) {
580 energy += pref * box_dipole.value().norm2();
581 }
582 }
583 energy *= prefactor;
584#ifdef NPT
585 if (npt_flag) {
586 npt_add_virial_contribution(energy);
587 }
588#endif
589 if (not energy_flag) {
590 energy = 0.;
591 }
592 return energy;
593 }
594
595 return 0.;
596}

639 std::tuple<double, double, double, double>
640 calculate_accuracy(Utils::Vector3i const &mesh, int cao,
641 double r_cut_iL) const override {
642
643 auto const &box_geo = *m_system.box_geo;
644 double alpha_L, rs_err, ks_err;
645
646 /* calc maximal real space error for setting */
647 rs_err = p3m_real_space_error(m_prefactor, r_cut_iL, p3m.sum_qpart,
648 p3m.sum_q2, 0., box_geo.length());
649
650 if (std::numbers::sqrt2 * rs_err > p3m.params.accuracy) {
651 /* assume rs_err = ks_err -> rs_err = accuracy/sqrt(2.0) -> alpha_L */
652 alpha_L = sqrt(log(std::numbers::sqrt2 * rs_err / p3m.params.accuracy)) /
653 r_cut_iL;
654 } else {
655 /* even alpha=0 is ok, however, we cannot choose it since it kills the
656 k-space error formula.
657 Anyways, this very likely NOT the optimal solution */
658 alpha_L = 0.1;
659 }
660
661 /* calculate real-space and k-space error for this alpha_L */
662 rs_err = p3m_real_space_error(m_prefactor, r_cut_iL, p3m.sum_qpart,
663 p3m.sum_q2, alpha_L, box_geo.length());
664#ifdef CUDA
665 auto const &solver = m_system.coulomb.impl->solver;
666 if (has_actor_of_type<CoulombP3MGPU>(solver)) {
667 if (this_node == 0) {
668 ks_err =
669 p3m_k_space_error_gpu(m_prefactor, mesh.data(), cao, p3m.sum_qpart,
670 p3m.sum_q2, alpha_L, box_geo.length().data());
671 }
672 boost::mpi::broadcast(comm_cart, ks_err, 0);
673 } else
674#endif
675 ks_err = p3m_k_space_error(m_prefactor, mesh, cao, p3m.sum_qpart,
676 p3m.sum_q2, alpha_L, box_geo.length());
677
678 return {Utils::Vector2d{rs_err, ks_err}.norm(), rs_err, ks_err, alpha_L};
679 }

681 void determine_mesh_limits() override {
682 auto const &box_geo = *m_system.box_geo;
683 auto const mesh_density =
684 static_cast<double>(p3m.params.mesh[0]) * box_geo.length_inv()[0];
685
686 if (p3m.params.mesh == Utils::Vector3i::broadcast(-1)) {
687 /* avoid using more than 1 GB of FFT arrays */
688 auto const normalized_box_dim = std::cbrt(box_geo.volume());
689 auto const max_npart_per_dim = 512.;
690 /* simple heuristic to limit the tried meshes if the accuracy cannot
691 be obtained with smaller meshes, but normally not all these
692 meshes have to be tested */
693 auto const min_npart_per_dim = std::min(
694 max_npart_per_dim, std::cbrt(static_cast<double>(p3m.sum_qpart)));
695 m_mesh_density_min = min_npart_per_dim / normalized_box_dim;
696 m_mesh_density_max = max_npart_per_dim / normalized_box_dim;
697 m_tune_mesh = true;
698 } else {
699 m_mesh_density_min = m_mesh_density_max = mesh_density;
700 assert(p3m.params.mesh[0] >= 1);
701 if (p3m.params.mesh[1] == -1 and p3m.params.mesh[2] == -1) {
702 // determine the two missing values by rescaling by the box length
703 for (auto i : {1, 2}) {
704 p3m.params.mesh[i] =
705 static_cast<int>(std::round(mesh_density * box_geo.length()[i]));
706 // make the mesh even in all directions
707 p3m.params.mesh[i] += p3m.params.mesh[i] % 2;
708 }
709 }
710 m_logger->report_fixed_mesh(p3m.params.mesh);
711 }
712 }
713
714 TuningAlgorithm::Parameters get_time() override {
715 auto const &box_geo = *m_system.box_geo;
716 auto const &solver = m_system.coulomb.impl->solver;
717 auto tuned_params = TuningAlgorithm::Parameters{};
718 auto time_best = time_sentinel;
719 auto mesh_density = m_mesh_density_min;
720 while (mesh_density <= m_mesh_density_max) {
721 auto trial_params = TuningAlgorithm::Parameters{};
722 if (m_tune_mesh) {
723 for (auto i : {0, 1, 2}) {
724 trial_params.mesh[i] =
725 static_cast<int>(std::round(box_geo.length()[i] * mesh_density));
726 // make the mesh even in all directions
727 trial_params.mesh[i] += trial_params.mesh[i] % 2;
728 }
729 } else {
730 trial_params.mesh = p3m.params.mesh;
731 }
732 trial_params.cao = cao_best;
733
734 auto const trial_time =
735 get_m_time(trial_params.mesh, trial_params.cao, trial_params.r_cut_iL,
736 trial_params.alpha_L, trial_params.accuracy);
737
738 if (trial_time >= 0.) {
739 /* the optimum r_cut for this mesh is the upper limit for higher meshes,
740 everything else is slower */
741 if (has_actor_of_type<CoulombP3M>(solver)) {
742 m_r_cut_iL_max = trial_params.r_cut_iL;
743 }
744
745 if (trial_time < time_best) {
746 /* new optimum */
747 reset_n_trials();
748 tuned_params = trial_params;
749 time_best = tuned_params.time = trial_time;
750 } else if (trial_time > time_best + time_granularity or
751 get_n_trials() > max_n_consecutive_trials) {
752 /* no hope of further optimisation */
753 break;
754 }
755 }
756 mesh_density += 0.1;
757 }
758 return tuned_params;
759 }
760};

762void CoulombP3M::tune() {
763 auto &system = get_system();
764 auto const &box_geo = *system.box_geo;
765 if (p3m.params.alpha_L == 0. and p3m.params.alpha != 0.) {
766 p3m.params.alpha_L = p3m.params.alpha * box_geo.length()[0];
767 }
768 if (p3m.params.r_cut_iL == 0. and p3m.params.r_cut != 0.) {
769 p3m.params.r_cut_iL = p3m.params.r_cut * box_geo.length_inv()[0];
770 }
771 if (not is_tuned()) {
772 count_charged_particles();
773 if (p3m.sum_qpart == 0) {
774 throw std::runtime_error(
775 "CoulombP3M: no charged particles in the system");
776 }
777 try {
778 CoulombTuningAlgorithm parameters(system, p3m, prefactor, tune_timings);
779 parameters.setup_logger(tune_verbose);
780 // parameter ranges
781 parameters.determine_mesh_limits();
782 parameters.determine_r_cut_limits();
783 parameters.determine_cao_limits(7);
784 // run tuning algorithm
785 parameters.tune();
786 m_is_tuned = true;
787 system.on_coulomb_change();
788 } catch (...) {
789 p3m.params.tuning = false;
790 throw;
791 }
792 }
793 init();
794}

796void CoulombP3M::sanity_checks_boxl() const {
797 auto const &system = get_system();
798 auto const &box_geo = *system.box_geo;
799 auto const &local_geo = *system.local_geo;
800 for (auto i = 0u; i < 3u; i++) {
801 /* check k-space cutoff */
802 if (p3m.params.cao_cut[i] >= box_geo.length_half()[i]) {
803 std::stringstream msg;
804 msg << "P3M_init: k-space cutoff " << p3m.params.cao_cut[i]
805 << " is larger than half of box dimension " << box_geo.length()[i];
806 throw std::runtime_error(msg.str());
807 }
808 if (p3m.params.cao_cut[i] >= local_geo.length()[i]) {
809 std::stringstream msg;
810 msg << "P3M_init: k-space cutoff " << p3m.params.cao_cut[i]
811 << " is larger than local box dimension " << local_geo.length()[i];
812 throw std::runtime_error(msg.str());
813 }
814 }
815
816 if (p3m.params.epsilon != P3M_EPSILON_METALLIC) {
817 if ((box_geo.length()[0] != box_geo.length()[1]) or
818 (box_geo.length()[1] != box_geo.length()[2]) or
819 (p3m.params.mesh[0] != p3m.params.mesh[1]) or
820 (p3m.params.mesh[1] != p3m.params.mesh[2])) {
821 throw std::runtime_error(
822 "CoulombP3M: non-metallic epsilon requires cubic box");
823 }
824 }
