83 class ParticleCoupling {
84  LBThermostat const &m_thermostat;
85  LB::Solver &m_lb;
86  BoxGeometry const &m_box_geo;
87  LocalBox const &m_local_box;
88  double m_time_step;
89  double m_noise_pref_wo_gamma;
90  bool m_thermalized;
91 
92 public:
93  ParticleCoupling(LBThermostat const &thermostat, LB::Solver &lb,
94  BoxGeometry const &box_geo, LocalBox const &local_box,
95  double time_step)
96  : m_thermostat{thermostat}, m_lb{lb}, m_box_geo{box_geo},
97  m_local_box{local_box}, m_time_step{time_step} {
98  /* Eq. (16) @cite ahlrichs99a, without the gamma term.
99  * The factor 12 comes from the fact that we use random numbers
100  * from -0.5 to 0.5 (equally distributed) which have variance 1/12.
101  * The time step comes from the discretization.
102  */
103  auto constexpr variance_inv = 12.;
104  auto const kT = lb.get_kT() * Utils::sqr(lb.get_lattice_speed());
105  m_thermalized = (kT != 0.);
106  m_noise_pref_wo_gamma = std::sqrt(variance_inv * 2. * kT / time_step);
107  }

125 /**
126  * @brief Keep track of ghost particles that have already been coupled once.
127  * In certain cases, there may be more than one ghost for the same particle.
128  * To make sure these are only coupled once, ghosts' ids are recorded.
129  */
130 class CouplingBookkeeping {
131  std::unordered_set<int> m_coupled_ghosts;
132  CellStructure const &m_cell_structure;
133 
134  /** @brief Check if there is locally a real particle for the given ghost. */
135  bool is_ghost_for_local_particle(Particle const &p) const {
136  return not m_cell_structure.get_local_particle(p.id())->is_ghost();
137  }
138 
139 public:
140  explicit CouplingBookkeeping(CellStructure const &cell_structure)
141  : m_cell_structure{cell_structure} {}
142 
143  /** @brief Determine if a given particle should be coupled. */
144  bool should_be_coupled(Particle const &p) {
145  auto const propagation = p.propagation();
146  if ((propagation & PropagationMode::TRANS_LB_MOMENTUM_EXCHANGE) == 0 and
147  (propagation & PropagationMode::SYSTEM_DEFAULT) == 0 and
148  (propagation & PropagationMode::TRANS_LB_TRACER) == 0) {
149  return false;
150  }
151 #ifdef VIRTUAL_SITES_RELATIVE
152  if ((propagation & PropagationMode::TRANS_LB_MOMENTUM_EXCHANGE) == 0 and
153  propagation & (PropagationMode::TRANS_VS_RELATIVE |
154  PropagationMode::ROT_VS_RELATIVE)) {
155  return false;
156  }
157 #endif
158  // real particles: always couple
159  if (not p.is_ghost()) {
160  return true;
161  }
162  // ghosts: check we don't have the corresponding real particle on the same
163  // node, and that a ghost for the same particle hasn't been coupled already
164  if (m_coupled_ghosts.count(p.id()) != 0 or is_ghost_for_local_particle(p)) {
165  return false;
166  }
167  m_coupled_ghosts.insert(p.id());
168  return true;
169  }
170 };
