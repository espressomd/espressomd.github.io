32template <class MemorySpace, class AlgorithmTag, class LayoutTag,
33 class BuildTag = Cabana::TeamVectorOpTag>
34class CustomVerletList : public Cabana::VerletList<MemorySpace, AlgorithmTag,
35 LayoutTag, BuildTag> {
36public:
37 CustomVerletList() = default;
38 CustomVerletList(std::size_t const begin, std::size_t const end,
39 std::size_t const max_neigh) {
40 initializeData(end - begin, max_neigh);
41 }
42
43 Kokkos::View<int *, MemorySpace> counts;
44 Kokkos::View<int **, Kokkos::LayoutRight, MemorySpace> neighbors;
45
46 // Method to initialize _data without filling neighbors
47 KOKKOS_INLINE_FUNCTION
48 void initializeData(std::size_t const num_particles,
49 std::size_t const max_neigh) {
50 counts = Kokkos::View<int *, MemorySpace>("num_neighbors", num_particles);
51 neighbors = Kokkos::View<int **, Kokkos::LayoutRight, MemorySpace>(
52 Kokkos::ViewAllocateWithoutInitializing("neighbors"), num_particles,
53 max_neigh);
54 }
55
56 // Method to add a neighbor
57 KOKKOS_INLINE_FUNCTION
58 void addNeighborAtomicLB(int pid, int nid) {
59 auto count = counts(pid);
60 auto count_n = counts(nid);
61
62 if (count > count_n) {
63 std::swap(pid, nid);
64 }
65 count = Kokkos::atomic_fetch_add(&counts(pid), 1);
66 assert(count < neighbors.extent(1));
67 neighbors(pid, count) = nid;
68 }
69
70 // Thread safe but non atomic method to add a neighbor
71 KOKKOS_INLINE_FUNCTION
72 void addNeighbor(int pid, int nid) {
73 auto const count = counts(pid);
74
75 assert(count < neighbors.extent(1));
76 neighbors(pid, count) = nid;
77 counts(pid) += 1;
78 }
79
80 // Non atomic and load balancing method to add a neighbor
81 KOKKOS_INLINE_FUNCTION
82 void addNeighborLB(int pid, int nid) {
83 auto count = counts(pid);
84 auto count_n = counts(nid);
85
86 if (count > count_n) {
87 std::swap(pid, nid);
88 count = counts(pid);
89 }
90 assert(count < neighbors.extent(1));
91 neighbors(pid, count) = nid;
92 counts(pid) += 1;
93 }
94
95 // Sorting a neighbor
96 KOKKOS_INLINE_FUNCTION
97 void sortNeighbors() {
98 Kokkos::parallel_for(
99 "custom_verlet_list::sort_neighbors",
100 Kokkos::RangePolicy<Kokkos::DefaultExecutionSpace>(0, counts.size()),
101 [&](const int i) {
102 const int count = counts(i);
103 auto *ptr = &neighbors(i, 0);
104 std::sort(ptr, ptr + count);
105 });
106 Kokkos::fence();
107 }
108
109 // Find max counts
110 KOKKOS_INLINE_FUNCTION
111 auto get_variance_max_counts(auto &ostream) {
112 auto max_counts = 0l;
113 auto ave_counts = 0l;
114 auto ave_sq_counts = 0l;
115 for (int pid = 0; pid < counts.extent(0); ++pid) {
116 auto const count = static_cast<long>(counts(pid));
117 if (max_counts < count)
118 max_counts = count;
119 ave_counts += count;
120 ave_sq_counts += count * count;
121 }
122 if (counts.extent(0) != 0) {
123 ave_counts /= static_cast<long>(counts.extent(0));
124 ave_sq_counts /= static_cast<long>(counts.extent(0));
125 ave_sq_counts -= ave_counts * ave_counts;
126 }
127 ostream << "max:" << max_counts << " ave:" << ave_counts
128 << " var:" << ave_sq_counts << std::endl;
129 return static_cast<int>(max_counts);
130 }
131
132 KOKKOS_INLINE_FUNCTION
133 auto get_max_counts() {
134 int max_counts;
135 Kokkos::Max<int> max_reduce(max_counts);
136 Kokkos::parallel_reduce(
137 "custom_verlet_list::reduce_max",
138 Kokkos::RangePolicy<Kokkos::DefaultExecutionSpace>(0, counts.size()),
139 [&](const int i, int &value) {
140 if (counts(i) > value)
141 value = counts(i);
142 },
143 max_reduce);
144 Kokkos::fence();
145 return max_counts;
146 }
147};

149template <class MemorySpace, class AlgorithmTag, class BuildTag>
150class Cabana::NeighborList<CustomVerletList<MemorySpace, AlgorithmTag,
151 Cabana::VerletLayout2D, BuildTag>> {
152public:
153 //! Kokkos memory space.
154 using memory_space = MemorySpace;
155 //! Neighbor list type.
156 using list_type = CustomVerletList<MemorySpace, AlgorithmTag,
157 Cabana::VerletLayout2D, BuildTag>;
158
159 //! Get the total number of neighbors across all particles.
160 KOKKOS_INLINE_FUNCTION
161 static std::size_t totalNeighbor(list_type const &list) {
162 std::size_t const num_p = list.counts.size();
163 std::size_t total_n = 0;
164 for (std::size_t i = 0; i < num_p; ++i)
165 total_n += list.counts(i);
166 return total_n;
167 }
168
169 //! Get the maximum number of neighbors per particle.
170 KOKKOS_INLINE_FUNCTION
171 static std::size_t maxNeighbor(list_type const &list) {
172 // Stored during neighbor search.
173 return list.max_n;
174 }
175
176 //! Get the number of neighbors for a given particle index.
177 KOKKOS_INLINE_FUNCTION
178 static std::size_t numNeighbor(list_type const &list,
179 std::size_t const particle_index) {
180 return list.counts(particle_index);
181 }
182
183 //! Get the id for a neighbor for a given particle index and the index of
184 //! the neighbor relative to the particle.
185 KOKKOS_INLINE_FUNCTION
186 static std::size_t getNeighbor(list_type const &list,
187 std::size_t const particle_index,
188 std::size_t const count) {
189 return list.neighbors(particle_index, count);
190 }
191};
