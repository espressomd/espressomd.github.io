36 template <class SIClass, class CoreClass> Actor<SIClass, CoreClass>::Actor() {
37  add_parameters({
38  {"prefactor", AutoParameter::read_only,
39  [this]() { return m_actor->prefactor; }},
40  {"check_neutrality",
41  [this](Variant const &value) {
42  auto const flag = get_value<bool>(value);
43  auto &tolerance = m_actor->charge_neutrality_tolerance;
44  if (flag) {
45  if (tolerance == -1.) {
46  tolerance = m_actor->charge_neutrality_tolerance_default;
47  }
48  } else {
49  tolerance = -1.;
50  }
51  },
52  [this]() {
53  auto const tolerance = m_actor->charge_neutrality_tolerance;
54  return Variant{tolerance != -1.};
55  }},
56  {"charge_neutrality_tolerance",
57  [this](Variant const &value) {
58  auto &tolerance = m_actor->charge_neutrality_tolerance;
59  if (is_none(value)) {
60  tolerance = -1.;
61  } else {
62  auto const new_tolerance = get_value<double>(value);
63  if (new_tolerance < 0.) {
64  if (context()->is_head_node()) {
65  throw std::domain_error(
66  "Parameter 'charge_neutrality_tolerance' must be >= 0");
67  }
68  throw Exception("");
69  }
70  tolerance = new_tolerance;
71  }
72  },
73  [this]() {
74  auto const tolerance = m_actor->charge_neutrality_tolerance;
75  if (tolerance == -1.) {
76  return make_variant(none);
77  }
78  return Variant{tolerance};
79  }},
80  });
81 }
