57 double ShapeBasedConstraint::min_dist(BoxGeometry const &box_geo,
58  ParticleRange const &particles) const {
59  auto global_mindist = std::numeric_limits<double>::infinity();
60 
61  auto const local_mindist = std::accumulate(
62  particles.begin(), particles.end(),
63  std::numeric_limits<double>::infinity(),
64  [this, &box_geo](double min, Particle const &p) {
65  auto const &ia_params = get_ia_param(p.type());
66  if (is_active(ia_params)) {
67  double dist;
68  Utils::Vector3d vec;
69  m_shape->calculate_dist(box_geo.folded_position(p.pos()), dist, vec);
70  return std::min(min, dist);
71  }
72  return min;
73  });
74  boost::mpi::reduce(comm_cart, local_mindist, global_mindist,
75  boost::mpi::minimum<double>(), 0);
76  return global_mindist;
77 }

79 ParticleForce ShapeBasedConstraint::force(Particle const &p,
80  Utils::Vector3d const &folded_pos,
81  double) {
82  ParticleForce pf{};
83  auto const &ia_params = get_ia_param(p.type());
84 
85  if (is_active(ia_params)) {
86  double dist = 0.;
87  Utils::Vector3d dist_vec;
88  m_shape->calculate_dist(folded_pos, dist, dist_vec);
89  auto const coulomb_kernel = m_system.coulomb.pair_force_kernel();
90 
91 #ifdef DPD
92  Utils::Vector3d dpd_force{};
93 #endif
94  Utils::Vector3d outer_normal_vec{};
95 
96  if (dist > 0) {
97  outer_normal_vec = -dist_vec / dist;
98  pf = calc_central_radial_force(p, part_rep, ia_params, dist_vec, dist) +
99  calc_central_radial_charge_force(p, part_rep, ia_params, dist_vec,
100  dist, get_ptr(coulomb_kernel)) +
101  calc_non_central_force(p, part_rep, ia_params, dist_vec, dist);
102 
103 #ifdef DPD
104  if (m_system.thermostat->thermo_switch & THERMO_DPD) {
105  dpd_force = dpd_pair_force(p, part_rep, *m_system.thermostat->dpd,
106  *m_system.box_geo, ia_params, dist_vec, dist,
107  dist * dist);
108  // Additional use of DPD here requires counter increase
109  m_system.thermostat->dpd->rng_increment();
110  }
111 #endif
112  } else if (m_penetrable && (dist <= 0)) {
113  if ((!m_only_positive) && (dist < 0)) {
114  pf =
115  calc_central_radial_force(p, part_rep, ia_params, dist_vec, -dist) +
116  calc_central_radial_charge_force(p, part_rep, ia_params, dist_vec,
117  -dist, get_ptr(coulomb_kernel)) +
118  calc_non_central_force(p, part_rep, ia_params, dist_vec, -dist);
119 
120 #ifdef DPD
121  if (m_system.thermostat->thermo_switch & THERMO_DPD) {
122  dpd_force = dpd_pair_force(p, part_rep, *m_system.thermostat->dpd,
123  *m_system.box_geo, ia_params, dist_vec,
124  dist, dist * dist);
125  // Additional use of DPD here requires counter increase
126  m_system.thermostat->dpd->rng_increment();
127  }
128 #endif
129  }
130  } else {
131  runtimeErrorMsg() << "Constraint violated by particle " << p.id()
132  << " dist " << dist;
133  }
134 
135 #ifdef ROTATION
136  part_rep.torque() += calc_opposing_force(pf, dist_vec).torque;
137 #endif
138 #ifdef DPD
139  pf.f += dpd_force;
140 #endif
141  m_local_force -= pf.f;
142  m_outer_normal_force -= outer_normal_vec * pf.f;
143  }
144  return pf;
145 }

147 void ShapeBasedConstraint::add_energy(const Particle &p,
148  const Utils::Vector3d &folded_pos, double,
149  Observable_stat &obs_energy) const {
150  double energy = 0.0;
151  auto const &ia_params = get_ia_param(p.type());
152 
153  if (is_active(ia_params)) {
154  auto const coulomb_kernel = m_system.coulomb.pair_energy_kernel();
155  double dist = 0.0;
156  Utils::Vector3d vec;
157  m_shape->calculate_dist(folded_pos, dist, vec);
158  if (dist > 0) {
159  energy = calc_non_bonded_pair_energy(p, part_rep, ia_params, vec, dist,
160  get_ptr(coulomb_kernel));
161  } else if ((dist <= 0) && m_penetrable) {
162  if (!m_only_positive && (dist < 0)) {
163  energy = calc_non_bonded_pair_energy(p, part_rep, ia_params, vec, -dist,
164  get_ptr(coulomb_kernel));
165  }
166  } else {
167  runtimeErrorMsg() << "Constraint violated by particle " << p.id();
168  }
169  }
170  // NOLINTNEXTLINE(clang-analyzer-cplusplus.NewDeleteLeaks)
171  if (part_rep.type() >= 0) {
172  obs_energy.add_non_bonded_contribution(
173  p.type(), part_rep.type(), p.mol_id(), part_rep.mol_id(), energy);
174  }
175 }
