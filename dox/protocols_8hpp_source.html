38 /** Lees-Edwards protocol for linear shearing */
39 struct LinearShear {
40  LinearShear()
41  : m_initial_pos_offset{0.}, m_shear_velocity{0.}, m_time_0{0.} {}
42  LinearShear(double initial_offset, double shear_velocity, double time_0)
43  : m_initial_pos_offset{initial_offset},
44  m_shear_velocity{shear_velocity}, m_time_0{time_0} {}
45  double shear_velocity(double time) const { return m_shear_velocity; }
46  double pos_offset(double time) const {
47  return m_initial_pos_offset + (time - m_time_0) * m_shear_velocity;
48  }
49  double m_initial_pos_offset;
50  double m_shear_velocity;
51  double m_time_0;
52 };

54 /** Lees-Edwards protocol for oscillatory shearing */
55 struct OscillatoryShear {
56  OscillatoryShear()
57  : m_initial_pos_offset{0.}, m_amplitude{0.}, m_omega{0.}, m_time_0{0.} {}
58  OscillatoryShear(double initial_offset, double amplitude, double omega,
59  double time_0)
60  : m_initial_pos_offset{initial_offset},
61  m_amplitude{amplitude}, m_omega{omega}, m_time_0{time_0} {}
62  double pos_offset(double time) const {
63  return m_initial_pos_offset +
64  m_amplitude * std::sin(m_omega * (time - m_time_0));
65  }
66  double shear_velocity(double time) const {
67  return m_omega * m_amplitude * std::cos(m_omega * (time - m_time_0));
68  }
69  double m_initial_pos_offset;
70  double m_amplitude;
71  double m_omega;
72  double m_time_0;
73 };
