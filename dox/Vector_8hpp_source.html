78 void swap(Vector &rhs) { std::ranges::swap_ranges(*this, rhs); }

80private:
81 constexpr void copy_init(T const *data) noexcept {
82 auto it = begin();
83 for (std::size_t i{0}; i != N; ++i) {
84 *it++ = data[i];
85 }
86 }
87
88public:
89 // range-based ctor that excludes Vector<T,N> to avoid ambiguous calls with
90 // the copy ctor, move ctor and cast operator; std::ranges::input_range is
91 // not used due to conflicts with move assignment in recursive variant types
92 template <class Range>
93 requires(not is_vector<std::remove_cvref_t<Range>>::value)
94 explicit constexpr Vector(Range &&rng)
95 : Vector(std::begin(rng), std::end(rng)) {}

97#if __cpp_lib_containers_ranges
98 template <std::ranges::input_range Range>
99 Vector(std::from_range_t, Range &&rng)
100 : Vector(std::begin(rng), std::end(rng)) {}
101#endif

103 explicit constexpr Vector(T const (&v)[N]) noexcept : Base() {
104 copy_init(std::begin(v));
105 }
106
107 constexpr Vector(std::initializer_list<T> v) : Base() {
108 if (N != v.size()) {
109 throw std::length_error(
110 "Construction of Vector from Container of wrong length.");
111 }
112 copy_init(v.begin());
113 }
114
115 template <typename InputIterator>
116 Vector(InputIterator first, InputIterator last) : Base() {
117 if (std::distance(first, last) == N) {
118 std::copy_n(first, N, begin());
119 } else {
120 throw std::length_error(
121 "Construction of Vector from Container of wrong length.");
122 }
123 }

125 /** @brief Create a vector that has all entries set to the same value. */
126 DEVICE_QUALIFIER static constexpr Vector<T, N>
127 broadcast(typename Base::value_type const &value) noexcept {
128 Vector<T, N> ret;
129 for (std::size_t i = 0u; i != N; ++i) {
130 ret[i] = value;
131 }
132 return ret;
133 }

145 template <class U> explicit operator Vector<U, N>() const {
146 Vector<U, N> ret;
147
148 std::ranges::transform(*this, ret.begin(),
149 [](T const &e) { return static_cast<U>(e); });

154 constexpr T norm2() const { return (*this) * (*this); }
155 T norm() const { return std::sqrt(norm2()); }
156
157 /*
158 * @brief Normalize the vector.
159 *
160 * Normalize the vector by its length,
161 * if not zero, otherwise the vector is unchanged.
162 */
163 Vector &normalize() {
164 auto const l = norm();
165 if (l != T(0)) {
166 for (std::size_t i = 0u; i < N; ++i)
167 this->operator[](i) /= l;
168 }
169
170 return *this;
171 }
172
173 Vector normalized() const { return (*this) / (*this).norm(); }

192template <std::size_t N, typename T, typename U, typename Op>
193auto binary_op(Vector<T, N> const &a, Vector<U, N> const &b, Op op) {
194 // we must use the non-range version std::transform for Vector<T, N> because:
195 // GCC 12 cannot inspect -Wmaybe-uninitialized through std::ranges::transform
196 // Clang/Xcode libc++ cannot select the binary form of std::ranges::transform
197 using R = decltype(std::declval<T>() + std::declval<U>());
198 Vector<R, N> ret;
199
200 // NOLINTNEXTLINE(modernize-use-ranges)
201 std::transform(std::begin(a), std::end(a), std::begin(b), std::begin(ret),
202 op);
203
204 return ret;
205}

207template <std::size_t N, typename T, typename Op>
208constexpr bool all_of(Vector<T, N> const &a, Vector<T, N> const &b, Op op) {
209 for (std::size_t i = 0u; i < N; ++i) {
210 if (not op(a[i], b[i])) {
211 return false;
212 }
213 }
214 return true;
215}

329/* Scalar product */
330template <std::size_t N, typename T, class U>
331 requires(not(detail::is_vector<T>::value or detail::is_vector<U>::value))
332auto constexpr operator*(Vector<T, N> const &a, Vector<U, N> const &b) {
333 using R = decltype(std::declval<T>() * std::declval<U>());
334 // std::inner_product isn't always inlined on Intel CPUs even with -O3,
335 // but a for loop can be inlined
336 R acc{};
337 for (std::size_t i = 0u; i < N; ++i) {
338 acc += a[i] * b[i];
339 }
340 return acc;
341}

363 return {a[1] * b[2] - a[2] * b[1], a[2] * b[0] - a[0] * b[2],

372template <class T, class U, std::size_t N>
373auto hadamard_product(Vector<T, N> const &a, Vector<U, N> const &b) {
374 using R = decltype(std::declval<T>() * std::declval<U>());
375 auto constexpr proj = std::identity{}; // required by Clang/Xcode libc++

406template <typename T> Vector<T, 3> unit_vector(unsigned int i) {
407 if (i == 0u)
