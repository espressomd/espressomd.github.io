130 std::string serialize_parameters(Variant const &pack) {
131  auto const parameters = boost::apply_visitor(GetParameterList{}, pack);
132  if (parameters.empty()) {
133  throw std::invalid_argument(
134  "ScaFaCoS methods require at least 1 parameter");
135  }
136  auto const visitor = ConvertToStringVector{};
137  std::string method_params = "";
138  for (auto const &kv : parameters) {
139  method_params += "," + kv.first;
140  for (auto const &value : boost::apply_visitor(visitor, kv.second)) {
141  method_params += "," + value;
142  }
143  }
144  return method_params.substr(1);
145 }

158 std::unordered_map<std::string, Variant>
159 deserialize_parameters(std::string const &parameters) {
160  /*
161  * ScaFaCoS parameters are serialized to a comma-separated string.
162  * Key-value pairs can be split with a look ahead: when the next
163  * item is a string, it is a parameter name and the current list
164  * of arithmetic values belong to the current parameter name.
165  * The only exception is string-valued parameters; in that case
166  * the current list of arithmetic values is empty.
167  */
168  auto const numbers = std::string("-0123456789");
169  std::unordered_map<std::string, Variant> method_params{};
170  std::vector<std::string> flat_array;
171  // Clang 10 false positive: https://github.com/boostorg/algorithm/issues/63
172  // NOLINTNEXTLINE(clang-analyzer-cplusplus.NewDeleteLeaks)
173  boost::split(flat_array, parameters, boost::is_any_of(","));
174  for (auto it = flat_array.begin(); it != flat_array.end();) {
175  auto const parameter_name = *it;
176  auto parameter_list = std::vector<Variant>{};
177  for (++it; it != flat_array.end(); ++it) {
178  if ((numbers.find(it->front()) == std::string::npos) and
179  not parameter_list.empty()) {
180  break;
181  }
182  auto result = Variant{*it};
183  if (auto converted = string_to_number<int>(*it)) {
184  result = Variant{*converted};
185  } else if (auto converted = string_to_number<double>(*it)) {
186  result = Variant{*converted};
187  }
188  parameter_list.emplace_back(result);
189  }
190  assert(not parameter_list.empty());
191  if (parameter_list.size() == 1ul) {
192  method_params[parameter_name] = parameter_list.front();
193  } else {
194  method_params[parameter_name] = Variant{std::move(parameter_list)};
195  }
196  }
197  return method_params;
198 }
