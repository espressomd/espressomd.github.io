52struct ConvertToStringVector {
53 using result_type = std::vector<std::string>;
54
55 auto operator()(std::string const &value) const { return result_type{value}; }
56
57 template <typename T> result_type operator()(T const &value) const {
58 if constexpr (std::is_arithmetic_v<T>) {
59 return operator()(to_str(value));
60 }
61 throw std::runtime_error("Cannot convert " + Utils::demangle<T>());
62 }
63
64 auto operator()(result_type const &values) const { return values; }
65
66 template <typename T> auto operator()(std::vector<T> const &values) const {
67 result_type values_str;
68 for (auto const &v : values) {
69 if constexpr (std::is_same_v<T, Variant>) {
70 values_str.emplace_back(std::visit(*this, v).front());
71 } else {
72 values_str.emplace_back(to_str(v));
73 }
74 }
75 return values_str;
76 }
77
78private:
79 template <typename T>
80 requires std::is_arithmetic_v<T>
81 std::string to_str(T const &value) const {
82 std::ostringstream serializer;
83 if constexpr (std::is_floating_point_v<T>) {
84 serializer << std::scientific << std::setprecision(17);
85 }
86 serializer << value;
87 return serializer.str();
88 }
89};

112std::string serialize_parameters(Variant const &pack) {
113 auto const parameters = std::visit(GetParameterList(), pack);
114 if (parameters.empty()) {
115 throw std::invalid_argument(
116 "ScaFaCoS methods require at least 1 parameter");
117 }
118 auto const visitor = ConvertToStringVector();
119 std::string method_params = "";
120 for (auto const &[name, values] : parameters) {
121 method_params += "," + name;
122 for (auto const &value : std::visit(visitor, values)) {
123 method_params += "," + value;
124 }
125 }
126 return method_params.substr(1);
127}

129template <typename T>
130std::optional<Variant> string_to_number(std::string const &s) {
131 auto deserializer = std::istringstream(s);
132 T result;
133 deserializer >> result;
134 if (deserializer.fail() or not deserializer.eof()) {
135 return {};
136 }
137 return Variant{result};
138}

140std::unordered_map<std::string, Variant>
141deserialize_parameters(std::string const &parameters) {
142 /*
143 * ScaFaCoS parameters are serialized to a comma-separated string.
144 * Key-value pairs can be split with a look ahead: when the next
145 * item is a string, it is a parameter name and the current list
146 * of arithmetic values belong to the current parameter name.
147 * The only exception is string-valued parameters; in that case
148 * the current list of arithmetic values is empty.
149 */
150 auto const numbers = std::string("-0123456789");
151 std::unordered_map<std::string, Variant> method_params{};
152 std::vector<std::string> flat_array;
153 std::istringstream buffer;
154 buffer.str(parameters);
155 for (std::string line; std::getline(buffer, line, ',');) {
156 flat_array.emplace_back(line);
157 }
158 for (auto it = flat_array.begin(); it != flat_array.end();) {
159 auto const parameter_name = *it;
160 auto parameter_list = std::vector<Variant>{};
161 for (++it; it != flat_array.end(); ++it) {
162 if ((numbers.find(it->front()) == std::string::npos) and
163 not parameter_list.empty()) {
164 break;
165 }
166 auto result = Variant{*it};
167 if (auto converted = string_to_number<int>(*it)) {
168 result = Variant{*converted};
169 } else if (auto converted = string_to_number<double>(*it)) {
170 result = Variant{*converted};
171 }
172 parameter_list.emplace_back(result);
173 }
174 assert(not parameter_list.empty());
175 if (parameter_list.size() == 1ul) {
176 method_params[parameter_name] = parameter_list.front();
177 } else {
178 method_params[parameter_name] = Variant{std::move(parameter_list)};
179 }
180 }
181 return method_params;
182}
