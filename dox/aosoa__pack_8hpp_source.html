56 void resize(std::size_t num_particles) {
57 if (position.extent(0) == 0) {
58 // First allocation
59 position = PositionViewType("position", num_particles);
60#ifdef ESPRESSO_ELECTROSTATICS
61 charge = ChargeViewType("charge", num_particles);
62#endif
63 id = IdViewType("id", num_particles);
64 type = TypeViewType("type", num_particles);
65#ifdef ESPRESSO_DPD
66 velocity = PositionViewType("velocity", num_particles);
67#endif
68#if defined(ESPRESSO_GAY_BERNE) or defined(ESPRESSO_DIPOLES)
69 director = DirectorViewType("director", num_particles);
70#endif
71#ifdef ESPRESSO_DIPOLES
72 dipm = DipmViewType("dipm", num_particles);
73#endif
74 } else {
75 // Reallocation
76 Kokkos::realloc(position, num_particles);
77#ifdef ESPRESSO_ELECTROSTATICS
78 Kokkos::realloc(charge, num_particles);
79#endif
80 Kokkos::realloc(id, num_particles);
81 Kokkos::realloc(type, num_particles);
82#ifdef ESPRESSO_DPD
83 Kokkos::realloc(velocity, num_particles);
84#endif
85#if defined(ESPRESSO_GAY_BERNE) or defined(ESPRESSO_DIPOLES)
86 Kokkos::realloc(director, num_particles);
87#endif
88#ifdef ESPRESSO_DIPOLES
89 Kokkos::realloc(dipm, num_particles);
90#endif
91 }
92 }
