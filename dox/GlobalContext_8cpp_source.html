20/** @file
21 *
22 * Infrastructure to synchronize objects created on the head node with
23 * their corresponding remote copies. Methods of script interface
24 * classes may throw exceptions of type @ref ScriptInterface::Exception.
25 * These exceptions will halt the flow of the program on the head node.
26 * The same exceptions will be thrown in the remote copies but will
27 * be silenced, since they are redundant. Other types of exceptions
28 * are not silenced.
29 *
30 * Implementation of @ref GlobalContext.hpp.
31 */

91std::shared_ptr<ObjectHandle>
92GlobalContext::make_shared(std::string const &name,
93 const VariantMap &parameters) {
94 assert(is_head_node());
95 auto sp = m_node_local_context->factory().make(name);
96 set_context(sp.get());
97
98 auto const id = ObjectId(sp.get());
99 remote_make_handle(id, name, parameters);
100
101 sp->construct(parameters);
102
103 return {sp.release(),
104 /* Custom deleter, we keep the corresponding global context,
105 * as well as the original deleter for the object. */
106 [global_context = this, deleter = sp.get_deleter()](ObjectHandle *o) {
107 /* Tell the other nodes before invoking the destructor, this is
108 * required
109 * to have synchronous destructors, which is needed by some client
110 * code. */
111 global_context->cb_delete_handle(ObjectId(o));
112
113 /* Locally destroy the object. */
114 deleter(o);
115 }};
116}
