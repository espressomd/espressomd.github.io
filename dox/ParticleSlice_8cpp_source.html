45static void set_particles_bonds(
46 std::vector<int> const &pids,
47 std::vector<std::vector<int>> const &all_bonds_ids,
48 std::vector<std::vector<std::vector<int>>> const &all_bonds_partner_ids,
49 ::CellStructure &cell_structure, ::System::System &system) {
50 for (std::size_t i = 0; i < pids.size(); ++i) {
51 auto const pid = pids[i];
52 auto const bonds_ids = all_bonds_ids[i];
53 auto const bonds_partner_ids = all_bonds_partner_ids[i];
54 // Remove old bonds
55 auto p = cell_structure.get_local_particle(pid);
56 if (p != nullptr and not p->is_ghost()) {
57 p->bonds().clear();
58 }
59 // Add new bonds
60 for (std::size_t j = 0; j < bonds_ids.size(); ++j) {
61 std::vector<int> particle_ids = {pid};
62 std::ranges::copy(bonds_partner_ids[j], std::back_inserter(particle_ids));
63 ::add_bond(system, bonds_ids[j], particle_ids);
64 system.on_particle_change();
65 }
66 }
67}

69#ifdef ESPRESSO_EXCLUSIONS
70static void
71set_particles_exclusions(std::vector<int> const &pids,
72 std::vector<std::vector<int>> const &exclusion_lists,
73 boost::mpi::communicator const &comm, Context *context,
74 ::CellStructure &cell_structure,
75 ::System::System &system) {
76 for (std::size_t i = 0; i < pids.size(); ++i) {
77 auto const pid = pids[i];
78 auto const &exclusion_list = exclusion_lists[i];
79 for (auto const excluded_pid : exclusion_list) { // collective communication
80 particle_exclusion_sanity_checks(pid, excluded_pid, cell_structure, comm);
81 }
82 auto p = cell_structure.get_local_particle(pid);
83 if (p != nullptr and not p->is_ghost()) {
84 // Remove all excluded ids of this particle
85 for (auto const old_excluded_pid : p->exclusions()) {
86 local_remove_exclusion(pid, old_excluded_pid, cell_structure);
87 }
88 // Add new excluded ids for this particle
89 for (auto const excluded_pid : exclusion_list) {
90 if (not p->has_exclusion(excluded_pid)) {
91 local_add_exclusion(pid, excluded_pid, cell_structure);
92 }
93 }
94 }
95 }
96 system.on_particle_change();
97}
98#endif // ESPRESSO_EXCLUSIONS

111static void set_particles_types(std::vector<int> const &pids,
112 std::vector<int> const &types,
113 CellStructure &cell_structure,
114 ::System::System &system) {
115 for (std::size_t i = 0; i < pids.size(); ++i) {
116 auto const pid = pids[i];
117 auto p = cell_structure.get_local_particle(pid);
118 if (p != nullptr and not p->is_ghost()) {
119 auto const old_type = p->type();
120 auto const &new_type = types[i];
121 if (new_type < 0) {
122 throw std::domain_error(error_msg("type", "must be an integer >= 0"));
123 }
124 system.nonbonded_ias->make_particle_type_exist(new_type);
125 on_particle_type_change(pid, old_type, new_type);
126 p->type() = new_type;
127 }
128 }
129}

147void ParticleSlice::do_construct(VariantMap const &params) {
148 if (params.contains("__cell_structure")) {
149 auto so = get_value<std::shared_ptr<CellSystem::CellSystem>>(
150 params, "__cell_structure");
151 so->configure(*this);
152 m_cell_structure = so;
153 }
154 if (params.contains("__bonded_ias")) {
155 m_bonded_ias = get_value<std::shared_ptr<Interactions::BondedInteractions>>(
156 params, "__bonded_ias");
157 }
158 m_id_selection = get_value<std::vector<int>>(params, "id_selection");
159 m_chunk_size = get_value_or<int>(params, "prefetch_chunk_size", 10000);
160 if (not context()->is_head_node()) {
161 return;
162 }
163 for (auto const pid : m_id_selection) {
164 if (not particle_exists(pid)) {
165 throw std::out_of_range("Particle does not exist: " +
166 std::to_string(pid));
167 }
168 }
169}

171Variant ParticleSlice::do_call_method(std::string const &name,
172 VariantMap const &params) {
173 if (name == "set_param_parallel") {
174 auto const param_name = get_value<std::string>(params, "name");
175 if (not params.contains("values")) {
176 context()->parallel_try_catch([&]() {
177 if (param_name == "bonds") {
178 if (not params.contains("all_bonds_ids")) {
179 throw Exception("Parameter 'all_bonds_ids' is missing");
180 }
181 if (not params.contains("all_bonds_partner_ids")) {
182 throw Exception("Parameter 'all_bonds_partner_ids' is missing");
183 }
184 } else {
185 throw Exception("Parameter 'values' is missing");
186 }
187 });
188 }
189 // Handle parameters with special setters
190 if (m_special_parameters.contains(param_name)) {
191 context()->parallel_try_catch([&]() {
192 if (param_name == "pos") {
193 set_particles_positions(
194 m_id_selection,
195 get_value<std::vector<Utils::Vector3d>>(params, "values"));
196 } else if (param_name == "type") {
197 set_particles_types(m_id_selection,
198 get_value<std::vector<int>>(params, "values"),
199 *get_cell_structure(), *get_system());
200 }
201#ifdef ESPRESSO_ELECTROSTATICS
202 else if (param_name == "q") {
203 std::vector<double> charges;
204 if (is_type<std::vector<int>>(params.at("values"))) {
205 auto tmp = get_value<std::vector<int>>(params, "values");
206 charges = std::vector<double>(tmp.begin(), tmp.end());
207 } else {
208 charges = get_value<std::vector<double>>(params, "values");
209 }
210 set_particles_charges(m_id_selection, charges, *get_cell_structure(),
211 *get_system());
212
213 }
214#endif // ESPRESSO_ELECTROSTATICS
215#ifdef ESPRESSO_EXCLUSIONS
216 else if (param_name == "exclusions") {
217 auto const excluded_pids =
218 get_value<std::vector<std::vector<int>>>(params, "values");
219 set_particles_exclusions(
220 m_id_selection,
221 get_value<std::vector<std::vector<int>>>(params, "values"),
222 context()->get_comm(), context(), *get_cell_structure(),
223 *get_system());
224 }
225#endif // ESPRESSO_EXCLUSIONS
226 else if (param_name == "bonds") {
227 set_particles_bonds(
228 m_id_selection,
229 get_value<std::vector<std::vector<int>>>(params, "all_bonds_ids"),
230 get_value<std::vector<std::vector<std::vector<int>>>>(
231 params, "all_bonds_partner_ids"),
232 *get_cell_structure(), *get_system());
233 }
234 });
235 } else {
236 // Handle generic parameters
237 context()->parallel_try_catch([&]() {
238 std::visit(
239 [&](auto &&vals) {
240 SetParticleParametersVisitor{}(m_id_selection, param_name, vals,
241 context(), m_cell_structure.lock(),
242 m_bonded_ias.lock());
243 },
244 params.at("values"));
245 });
246 }
247 return {};
248 }
249 if (not context()->is_head_node()) {
250 return {};
251 }
252 if (name == "prefetch_particle_data") {
253 auto p_ids = get_value<std::vector<int>>(params, "chunk");
254 prefetch_particle_data(p_ids);
255 return {};
256 }
257 if (name == "get_particle") {
258 return context()->make_shared(
259 "Particles::ParticleHandle",
260 {{"id", get_value<int>(params, "p_id")},
261 {"__cell_structure", m_cell_structure.lock()},
262 {"__bonded_ias", m_bonded_ias.lock()}});
263 }
264 return {};
265}
