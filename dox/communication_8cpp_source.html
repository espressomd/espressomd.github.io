81void init(std::shared_ptr<boost::mpi::environment> mpi_env) {
82#ifdef SHARED_MEMORY_PARALLELISM
83 const char *const env_omp_num_threads = std::getenv("OMP_NUM_THREADS");

126void Communicator::init_comm_cart() {
127 auto constexpr reorder = false;
128 comm = Utils::Mpi::cart_create(comm, node_grid, reorder);
129 this_node = comm.rank();
130 // check topology validity
131 std::ignore = Utils::Mpi::cart_neighbors<3>(comm);
132}

134void Communicator::full_initialization() {
135 assert(this_node == -1);
136 assert(size == -1);
137 MPI_Comm_size(MPI_COMM_WORLD, &size);
138 node_grid = Utils::Mpi::dims_create<3>(size);
139 init_comm_cart();
140}
