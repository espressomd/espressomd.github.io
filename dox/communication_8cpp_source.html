93CommunicationEnvironment::CommunicationEnvironment(
94 std::shared_ptr<boost::mpi::environment> mpi_env)
95 : m_mpi_env{std::move(mpi_env)} {
96#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
97 auto const num_threads_env = get_env_variable("OMP_NUM_THREADS");
98 if (not num_threads_env or num_threads_env->empty()) {
99 omp_set_num_threads(1);
100 }
101#endif
102
103 m_is_mpi_gpu_aware = false;
104
105#if defined(OPEN_MPI)
106#if defined(OMPI_HAVE_MPI_EXT_ROCM) && OMPI_HAVE_MPI_EXT_ROCM
107 m_is_mpi_gpu_aware |= static_cast<bool>(MPIX_Query_rocm_support());
108#endif
109#if defined(OMPI_HAVE_MPI_EXT_CUDA) && OMPI_HAVE_MPI_EXT_CUDA
110 m_is_mpi_gpu_aware |= static_cast<bool>(MPIX_Query_cuda_support());
111#endif
112#endif // defined(OPEN_MPI)
113
114#if defined(MPICH)
115 auto const mpich_gpu_env = get_env_variable("MPIR_CVAR_ENABLE_GPU");
116 m_is_mpi_gpu_aware |= (mpich_gpu_env and *mpich_gpu_env == "1");
117#endif // defined(MPICH)
118
119#if defined(_CRAYC) or defined(__cray__)
120 auto const cray_mpich_gpu_env = get_env_variable("MPICH_GPU_SUPPORT_ENABLED");
121 m_is_mpi_gpu_aware |= (cray_mpich_gpu_env and *cray_mpich_gpu_env == "1");
122#endif // defined(_CRAYC) or defined(__cray__)
123
124#ifdef ESPRESSO_WALBERLA
125 walberla::mpi_init();
126#endif
127
128 communicator.full_initialization();
129
130 m_callbacks =
131 std::make_shared<Communication::MpiCallbacks>(comm_cart, m_mpi_env);
132
133 ErrorHandling::init_error_handling(comm_cart);
134
135#ifdef ESPRESSO_CUDA
136 cuda_on_program_start();
137#endif
138
139#ifdef ESPRESSO_FFTW
140 fft_on_program_start();

144 kokkos_handle = std::make_shared<KokkosHandle>();

146}

166void Communicator::init_comm_cart() {
167 auto constexpr reorder = false;
168 comm = Utils::Mpi::cart_create(comm, node_grid, reorder);
169 this_node = comm.rank();
170 // check topology validity
171 std::ignore = Utils::Mpi::cart_neighbors<3>(comm);
172#ifdef ESPRESSO_WALBERLA
173 walberla::mpi_reinit(node_grid.data());
174#endif
175}
176
177void Communicator::full_initialization() {
178 assert(this_node == -1);
179 assert(size == -1);
180 MPI_Comm_size(MPI_COMM_WORLD, &size);
181 node_grid = Utils::Mpi::dims_create<3>(size);
182 init_comm_cart();
183}

185void Communicator::set_node_grid(Utils::Vector3i const &value) {
