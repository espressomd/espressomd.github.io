39/**
40 * @brief Base class for the electrostatics and magnetostatics P3M algorithms.
41 * Contains a handle to the FFT backend, information about the local mesh,
42 * the differential operator, and various buffers.
43 */
44struct p3m_data_struct {
45 explicit p3m_data_struct(P3MParameters &&parameters)
46 : params{std::move(parameters)} {}
47
48 /** @brief P3M base parameters. */
49 P3MParameters params;
50 /** @brief Local mesh properties. */
51 P3MLocalMesh local_mesh;
52
53 /**
54 * @brief Spatial differential operator in k-space.
55 * We use an i*k differentiation.
56 */
57 std::array<std::vector<int>, 3> d_op;
58
59 /** Calculate the Fourier transformed differential operator.
60 * Remark: This is done on the level of n-vectors and not k-vectors,
61 * i.e. the prefactor @f$ 2i\pi/L @f$ is missing!
62 */
63 void calc_differential_operator() {
64 d_op = detail::calc_meshift(params.mesh, true);
65 }
66};

68template <typename FloatType>
69struct p3m_data_struct_fft : public p3m_data_struct {
70 using p3m_data_struct::p3m_data_struct;
71 using value_type = FloatType;
72 /** @brief Local mesh FFT buffers. */
73 P3MFFTMesh<FloatType> mesh;
74
75 /** Force optimised influence function (k-space) */
76 std::vector<FloatType> g_force;
77 /** Energy optimised influence function (k-space) */
78 std::vector<FloatType> g_energy;
79 /** FFT backend. */
80 std::unique_ptr<FFTBackend<FloatType>> fft;
81
82 template <typename T, class... Args> void make_fft_instance(Args... args) {
83 assert(fft == nullptr);
84 fft = std::make_unique<T>(*this, args...);
85 }
86};

88/**
89 * @brief API for the FFT backend of the P3M algorithm.
90 * Any FFT backend must implement this interface.
91 * The backend can read some members of @ref p3m_data_struct
92 * but can only modify the FFT buffers in @ref P3MFFTMesh.
93 */
94template <typename FloatType> class FFTBackend {
95protected:
96 P3MParameters const &params;
97 P3MLocalMesh const &local_mesh;
98 P3MFFTMesh<FloatType> &mesh;
99
100public:
101 bool check_complex_residuals = false;
102 explicit FFTBackend(p3m_data_struct_fft<FloatType> &obj)
103 : params{obj.params}, local_mesh{obj.local_mesh}, mesh{obj.mesh} {}
104 virtual ~FFTBackend() = default;
105 /** @brief Initialize the FFT plans and buffers. */
106 virtual void init_fft() = 0;
107 /** @brief Initialize the halo buffers. */
108 virtual void init_halo() = 0;
109 /** @brief Carry out the forward FFT of the scalar mesh. */
110 virtual void perform_scalar_fwd_fft() = 0;
111 /** @brief Carry out the forward FFT of the vector meshes. */
112 virtual void perform_vector_fwd_fft() = 0;
113 /** @brief Carry out the backward FFT of the scalar mesh. */
114 virtual void perform_scalar_back_fft() = 0;
115 /** @brief Carry out the backward FFT of the vector meshes. */
116 virtual void perform_vector_back_fft() = 0;
117 /** @brief Update scalar mesh halo with data from neighbors (accumulation). */
118 virtual void perform_scalar_halo_gather() = 0;
119 /** @brief Update vector mesh halo with data from neighbors (accumulation). */
120 virtual void perform_vector_halo_gather() = 0;
121 /** @brief Update scalar mesh halo of all neighbors. */
122 virtual void perform_scalar_halo_spread() = 0;
123 /** @brief Update vector mesh halo of all neighbors. */
124 virtual void perform_vector_halo_spread() = 0;
125 /** @brief Get indices of the k-space data layout. */
126 virtual std::tuple<int, int, int> get_permutations() const = 0;
127};
