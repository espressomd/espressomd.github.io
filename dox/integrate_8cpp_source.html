96/**
97 * @brief Update the Lees-Edwards parameters of the box geometry
98 * for the current simulation time.
99 */
100void LeesEdwards::update_box_params(BoxGeometry &box_geo, double sim_time) {
101 if (box_geo.type() == BoxType::LEES_EDWARDS) {
102 assert(m_protocol != nullptr);
103 box_geo.lees_edwards_update(get_pos_offset(sim_time, *m_protocol),
104 get_shear_velocity(sim_time, *m_protocol));
105 }
106}

108void LeesEdwards::set_protocol(std::shared_ptr<ActiveProtocol> protocol) {
109 auto &system = get_system();
110 auto &cell_structure = *system.cell_structure;
111 auto &box_geo = *system.box_geo;
112 box_geo.set_type(BoxType::LEES_EDWARDS);
113 m_protocol = std::move(protocol);
114 update_box_params(box_geo, system.get_sim_time());
115 system.propagation->recalc_forces = true;
116 cell_structure.set_resort_particles(Cells::RESORT_LOCAL);
117}
118
119void LeesEdwards::unset_protocol() {
120 auto &system = get_system();
121 auto &cell_structure = *system.cell_structure;
122 auto &box_geo = *system.box_geo;
123 m_protocol = nullptr;
124 box_geo.set_type(BoxType::CUBOID);
125 system.propagation->recalc_forces = true;
126 cell_structure.set_resort_particles(Cells::RESORT_LOCAL);
127}

131void Propagation::update_default_propagation(int thermo_switch) {
132 switch (integ_switch) {
133 case INTEG_METHOD_STEEPEST_DESCENT:
134 default_propagation = PropagationMode::NONE;
135 break;
136 case INTEG_METHOD_NVT:
137 case INTEG_METHOD_SYMPLECTIC_EULER: {
138 // NOLINTNEXTLINE(bugprone-branch-clone)
139 if ((thermo_switch & THERMO_LB) and (thermo_switch & THERMO_LANGEVIN)) {
140 default_propagation = PropagationMode::TRANS_LB_MOMENTUM_EXCHANGE;
141#ifdef ESPRESSO_ROTATION
142 default_propagation |= PropagationMode::ROT_LANGEVIN;
143#endif
144 } else if (thermo_switch & THERMO_LB) {
145 default_propagation = PropagationMode::TRANS_LB_MOMENTUM_EXCHANGE;
146#ifdef ESPRESSO_ROTATION
147 default_propagation |= PropagationMode::ROT_EULER;
148#endif
149 } else if (thermo_switch & THERMO_LANGEVIN) {
150 default_propagation = PropagationMode::TRANS_LANGEVIN;
151#ifdef ESPRESSO_ROTATION
152 default_propagation |= PropagationMode::ROT_LANGEVIN;
153#endif
154 } else {
155 default_propagation = PropagationMode::TRANS_NEWTON;
156#ifdef ESPRESSO_ROTATION
157 default_propagation |= PropagationMode::ROT_EULER;
158#endif
159 }
160 break;
161 }
162#ifdef ESPRESSO_NPT
163 case INTEG_METHOD_NPT_ISO_AND:
164 case INTEG_METHOD_NPT_ISO_MTK:
165 default_propagation = PropagationMode::TRANS_LANGEVIN_NPT;
166 break;
167#endif
168 case INTEG_METHOD_BD:
169 default_propagation = PropagationMode::TRANS_BROWNIAN;
170#ifdef ESPRESSO_ROTATION
171 default_propagation |= PropagationMode::ROT_BROWNIAN;
172#endif
173 break;
174#ifdef ESPRESSO_STOKESIAN_DYNAMICS
175 case INTEG_METHOD_SD:
176 default_propagation = PropagationMode::TRANS_STOKESIAN;
177 break;
178#endif // ESPRESSO_STOKESIAN_DYNAMICS
179 default:
180 throw std::runtime_error("Unknown value for integ_switch");
181 }
182}

184void System::System::update_used_propagations() {
185 int used_propagations = PropagationMode::NONE;
186 for (auto &p : cell_structure->local_particles()) {
187 used_propagations |= p.propagation();
188 }
189 if (used_propagations & PropagationMode::SYSTEM_DEFAULT) {
190 used_propagations |= propagation->default_propagation;
191 }
192 used_propagations = boost::mpi::all_reduce(::comm_cart, used_propagations,
193 std::bit_or<int>());
194 propagation->used_propagations = used_propagations;
195}

197void System::System::integrator_sanity_checks() const {
198 auto const thermo_switch = thermostat->thermo_switch;
199 if (time_step <= 0.) {
200 runtimeErrorMsg() << "time_step not set";
201 }
202 if (propagation->integ_switch == INTEG_METHOD_STEEPEST_DESCENT) {
203 if (thermo_switch != THERMO_OFF) {
204 runtimeErrorMsg()
205 << "The steepest descent integrator is incompatible with thermostats";
206 }
207 }
208 if (propagation->integ_switch == INTEG_METHOD_NVT) {
209 if (thermo_switch & (THERMO_NPT_ISO | THERMO_BROWNIAN | THERMO_SD)) {
210 runtimeErrorMsg() << "The VV integrator is incompatible with the "
211 "currently active combination of thermostats";
212 }
213 }
214#ifdef ESPRESSO_NPT
215 if (propagation->used_propagations & PropagationMode::TRANS_LANGEVIN_NPT) {
216 if (thermo_switch != THERMO_NPT_ISO) {
217 runtimeErrorMsg() << "The NpT integrator requires the NpT thermostat";
218 }
219 if (box_geo->type() == BoxType::LEES_EDWARDS) {
220 runtimeErrorMsg() << "The NpT integrator cannot use Lees-Edwards";
221 }
222 try {
223 nptiso->coulomb_dipole_sanity_checks(*this);
224 } catch (std::runtime_error const &err) {
225 runtimeErrorMsg() << err.what();
226 }
227 }
228#endif
229 if (propagation->used_propagations & PropagationMode::TRANS_BROWNIAN) {
230 if (thermo_switch != THERMO_BROWNIAN) {
231 runtimeErrorMsg() << "The BD integrator requires the BD thermostat";
232 }
233 }
234 if (propagation->used_propagations & PropagationMode::TRANS_STOKESIAN) {
235#ifdef ESPRESSO_STOKESIAN_DYNAMICS
236 if (thermo_switch != THERMO_SD) {
237 runtimeErrorMsg() << "The SD integrator requires the SD thermostat";
238 }
239#endif
240 }
241 if (lb.is_solver_set() and (propagation->used_propagations &
242 (PropagationMode::TRANS_LB_MOMENTUM_EXCHANGE |
243 PropagationMode::TRANS_LB_TRACER))) {
244 if (thermostat->lb == nullptr) {
245 runtimeErrorMsg() << "The LB integrator requires the LB thermostat";
246 }
247 }
248 if (bonded_ias->get_n_thermalized_bonds() >= 1 and
249 (thermostat->thermalized_bond == nullptr or
250 (thermo_switch & THERMO_BOND) == 0)) {
251 runtimeErrorMsg()
252 << "Thermalized bonds require the thermalized_bond thermostat";
253 }
254
255#ifdef ESPRESSO_ROTATION
256 for (auto const &p : cell_structure->local_particles()) {
257 using namespace PropagationMode;
258 if (p.can_rotate() and not p.is_virtual() and
259 (p.propagation() & (SYSTEM_DEFAULT | ROT_EULER | ROT_LANGEVIN |
260 ROT_BROWNIAN | ROT_STOKESIAN)) == 0) {
261 runtimeErrorMsg()
262 << "Rotating particles must have a rotation propagation mode enabled";
263 break;
264 }
265 }
266#endif
267}

269#ifdef ESPRESSO_WALBERLA
270void walberla_tau_sanity_checks(std::string method, double tau,
271 double time_step) {
272 if (time_step <= 0.) {
273 return;
274 }
275 // use float epsilon since tau may be a float
276 auto const eps = static_cast<double>(std::numeric_limits<float>::epsilon());
277 if ((tau - time_step) / (tau + time_step) < -eps)
278 throw std::invalid_argument(method + " tau (" + std::to_string(tau) +
279 ") must be >= MD time_step (" +
280 std::to_string(time_step) + ")");
281 auto const factor = tau / time_step;
282 if (std::fabs(std::round(factor) - factor) / factor > eps)
283 throw std::invalid_argument(method + " tau (" + std::to_string(tau) +
284 ") must be an integer multiple of the "
285 "MD time_step (" +
286 std::to_string(time_step) + "). Factor is " +
287 std::to_string(factor));
288}

290void walberla_agrid_sanity_checks(std::string method,
291 Utils::Vector3d const &geo_left,
292 Utils::Vector3d const &geo_right,
293 Utils::Vector3d const &lattice_left,
294 Utils::Vector3d const &lattice_right,
295 double agrid) {
296 // waLBerla and ESPResSo must agree on domain decomposition
297 auto const tol = agrid / 1E6;
298 if ((lattice_left - geo_left).norm2() > tol or
299 (lattice_right - geo_right).norm2() > tol) {
300 runtimeErrorMsg() << "\nMPI rank " << ::this_node << ": "
301 << "left ESPResSo: [" << geo_left << "], "
302 << "left waLBerla: [" << lattice_left << "]"
303 << "\nMPI rank " << ::this_node << ": "
304 << "right ESPResSo: [" << geo_right << "], "
305 << "right waLBerla: [" << lattice_right << "]"
306 << "\nfor method: " << method;
307 throw std::runtime_error(
308 "waLBerla and ESPResSo disagree about domain decomposition.");
309 }
310}

313static void resort_particles_if_needed(System::System &system) {
314 auto &cell_structure = *system.cell_structure;
315 auto const offset = LeesEdwards::verlet_list_offset(
316 *system.box_geo, cell_structure.get_le_pos_offset_at_last_resort());
317 if (cell_structure.check_resort_required(offset)) {

322/** @brief Calls the hook for propagation kernels before the force calculation
323 * @return whether or not to stop the integration loop early.
324 */
325static bool integrator_step_1(CellStructure &cell_structure,
326 Propagation const &propagation,
327 System::System &system, double time_step) {
328 // steepest decent
329 if (propagation.integ_switch == INTEG_METHOD_STEEPEST_DESCENT)
330 return steepest_descent_step(cell_structure.local_particles());
331
332 auto const &thermostat = *system.thermostat;
333 auto const kT = thermostat.kT;
334 cell_structure.for_each_local_particle([&](Particle &p) {
335#ifdef ESPRESSO_VIRTUAL_SITES
336 // virtual sites are updated later in the integration loop
337 if (p.is_virtual())
338 return;
339#endif
340 if (propagation.integ_switch == INTEG_METHOD_SYMPLECTIC_EULER) {
341 if (propagation.should_propagate_with(
342 p, PropagationMode::TRANS_LB_MOMENTUM_EXCHANGE))
343 symplectic_euler_propagator_1(p, time_step);
344 if (propagation.should_propagate_with(p, PropagationMode::TRANS_NEWTON))
345 symplectic_euler_propagator_1(p, time_step);
346#ifdef ESPRESSO_ROTATION
347 if (propagation.should_propagate_with(p, PropagationMode::ROT_EULER))
348 symplectic_euler_rotator_1(p, time_step);
349#endif
350 if (propagation.should_propagate_with(p, PropagationMode::TRANS_LANGEVIN))
351 symplectic_euler_propagator_1(p, time_step);
352#ifdef ESPRESSO_ROTATION
353 if (propagation.should_propagate_with(p, PropagationMode::ROT_LANGEVIN))
354 symplectic_euler_rotator_1(p, time_step);
355#endif
356 } else {
357 if (propagation.should_propagate_with(
358 p, PropagationMode::TRANS_LB_MOMENTUM_EXCHANGE))
359 velocity_verlet_propagator_1(p, time_step);
360 if (propagation.should_propagate_with(p, PropagationMode::TRANS_NEWTON))
361 velocity_verlet_propagator_1(p, time_step);
362#ifdef ESPRESSO_ROTATION
363 if (propagation.should_propagate_with(p, PropagationMode::ROT_EULER))
364 velocity_verlet_rotator_1(p, time_step);
365#endif
366 if (propagation.should_propagate_with(p, PropagationMode::TRANS_LANGEVIN))
367 velocity_verlet_propagator_1(p, time_step);
368#ifdef ESPRESSO_ROTATION
369 if (propagation.should_propagate_with(p, PropagationMode::ROT_LANGEVIN))
370 velocity_verlet_rotator_1(p, time_step);
371#endif
372 }
373 if (propagation.should_propagate_with(p, PropagationMode::TRANS_BROWNIAN))
374 brownian_dynamics_propagator(*thermostat.brownian, p, time_step, kT);
375#ifdef ESPRESSO_ROTATION
376 if (propagation.should_propagate_with(p, PropagationMode::ROT_BROWNIAN))
377 brownian_dynamics_rotator(*thermostat.brownian, p, time_step, kT);
378#endif
379 });
380
381#ifdef ESPRESSO_NPT
382 if ((propagation.used_propagations & PropagationMode::TRANS_LANGEVIN_NPT) and
383 (propagation.default_propagation & PropagationMode::TRANS_LANGEVIN_NPT)) {
384 auto pred = PropagationPredicateNPT(propagation.default_propagation);
385 if (propagation.integ_switch == INTEG_METHOD_NPT_ISO_AND) {
386 velocity_verlet_npt_Andersen_step_1(
387 cell_structure.local_particles().filter(pred), *thermostat.npt_iso,
388 time_step, system);
389 } else if (propagation.integ_switch == INTEG_METHOD_NPT_ISO_MTK) {
390 velocity_verlet_npt_MTK_step_1(
391 cell_structure.local_particles().filter(pred), *thermostat.npt_iso,
392 time_step, system);
393 }
394 }
395#endif
396
397#ifdef ESPRESSO_STOKESIAN_DYNAMICS
398 if ((propagation.used_propagations & PropagationMode::TRANS_STOKESIAN) and
399 (propagation.default_propagation & PropagationMode::TRANS_STOKESIAN)) {
400 auto pred = PropagationPredicateStokesian(propagation.default_propagation);
401 stokesian_dynamics_step_1(cell_structure.local_particles().filter(pred),
402 *thermostat.stokesian, time_step, kT);
403 }
404#endif // ESPRESSO_STOKESIAN_DYNAMICS
405
406 return false;
407}

409static void integrator_step_2(CellStructure &cell_structure,
410 Propagation const &propagation,
411 [[maybe_unused]] System::System &system,
412 double time_step) {
413 if (propagation.integ_switch == INTEG_METHOD_STEEPEST_DESCENT)
414 return;
415
416 cell_structure.for_each_local_particle([&](Particle &p) {
417#ifdef ESPRESSO_VIRTUAL_SITES
418 // virtual sites are updated later in the integration loop
419 if (p.is_virtual())
420 return;
421#endif
422 if (propagation.integ_switch == INTEG_METHOD_SYMPLECTIC_EULER) {
423 if (propagation.should_propagate_with(
424 p, PropagationMode::TRANS_LB_MOMENTUM_EXCHANGE))
425 symplectic_euler_propagator_2(p, time_step);
426 if (propagation.should_propagate_with(p, PropagationMode::TRANS_NEWTON))
427 symplectic_euler_propagator_2(p, time_step);
428#ifdef ESPRESSO_ROTATION
429 if (propagation.should_propagate_with(p, PropagationMode::ROT_EULER))
430 symplectic_euler_rotator_2(p, time_step);
431#endif
432 if (propagation.should_propagate_with(p, PropagationMode::TRANS_LANGEVIN))
433 symplectic_euler_propagator_2(p, time_step);
434#ifdef ESPRESSO_ROTATION
435 if (propagation.should_propagate_with(p, PropagationMode::ROT_LANGEVIN))
436 symplectic_euler_rotator_2(p, time_step);
437#endif
438 } else {
439 if (propagation.should_propagate_with(
440 p, PropagationMode::TRANS_LB_MOMENTUM_EXCHANGE))
441 velocity_verlet_propagator_2(p, time_step);
442 if (propagation.should_propagate_with(p, PropagationMode::TRANS_NEWTON))
443 velocity_verlet_propagator_2(p, time_step);
444#ifdef ESPRESSO_ROTATION
445 if (propagation.should_propagate_with(p, PropagationMode::ROT_EULER))
446 velocity_verlet_rotator_2(p, time_step);
447#endif
448 if (propagation.should_propagate_with(p, PropagationMode::TRANS_LANGEVIN))
449 velocity_verlet_propagator_2(p, time_step);
450#ifdef ESPRESSO_ROTATION
451 if (propagation.should_propagate_with(p, PropagationMode::ROT_LANGEVIN))
452 velocity_verlet_rotator_2(p, time_step);
453#endif
454 }
455 });
456
457#ifdef ESPRESSO_NPT
458 if ((propagation.used_propagations & PropagationMode::TRANS_LANGEVIN_NPT) and
459 (propagation.default_propagation & PropagationMode::TRANS_LANGEVIN_NPT)) {
460 auto pred = PropagationPredicateNPT(propagation.default_propagation);
461 if (propagation.integ_switch == INTEG_METHOD_NPT_ISO_AND) {
462 velocity_verlet_npt_Andersen_step_2(
463 cell_structure.local_particles().filter(pred), time_step, system);
464 } else if (propagation.integ_switch == INTEG_METHOD_NPT_ISO_MTK) {
465 velocity_verlet_npt_MTK_step_2(
466 cell_structure.local_particles().filter(pred), time_step, system);
467 }
468 }
469#endif
470}

472int System::System::integrate(int n_steps, int reuse_forces) {
473#ifdef ESPRESSO_CALIPER
474 CALI_CXX_MARK_FUNCTION;
475#endif
476 auto &propagation = *this->propagation;
477#ifdef ESPRESSO_VIRTUAL_SITES_RELATIVE
478 auto const has_vs_rel = [&propagation]() {
479 return propagation.used_propagations & (PropagationMode::ROT_VS_RELATIVE |
480 PropagationMode::TRANS_VS_RELATIVE);
481 };
482#endif
483#ifdef ESPRESSO_BOND_CONSTRAINT
484 auto const n_rigid_bonds = bonded_ias->get_n_rigid_bonds();
485#endif
486
487 // Prepare particle structure and run sanity checks of all active algorithms
488 propagation.update_default_propagation(thermostat->thermo_switch);
489 update_used_propagations();
490 on_integration_start();
491
492 // If any method vetoes (e.g. P3M not initialized), immediately bail out
493 if (check_runtime_errors(comm_cart))
494 return INTEG_ERROR_RUNTIME;
495
496 // Additional preparations for the first integration step
497 if (reuse_forces == INTEG_REUSE_FORCES_NEVER or
498 ((reuse_forces != INTEG_REUSE_FORCES_ALWAYS) and
499 propagation.recalc_forces)) {
500#ifdef ESPRESSO_CALIPER
501 CALI_MARK_BEGIN("Initial Force Calculation");
502#endif
503 thermostat->lb_coupling_deactivate();
504
505#ifdef ESPRESSO_VIRTUAL_SITES_RELATIVE
506 if (has_vs_rel()) {
507 vs_relative_update_particles(*cell_structure, *box_geo);
508 }
509#endif
510
511 // Communication step: distribute ghost positions
512 cell_structure->update_ghosts_and_resort_particle(get_global_ghost_flags());
513
514 calculate_forces();
515
516 if (propagation.integ_switch != INTEG_METHOD_STEEPEST_DESCENT) {
517#ifdef ESPRESSO_ROTATION
518 convert_initial_torques(cell_structure->local_particles());
519#endif
520 }
521
522#ifdef ESPRESSO_CALIPER
523 CALI_MARK_END("Initial Force Calculation");
524#endif
525 }
526
527 thermostat->lb_coupling_activate();
528
529 if (check_runtime_errors(comm_cart))
530 return INTEG_ERROR_RUNTIME;
531
532 // Keep track of the number of Verlet updates (i.e. particle resorts)
533 int n_verlet_updates = 0;
534
535 // Keep track of whether an interrupt signal was caught (only in singleton
536 // mode, since signal handlers are unreliable with more than 1 MPI rank)
537 auto const singleton_mode = comm_cart.size() == 1;
538 auto caught_sigint = false;
539 auto caught_error = false;
540
541 auto lb_active = false;
542 auto ek_active = false;
543 if (propagation.integ_switch != INTEG_METHOD_STEEPEST_DESCENT) {
544 lb_active = lb.is_solver_set();
545 ek_active = ek.is_ready_for_propagation();
546 }
547 auto const calc_md_steps_per_tau = [this](double tau) {
548 return static_cast<int>(std::round(tau / time_step));
549 };
550
551#ifdef ESPRESSO_VALGRIND
552 CALLGRIND_START_INSTRUMENTATION;
553#endif
554 // Integration loop
555#ifdef ESPRESSO_CALIPER
556 CALI_CXX_MARK_LOOP_BEGIN(integration_loop, "Integration loop");
557#endif
558 int integrated_steps = 0;
559 for (int step = 0; step < n_steps; step++) {
560#ifdef ESPRESSO_CALIPER
561 CALI_CXX_MARK_LOOP_ITERATION(integration_loop, step);
562#endif
563
564#ifdef ESPRESSO_BOND_CONSTRAINT
565 if (n_rigid_bonds)
566 save_old_position(cell_structure->local_particles(),
567 cell_structure->ghost_particles());
568#endif
569
570 lees_edwards->update_box_params(*box_geo, sim_time);
571 bool early_exit =
572 integrator_step_1(*cell_structure, propagation, *this, time_step);
573 if (early_exit)
574 break;
575
576 sim_time += time_step;
577 if (box_geo->type() == BoxType::LEES_EDWARDS) {
578 auto const kernel = LeesEdwards::Push{*box_geo};
579 cell_structure->for_each_local_particle(
580 [&kernel](Particle &p) { kernel(p); });
581 }
582
583#ifdef ESPRESSO_NPT
584 if (not has_npt_enabled())
585#endif
586 {
587 resort_particles_if_needed(*this);
588 }
589
590 // Propagate philox RNG counters
591 thermostat->philox_counter_increment();
592
593#ifdef ESPRESSO_BOND_CONSTRAINT
594 // Correct particle positions that participate in a rigid/constrained bond
595 if (n_rigid_bonds) {
596 correct_position_shake(*cell_structure, *box_geo, *bonded_ias);
597 }
598#endif
599
600#ifdef ESPRESSO_VIRTUAL_SITES_RELATIVE
601 if (has_vs_rel()) {
602#ifdef ESPRESSO_NPT
603 if (has_npt_enabled()) {
604 cell_structure->update_ghosts_and_resort_particle(
605 Cells::DATA_PART_PROPERTIES);
606 }
607#endif // ESPRESSO_NPT
608 vs_relative_update_particles(*cell_structure, *box_geo);
609 }
610#endif // ESPRESSO_VIRTUAL_SITES_RELATIVE
611
612 if (cell_structure->get_resort_particles() >= Cells::RESORT_LOCAL)
613 n_verlet_updates++;
614
615 // Communication step: distribute ghost positions
616 cell_structure->update_ghosts_and_resort_particle(get_global_ghost_flags());
617
618 calculate_forces();
619
620#ifdef ESPRESSO_VIRTUAL_SITES_INERTIALESS_TRACERS
621 if (thermostat->lb and
622 (propagation.used_propagations & PropagationMode::TRANS_LB_TRACER)) {
623 lb_tracers_add_particle_force_to_fluid(*cell_structure, *box_geo,
624 *local_geo, lb);
625 }
626#endif
627 integrator_step_2(*cell_structure, propagation, *this, time_step);
628 if (propagation.integ_switch == INTEG_METHOD_BD) {
629 resort_particles_if_needed(*this);
630 }
631 if (box_geo->type() == BoxType::LEES_EDWARDS) {
632 auto const kernel = LeesEdwards::UpdateOffset{*box_geo};
633 cell_structure->for_each_local_particle(
634 [&kernel](Particle &p) { kernel(p); });
635 }
636#ifdef ESPRESSO_BOND_CONSTRAINT
637 if (n_rigid_bonds) {
638 correct_velocity_shake(*cell_structure, *box_geo, *bonded_ias);
639 }
640#endif
641
642 // propagate one-step functionalities
643 if (propagation.integ_switch != INTEG_METHOD_STEEPEST_DESCENT) {
644 if (lb_active and ek_active) {
645 // assume that they are coupled, which is not necessarily true
646 auto const md_steps_per_lb_step = calc_md_steps_per_tau(lb.get_tau());
647 auto const md_steps_per_ek_step = calc_md_steps_per_tau(ek.get_tau());
648
649 if (md_steps_per_lb_step != md_steps_per_ek_step) {
650 runtimeErrorMsg()
651 << "LB and EK are active but with different time steps.";
652 }
653
654 assert(lb.is_gpu() == ek.is_gpu());
655 assert(propagation.lb_skipped_md_steps ==
656 propagation.ek_skipped_md_steps);
657
658 propagation.lb_skipped_md_steps += 1;
659 propagation.ek_skipped_md_steps += 1;
660 if (propagation.lb_skipped_md_steps >= md_steps_per_lb_step) {
661 propagation.lb_skipped_md_steps = 0;
662 propagation.ek_skipped_md_steps = 0;
663 lb.propagate();
664 lb.ghost_communication_vel();
665 ek.propagate();
666 }
667 } else if (lb_active) {
668 auto const md_steps_per_lb_step = calc_md_steps_per_tau(lb.get_tau());
669 propagation.lb_skipped_md_steps += 1;
670 if (propagation.lb_skipped_md_steps >= md_steps_per_lb_step) {
671 propagation.lb_skipped_md_steps = 0;
672 lb.propagate();
673 }
674 } else if (ek_active) {
675 auto const md_steps_per_ek_step = calc_md_steps_per_tau(ek.get_tau());
676 propagation.ek_skipped_md_steps += 1;
677 if (propagation.ek_skipped_md_steps >= md_steps_per_ek_step) {
678 propagation.ek_skipped_md_steps = 0;
679 ek.propagate();
680 }
681 }
682 if (lb_active and (propagation.used_propagations &
683 PropagationMode::TRANS_LB_MOMENTUM_EXCHANGE)) {
684 thermostat->lb->rng_increment();
685 }
686
687#ifdef ESPRESSO_VIRTUAL_SITES_INERTIALESS_TRACERS
688 if (thermostat->lb and
689 (propagation.used_propagations & PropagationMode::TRANS_LB_TRACER)) {
690 if (lb_active) {
691 lb.ghost_communication_vel();
692 }
693 lb_tracers_propagate(*cell_structure, lb, time_step);
694 }
695#endif
696
697#ifdef ESPRESSO_COLLISION_DETECTION
698 collision_detection->handle_collisions();
699#endif
700 bond_breakage->process_queue(*this);
701 }
702
703 integrated_steps++;
704
705 if (check_runtime_errors(comm_cart)) {
706 caught_error = true;
707 break;
708 }
709
710 // Check if SIGINT has been caught.
711 if (singleton_mode and ctrl_C == 1) {
712 caught_sigint = true;
713 break;
714 }
715
716 } // for-loop over integration steps
717 if (lb_active) {
718 lb.ghost_communication();
719 }
720 lees_edwards->update_box_params(*box_geo, sim_time);
721#ifdef ESPRESSO_CALIPER
722 CALI_CXX_MARK_LOOP_END(integration_loop);
723#endif
724
725#ifdef ESPRESSO_VALGRIND
726 CALLGRIND_STOP_INSTRUMENTATION;
727#endif
728
729#ifdef ESPRESSO_VIRTUAL_SITES_RELATIVE
730 if (has_vs_rel()) {
731 vs_relative_update_particles(*cell_structure, *box_geo);
732 }
733#endif
734
735 // Verlet list statistics
736 cell_structure->update_verlet_stats(n_steps, n_verlet_updates);
737
738#ifdef ESPRESSO_NPT
739 if (has_npt_enabled()) {
740 synchronize_npt_state();
741 }
742#endif
743 if (caught_sigint) {
744 ctrl_C = 0;
745 return INTEG_ERROR_SIGINT;
746 }
747 if (caught_error) {
748 return INTEG_ERROR_RUNTIME;
749 }
750#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
751 if (boost::mpi::all_reduce(::comm_cart, not cell_structure->use_verlet_list,
752 std::logical_or<>())) {
753 cell_structure->use_verlet_list = false;
754 }
755#endif
756 return integrated_steps;
757}

759int System::System::integrate_with_signal_handler(int n_steps, int reuse_forces,
760 bool update_accumulators) {
761 assert(n_steps >= 0);
762
763 // Override the signal handler so that the integrator obeys Ctrl+C
764 SignalHandler sa(SIGINT, [](int) { ctrl_C = 1; });
765
766 /* if skin wasn't set, do an educated guess now */
767 if (not cell_structure->is_verlet_skin_set()) {
768 try {
769 cell_structure->set_verlet_skin_heuristic();
770 } catch (...) {
771 if (comm_cart.rank() == 0) {
772 throw;
773 }
774 return INTEG_ERROR_RUNTIME;
775 }
776 }
777
778 if (not update_accumulators or n_steps == 0) {
779 return integrate(n_steps, reuse_forces);
780 }
781
782 for (int i = 0; i < n_steps;) {
783 /* Integrate to either the next accumulator update, or the
784 * end, depending on what comes first. */
785 auto const steps =
786 std::min((n_steps - i), auto_update_accumulators->next_update());
787
788 auto const local_retval = integrate(steps, reuse_forces);
789
790 // make sure all ranks exit when one rank fails
791 std::remove_const_t<decltype(local_retval)> global_retval;
792 boost::mpi::all_reduce(comm_cart, local_retval, global_retval,
793 std::plus<int>());
794 if (global_retval < 0) {
795 return global_retval; // propagate error code
796 }
797
798 reuse_forces = INTEG_REUSE_FORCES_ALWAYS;
799
800 (*auto_update_accumulators)(comm_cart, steps);
801
802 i += steps;
803 }
804
805 return 0;
806}
