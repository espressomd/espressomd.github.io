104 struct ShortRangePressureKernel
105  : public boost::static_visitor<boost::optional<std::function<Utils::Matrix<
106  double, 3, 3>(double, Utils::Vector3d const &, double)>>> {
107 
108  using kernel_type = result_type::value_type;
109 
110 #ifdef ELECTROSTATICS
111  template <typename T,
112  std::enable_if_t<traits::has_pressure<T>::value> * = nullptr>
113  result_type operator()(std::shared_ptr<T> const &ptr) const {
114  result_type pressure_kernel = {};
115  if (auto const force_kernel_opt = pair_force_kernel()) {
116  pressure_kernel =
117  kernel_type{[force_kernel = *force_kernel_opt](
118  double q1q2, Utils::Vector3d const &d, double dist) {
119  auto const force = force_kernel(q1q2, d, dist);
120  return Utils::tensor_product(force, d);
121  }};
122  }
123  return pressure_kernel;
124  }

134 struct ShortRangeEnergyKernel
135  : public boost::static_visitor<boost::optional<
136  std::function<double(Particle const &, Particle const &, double,
137  Utils::Vector3d const &, double)>>> {
138 
139  using kernel_type = result_type::value_type;
140 
141 #ifdef ELECTROSTATICS
142  template <typename T>
143  result_type operator()(std::shared_ptr<T> const &ptr) const {
144  auto const &actor = *ptr;
145  return kernel_type{[&actor](Particle const &, Particle const &, double q1q2,
146  Utils::Vector3d const &, double dist) {
147  return actor.pair_energy(q1q2, dist);
148  }};
149  }
150 #ifdef P3M
151  result_type
152  operator()(std::shared_ptr<ElectrostaticLayerCorrection> const &ptr) const {
153  auto const &actor = *ptr;
154  auto const energy_kernel = boost::apply_visitor(*this, actor.base_solver);
155  return kernel_type{[&actor, energy_kernel](
156  Particle const &p1, Particle const &p2, double q1q2,
157  Utils::Vector3d const &d, double dist) {
158  auto energy = 0.;
159  if (energy_kernel) {
160  energy = (*energy_kernel)(p1, p2, q1q2, d, dist);
161  }
162  return energy + actor.pair_energy_correction(q1q2, p1, p2);
163  }};
164  }
165 #endif // P3M
166 #ifdef MMM1D_GPU
167  result_type operator()(std::shared_ptr<CoulombMMM1DGpu> const &) const {
168  return {};
169  }
170 #endif // MMM1D_GPU
171  result_type operator()(std::shared_ptr<CoulombMMM1D> const &actor) const {
172  return kernel_type{[&actor](Particle const &, Particle const &, double q1q2,
173  Utils::Vector3d const &d, double dist) {
174  return actor->pair_energy(q1q2, d, dist);
175  }};
176  }
177 #endif // ELECTROSTATICS
178 };
