32double BondedInteractionsMap::maximal_cutoff() const {
33 auto const max_cut_bonded = std::accumulate(
34 begin(), end(), bonded_inactive_cutoff, [](auto max_cut, auto const &kv) {
35 auto constexpr visitor = [](auto const &bond) { return bond.cutoff(); };
36 return std::max(max_cut, std::visit(visitor, *kv.second));
37 });
38
39 /* Check if there are dihedrals */
40 auto const any_dihedrals = std::ranges::any_of(*this, [](auto const &kv) {
41 return (std::holds_alternative<DihedralBond>(*kv.second) or
42 std::holds_alternative<TabulatedDihedralBond>(*kv.second));
43 });
44
45 /* dihedrals: the central particle is indirectly connected to the fourth
46 * particle via the third particle, so we have to double the cutoff */
47 return (any_dihedrals) ? 2 * max_cut_bonded : max_cut_bonded;
48}

50void BondedInteractionsMap::on_ia_change() {
51 n_thermalized_bonds = 0;
52#ifdef ESPRESSO_BOND_CONSTRAINT
53 n_rigid_bonds = 0;
54#endif
55 for (auto const &bond : std::views::elements<1>(*this)) {
56 if (std::holds_alternative<ThermalizedBond>(*bond)) {
57 ++n_thermalized_bonds;
58 }
59#ifdef ESPRESSO_BOND_CONSTRAINT
60 if (std::holds_alternative<RigidBond>(*bond)) {
61 ++n_rigid_bonds;
62 }
63#endif
64 }
65 if (auto system = m_system.lock()) {
66 system->on_short_range_ia_change();
67 system->on_thermostat_param_change(); // thermalized bonds
68 }
69}

71void BondedInteractionsMap::activate_bond(mapped_type const &ptr) {
72 auto &system = get_system();
73 if (auto bond = std::get_if<ThermalizedBond>(ptr.get())) {
74 bond->set_thermostat_view(system.thermostat);
75 }
76 if (auto bond = std::get_if<IBMVolCons>(ptr.get())) {
77 system.immersed_boundaries->register_softID(*bond);
78 }
79 if (auto bond = std::get_if<IBMTriel>(ptr.get())) {
80 bond->initialize(*system.box_geo, *system.cell_structure);
81 }
82 if (auto bond = std::get_if<IBMTribend>(ptr.get())) {
83 bond->initialize(*system.box_geo, *system.cell_structure);
84 }
85}

87void BondedInteractionsMap::deactivate_bond(mapped_type const &ptr) {
88 if (auto bond = std::get_if<ThermalizedBond>(ptr.get())) {
89 bond->unset_thermostat_view();
90 n_thermalized_bonds = -1;

92 if (auto bond = std::get_if<IBMVolCons>(ptr.get())) {
93 bond->unset_volumes_view();
94 }
95#ifdef ESPRESSO_BOND_CONSTRAINT
96 if (std::get_if<RigidBond>(ptr.get())) {
97 n_rigid_bonds = -1;
