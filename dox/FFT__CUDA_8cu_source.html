48template <typename FloatType>
49__global__ void
50create_greens_function(gpu::FieldAccessor<FloatType> greens_function, int x,
51 int y, int z) {
52 using RealType = std::conditional<std::is_same<FloatType, float>::value,
53 cufftReal, cufftDoubleReal>::type;
54 greens_function.set(blockIdx, threadIdx);
55 unsigned int index =
56 greens_function.getLinearIndex(blockIdx, threadIdx, gridDim, blockDim);
57 unsigned int tmp;
58 unsigned int coord[3];
59
60 coord[0] = index % x;
61 tmp = index / x;
62 coord[1] = tmp % y;
63 coord[2] = tmp / y;
64 if (index < z * y * x) {
65 if (index == 0) {
66 // setting 0th Fourier mode to 0 enforces charge neutrality
67 greens_function.get(0u) = 0.0f;
68 } else {
69 constexpr RealType two_pi = 2.0f * M_PI;
70 greens_function.get(0u) = -0.5f /
71 (cos(two_pi * static_cast<RealType>(coord[0]) /
72 static_cast<RealType>(x)) +
73 cos(two_pi * static_cast<RealType>(coord[1]) /
74 static_cast<RealType>(y)) +
75 cos(two_pi * static_cast<RealType>(coord[2]) /
76 static_cast<RealType>(z)) -
77 3.0f) /
78 static_cast<RealType>(x * y * z);
79 }
80 }
81}

83template <typename FloatType, typename ComplexType>
84__global__ void
85multiply_by_greens_function(gpu::FieldAccessor<ComplexType> potential,
86 gpu::FieldAccessor<FloatType> greens_function) {
87 potential.set(blockIdx, threadIdx);
88 greens_function.set(blockIdx, threadIdx);
89 if (potential.isValidPosition() && greens_function.isValidPosition()) {
90 potential.get(0u) =
91 ComplexType(potential.get(0u).x * greens_function.get(0u),
92 potential.get(0u).y * greens_function.get(0u));
93 }
94}

96template <typename FloatType>
97__global__ void add_fields_with_factor(gpu::FieldAccessor<FloatType> field_out,
98 gpu::FieldAccessor<FloatType> field_add,
99 const FloatType factor) {
100 field_out.set(blockIdx, threadIdx);
101 field_add.set(blockIdx, threadIdx);
102 if (field_out.isValidPosition() && field_add.isValidPosition()) {
103 field_out.get(0u) += field_add.get(0u) * factor;
104 }
105}

107template <typename FloatType>
108__global__ void move_field(gpu::FieldAccessor<FloatType> dest_field,
109 gpu::FieldAccessor<FloatType> src_field) {
110 dest_field.set(blockIdx, threadIdx);
111 src_field.set(blockIdx, threadIdx);
112 if (dest_field.isValidPosition() && src_field.isValidPosition()) {
113 dest_field.get(0u) = src_field.get(0u);
114 src_field.get(0u) = 0.0;
115 }
116}

126template <typename FloatType>
127FFT_CUDA<FloatType>::FFT_CUDA(std::shared_ptr<LatticeWalberla> lattice,
128 double permittivity)
129 : m_lattice(std::move(lattice)), m_permittivity(permittivity),
130 kernel_greens(gpu::make_kernel(
131 multiply_by_greens_function<FloatType, ComplexType>)),
132 kernel_move_fields(gpu::make_kernel(move_field<FloatType>)) {
133 m_blocks = get_lattice().get_blocks();
134
135 m_potential_field_id = gpu::addGPUFieldToStorage<PotentialField>(
136 get_lattice().get_blocks(), "potential field", 1, field::fzyx, 0, false);
137 m_potential_field_with_ghosts_id = gpu::addGPUFieldToStorage<PotentialField>(
138 get_lattice().get_blocks(), "potential field with ghosts", 1, field::fzyx,
139 get_lattice().get_ghost_layers());
140 m_greens_function_field_id = gpu::addGPUFieldToStorage<GreenFunctionField>(
141 get_lattice().get_blocks(), "greens function", 1, field::fzyx, 0, false);
142 m_potential_fourier_id = gpu::addGPUFieldToStorage<PotentialFourier>(
143 get_lattice().get_blocks(), "fourier field", 1, field::fzyx, 0, false);
144 reset_charge_field();
145
146 heffte = std::make_shared<heffte_container>();
147 auto dim = get_lattice().get_grid_dimensions();
148 auto offset_vec = Utils::Vector3i({1, 1, 1});
149 auto order = Utils::Vector3i({0, 1, 2});
150 heffte->m_box_in = std::make_shared<heffte::box3d<>>(
151 to_array(Utils::Vector3i({0, 0, 0})), to_array(dim - offset_vec),
152 to_array(order));
153 heffte->m_box_out = std::make_shared<heffte::box3d<>>(
154 to_array(Utils::Vector3i({0, 0, 0})), to_array(dim - offset_vec),
155 to_array(order));
156 heffte->m_fft = std::make_shared<heffte::fft3d<heffte::backend::cufft>>(
157 *(heffte->m_box_in), *(heffte->m_box_out), MPI_COMM_WORLD);
158 heffte->m_buffer = std::make_shared<
159 heffte::fft3d<heffte::backend::cufft>::buffer_container<ComplexType>>(
160 heffte->m_fft->size_workspace());
161
162 auto block = get_lattice().get_blocks()->getBlock(0, 0, 0);
163 auto green_field =
164 block->template getData<GreenFunctionField>(m_greens_function_field_id);
165 auto kernel = gpu::make_kernel(create_greens_function<FloatType>);
166 kernel.addFieldIndexingParam(
167 gpu::FieldIndexing<FloatType>::xyz(*green_field));
168 kernel.addParam(dim[0]);
169 kernel.addParam(dim[1]);
170 kernel.addParam(dim[2]);
171 kernel();
172
173 for (auto &block : *get_lattice().get_blocks()) {
174 auto potential =
175 block.template getData<PotentialField>(m_potential_field_id);
176 auto potential_ghosts = block.template getData<PotentialField>(
177 m_potential_field_with_ghosts_id);
178 auto green =
179 block.template getData<GreenFunctionField>(m_greens_function_field_id);
180 auto fourier =
181 block.template getData<PotentialFourier>(m_potential_fourier_id);
182
183 kernel_greens =
184 gpu::make_kernel(multiply_by_greens_function<FloatType, ComplexType>);
185 kernel_greens.addFieldIndexingParam(
186 gpu::FieldIndexing<ComplexType>::allInner(*fourier));
187 kernel_greens.addFieldIndexingParam(
188 gpu::FieldIndexing<FloatType>::allInner(*green));
189
190 kernel_move_fields = gpu::make_kernel(move_field<FloatType>);
191 kernel_move_fields.addFieldIndexingParam(
192 gpu::FieldIndexing<FloatType>::xyz(*potential_ghosts));
193 kernel_move_fields.addFieldIndexingParam(
194 gpu::FieldIndexing<FloatType>::xyz(*potential));
195 }
196
197 m_full_communication =
198 std::make_shared<FullCommunicator>(get_lattice().get_blocks());
199 m_full_communication->addPackInfo(
200 std::make_shared<gpu::communication::MemcpyPackInfo<PotentialField>>(
201 m_potential_field_with_ghosts_id));
202}

204template <typename FloatType> void FFT_CUDA<FloatType>::reset_charge_field() {
205 // the FFT-solver re-uses the potential field for the charge
206 auto const potential_id = walberla::BlockDataID(m_potential_field_id);
207
208 for (auto &block : *get_lattice().get_blocks()) {
209 auto field = block.template getData<PotentialField>(potential_id);
210 ek::accessor::Scalar::initialize(field, FloatType_c(0.0));
211 }
212}

214template <typename FloatType>
215void FFT_CUDA<FloatType>::add_fields(PotentialField *field_out,
216 gpu::GPUField<FloatType> *field_add,
217 FloatType factor) {
218 auto kernel = gpu::make_kernel(add_fields_with_factor<FloatType>);
219 kernel.addFieldIndexingParam(gpu::FieldIndexing<FloatType>::xyz(*field_out));
220 kernel.addFieldIndexingParam(gpu::FieldIndexing<FloatType>::xyz(*field_add));
221 kernel.addParam(factor);
222 kernel();
223}

225template <typename FloatType>
226void FFT_CUDA<FloatType>::add_charge_to_field(std::size_t id, double valency,
227 bool is_double_precision) {
228 auto const factor = FloatType_c(valency) / FloatType_c(get_permittivity());
229 const auto density_id = walberla::BlockDataID(id);
230 for (auto &block : *get_lattice().get_blocks()) {
231 auto field = block.template getData<PotentialField>(m_potential_field_id);
232 // TODO do we enforce, that the FloatType of the Charge and the
233 // species density is the same?
234 auto density_field =
235 block.template getData<gpu::GPUField<FloatType>>(density_id);
236 add_fields(field, density_field, FloatType_c(factor));
237 }
238}

240template <typename FloatType> void FFT_CUDA<FloatType>::solve() {
241 for (auto &block : *get_lattice().get_blocks()) {
242 auto potential =
243 block.template getData<PotentialField>(m_potential_field_id);
244 auto fourier =
245 block.template getData<PotentialFourier>(m_potential_fourier_id);
246 FloatType *_data_potential = potential->dataAt(0, 0, 0, 0);
247 ComplexType *_data_fourier = fourier->dataAt(0, 0, 0, 0);
248 heffte->m_fft->forward(_data_potential, _data_fourier,
249 heffte->m_buffer->data());
250 kernel_greens();
251 heffte->m_fft->backward(_data_fourier, _data_potential,
252 heffte->m_buffer->data());
253 kernel_move_fields();
254 ghost_communication();
255 }
256}
