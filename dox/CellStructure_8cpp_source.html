97static auto estimate_max_counts(double pair_cutoff,
98 std::size_t number_of_unique_particles) {
99 if (std::isinf(pair_cutoff)) {
100 return number_of_unique_particles;
101 }
102 if (pair_cutoff < 0.) {
103 pair_cutoff = 0.;
104 }
105 auto const volume = Utils::int_pow<3>(pair_cutoff);
106 auto max_counts = static_cast<std::size_t>(std::ceil(8. * volume));
107 std::size_t constexpr threshold_num = 16;
108 if (max_counts < threshold_num) {
109 max_counts = std::min(threshold_num, number_of_unique_particles);
110 }
111 return max_counts;
112}

114void CellStructure::rebuild_local_properties(double const pair_cutoff) {
115 assert(m_kokkos_handle);
116 using execution_space = Kokkos::DefaultExecutionSpace;
117 auto const num_threads = execution_space().concurrency();
118 auto const num_part = get_unique_particles().size();
119 auto max_counts = estimate_max_counts(pair_cutoff, num_part);
120#ifdef ESPRESSO_COLLISION_DETECTION
121 auto const &system = get_system();
122 if (system.has_collision_detection_enabled()) {
123 // TODO: use other types of Verlet list data structures
124 max_counts = num_part * 2ul;
125 }
126#endif
127 if (m_local_force) { // local properties are reallocated
128 Kokkos::realloc(get_local_force(), num_part, num_threads);
129#ifdef ESPRESSO_ROTATION
130 Kokkos::realloc(get_local_torque(), num_part, num_threads);
131#endif
132 Kokkos::realloc(get_id_to_index(), get_cached_max_local_particle_id() + 1);
133 Kokkos::deep_copy(get_id_to_index(), -1);
134 // Resize particle views using AoSoA_pack's resize method
135 m_aosoa->resize(num_part);
136 m_verlet_list_cabana->reallocData(num_part, max_counts);
137 } else { // local properties are initialized
138 m_local_force =
139 std::make_unique<ForceType>("local_force", num_part, num_threads);
140#ifdef ESPRESSO_ROTATION
141 m_local_torque =
142 std::make_unique<ForceType>("local_torque", num_part, num_threads);
143#endif
144 m_id_to_index = std::make_unique<Kokkos::View<int *>>(
145 Kokkos::ViewAllocateWithoutInitializing("id_to_index"),
146 get_cached_max_local_particle_id() + 1);
147 Kokkos::deep_copy(get_id_to_index(), -1);
148 // Create AoSoA_pack and initialize with resize
149 m_aosoa = std::make_unique<AoSoA_pack>();
150 m_aosoa->resize(num_part);
151
152 m_verlet_list_cabana =
153 std::make_unique<ListType>(0ul, num_part, max_counts);
154 }
155#ifdef ESPRESSO_NPT
156 m_local_virial = std::make_unique<VirialType>("local_virial", num_threads);
157#endif
158}

174void CellStructure::set_index_map() {
175 auto &unique_particles = m_unique_particles;
176 unique_particles.clear();
177 unique_particles.resize(count_local_particles());
178 std::unordered_set<int> registered_index{};
179 using execution_space = Kokkos::DefaultExecutionSpace;
180 int n_threads = execution_space().concurrency();
181 std::vector<int> max_ids(n_threads);
182 enumerate_local_particles(
183 *this, [&unique_particles, &max_ids](std::size_t index, Particle &p) {
184 unique_particles[index] = &p;
185 const int thread_num = omp_get_thread_num();
186 max_ids[thread_num] = std::max(p.id(), max_ids[thread_num]);
187 });
188 int max_id = *(std::max_element(max_ids.begin(), max_ids.end()));
189 for (auto &p : ghost_particles()) {
190 auto const *local_particle = get_local_particle(p.id());
191 if (not local_particle) {
192 continue;
193 }
194 if (not local_particle->is_ghost()) {
195 continue;
196 }
197 if (registered_index.contains(p.id())) {
198 continue;
199 }
200 registered_index.insert(p.id());
201 unique_particles.emplace_back(&p);
202 max_id = std::max(p.id(), max_id);
203 }
204 registered_index.clear();
205 m_cached_max_local_particle_id = max_id;
206}

213void CellStructure::check_particle_index() const {
214 auto const max_id = get_max_local_particle_id();
215
216 for (auto const &p : local_particles()) {
217 auto const id = p.id();
218
219 if (id < 0 || id > max_id) {
220 throw std::runtime_error("Particle id out of bounds.");
221 }
222
223 if (get_local_particle(id) != &p) {
224 throw std::runtime_error("Invalid local particle index entry.");
225 }
226 }
227
228 /* checks: local particle id */
229 std::size_t local_part_cnt = 0u;
230 for (int n = 0; n < get_max_local_particle_id() + 1; n++) {
231 if (get_local_particle(n) != nullptr) {
232 local_part_cnt++;
233 if (get_local_particle(n)->id() != n) {
234 throw std::runtime_error("local_particles part has corrupted id.");
235 }
236 }
237 }
238
239 if (local_part_cnt != local_particles().size()) {
240 throw std::runtime_error(
241 std::to_string(local_particles().size()) + " parts in cells but " +
242 std::to_string(local_part_cnt) + " parts in local_particles");
243 }
244}

257void CellStructure::remove_particle(int id) {
258 auto remove_all_bonds_to = [id](BondList &bl) {
259 for (auto it = bl.begin(); it != bl.end();) {
260 if (Utils::contains(it->partner_ids(), id)) {
261 it = bl.erase(it);
262 } else {
263 std::advance(it, 1);
264 }
265 }
266 };
267
268 for (auto cell : decomposition().local_cells()) {
269 auto &parts = cell->particles();
270 for (auto it = parts.begin(); it != parts.end();) {
271 if (it->id() == id) {
272 it = parts.erase(it);
273 update_particle_index(id, nullptr);
274 update_particle_index(parts);
275 } else {
276 remove_all_bonds_to(it->bonds());
277 it++;
278 }
279 }
280 }

293Particle *CellStructure::add_particle(Particle &&p) {
294 auto const sort_cell = particle_to_cell(p);
295 /* There is always at least one cell, so if the particle
296 * does not belong to a cell on this node we can put it there. */
297 auto cell = sort_cell ? sort_cell : decomposition().local_cells()[0];
298
299 /* If the particle isn't local a global resort may be
300 * needed, otherwise a local resort if sufficient. */
301 set_resort_particles(sort_cell ? Cells::RESORT_LOCAL : Cells::RESORT_GLOBAL);
302
303 return std::addressof(
304 append_indexed_particle(cell->particles(), std::move(p)));
305}

373void CellStructure::resort_particles(bool global_flag) {
374 invalidate_ghosts();
375
376 static std::vector<ParticleChange> diff;
377 diff.clear();
378
379 m_decomposition->resort(global_flag, diff);
380
381 for (auto d : diff) {
382 std::visit(UpdateParticleIndexVisitor{this}, d);
383 }
384
385 auto const &lebc = get_system().box_geo->lees_edwards_bc();
386 m_rebuild_verlet_list = true;
387 m_rebuild_verlet_list_cabana = true;
388 m_le_pos_offset_at_last_resort = lebc.pos_offset;
389
390#ifdef ESPRESSO_ADDITIONAL_CHECKS
391 check_particle_index();
392 check_particle_sorting();
393#endif
394}

441void CellStructure::set_verlet_skin_heuristic() {
442 assert(not is_verlet_skin_set());
443 auto const max_cut = get_system().maximal_cutoff();
444 if (max_cut <= 0.) {
445 throw std::runtime_error(
446 "cannot automatically determine skin, please set it manually");
447 }
448 /* maximal skin that can be used without resorting is the maximal
449 * range of the cell system minus what is needed for interactions. */
450 auto const max_range = std::ranges::min(max_cutoff());
451 auto const new_skin = std::min(0.4 * max_cut, max_range - max_cut);
452 set_verlet_skin(new_skin);
453}

455void CellStructure::update_ghosts_and_resort_particle(unsigned data_parts) {
456 /* data parts that are only updated on resort */
457 auto constexpr resort_only_parts =
458 Cells::DATA_PART_PROPERTIES | Cells::DATA_PART_BONDS;
459
460 auto const global_resort = boost::mpi::all_reduce(
461 ::comm_cart, m_resort_particles, std::bit_or<unsigned>());
462
463 if (global_resort != Cells::RESORT_NONE) {
464 auto const do_global_resort = (global_resort & Cells::RESORT_GLOBAL) != 0;
465
466 /* Resort cell system */
467 resort_particles(do_global_resort);
468 ghosts_count();
469 ghosts_update(data_parts);
470
471 /* Add the ghost particles to the index if we don't already
472 * have them. */
473 for (auto &p : ghost_particles()) {
474 if (get_local_particle(p.id()) == nullptr) {
475 update_particle_index(p.id(), &p);
476 }
477 }
478
479 /* Particles are now sorted */
480 clear_resort_particles();
481 } else {
482 /* Communication step: ghost information */
483 ghosts_update(data_parts & ~resort_only_parts);
484 }
485}

505bool CellStructure::check_resort_required(
506 Utils::Vector3d const &additional_offset) const {
507 auto const lim = Utils::sqr(m_verlet_skin / 2.) - additional_offset.norm2();
508
509 Reduction::AddPartialResultKernel<bool> add_partial =
510 [lim](bool &result, Particle const &p) {
511 if ((p.pos() - p.pos_at_last_verlet_update()).norm2() > lim) {
512 result = true;
513 }
514 };
515
516 Reduction::ReductionOp<bool> reduce_op = [](bool &acc, bool const &val) {
517 acc |= val;
518 };
519
520 return reduce_over_local_particles(*this, add_partial, reduce_op);
