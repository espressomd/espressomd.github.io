50 void CellStructure::check_particle_index() {
51  auto const max_id = get_max_local_particle_id();
52 
53  for (auto const &p : local_particles()) {
54  auto const id = p.id();
55 
56  if (id < 0 || id > max_id) {
57  throw std::runtime_error("Particle id out of bounds.");
58  }
59 
60  if (get_local_particle(id) != &p) {
61  throw std::runtime_error("Invalid local particle index entry.");
62  }
63  }
64 
65  /* checks: local particle id */
66  int local_part_cnt = 0;
67  for (int n = 0; n < get_max_local_particle_id() + 1; n++) {
68  if (get_local_particle(n) != nullptr) {
69  local_part_cnt++;
70  if (get_local_particle(n)->id() != n) {
71  throw std::runtime_error("local_particles part has corrupted id.");
72  }
73  }
74  }
75 
76  if (local_part_cnt != local_particles().size()) {
77  throw std::runtime_error(
78  std::to_string(local_particles().size()) + " parts in cells but " +
79  std::to_string(local_part_cnt) + " parts in local_particles");
80  }
81 }

98 void CellStructure::remove_particle(int id) {
99  auto remove_all_bonds_to = [id](BondList &bl) {
100  for (auto it = bl.begin(); it != bl.end();) {
101  if (Utils::contains(it->partner_ids(), id)) {
102  it = bl.erase(it);
103  } else {
104  std::advance(it, 1);
105  }
106  }
107  };
108 
109  for (auto c : decomposition().local_cells()) {
110  auto &parts = c->particles();
111 
112  for (auto it = parts.begin(); it != parts.end();) {
113  if (it->id() == id) {
114  it = parts.erase(it);
115  update_particle_index(id, nullptr);
116  update_particle_index(parts);
117  } else {
118  remove_all_bonds_to(it->bonds());
119  it++;
120  }
121  }
122  }
123 }

136 Particle *CellStructure::add_particle(Particle &&p) {
137  auto const sort_cell = particle_to_cell(p);
138  /* There is always at least one cell, so if the particle
139  * does not belong to a cell on this node we can put it there. */
140  auto cell = sort_cell ? sort_cell : local_cells()[0];
141 
142  /* If the particle isn't local a global resort may be
143  * needed, otherwise a local resort if sufficient. */
144  set_resort_particles(sort_cell ? Cells::RESORT_LOCAL : Cells::RESORT_GLOBAL);
145 
146  return std::addressof(
147  append_indexed_particle(cell->particles(), std::move(p)));
148 }

236 void CellStructure::resort_particles(bool global_flag, BoxGeometry const &box) {
237  invalidate_ghosts();
238 
239  static std::vector<ParticleChange> diff;
240  diff.clear();
241 
242  m_decomposition->resort(global_flag, diff);
243 
244  for (auto d : diff) {
245  boost::apply_visitor(UpdateParticleIndexVisitor{this}, d);
246  }
247 
248  m_rebuild_verlet_list = true;
249  m_le_pos_offset_at_last_resort = box.lees_edwards_bc().pos_offset;
250 
251 #ifdef ADDITIONAL_CHECKS
252  check_particle_index();
253  check_particle_sorting();
254 #endif
255 }
