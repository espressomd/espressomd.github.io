56 void CellStructure::check_particle_index() const {
57  auto const max_id = get_max_local_particle_id();
58 
59  for (auto const &p : local_particles()) {
60  auto const id = p.id();
61 
62  if (id < 0 || id > max_id) {
63  throw std::runtime_error("Particle id out of bounds.");
64  }
65 
66  if (get_local_particle(id) != &p) {
67  throw std::runtime_error("Invalid local particle index entry.");
68  }
69  }
70 
71  /* checks: local particle id */
72  int local_part_cnt = 0;
73  for (int n = 0; n < get_max_local_particle_id() + 1; n++) {
74  if (get_local_particle(n) != nullptr) {
75  local_part_cnt++;
76  if (get_local_particle(n)->id() != n) {
77  throw std::runtime_error("local_particles part has corrupted id.");
78  }
79  }
80  }
81 
82  if (local_part_cnt != local_particles().size()) {
83  throw std::runtime_error(
84  std::to_string(local_particles().size()) + " parts in cells but " +
85  std::to_string(local_part_cnt) + " parts in local_particles");
86  }
87 }

100 void CellStructure::remove_particle(int id) {
101  auto remove_all_bonds_to = [id](BondList &bl) {
102  for (auto it = bl.begin(); it != bl.end();) {
103  if (Utils::contains(it->partner_ids(), id)) {
104  it = bl.erase(it);
105  } else {
106  std::advance(it, 1);
107  }
108  }
109  };
110 
111  for (auto c : decomposition().local_cells()) {
112  auto &parts = c->particles();
113 
114  for (auto it = parts.begin(); it != parts.end();) {
115  if (it->id() == id) {
116  it = parts.erase(it);
117  update_particle_index(id, nullptr);
118  update_particle_index(parts);
119  } else {
120  remove_all_bonds_to(it->bonds());
121  it++;
122  }
123  }
124  }
125 }

138 Particle *CellStructure::add_particle(Particle &&p) {
139  auto const sort_cell = particle_to_cell(p);
140  /* There is always at least one cell, so if the particle
141  * does not belong to a cell on this node we can put it there. */
142  auto cell = sort_cell ? sort_cell : decomposition().local_cells()[0];
143 
144  /* If the particle isn't local a global resort may be
145  * needed, otherwise a local resort if sufficient. */
146  set_resort_particles(sort_cell ? Cells::RESORT_LOCAL : Cells::RESORT_GLOBAL);
147 
148  return std::addressof(
149  append_indexed_particle(cell->particles(), std::move(p)));
150 }

218 void CellStructure::resort_particles(bool global_flag) {
219  invalidate_ghosts();
220 
221  static std::vector<ParticleChange> diff;
222  diff.clear();
223 
224  m_decomposition->resort(global_flag, diff);
225 
226  for (auto d : diff) {
227  boost::apply_visitor(UpdateParticleIndexVisitor{this}, d);
228  }
229 
230  auto const &lebc = get_system().box_geo->lees_edwards_bc();
231  m_rebuild_verlet_list = true;
232  m_le_pos_offset_at_last_resort = lebc.pos_offset;
233 
234 #ifdef ADDITIONAL_CHECKS
235  check_particle_index();
236  check_particle_sorting();
237 #endif
238 }

283 void CellStructure::set_verlet_skin_heuristic() {
284  assert(not is_verlet_skin_set());
285  auto const max_cut = get_system().maximal_cutoff();
286  if (max_cut <= 0.) {
287  throw std::runtime_error(
288  "cannot automatically determine skin, please set it manually");
289  }
290  /* maximal skin that can be used without resorting is the maximal
291  * range of the cell system minus what is needed for interactions. */
292  auto const max_range = *boost::min_element(max_cutoff());
293  auto const new_skin = std::min(0.4 * max_cut, max_range - max_cut);
294  set_verlet_skin(new_skin);
295 }

297 void CellStructure::update_ghosts_and_resort_particle(unsigned data_parts) {
298  /* data parts that are only updated on resort */
299  auto constexpr resort_only_parts =
300  Cells::DATA_PART_PROPERTIES | Cells::DATA_PART_BONDS;
301 
302  auto const global_resort = boost::mpi::all_reduce(
303  ::comm_cart, m_resort_particles, std::bit_or<unsigned>());
304 
305  if (global_resort != Cells::RESORT_NONE) {
306  auto const do_global_resort = (global_resort & Cells::RESORT_GLOBAL) != 0;
307 
308  /* Resort cell system */
309  resort_particles(do_global_resort);
310  ghosts_count();
311  ghosts_update(data_parts);
312 
313  /* Add the ghost particles to the index if we don't already
314  * have them. */
315  for (auto &p : ghost_particles()) {
316  if (get_local_particle(p.id()) == nullptr) {
317  update_particle_index(p.id(), &p);
318  }
319  }
320 
321  /* Particles are now sorted */
322  clear_resort_particles();
323  } else {
324  /* Communication step: ghost information */
325  ghosts_update(data_parts & ~resort_only_parts);
326  }
327 }
