40/**
41 * @brief Vector pointing from the real particle to the virtual site.
42 *
43 * @return Relative distance.
44 */
45static auto connection_vector(Particle const &p_ref, Particle const &p_vs) {
46 // Calculate the quaternion defining the orientation of the vector connecting
47 // the virtual site and the real particle
48 // This is obtained, by multiplying the quaternion representing the director
49 // of the real particle with the quaternion of the virtual particle, which
50 // specifies the relative orientation.
51 auto const director = Utils::convert_quaternion_to_director(
52 p_ref.quat() * p_vs.vs_relative().rel_orientation)
53 .normalize();
54
55 return p_vs.vs_relative().distance * director;
56}

58/**
59 * @brief Velocity of the virtual site
60 * @param p_ref Reference particle for the virtual site.
61 * @param p_vs Virtual site.
62 * @return Velocity of the virtual site.
63 */
64static Utils::Vector3d velocity(Particle const &p_ref, Particle const &p_vs) {
65 auto const d = connection_vector(p_ref, p_vs);
66
67 // Get omega of real particle in space-fixed frame
68 auto const omega_space_frame =
69 convert_vector_body_to_space(p_ref, p_ref.omega());
70 // Obtain velocity from v = v_real particle + omega_real_particle * director
71 return vector_product(omega_space_frame, d) + p_ref.v();
72}

74/**
75 * @brief Get real particle tracked by a virtual site.
76 *
77 * @param cell_structure Cell structure.
78 * @param p Virtual site.
79 * @return Pointer to real particle.
80 */
81static Particle *get_reference_particle(CellStructure &cell_structure,
82 Particle const &p) {
83 auto const &vs_rel = p.vs_relative();
84 if (vs_rel.to_particle_id == -1) {
85 runtimeErrorMsg() << "Particle with id " << p.id()
86 << " is a dangling virtual site";
87 return nullptr;
88 }
89 auto p_ref_ptr = cell_structure.get_local_particle(vs_rel.to_particle_id);
90 if (!p_ref_ptr) {
91 runtimeErrorMsg() << "No real particle with id " << vs_rel.to_particle_id
92 << " for virtual site with id " << p.id();
93 }
94 return p_ref_ptr;
95}

97/**
98 * @brief Constraint force to hold the particle at its prescribed position.
99 *
100 * @param p_ref Reference particle.
101 * @param p_vs Virtual site.
102 * @return Constraint force.
103 */
104static auto constraint_stress(Particle const &p_ref, Particle const &p_vs) {
105 /* The constraint force is minus the force on the particle, make it force
106 * free. The counter force is translated by the connection vector to the
107 * real particle, hence the virial stress is */
108 return tensor_product(-p_vs.force(), connection_vector(p_ref, p_vs));
109}

111static bool is_vs_relative_trans(Particle const &p) {
112 return p.propagation() & PropagationMode::TRANS_VS_RELATIVE;
113}
114static bool is_vs_relative_rot(Particle const &p) {
115 return p.propagation() & PropagationMode::ROT_VS_RELATIVE;
116}

118static bool is_vs_relative(Particle const &p) {
119 return (is_vs_relative_trans(p) or is_vs_relative_rot(p));

122void vs_relative_update_particles(CellStructure &cell_structure,
123 BoxGeometry const &box_geo) {
124 cell_structure.ghosts_update(Cells::DATA_PART_POSITION |
125 Cells::DATA_PART_MOMENTUM);
126
127 cell_structure.for_each_local_particle([&](Particle &p) {
128 if (!is_vs_relative(p)) {
129 return;
130 }
131
132 auto const *p_ref_ptr = get_reference_particle(cell_structure, p);
133 if (!p_ref_ptr)
134 return;
135
136 auto const &p_ref = *p_ref_ptr;
137
138 // position update
139 if (is_vs_relative_trans(p)) {
140 p.image_box() = p_ref.image_box();
141 p.pos() = p_ref.pos() + connection_vector(p_ref, p);
142 p.v() = velocity(p_ref, p);
143
144 if (box_geo.type() == BoxType::LEES_EDWARDS) {
145 auto push = LeesEdwards::Push(box_geo);
146 push(p, 1); // includes a position fold
147 } else {
148 box_geo.fold_position(p.pos(), p.image_box());
149 }
150 }
151
152 // Orientation update
153 if (is_vs_relative_rot(p)) {
154 p.quat() = p_ref.quat() * p.vs_relative().quat;
155 }
156 });
157
158 if (cell_structure.check_resort_required()) {
159 cell_structure.set_resort_particles(Cells::RESORT_LOCAL);
160 }
161}

163// Distribute forces that have accumulated on virtual particles to the
164// associated real particles
165void vs_relative_back_transfer_forces_and_torques(
166 CellStructure &cell_structure) {
167 cell_structure.ghosts_reduce_forces();
168
169 init_forces_ghosts(cell_structure);
170
171 // Iterate over all the particles in the local cells
172 cell_structure.for_each_local_particle([&](Particle &p) {
173 if (!is_vs_relative(p))
174 return;
175
176 auto *p_ref_ptr = get_reference_particle(cell_structure, p);
177 assert(p_ref_ptr != nullptr);
178
179 auto &p_ref = *p_ref_ptr;
180 if (is_vs_relative_trans(p)) {
181 p_ref.force() += p.force();
182 p_ref.torque() += vector_product(connection_vector(p_ref, p), p.force());
183 }
184
185 if (is_vs_relative_rot(p)) {
186 p_ref.torque() += p.torque();
187 }
188 });
189}

191// Rigid body contribution to scalar pressure and pressure tensor
192Utils::Matrix<double, 3, 3>
193vs_relative_pressure_tensor(CellStructure const &cell_structure) {
194 Utils::Matrix<double, 3, 3> pressure_tensor = {};
195
196 for (auto const &p : cell_structure.local_particles()) {
197 if (is_vs_relative_trans(p)) {
198 if (auto const *p_ref_ptr = cell_structure.get_local_particle(
199 p.vs_relative().to_particle_id)) {
200 pressure_tensor += constraint_stress(*p_ref_ptr, p);
201 }
202 }
203 }
204
205 return pressure_tensor;
206}
