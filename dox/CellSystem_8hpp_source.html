78  std::unordered_map<std::string, CellStructureType> const cs_name_to_type = {
79  {"regular_decomposition", CellStructureType::CELL_STRUCTURE_REGULAR},
80  {"n_square", CellStructureType::CELL_STRUCTURE_NSQUARE},
81  {"hybrid_decomposition", CellStructureType::CELL_STRUCTURE_HYBRID},

84 public:
85  CellSystem() {
86  add_parameters({
87  {"use_verlet_lists", cell_structure.use_verlet_list},
88  {"node_grid",
89  [this](Variant const &v) {
90  context()->parallel_try_catch([&v]() {
91  auto const error_msg = std::string("Parameter 'node_grid'");
92  auto const vec = get_value<std::vector<int>>(v);
93  if (vec.size() != 3ul) {
94  throw std::invalid_argument(error_msg + " must be 3 ints");
95  }
96  auto const old_node_grid = ::node_grid;
97  auto const new_node_grid = Utils::Vector3i{vec.begin(), vec.end()};
98  auto const n_nodes_old = Utils::product(old_node_grid);
99  auto const n_nodes_new = Utils::product(new_node_grid);
100  if (n_nodes_new != n_nodes_old) {
101  std::stringstream reason;
102  reason << ": MPI world size " << n_nodes_old << " incompatible "
103  << "with new node grid [" << new_node_grid << "]";
104  throw std::invalid_argument(error_msg + reason.str());
105  }
106  try {
107  ::node_grid = new_node_grid;
108  on_node_grid_change();
109  } catch (...) {
110  ::node_grid = old_node_grid;
111  on_node_grid_change();
112  throw;
113  }
114  });
115  },
116  []() { return pack_vector(::node_grid); }},
117  {"skin",
118  [this](Variant const &v) {
119  auto const new_skin = get_value<double>(v);
120  if (new_skin < 0.) {
121  if (context()->is_head_node()) {
122  throw std::domain_error("Parameter 'skin' must be >= 0");
123  }
124  throw Exception("");
125  }
126  mpi_set_skin_local(new_skin);
127  },
128  []() { return ::skin; }},
129  {"decomposition_type", AutoParameter::read_only,
130  [this]() {
131  return cs_type_to_name.at(cell_structure.decomposition_type());
132  }},
133  {"n_square_types", AutoParameter::read_only,
134  []() {
135  if (cell_structure.decomposition_type() !=
136  CellStructureType::CELL_STRUCTURE_HYBRID) {
137  return Variant{none};
138  }
139  auto const hd = get_hybrid_decomposition();
140  auto const ns_types = hd.get_n_square_types();
141  return Variant{std::vector<int>(ns_types.begin(), ns_types.end())};
142  }},
143  {"cutoff_regular", AutoParameter::read_only,
144  []() {
145  if (cell_structure.decomposition_type() !=
146  CellStructureType::CELL_STRUCTURE_HYBRID) {
147  return Variant{none};
148  }
149  auto const hd = get_hybrid_decomposition();
150  return Variant{hd.get_cutoff_regular()};
151  }},
152  {"max_cut_nonbonded", AutoParameter::read_only,
153  maximal_cutoff_nonbonded},
154  {"max_cut_bonded", AutoParameter::read_only, maximal_cutoff_bonded},
155  {"interaction_range", AutoParameter::read_only, interaction_range},
156  });
157  }

159  void do_construct(VariantMap const &params) override {
160  // the core cell structure has a default constructor; params is empty
161  // during default construction for the python class, and not empty
162  // during checkpointing
163  if (params.count("decomposition_type")) {
164  auto const cs_name = get_value<std::string>(params, "decomposition_type");
165  auto const cs_type = cs_name_to_type.at(cs_name);
166  initialize(cs_type, params);
167  do_set_parameter("skin", params.at("skin"));
168  do_set_parameter("node_grid", params.at("node_grid"));
169  }
170  }
171 
172  Variant do_call_method(std::string const &name,
173  VariantMap const &params) override {
174  if (name == "initialize") {
175  auto const cs_name = get_value<std::string>(params, "name");
176  auto const cs_type = cs_name_to_type.at(cs_name);
177  initialize(cs_type, params);
178  return {};
179  }
180  if (name == "resort") {
181  auto const global_flag = get_value_or<bool>(params, "global_flag", true);
182  return mpi_resort_particles(global_flag);
183  }
184  if (name == "get_state") {
185  auto state = get_parameters();
186  auto const cs_type = cell_structure.decomposition_type();
187  if (cs_type == CellStructureType::CELL_STRUCTURE_REGULAR) {
188  auto const rd = get_regular_decomposition();
189  state["cell_grid"] = pack_vector(rd.cell_grid);
190  state["cell_size"] = Variant{rd.cell_size};
191  } else if (cs_type == CellStructureType::CELL_STRUCTURE_HYBRID) {
192  auto const hd = get_hybrid_decomposition();
193  state["cell_grid"] = pack_vector(hd.get_cell_grid());
194  state["cell_size"] = Variant{hd.get_cell_size()};
195  mpi_resort_particles(true); // needed to get correct particle counts
196  state["parts_per_decomposition"] =
197  Variant{std::unordered_map<std::string, Variant>{
198  {"regular", hd.count_particles_in_regular()},
199  {"n_square", hd.count_particles_in_n_square()}}};
200  }
201  state["verlet_reuse"] = get_verlet_reuse();
202  state["n_nodes"] = ::n_nodes;
203  return state;
204  }
205  if (name == "get_pairs") {
206  std::vector<Variant> out;
207  context()->parallel_try_catch([&params, &out]() {
208  std::vector<std::pair<int, int>> pair_list;
209  auto const distance = get_value<double>(params, "distance");
210  if (boost::get<std::string>(&params.at("types")) != nullptr) {
211  auto const key = get_value<std::string>(params, "types");
212  if (key != "all") {
213  throw std::invalid_argument("Unknown argument types='" + key + "'");
214  }
215  pair_list = get_pairs(distance);
216  } else {
217  auto const types = get_value<std::vector<int>>(params, "types");
218  pair_list = get_pairs_of_types(distance, types);
219  }
220  std::transform(pair_list.begin(), pair_list.end(),
221  std::back_inserter(out),
222  [](std::pair<int, int> const &pair) {
223  return std::vector<int>{pair.first, pair.second};
224  });
225  });
226  return out;
227  }
228  if (name == "get_neighbors") {
229  std::vector<std::vector<int>> neighbors_global;
230  context()->parallel_try_catch([&neighbors_global, &params]() {
231  auto const dist = get_value<double>(params, "distance");
232  auto const pid = get_value<int>(params, "pid");
233  auto const ret = mpi_get_short_range_neighbors_local(pid, dist, true);
234  std::vector<int> neighbors_local;
235  if (ret) {
236  neighbors_local = *ret;
237  }
238  boost::mpi::gather(comm_cart, neighbors_local, neighbors_global, 0);
239  });
240  std::vector<int> neighbors;
241  for (auto const &neighbors_local : neighbors_global) {
242  if (not neighbors_local.empty()) {
243  neighbors = neighbors_local;
244  break;
245  }
246  }
247  return neighbors;
248  }
249  if (name == "non_bonded_loop_trace") {
250  std::vector<Variant> out;
251  auto const pair_list = non_bonded_loop_trace();
252  std::transform(pair_list.begin(), pair_list.end(),
253  std::back_inserter(out), [](PairInfo const &pair) {
254  return std::vector<Variant>{pair.id1, pair.id2,
255  pair.pos1, pair.pos2,
256  pair.vec21, pair.node};
257  });
258  return out;
259  }
260  if (name == "tune_skin") {
261  if (context()->is_head_node()) {
262  tune_skin(get_value<double>(params, "min_skin"),
263  get_value<double>(params, "max_skin"),
264  get_value<double>(params, "tol"),
265  get_value<int>(params, "int_steps"),
266  get_value_or<bool>(params, "adjust_max_skin", false));
267  }
268  return ::skin;
269  }
270  return {};
271  }

273 private:
274  /**
275  * @brief Resort the particles.
276  *
277  * This function resorts the particles on the nodes.
278  *
279  * @param global_flag If true a global resort is done, if false particles
280  * are only exchanges between neighbors.
281  * @return The number of particles on the nodes after the resort.
282  */
283  std::vector<int> mpi_resort_particles(bool global_flag) const {
284  cell_structure.resort_particles(global_flag, box_geo);
285  if (context()->is_head_node()) {
286  clear_particle_node();
287  }
288  auto const size = static_cast<int>(cell_structure.local_particles().size());
289  std::vector<int> n_part_per_node;
290  boost::mpi::gather(comm_cart, size, n_part_per_node, 0);
291  return n_part_per_node;
292  }

294  void initialize(CellStructureType const &cs_type,
295  VariantMap const &params) const {
296  auto const verlet = get_value_or<bool>(params, "use_verlet_lists", true);
297  cell_structure.use_verlet_list = verlet;
298  if (cs_type == CellStructureType::CELL_STRUCTURE_HYBRID) {
299  auto const cutoff_regular = get_value<double>(params, "cutoff_regular");
300  auto const ns_types =
301  get_value_or<std::vector<int>>(params, "n_square_types", {});
302  auto n_square_types = std::set<int>{ns_types.begin(), ns_types.end()};
303  set_hybrid_decomposition(std::move(n_square_types), cutoff_regular);
304  } else {
305  cells_re_init(cs_type);
306  }
307  }
308 };
