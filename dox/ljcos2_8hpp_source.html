49 /** Calculate Lennard-Jones cosine squared force factor */
50 inline double ljcos2_pair_force_factor(IA_parameters const &ia_params,
51  double dist) {
52  if (dist < (ia_params.ljcos2.cut + ia_params.ljcos2.offset)) {
53  auto const r_off = dist - ia_params.ljcos2.offset;
54  auto fac = 0.0;
55  if (r_off < ia_params.ljcos2.rchange) {
56  auto const frac6 = Utils::int_pow<6>(ia_params.ljcos2.sig / r_off);
57  fac =
58  48.0 * ia_params.ljcos2.eps * frac6 * (frac6 - 0.5) / (r_off * dist);
59  } else if (r_off < ia_params.ljcos2.rchange + ia_params.ljcos2.w) {
60  fac = -ia_params.ljcos2.eps * Utils::pi() / 2 / ia_params.ljcos2.w /
61  dist *
62  sin(Utils::pi() * (r_off - ia_params.ljcos2.rchange) /
63  ia_params.ljcos2.w);
64  }
65  return fac;
66  }
67  return 0.0;
68 }

70 /** Calculate Lennard-Jones cosine squared energy */
71 inline double ljcos2_pair_energy(IA_parameters const &ia_params, double dist) {
72  if (dist < (ia_params.ljcos2.cut + ia_params.ljcos2.offset)) {
73  auto const r_off = dist - ia_params.ljcos2.offset;
74  if (r_off < ia_params.ljcos2.rchange) {
75  auto const frac6 = Utils::int_pow<6>(ia_params.ljcos2.sig / r_off);
76  return 4.0 * ia_params.ljcos2.eps * (Utils::sqr(frac6) - frac6);
77  }
78  if (r_off < (ia_params.ljcos2.rchange + ia_params.ljcos2.w)) {
79  return -ia_params.ljcos2.eps / 2 *
80  (cos(Utils::pi() * (r_off - ia_params.ljcos2.rchange) /
81  ia_params.ljcos2.w) +
82  1);
83  }
84  }
85  return 0.0;
86 }
