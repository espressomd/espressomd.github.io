94 void update_mesh_views() {
95 auto const mesh_size_ptr = fft->get_mesh_size();
96 auto const mesh_start_ptr = fft->get_mesh_start();
97 for (auto i = 0u; i < 3u; ++i) {
98 mesh.size[i] = mesh_size_ptr[i];
99 mesh.start[i] = mesh_start_ptr[i];
100 }
101 mesh.stop = mesh.start + mesh.size;
102 fft_buffers->update_mesh_views(mesh);
103 }

157template <typename FloatType, Arch Architecture, class FFTConfig>
158struct DipolarP3MHeffte : public DipolarP3M {
159 ~DipolarP3MHeffte() override = default;
160
161 // heFFTe checks are only implemented for row-major data
162 static_assert(FFTConfig::r_space_order == Utils::MemoryOrder::ROW_MAJOR);
163 static_assert(FFTConfig::k_space_order == Utils::MemoryOrder::ROW_MAJOR);
164
165 using DipolarP3MStateClass = DipolarP3MState<FloatType, FFTConfig>;
166 /** @brief Dipolar P3M parameters. */
167 DipolarP3MStateClass &dp3m;
168
169private:
170#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
171 // kokkos handle must outlive kokkos data structures from other class members
172 std::shared_ptr<KokkosHandle> m_kokkos_handle;
173#endif
174 /** @brief Dipolar P3M meshes and FFT algorithm. */
175 std::unique_ptr<DipolarP3MStateClass> dp3m_impl;
176 TuningParameters tuning;
177 bool m_is_tuned;
178
179public:
180 DipolarP3MHeffte(std::unique_ptr<DipolarP3MStateClass> &&dp3m_state,
181 TuningParameters tuning_params, double prefactor)
182 : DipolarP3M(dp3m_state->params), dp3m{*dp3m_state},
183#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
184 m_kokkos_handle{::kokkos_handle},
185#endif
186 dp3m_impl{std::move(dp3m_state)}, tuning{std::move(tuning_params)} {
187
188 if (tuning.timings <= 0) {
189 throw std::domain_error("Parameter 'timings' must be > 0");
190 }
191 if (dp3m.params.mesh != Utils::Vector3i::broadcast(dp3m.params.mesh[0])) {
192 throw std::domain_error("DipolarP3M requires a cubic mesh");
193 }
194 m_is_tuned = not dp3m.params.tuning;
195 dp3m.params.tuning = false;
196 set_prefactor(prefactor);
197#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
198 dp3m.init_labels();
199#endif
200 }

233private:
234 void prepare_fft_mesh() {
235 dp3m.inter_weights.reset(dp3m.params.cao);
236#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
237 using execution_space = Kokkos::DefaultExecutionSpace;
238 auto const num_threads = execution_space().concurrency();
239 Kokkos::realloc(Kokkos::WithoutInitializing, dp3m.rs_fields_kokkos,
240 num_threads, 3, dp3m.local_mesh.size);
241 Kokkos::deep_copy(dp3m.rs_fields_kokkos, FloatType{0});
242#endif // ESPRESSO_SHARED_MEMORY_PARALLELISM
243 for (auto &rs_mesh_field : dp3m.mesh.rs_fields) {
244 std::ranges::fill(rs_mesh_field, FloatType{0});
245 }
246#ifdef ESPRESSO_DP3M_HEFFTE_CROSS_CHECKS
247 for (auto dir : {0u, 1u, 2u}) {
248 dp3m.heffte.rs_dipole_density[dir].resize(dp3m.local_mesh.size);
249 std::ranges::fill(dp3m.heffte.rs_dipole_density[dir], FloatType{0});
250 }
251#endif // ESPRESSO_DP3M_HEFFTE_CROSS_CHECKS
252 }
