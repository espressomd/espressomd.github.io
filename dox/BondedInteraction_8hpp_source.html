48 template <class T> class BondedInteractionInterface {
49 protected:
50  std::shared_ptr<::Bonded_IA_Parameters> m_bonded_ia;
51 
52 public:
53  std::shared_ptr<::Bonded_IA_Parameters> bonded_ia() { return m_bonded_ia; }
54  std::shared_ptr<const ::Bonded_IA_Parameters> bonded_ia() const {
55  return m_bonded_ia;
56  }
57 };
58 
59 class BondedInteraction : public AutoParameters<BondedInteraction>,
60  public BondedInteractionInterface<BondedInteraction> {
61  void do_construct(VariantMap const &params) override {
62  // Check if initialization "by id" or "by parameters"
63  if (params.find("bond_id") != params.end()) {
64  m_bonded_ia = ::bonded_ia_params.at(get_value<int>(params, "bond_id"));
65  } else {
66  construct_bond(params);
67  }
68  }
69 
70 private:
71  virtual void construct_bond(VariantMap const &params) {}
72 
73 public:
74  template <typename T>
75  bool operator==(const BondedInteractionInterface<T> &other) {
76  return m_bonded_ia == other.m_bonded_ia;
77  }
78 
79  Variant do_call_method(std::string const &name,
80  VariantMap const &params) override {
81  // this feature is needed to compare bonds
82  if (name == "get_address") {
83  return reinterpret_cast<std::size_t>(bonded_ia().get());
84  }
85  if (name == "get_num_partners") {
86  return number_of_partners(*bonded_ia());
87  }
88 
89  return {};
90  }
91 };
