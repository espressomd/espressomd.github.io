49/**
50 * @brief Cache for interpolation weights.
51 *
52 * This is a storage container for interpolation weights of
53 * type InterpolationWeights.
54 */
55class p3m_interpolation_cache {
56 int m_cao = 0;
57 /** Charge fractions for mesh assignment. */
58 std::vector<double> ca_frac;
59 /** index of first mesh point for charge assignment. */
60 std::vector<int> ca_fmp;
61
62public:
63 /**
64 * @brief Number of points in the cache.
65 * @return Number of points currently in the cache.
66 */
67 auto size() const { return ca_fmp.size(); }
68
69 /**
70 * @brief Charge assignment order the weights are for.
71 * @return The charge assignment order.
72 */
73 auto cao() const { return m_cao; }
74
75 /**
76 * @brief Fill cache with zero-initialized data.
77 * Meant for the parallel weight interpolation calculation.
78 * @param size Number of elements.
79 */
80 void zfill(std::size_t size) {
81 assert(ca_frac.empty());
82 assert(ca_fmp.empty());
83 ca_fmp.resize(size);
84 ca_frac.resize(size * static_cast<std::size_t>(m_cao * 3));
85 }
86
87 /**
88 * @brief Push back weights for one point.
89 *
90 * @tparam cao Interpolation order has to match the order
91 * set at last call to @ref p3m_interpolation_cache::reset.
92 * @param weights Interpolation weights to store.
93 */
94 template <int cao> void store(const InterpolationWeights<cao> &weights) {
95 assert(cao == m_cao);
96
97 ca_fmp.emplace_back(weights.ind);
98 auto it = std::back_inserter(ca_frac);
99 std::ranges::copy(weights.w_x, it);
100 std::ranges::copy(weights.w_y, it);
101 std::ranges::copy(weights.w_z, it);
102 }
103
104 /**
105 * @brief Insert weights for one point.
106 *
107 * @tparam cao Interpolation order has to match the order
108 * set at last call to @ref p3m_interpolation_cache::reset.
109 * @param p_index Particle index.
110 * @param weights Interpolation weights to store.
111 */
112 template <int cao>
113 void store_at(std::size_t p_index, const InterpolationWeights<cao> &weights) {
114 assert(cao == m_cao);
115 assert(p_index < size());
116
117 ca_fmp[p_index] = weights.ind;
118 auto it = ca_frac.begin();
119 std::advance(it, p_index * static_cast<std::size_t>(m_cao * 3));
120 std::ranges::copy(weights.w_x, it);
121 std::advance(it, m_cao);
122 std::ranges::copy(weights.w_y, it);
123 std::advance(it, m_cao);
124 std::ranges::copy(weights.w_z, it);
125 }
126
127 /**
128 * @brief Load entry from the cache.
129 *
130 * This loads an entry at an index from the cache,
131 * the entries are indexed by the order they were stored.
132 *
133 * @tparam cao Interpolation order has to match the order
134 * set at last call to @ref p3m_interpolation_cache::reset.
135 * @param p_index Index of the entry to load.
136 * @return Interpolation weights.
137 */
138 template <int cao> InterpolationWeights<cao> load(std::size_t p_index) const {
139 assert(cao == m_cao);
140 assert(p_index < size());
141
142 InterpolationWeights<cao> ret;
143 ret.ind = ca_fmp[p_index];
144
145 auto const view = std::span(std::as_const(ca_frac));
146 auto const offset = 3ul * p_index * static_cast<std::size_t>(cao);
147
148 std::ranges::copy(view.subspan(offset + 0ul * cao, cao), ret.w_x.begin());
149 std::ranges::copy(view.subspan(offset + 1ul * cao, cao), ret.w_y.begin());
150 std::ranges::copy(view.subspan(offset + 2ul * cao, cao), ret.w_z.begin());
151
152 return ret;
153 }
154
155 /**
156 * @brief Reset the cache.
157 *
158 * @param cao Interpolation order.
159 */
160 void reset(int cao) {
161 m_cao = cao;
162 ca_frac.clear();
163 ca_fmp.clear();
164 }
165};

167/**
168 * @brief Calculate the P-th order interpolation weights.
169 *
170 * As described in from @cite hockney88a 5-189 (or 8-61).
171 * The weights are also tabulated in @cite deserno98a @cite deserno98b.
172 */
173template <int cao, Utils::MemoryOrder memory_order>
174InterpolationWeights<cao>
175p3m_calculate_interpolation_weights(const Utils::Vector3d &position,
176 const Utils::Vector3d &ai,
177 P3MLocalMesh const &local_mesh) {
178 /** position shift for calc. of first assignment mesh point. */
179 auto const pos_shift = std::floor((cao - 1) / 2.0) - (cao % 2) / 2.0;
180
181 /* distance to nearest mesh point */
182 Utils::Vector3d dist;
183
184 /* nearest mesh point */
185 Utils::Vector3i nmp;
186
187 for (unsigned int d = 0; d < 3; d++) {
188 /* particle position in mesh coordinates */
189 auto const pos = ((position[d] - local_mesh.ld_pos[d]) * ai[d]) - pos_shift;
190
191 nmp[d] = static_cast<int>(pos);
192
193 /* distance to nearest mesh point */
194 dist[d] = (pos - nmp[d]) - 0.5;
195 }
196
197 InterpolationWeights<cao> ret;
198
199 /* 3d-array index of nearest mesh point */
200 ret.ind = Utils::get_linear_index<memory_order>(nmp, local_mesh.dim);
201
202 assert((nmp + Utils::Vector3i::broadcast(cao)) <= local_mesh.dim);
203 for (int i = 0; i < cao; i++) {
204 using Utils::bspline;
205
206 ret.w_x[i] = bspline<cao>(i, dist[0]);
207 ret.w_y[i] = bspline<cao>(i, dist[1]);
208 ret.w_z[i] = bspline<cao>(i, dist[2]);
209 }
210
211 return ret;
212}

214/**
215 * @brief P3M grid interpolation.
216 *
217 * This runs an kernel for every interpolation point
218 * in a set of interpolation weights with the linear
219 * grid index and the weight of the point as arguments.
220 *
221 * @param local_mesh Mesh info.
222 * @param weights Set of weights
223 * @param kernel The kernel to run.
224 */
225template <int cao, class Kernel>
226void p3m_interpolate(P3MLocalMesh const &local_mesh,
227 InterpolationWeights<cao> const &weights, Kernel kernel) {
228 auto q_ind = weights.ind;
229 for (int i0 = 0; i0 < cao; i0++) {
230 auto const w_x = weights.w_x[i0];
231 for (int i1 = 0; i1 < cao; i1++) {
232 auto const w_xy = w_x * weights.w_y[i1];
233 for (int i2 = 0; i2 < cao; i2++) {
234 auto const w_xyz = w_xy * weights.w_z[i2];
235 kernel(q_ind, w_xyz);
236 q_ind++;
237 }
238 q_ind += local_mesh.q_2_off;
239 }
240 q_ind += local_mesh.q_21_off;
241 }
