30/** Observable for the pressure and energy. */
31class Observable_stat {
32 /** Array for observables on each node. */
33 std::vector<double> m_data;
34 /** Number of doubles per data item */
35 std::size_t m_chunk_size;
36
37 std::size_t get_non_bonded_offset(int type1, int type2) const;
38
39 /** Get contribution from a non-bonded interaction */
40 auto get_non_bonded_contribution(std::span<double> view, int type1,
41 int type2) const {
42 auto const offset = get_non_bonded_offset(type1, type2);
43 return view.subspan(offset * m_chunk_size, m_chunk_size);
44 }
45
46public:
47 Observable_stat(std::size_t chunk_size, std::size_t n_bonded, int max_type);
48
49 auto get_chunk_size() const { return m_chunk_size; }
50
51 /** Accumulate values.
52 * @param acc Initial value for the accumulator.
53 * @param column Which column to sum up (only relevant for multi-dimensional
54 * observables).
55 */
56 double accumulate(double acc = 0.0, std::size_t column = 0ul) const {
57 assert(column < m_chunk_size);
58 if (m_chunk_size == 1ul)
59 return std::accumulate(m_data.begin(), m_data.end(), acc);
60
61 for (auto it = m_data.begin() + static_cast<std::ptrdiff_t>(column);
62 it < m_data.end(); it += static_cast<std::ptrdiff_t>(m_chunk_size))
63 acc += *it;
64 return acc;
65 }
66
67 /** Rescale values */
68 void rescale(double volume) {
69 std::ranges::transform(m_data, m_data.begin(),
70 std::bind_front(std::multiplies{}, 1. / volume));
71 }
72
73 /** Contribution from linear kinetic energy. */
74 std::span<double> kinetic_lin;
75 /** Contribution from angular kinetic energy. */
76 std::span<double> kinetic_rot;
77 /** Contribution(s) from bonded interactions. */
78 std::span<double> bonded;

92 std::span<double> non_bonded_inter;
93
94 /** Get contribution from a bonded interaction */
95 std::span<double> bonded_contribution(int bond_id) const {
96 auto const offset = m_chunk_size * static_cast<std::size_t>(bond_id);
97 return {bonded.data() + offset, m_chunk_size};
98 }
99
100 void add_non_bonded_contribution(int type1, int type2, int molid1, int molid2,
101 std::span<const double> data) {
102 assert(data.size() == m_chunk_size);
103 auto const view = (molid1 == molid2) ? non_bonded_intra : non_bonded_inter;
104 auto const dest = get_non_bonded_contribution(view, type1, type2);
105
106 std::ranges::transform(dest, data, dest.begin(), std::plus{});
107 }
108
109 void add_non_bonded_contribution(int type1, int type2, int molid1, int molid2,
110 double data) {
111 add_non_bonded_contribution(type1, type2, molid1, molid2, {&data, 1});
112 }
113
114 /** Get contribution from a non-bonded intramolecular interaction */
115 auto non_bonded_intra_contribution(int type1, int type2) const {
116 return get_non_bonded_contribution(non_bonded_intra, type1, type2);
117 }
118
119 /** Get contribution from a non-bonded intermolecular interaction */
120 auto non_bonded_inter_contribution(int type1, int type2) const {
121 return get_non_bonded_contribution(non_bonded_inter, type1, type2);
122 }
123
124 /** MPI reduction. */
125 void mpi_reduce();
126};
