222 /**
223 * @brief Convenience function to add a field with a custom allocator.
224 *
225 * When vectorization is off, let waLBerla decide which memory allocator
226 * to use. When vectorization is on, the aligned memory allocator is
227 * required, otherwise <tt>cpu_vectorize_info["assume_aligned"]</tt> will
228 * trigger assertions. That is because for single-precision kernels the
229 * waLBerla heuristic in <tt>src/field/allocation/FieldAllocator.h</tt>
230 * will fall back to @c StdFieldAlloc, yet @c AllocateAligned is needed
231 * for intrinsics to work.
232 */
233 template <typename Field>
234 auto add_to_storage(std::string const tag, FloatType value) {
235 auto const &blocks = m_lattice->get_blocks();
236 auto const n_ghost_layers = m_lattice->get_ghost_layers();
237 if constexpr (Architecture == lbmpy::Arch::CPU) {
238 return field::addToStorage<Field>(blocks, tag, FloatType{value},
239 field::fzyx, n_ghost_layers);
240 }
241#if defined(__CUDACC__)
242 else {
243 auto field_id = gpu::addGPUFieldToStorage<GPUField>(
244 blocks, tag, Field::F_SIZE, field::fzyx, n_ghost_layers);
245 if constexpr (std::is_same_v<Field, _DensityField>) {
246 for (auto block = blocks->begin(); block != blocks->end(); ++block) {
247 auto field = block->template getData<GPUField>(field_id);
248 ek::accessor::Scalar::initialize(field, FloatType{value});
249 }
250 } else if constexpr (std::is_same_v<Field, _FluxField>) {
251 for (auto block = blocks->begin(); block != blocks->end(); ++block) {
252 auto field = block->template getData<GPUField>(field_id);
253 ek::accessor::Flux::initialize(field,
254 std::array<FloatType, FluxCount>{});
255 }
256 }
257 return field_id;
258 }
259#endif

287 EKinWalberlaImpl(std::shared_ptr<LatticeWalberla> lattice, double diffusion,
288 double kT, double valency, Utils::Vector3d const &ext_efield,
289 double density, bool advection, bool friction_coupling,
290 bool thermalized, unsigned int seed)
291 : m_diffusion(FloatType_c(diffusion)), m_kT(FloatType_c(kT)),
292 m_valency(FloatType_c(valency)), m_ext_efield(ext_efield),
293 m_advection(advection), m_friction_coupling(friction_coupling),
294 m_seed(seed), m_lattice(std::move(lattice)) {
295
296 auto const &blocks = m_lattice->get_blocks();
297 auto const n_ghost_layers = m_lattice->get_ghost_layers();
298
299 m_density_field_id =
300 add_to_storage<_DensityField>("density field", FloatType_c(density));
301 m_flux_field_id =
302 add_to_storage<_FluxField>("flux field", FloatType_c(0.0));
303
304 m_continuity =
305 std::make_unique<ContinuityKernel>(m_flux_field_id, m_density_field_id);
306
307#if defined(__CUDACC__)
308 m_host_field_allocator =
309 std::make_shared<gpu::HostFieldAllocator<FloatType>>();
310#endif
311
312 if (thermalized) {
313 set_diffusion_kernels(*m_lattice, seed);
314 } else {
315 set_diffusion_kernels();
316 }
317
318 // Init boundary related stuff
319 m_flag_field_density_id = field::addFlagFieldToStorage<FlagField>(
320 blocks, "flag field density", n_ghost_layers);
321 reset_density_boundary_handling(blocks);
322
323 m_flag_field_flux_id = field::addFlagFieldToStorage<FlagField>(
324 blocks, "flag field flux", n_ghost_layers);
325 reset_flux_boundary_handling(blocks);
326
327 m_full_communication = std::make_shared<FullCommunicator>(blocks);
328 m_full_communication->addPackInfo(
329 std::make_shared<PackInfo<DensityField>>(m_density_field_id));
330 }

562public:
563 void integrate(std::size_t potential_id, std::size_t velocity_id,
564 std::size_t force_id, double lb_density) override {
565
566 updated_boundary_fields();
567
568 if (get_diffusion() == 0.)
569 return;
570
571 if (get_valency() != 0.) {
572 if (potential_id == walberla::BlockDataID{}) {
573 throw std::runtime_error("Walberla EK: electrostatic potential enabled "
574 "but no field accessible. potential id is " +
575 std::to_string(potential_id));
576 }
577 kernel_diffusion_electrostatic(potential_id);
578 } else {
579 kernel_diffusion();
580 }
581
582 kernel_migration();
583 kernel_boundary_flux();
584 // friction coupling
585 if (get_friction_coupling()) {
586 if (force_id == walberla::BlockDataID{}) {
587 throw std::runtime_error("Walberla EK: friction coupling enabled but "
588 "no force field accessible. force_id is " +
589 std::to_string(force_id) +
590 ". Hint: LB may be inactive.");
591 }
592 kernel_friction_coupling(force_id, lb_density);
593 }
594
595 if (get_advection()) {
596 if (velocity_id == walberla::BlockDataID{}) {
597 throw std::runtime_error("Walberla EK: advection enabled but no "
598 "velocity field accessible. velocity_id is " +
599 std::to_string(velocity_id) +
600 ". Hint: LB may be inactive.");
601 }
602 kernel_advection(velocity_id);
603 kernel_boundary_flux();
604 }
605 kernel_continuity();
606
607 // is this the expected behavior when reactions are included?
608 kernel_boundary_density();
609
610 // Handle VTK writers
611 integrate_vtk_writers();
612 }
