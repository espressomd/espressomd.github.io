147 [[nodiscard]] std::optional<CellInterval>
148 get_interval(Utils::Vector3i const &lower_corner,
149 Utils::Vector3i const &upper_corner) const {
150 auto const &lattice = get_lattice();
151 auto const &cell_min = lower_corner;
152 auto const cell_max = upper_corner - Utils::Vector3i::broadcast(1);
153 auto const lower_bc = get_block_and_cell(lattice, cell_min, true);
154 auto const upper_bc = get_block_and_cell(lattice, cell_max, true);
155 if (not lower_bc or not upper_bc) {
156 return std::nullopt;
157 }
158 assert(&(*(lower_bc->block)) == &(*(upper_bc->block)));
159 return {CellInterval(lower_bc->cell, upper_bc->cell)};
160 }

182public:
183 EKinWalberlaImpl(std::shared_ptr<LatticeWalberla> lattice, double diffusion,
184 double kT, double valency, Utils::Vector3d const &ext_efield,
185 double density, bool advection, bool friction_coupling,
186 bool thermalized, unsigned int seed)
187 : m_diffusion(FloatType_c(diffusion)), m_kT(FloatType_c(kT)),
188 m_valency(FloatType_c(valency)), m_ext_efield(ext_efield),
189 m_advection(advection), m_friction_coupling(friction_coupling),
190 m_seed(seed), m_lattice(std::move(lattice)) {
191
192 auto const &blocks = m_lattice->get_blocks();
193 auto const n_ghost_layers = m_lattice->get_ghost_layers();
194
195 m_density_field_id = field::addToStorage<DensityField>(
196 blocks, "density field", FloatType_c(density), field::fzyx,
197 n_ghost_layers);
198 m_density_field_flattened_id =
199 field::addFlattenedShallowCopyToStorage<DensityField>(
200 blocks, m_density_field_id, "flattened density field");
201 m_flux_field_id = field::addToStorage<FluxField>(
202 blocks, "flux field", FloatType{0}, field::fzyx, n_ghost_layers);
203 m_flux_field_flattened_id =
204 field::addFlattenedShallowCopyToStorage<FluxField>(
205 blocks, m_flux_field_id, "flattened flux field");
206
207 m_continuity = std::make_unique<ContinuityKernel>(
208 m_flux_field_flattened_id, m_density_field_flattened_id);
209
210 if (thermalized) {
211 set_diffusion_kernels(*m_lattice, seed);
212 } else {
213 set_diffusion_kernels();
214 }
215
216 // Init boundary related stuff
217 m_flag_field_density_id = field::addFlagFieldToStorage<FlagField>(
218 blocks, "flag field density", n_ghost_layers);
219 reset_density_boundary_handling(blocks);
220
221 m_flag_field_flux_id = field::addFlagFieldToStorage<FlagField>(
222 blocks, "flag field flux", n_ghost_layers);
223 reset_flux_boundary_handling(blocks);
224
225 m_full_communication = std::make_shared<FullCommunicator>(blocks);
226 m_full_communication->addPackInfo(
227 std::make_shared<field::communication::PackInfo<DensityField>>(
228 m_density_field_id));
229 }

323 void set_diffusion_kernels() {
324 auto kernel = DiffusiveFluxKernelUnthermalized(m_flux_field_flattened_id,
325 m_density_field_flattened_id,
326 FloatType_c(m_diffusion));
327 m_diffusive_flux = std::make_unique<DiffusiveFluxKernel>(std::move(kernel));
328
329 auto kernel_electrostatic = DiffusiveFluxKernelElectrostaticUnthermalized(
330 m_flux_field_flattened_id, BlockDataID{}, m_density_field_flattened_id,
331 FloatType_c(m_diffusion), FloatType_c(m_ext_efield[0]),
332 FloatType_c(m_ext_efield[1]), FloatType_c(m_ext_efield[2]),
333 FloatType_c(m_kT), FloatType_c(m_valency));
334
335 m_diffusive_flux_electrostatic =
336 std::make_unique<DiffusiveFluxKernelElectrostatic>(
337 std::move(kernel_electrostatic));
338 }
339
340 void set_diffusion_kernels(LatticeWalberla const &lattice,
341 unsigned int seed) {
342 auto const grid_dim = lattice.get_grid_dimensions();
343
344 auto kernel = DiffusiveFluxKernelThermalized(
345 m_flux_field_flattened_id, m_density_field_flattened_id,
346 FloatType_c(m_diffusion), grid_dim[0], grid_dim[1], grid_dim[2], seed,
347 0);
348
349 auto kernel_electrostatic = DiffusiveFluxKernelElectrostaticThermalized(
350 m_flux_field_flattened_id, BlockDataID{}, m_density_field_flattened_id,
351 FloatType_c(m_diffusion), FloatType_c(m_ext_efield[0]),
352 FloatType_c(m_ext_efield[1]), FloatType_c(m_ext_efield[2]), grid_dim[0],
353 grid_dim[1], grid_dim[2], FloatType_c(m_kT), seed, 0,
354 FloatType_c(m_valency));
355
356 auto const blocks = lattice.get_blocks();
357
358 for (auto &block : *blocks) {
359 kernel.configure(blocks, &block);
360 kernel_electrostatic.configure(blocks, &block);
361 }
362
363 m_diffusive_flux = std::make_unique<DiffusiveFluxKernel>(std::move(kernel));
364 m_diffusive_flux_electrostatic =
365 std::make_unique<DiffusiveFluxKernelElectrostatic>(
366 std::move(kernel_electrostatic));
367 }

387 void kernel_diffusion() {
388 for (auto &block : *m_lattice->get_blocks()) {

398 std::get_if<DiffusiveFluxKernelElectrostaticThermalized>(
399 &*m_diffusive_flux_electrostatic);
400 kernel_electrostatic->setTime_step(kernel_electrostatic->getTime_step() +
401 1u);
402 }

425 void kernel_diffusion_electrostatic(std::size_t const potential_id) {
426 auto const phiID = BlockDataID(potential_id);
427 std::visit([phiID](auto &kernel) { kernel.setPhiID(phiID); },
428 *m_diffusive_flux_electrostatic);
429
430 for (auto &block : *m_lattice->get_blocks()) {
431 std::visit([&block](auto &kernel) { kernel.run(&block); },
432 *m_diffusive_flux_electrostatic);
433 }
434
435 if (auto *kernel_electrostatic =
436 std::get_if<DiffusiveFluxKernelElectrostaticThermalized>(
437 &*m_diffusive_flux_electrostatic)) {
438 kernel_electrostatic->setTime_step(kernel_electrostatic->getTime_step() +
439 1u);
440
441 auto *kernel =
442 std::get_if<DiffusiveFluxKernelThermalized>(&*m_diffusive_flux);
443 kernel->setTime_step(kernel->getTime_step() + 1u);
444 }
445 }

465public:
466 void integrate(std::size_t potential_id, std::size_t velocity_id,
467 std::size_t force_id, double lb_density) override {
468
469 updated_boundary_fields();
470
471 if (get_diffusion() == 0.)
472 return;
473
474 if (get_valency() != 0.) {
475 if (potential_id == walberla::BlockDataID{}) {
476 throw std::runtime_error("Walberla EK: electrostatic potential enabled "
477 "but no field accessible. potential id is " +
478 std::to_string(potential_id));
479 }
480 kernel_diffusion_electrostatic(potential_id);
481 } else {
482 kernel_diffusion();
483 }
484
485 kernel_migration();
486 kernel_boundary_flux();
487 // friction coupling
488 if (get_friction_coupling()) {
489 if (force_id == walberla::BlockDataID{}) {
490 throw std::runtime_error("Walberla EK: friction coupling enabled but "
491 "no force field accessible. force_id is " +
492 std::to_string(force_id) +
493 ". Hint: LB may be inactive.");
494 }
495 kernel_friction_coupling(force_id, lb_density);
496 }
497
498 if (get_advection()) {
499 if (velocity_id == walberla::BlockDataID{}) {
500 throw std::runtime_error("Walberla EK: advection enabled but no "
501 "velocity field accessible. velocity_id is " +
502 std::to_string(velocity_id) +
503 ". Hint: LB may be inactive.");
504 }
505 kernel_advection(velocity_id);
506 }
507 kernel_continuity();
508
509 // is this the expected behavior when reactions are included?
510 kernel_boundary_density();
511
512 // Handle VTK writers
513 integrate_vtk_writers();
514 }

547 [[nodiscard]] std::vector<double>
548 get_slice_density(Utils::Vector3i const &lower_corner,
549 Utils::Vector3i const &upper_corner) const override {
550 std::vector<double> out;
551 if (auto const ci = get_interval(lower_corner, upper_corner)) {
552 auto const &lattice = get_lattice();
553 auto const &block = *(lattice.get_blocks()->begin());
554 auto const density_field =
555 block.template getData<DensityField>(m_density_field_id);
556 auto const lower_cell = ci->min();
557 auto const upper_cell = ci->max();
558 auto const n_values = ci->numCells();
559 out.reserve(n_values);
560 for (auto x = lower_cell.x(); x <= upper_cell.x(); ++x) {
561 for (auto y = lower_cell.y(); y <= upper_cell.y(); ++y) {
562 for (auto z = lower_cell.z(); z <= upper_cell.z(); ++z) {
563 out.emplace_back(density_field->get(Cell{x, y, z}));
564 }
565 }
566 }
567 assert(out.size() == n_values);
568 }
569 return out;
570 }

580 auto it = density.begin();
581 auto const lower_cell = ci->min();

583 assert(density.size() == ci->numCells());
584 for (auto x = lower_cell.x(); x <= upper_cell.x(); ++x) {
585 for (auto y = lower_cell.y(); y <= upper_cell.y(); ++y) {
586 for (auto z = lower_cell.z(); z <= upper_cell.z(); ++z) {
587 density_field->get(Cell{x, y, z}) = FloatType_c(*it);
588 ++it;
589 }
590 }
591 }
