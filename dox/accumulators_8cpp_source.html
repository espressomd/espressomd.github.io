30struct AutoUpdateAccumulator {
31 explicit AutoUpdateAccumulator(AccumulatorBase *acc)
32 : frequency(acc->delta_N()), counter(1), acc(acc) {}
33 int frequency;
34 int counter;
35 AccumulatorBase *acc;
36};

38std::vector<AutoUpdateAccumulator> auto_update_accumulators;

41void auto_update(boost::mpi::communicator const &comm, int steps) {
42 for (auto &acc : auto_update_accumulators) {
43 assert(steps <= acc.frequency);
44 acc.counter -= steps;
45 if (acc.counter <= 0) {
46 acc.acc->update(comm);
47 acc.counter = acc.frequency;
48 }
49
50 assert(acc.counter > 0);
51 }
52}
53
54int auto_update_next_update() {
55 return boost::accumulate(auto_update_accumulators,
56 std::numeric_limits<int>::max(),
57 [](int a, AutoUpdateAccumulator const &acc) {
58 return std::min(a, acc.counter);
59 });
60}

62namespace detail {
63struct MatchPredicate {
64 AccumulatorBase const *m_acc;
65 template <typename T> bool operator()(T const &a) const {
66 return a.acc == m_acc;
67 }
68};
69} // namespace detail

71void auto_update_add(AccumulatorBase *acc) {
72 assert(not auto_update_contains(acc));
73 auto_update_accumulators.emplace_back(acc);
74}
75
76void auto_update_remove(AccumulatorBase *acc) {
77 assert(auto_update_contains(acc));
78 std::erase_if(auto_update_accumulators, detail::MatchPredicate{acc});
79}
80
81bool auto_update_contains(AccumulatorBase const *acc) noexcept {
82 assert(acc);
83 return std::ranges::any_of(auto_update_accumulators,
84 detail::MatchPredicate{acc});
85}
