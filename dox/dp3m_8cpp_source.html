72 void DipolarP3M::count_magnetic_particles() {
73  int local_n = 0;
74  double local_mu2 = 0.;
75 
76  for (auto const &p : cell_structure.local_particles()) {
77  if (p.dipm() != 0.) {
78  local_mu2 += p.calc_dip().norm2();
79  local_n++;
80  }
81  }
82 
83  boost::mpi::all_reduce(comm_cart, local_mu2, dp3m.sum_mu2, std::plus<>());
84  boost::mpi::all_reduce(comm_cart, local_n, dp3m.sum_dip_part, std::plus<>());
85 }

104 double DipolarP3M::calc_average_self_energy_k_space() const {
105  auto const start = Utils::Vector3i{dp3m.fft.plan[3].start};
106  auto const size = Utils::Vector3i{dp3m.fft.plan[3].new_mesh};
107 
108  auto const node_phi = grid_influence_function_self_energy(
109  dp3m.params, start, start + size, dp3m.g_energy);
110 
111  double phi = 0.;
112  boost::mpi::reduce(comm_cart, node_phi, phi, std::plus<>(), 0);
113  phi /= 3. * box_geo.length()[0] * Utils::int_pow<3>(dp3m.params.mesh[0]);
114  return phi * Utils::pi();
115 }

117 void DipolarP3M::init() {
118  assert(dp3m.params.mesh >= Utils::Vector3i::broadcast(1));
119  assert(dp3m.params.cao >= 1 and dp3m.params.cao <= 7);
120  assert(dp3m.params.alpha > 0.);
121 
122  dp3m.params.cao3 = Utils::int_pow<3>(dp3m.params.cao);
123  dp3m.params.recalc_a_ai_cao_cut(box_geo.length());
124  dp3m.local_mesh.calc_local_ca_mesh(dp3m.params, local_geo, skin, 0.);
125 
126  dp3m.sm.resize(comm_cart, dp3m.local_mesh);
127 
128  int ca_mesh_size = fft_init(dp3m.local_mesh.dim, dp3m.local_mesh.margin,
129  dp3m.params.mesh, dp3m.params.mesh_off,
130  dp3m.ks_pnum, dp3m.fft, node_grid, comm_cart);
131  dp3m.rs_mesh.resize(ca_mesh_size);
132  dp3m.ks_mesh.resize(ca_mesh_size);
133 
134  for (auto &val : dp3m.rs_mesh_dip) {
135  val.resize(ca_mesh_size);
136  }
137 
138  dp3m.calc_differential_operator();
139 
140  /* fix box length dependent constants */
141  scaleby_box_l();
142 
143  count_magnetic_particles();
144 }

164 template <std::size_t cao> struct AssignDipole {
165  void operator()(dp3m_data_struct &dp3m, Utils::Vector3d const &real_pos,
166  Utils::Vector3d const &dip) const {
167  auto const weights = p3m_calculate_interpolation_weights<cao>(
168  real_pos, dp3m.params.ai, dp3m.local_mesh);
169  p3m_interpolate<cao>(dp3m.local_mesh, weights,
170  [&dip, &dp3m](int ind, double w) {
171  dp3m.rs_mesh_dip[0][ind] += w * dip[0];
172  dp3m.rs_mesh_dip[1][ind] += w * dip[1];
173  dp3m.rs_mesh_dip[2][ind] += w * dip[2];
174  });
175 
176  dp3m.inter_weights.store<cao>(weights);
177  }
178 };

181 void DipolarP3M::dipole_assign(ParticleRange const &particles) {
182  dp3m.inter_weights.reset(dp3m.params.cao);
183 
184  /* prepare local FFT mesh */
185  for (auto &i : dp3m.rs_mesh_dip)
186  for (int j = 0; j < dp3m.local_mesh.size; j++)
187  i[j] = 0.;
188 
189  for (auto const &p : particles) {
190  if (p.dipm() != 0.) {
191  Utils::integral_parameter<AssignDipole, 1, 7>(dp3m.params.cao, dp3m,
192  p.pos(), p.calc_dip());
193  }
194  }
195 }

198 template <std::size_t cao> struct AssignTorques {
199  void operator()(dp3m_data_struct const &dp3m, double prefac, int d_rs,
200  ParticleRange const &particles) const {
201 
202  /* magnetic particle index */
203  auto p_index = std::size_t{0ul};
204 
205  for (auto &p : particles) {
206  if (p.dipm() != 0.) {
207  auto const w = dp3m.inter_weights.load<cao>(p_index);
208 
209  Utils::Vector3d E{};
210  p3m_interpolate(dp3m.local_mesh, w,
211  [&E, &dp3m, d_rs](int ind, double w) {
212  E[d_rs] += w * dp3m.rs_mesh[ind];
213  });
214 
215  p.torque() -= vector_product(p.calc_dip(), prefac * E);
216  ++p_index;
217  }
218  }
219  }
220 };

222 template <std::size_t cao> struct AssignForces {
223  void operator()(dp3m_data_struct const &dp3m, double prefac, int d_rs,
224  ParticleRange const &particles) const {
225 
226  /* magnetic particle index */
227  auto p_index = std::size_t{0ul};
228 
229  for (auto &p : particles) {
230  if (p.dipm() != 0.) {
231  auto const w = dp3m.inter_weights.load<cao>(p_index);
232 
233  Utils::Vector3d E{};
234  p3m_interpolate(dp3m.local_mesh, w, [&E, &dp3m](int ind, double w) {
235  E[0] += w * dp3m.rs_mesh_dip[0][ind];
236  E[1] += w * dp3m.rs_mesh_dip[1][ind];
237  E[2] += w * dp3m.rs_mesh_dip[2][ind];
238  });
239 
240  p.force()[d_rs] += p.calc_dip() * prefac * E;
241  ++p_index;
242  }
243  }
244  }
245 };

248 double DipolarP3M::kernel(bool force_flag, bool energy_flag,
249  ParticleRange const &particles) {
250  int i, ind, j[3];
251  /* k-space energy */
252  double k_space_energy_dip = 0.;
253  double tmp0, tmp1;
254 
255  auto const dipole_prefac = prefactor / Utils::int_pow<3>(dp3m.params.mesh[0]);
256 
257  if (dp3m.sum_mu2 > 0) {
258  /* Gather information for FFT grid inside the nodes domain (inner local
259  * mesh) and perform forward 3D FFT (Charge Assignment Mesh). */
260  std::array<double *, 3> meshes = {{dp3m.rs_mesh_dip[0].data(),
261  dp3m.rs_mesh_dip[1].data(),
262  dp3m.rs_mesh_dip[2].data()}};
263 
264  dp3m.sm.gather_grid(Utils::make_span(meshes), comm_cart,
265  dp3m.local_mesh.dim);
266 
267  fft_perform_forw(dp3m.rs_mesh_dip[0].data(), dp3m.fft, comm_cart);
268  fft_perform_forw(dp3m.rs_mesh_dip[1].data(), dp3m.fft, comm_cart);
269  fft_perform_forw(dp3m.rs_mesh_dip[2].data(), dp3m.fft, comm_cart);
270  // Note: after these calls, the grids are in the order yzx and not xyz
271  // anymore!!!
272  }
273 
274  /* === k-space energy calculation === */
275  if (energy_flag) {
276  /*********************
277  Dipolar energy
278  **********************/
279  if (dp3m.sum_mu2 > 0) {
280  /* i*k differentiation for dipolar gradients:
281  * |(\Fourier{\vect{mu}}(k)\cdot \vect{k})|^2 */
282  ind = 0;
283  i = 0;
284  double node_k_space_energy_dip = 0.0;
285  for (j[0] = 0; j[0] < dp3m.fft.plan[3].new_mesh[0]; j[0]++) {
286  for (j[1] = 0; j[1] < dp3m.fft.plan[3].new_mesh[1]; j[1]++) {
287  for (j[2] = 0; j[2] < dp3m.fft.plan[3].new_mesh[2]; j[2]++) {
288  node_k_space_energy_dip +=
289  dp3m.g_energy[i] *
290  (Utils::sqr(
291  dp3m.rs_mesh_dip[0][ind] *
292  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] +
293  dp3m.rs_mesh_dip[1][ind] *
294  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] +
295  dp3m.rs_mesh_dip[2][ind] *
296  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]]) +
297  Utils::sqr(
298  dp3m.rs_mesh_dip[0][ind + 1] *
299  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] +
300  dp3m.rs_mesh_dip[1][ind + 1] *
301  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] +
302  dp3m.rs_mesh_dip[2][ind + 1] *
303  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]]));
304  ind += 2;
305  i++;
306  }
307  }
308  }
309  node_k_space_energy_dip *=
310  dipole_prefac * Utils::pi() * box_geo.length_inv()[0];
311  boost::mpi::reduce(comm_cart, node_k_space_energy_dip, k_space_energy_dip,
312  std::plus<>(), 0);
313 
314  if (dp3m.energy_correction == 0.)
315  calc_energy_correction();
316 
317  if (this_node == 0) {
318  /* self energy correction */
319  k_space_energy_dip -= prefactor * dp3m.sum_mu2 * 2. *
320  Utils::int_pow<3>(dp3m.params.alpha) *
321  Utils::sqrt_pi_i() / 3.;
322 
323  /* dipolar energy correction due to systematic Madelung-self effects */
324  auto const volume = box_geo.volume();
325  k_space_energy_dip += prefactor * dp3m.energy_correction / volume;
326  }
327  }
328  } // if (energy_flag)
329 
330  /* === k-space force calculation === */
331  if (force_flag) {
332  /****************************
333  * DIPOLAR TORQUES (k-space)
334  ****************************/
335  if (dp3m.sum_mu2 > 0) {
336  auto const two_pi_L_i = 2. * Utils::pi() * box_geo.length_inv()[0];
337  /* fill in ks_mesh array for torque calculation */
338  ind = 0;
339  i = 0;
340 
341  for (j[0] = 0; j[0] < dp3m.fft.plan[3].new_mesh[0]; j[0]++) { // j[0]=n_y
342  for (j[1] = 0; j[1] < dp3m.fft.plan[3].new_mesh[1];
343  j[1]++) { // j[1]=n_z
344  for (j[2] = 0; j[2] < dp3m.fft.plan[3].new_mesh[2];
345  j[2]++) { // j[2]=n_x
346  // tmp0 = Re(mu)*k, tmp1 = Im(mu)*k
347 
348  tmp0 = dp3m.rs_mesh_dip[0][ind] *
349  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] +
350  dp3m.rs_mesh_dip[1][ind] *
351  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] +
352  dp3m.rs_mesh_dip[2][ind] *
353  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]];
354 
355  tmp1 = dp3m.rs_mesh_dip[0][ind + 1] *
356  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] +
357  dp3m.rs_mesh_dip[1][ind + 1] *
358  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] +
359  dp3m.rs_mesh_dip[2][ind + 1] *
360  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]];
361 
362  /* the optimal influence function is the same for torques
363  and energy */
364  dp3m.ks_mesh[ind] = tmp0 * dp3m.g_energy[i];
365  dp3m.ks_mesh[ind + 1] = tmp1 * dp3m.g_energy[i];
366  ind += 2;
367  i++;
368  }
369  }
370  }
371 
372  /* Force component loop */
373  for (int d = 0; d < 3; d++) {
374  auto const d_rs = (d + dp3m.ks_pnum) % 3;
375  ind = 0;
376  for (j[0] = 0; j[0] < dp3m.fft.plan[3].new_mesh[0]; j[0]++) {
377  for (j[1] = 0; j[1] < dp3m.fft.plan[3].new_mesh[1]; j[1]++) {
378  for (j[2] = 0; j[2] < dp3m.fft.plan[3].new_mesh[2]; j[2]++) {
379  dp3m.rs_mesh[ind] =
380  dp3m.d_op[0][j[d] + dp3m.fft.plan[3].start[d]] *
381  dp3m.ks_mesh[ind];
382  ind++;
383  dp3m.rs_mesh[ind] =
384  dp3m.d_op[0][j[d] + dp3m.fft.plan[3].start[d]] *
385  dp3m.ks_mesh[ind];
386  ind++;
387  }
388  }
389  }
390 
391  /* Back FFT force component mesh */
392  fft_perform_back(dp3m.rs_mesh.data(), false, dp3m.fft, comm_cart);
393  /* redistribute force component mesh */
394  dp3m.sm.spread_grid(dp3m.rs_mesh.data(), comm_cart,
395  dp3m.local_mesh.dim);
396  /* Assign force component from mesh to particle */
397  Utils::integral_parameter<AssignTorques, 1, 7>(
398  dp3m.params.cao, dp3m, dipole_prefac * two_pi_L_i, d_rs, particles);
399  }
400 
401  /***************************
402  DIPOLAR FORCES (k-space)
403  ****************************/
404 
405  // Compute forces after torques because the algorithm below overwrites the
406  // grids dp3m.rs_mesh_dip !
407  // Note: I'll do here 9 inverse FFTs. By symmetry, we can reduce this
408  // number to 6 !
409  /* fill in ks_mesh array for force calculation */
410  ind = 0;
411  i = 0;
412  for (j[0] = 0; j[0] < dp3m.fft.plan[3].new_mesh[0]; j[0]++) { // j[0]=n_y
413  for (j[1] = 0; j[1] < dp3m.fft.plan[3].new_mesh[1];
414  j[1]++) { // j[1]=n_z
415  for (j[2] = 0; j[2] < dp3m.fft.plan[3].new_mesh[2];
416  j[2]++) { // j[2]=n_x
417  // tmp0 = Im(mu)*k, tmp1 = -Re(mu)*k
418  tmp0 = dp3m.rs_mesh_dip[0][ind + 1] *
419  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] +
420  dp3m.rs_mesh_dip[1][ind + 1] *
421  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] +
422  dp3m.rs_mesh_dip[2][ind + 1] *
423  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]];
424  tmp1 = dp3m.rs_mesh_dip[0][ind] *
425  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] +
426  dp3m.rs_mesh_dip[1][ind] *
427  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] +
428  dp3m.rs_mesh_dip[2][ind] *
429  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]];
430  dp3m.ks_mesh[ind] = tmp0 * dp3m.g_force[i];
431  dp3m.ks_mesh[ind + 1] = -tmp1 * dp3m.g_force[i];
432  ind += 2;
433  i++;
434  }
435  }
436  }
437 
438  /* Force component loop */
439  for (int d = 0; d < 3; d++) { /* direction in k-space: */
440  auto const d_rs = (d + dp3m.ks_pnum) % 3;
441  ind = 0;
442  for (j[0] = 0; j[0] < dp3m.fft.plan[3].new_mesh[0];
443  j[0]++) { // j[0]=n_y
444  for (j[1] = 0; j[1] < dp3m.fft.plan[3].new_mesh[1];
445  j[1]++) { // j[1]=n_z
446  for (j[2] = 0; j[2] < dp3m.fft.plan[3].new_mesh[2];
447  j[2]++) { // j[2]=n_x
448  tmp0 = dp3m.d_op[0][j[d] + dp3m.fft.plan[3].start[d]] *
449  dp3m.ks_mesh[ind];
450  dp3m.rs_mesh_dip[0][ind] =
451  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] * tmp0;
452  dp3m.rs_mesh_dip[1][ind] =
453  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] * tmp0;
454  dp3m.rs_mesh_dip[2][ind] =
455  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]] * tmp0;
456  ind++;
457  tmp0 = dp3m.d_op[0][j[d] + dp3m.fft.plan[3].start[d]] *
458  dp3m.ks_mesh[ind];
459  dp3m.rs_mesh_dip[0][ind] =
460  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] * tmp0;
461  dp3m.rs_mesh_dip[1][ind] =
462  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] * tmp0;
463  dp3m.rs_mesh_dip[2][ind] =
464  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]] * tmp0;
465  ind++;
466  }
467  }
468  }
469  /* Back FFT force component mesh */
470  fft_perform_back(dp3m.rs_mesh_dip[0].data(), false, dp3m.fft,
471  comm_cart);
472  fft_perform_back(dp3m.rs_mesh_dip[1].data(), false, dp3m.fft,
473  comm_cart);
474  fft_perform_back(dp3m.rs_mesh_dip[2].data(), false, dp3m.fft,
475  comm_cart);
476  /* redistribute force component mesh */
477  std::array<double *, 3> meshes = {{dp3m.rs_mesh_dip[0].data(),
478  dp3m.rs_mesh_dip[1].data(),
479  dp3m.rs_mesh_dip[2].data()}};
480 
481  dp3m.sm.spread_grid(Utils::make_span(meshes), comm_cart,
482  dp3m.local_mesh.dim);
483  /* Assign force component from mesh to particle */
484  Utils::integral_parameter<AssignForces, 1, 7>(
485  dp3m.params.cao, dp3m, dipole_prefac * Utils::sqr(two_pi_L_i), d_rs,
486  particles);
487  }
488  } /* if (dp3m.sum_mu2 > 0) */
489  } /* if (force_flag) */
490 
491  if (dp3m.params.epsilon != P3M_EPSILON_METALLIC) {
492  auto const surface_term =
493  calc_surface_term(force_flag, energy_flag, particles);
494  if (this_node == 0)
495  k_space_energy_dip += surface_term;
496  }
497 
498  return k_space_energy_dip;
499 }

501 double DipolarP3M::calc_surface_term(bool force_flag, bool energy_flag,
502  ParticleRange const &particles) {
503  auto const pref = prefactor * 4. * Utils::pi() / box_geo.volume() /
504  (2. * dp3m.params.epsilon + 1.);
505  auto const n_local_part = particles.size();
506 
507  // We put all the dipolar momenta in a the arrays mx,my,mz according to the
508  // id-number of the particles
509  std::vector<double> mx(n_local_part);
510  std::vector<double> my(n_local_part);
511  std::vector<double> mz(n_local_part);
512 
513  int ip = 0;
514  for (auto const &p : particles) {
515  auto const dip = p.calc_dip();
516  mx[ip] = dip[0];
517  my[ip] = dip[1];
518  mz[ip] = dip[2];
519  ip++;
520  }
521 
522  // we will need the sum of all dipolar momenta vectors
523  auto local_dip = Utils::Vector3d{};
524  for (int i = 0; i < n_local_part; i++) {
525  local_dip[0] += mx[i];
526  local_dip[1] += my[i];
527  local_dip[2] += mz[i];
528  }
529  auto const box_dip =
530  boost::mpi::all_reduce(comm_cart, local_dip, std::plus<>());
531 
532  double energy = 0.;
533  if (energy_flag) {
534  double sum_e = 0.;
535  for (int i = 0; i < n_local_part; i++) {
536  sum_e += mx[i] * box_dip[0] + my[i] * box_dip[1] + mz[i] * box_dip[2];
537  }
538  energy =
539  0.5 * pref * boost::mpi::all_reduce(comm_cart, sum_e, std::plus<>());
540  }
541 
542  if (force_flag) {
543 
544  std::vector<double> sumix(n_local_part);
545  std::vector<double> sumiy(n_local_part);
546  std::vector<double> sumiz(n_local_part);
547 
548  for (int i = 0; i < n_local_part; i++) {
549  sumix[i] = my[i] * box_dip[2] - mz[i] * box_dip[1];
550  sumiy[i] = mz[i] * box_dip[0] - mx[i] * box_dip[2];
551  sumiz[i] = mx[i] * box_dip[1] - my[i] * box_dip[0];
552  }
553 
554  ip = 0;
555  for (auto &p : particles) {
556  auto &torque = p.torque();
557  torque[0] -= pref * sumix[ip];
558  torque[1] -= pref * sumiy[ip];
559  torque[2] -= pref * sumiz[ip];
560  ip++;
561  }
562  }
563 
564  return energy;
565 }

610  std::tuple<double, double, double, double>
611  calculate_accuracy(Utils::Vector3i const &mesh, int cao,
612  double r_cut_iL) const override {
613 
614  double alpha_L, rs_err, ks_err;
615 
616  /* calc maximal real space error for setting */
617  rs_err = dp3m_real_space_error(box_geo.length()[0], r_cut_iL,
618  dp3m.sum_dip_part, dp3m.sum_mu2, 0.001);
619  // alpha cannot be zero for dipoles because real-space formula breaks down
620 
621  if (Utils::sqrt_2() * rs_err > dp3m.params.accuracy) {
622  /* assume rs_err = ks_err -> rs_err = accuracy/sqrt(2.0) -> alpha_L */
623  alpha_L = dp3m_rtbisection(
624  box_geo.length()[0], r_cut_iL, dp3m.sum_dip_part, dp3m.sum_mu2,
625  0.0001 * box_geo.length()[0], 5. * box_geo.length()[0], 0.0001,
626  dp3m.params.accuracy);
627  } else {
628  /* even alpha=0 is ok, however, we cannot choose it since it kills the
629  k-space error formula.
630  Anyways, this very likely NOT the optimal solution */
631  alpha_L = 0.1;
632  }
633 
634  /* calculate real-space and k-space error for this alpha_L */
635  rs_err = dp3m_real_space_error(box_geo.length()[0], r_cut_iL,
636  dp3m.sum_dip_part, dp3m.sum_mu2, alpha_L);
637  ks_err = dp3m_k_space_error(box_geo.length()[0], mesh[0], cao,
638  dp3m.sum_dip_part, dp3m.sum_mu2, alpha_L);
639 
640  return {Utils::Vector2d{rs_err, ks_err}.norm(), rs_err, ks_err, alpha_L};
641  }

643  void determine_mesh_limits() override {

659  TuningAlgorithm::Parameters get_time() override {
660  auto tuned_params = TuningAlgorithm::Parameters{};
661  auto time_best = time_sentinel;
662  for (auto tmp_mesh = m_mesh_min; tmp_mesh <= m_mesh_max; tmp_mesh += 2) {
663  auto trial_params = TuningAlgorithm::Parameters{};
664  trial_params.mesh = Utils::Vector3i::broadcast(tmp_mesh);
665  trial_params.cao = cao_best;
666 
667  auto const trial_time =
668  get_m_time(trial_params.mesh, trial_params.cao, trial_params.r_cut_iL,
669  trial_params.alpha_L, trial_params.accuracy);
670 
671  /* this mesh does not work at all */
672  if (trial_time < 0.)
673  continue;
674 
675  /* the optimum r_cut for this mesh is the upper limit for higher meshes,
676  everything else is slower */
677  m_r_cut_iL_max = trial_params.r_cut_iL;
678 
679  if (trial_time < time_best) {
680  /* new optimum */
681  reset_n_trials();
682  tuned_params = trial_params;
683  time_best = tuned_params.time = trial_time;
684  } else if (trial_time > time_best + time_granularity or
685  get_n_trials() > max_n_consecutive_trials) {
686  /* no hope of further optimisation */
687  break;
688  }
689  }
690  return tuned_params;
691  }

726 /** Calculate the k-space error of dipolar-P3M */
727 static double dp3m_k_space_error(double box_size, int mesh, int cao,
728  int n_c_part, double sum_q2, double alpha_L) {
729  double he_q = 0.;
730  auto const mesh_i = 1. / mesh;
731  auto const alpha_L_i = 1. / alpha_L;
732 
733  for (int nx = -mesh / 2; nx < mesh / 2; nx++)
734  for (int ny = -mesh / 2; ny < mesh / 2; ny++)
735  for (int nz = -mesh / 2; nz < mesh / 2; nz++)
736  if ((nx != 0) || (ny != 0) || (nz != 0)) {
737  auto const n2 = Utils::sqr(nx) + Utils::sqr(ny) + Utils::sqr(nz);
738  auto const cs = p3m_analytic_cotangent_sum(nx, mesh_i, cao) *
739  p3m_analytic_cotangent_sum(ny, mesh_i, cao) *
740  p3m_analytic_cotangent_sum(nz, mesh_i, cao);
741  double alias1, alias2;
742  dp3m_tune_aliasing_sums(nx, ny, nz, mesh, mesh_i, cao, alpha_L_i,
743  &alias1, &alias2);
744  double d = alias1 - Utils::sqr(alias2 / cs) /
745  Utils::int_pow<3>(static_cast<double>(n2));
746  /* at high precision, d can become negative due to extinction;
747  also, don't take values that have no significant digits left*/
748  if (d > 0 && (fabs(d / alias1) > ROUND_ERROR_PREC))
749  he_q += d;
750  }
751 
752  return 8. * Utils::sqr(Utils::pi()) / 3. * sum_q2 * sqrt(he_q / n_c_part) /
753  Utils::int_pow<4>(box_size);
754 }

756 /** Tuning dipolar-P3M */
757 void dp3m_tune_aliasing_sums(int nx, int ny, int nz, int mesh, double mesh_i,
758  int cao, double alpha_L_i, double *alias1,
759  double *alias2) {
760  using Utils::sinc;
761 
762  auto const factor1 = Utils::sqr(Utils::pi() * alpha_L_i);
763 
764  *alias1 = *alias2 = 0.;
765  for (int mx = -P3M_BRILLOUIN; mx <= P3M_BRILLOUIN; mx++) {
766  auto const nmx = nx + mx * mesh;
767  auto const fnmx = mesh_i * nmx;
768  for (int my = -P3M_BRILLOUIN; my <= P3M_BRILLOUIN; my++) {
769  auto const nmy = ny + my * mesh;
770  auto const fnmy = mesh_i * nmy;
771  for (int mz = -P3M_BRILLOUIN; mz <= P3M_BRILLOUIN; mz++) {
772  auto const nmz = nz + mz * mesh;
773  auto const fnmz = mesh_i * nmz;
774 
775  auto const nm2 = Utils::sqr(nmx) + Utils::sqr(nmy) + Utils::sqr(nmz);
776  auto const ex = std::exp(-factor1 * nm2);
777 
778  auto const U2 = pow(sinc(fnmx) * sinc(fnmy) * sinc(fnmz), 2. * cao);
779 
780  *alias1 += Utils::sqr(ex) * nm2;
781  *alias2 += U2 * ex * pow((nx * nmx + ny * nmy + nz * nmz), 3.) / nm2;
782  }
783  }
784  }
785 }

787 /** Calculate the value of the errors for the REAL part of the force in terms
788  * of the splitting parameter alpha of Ewald. Based on eq. (33) @cite wang01a.
789  *
790  * Please note that in this more refined approach we don't use
791  * eq. (37), but eq. (33) which maintains all the powers in alpha.
792  */
793 double dp3m_real_space_error(double box_size, double r_cut_iL, int n_c_part,
794  double sum_q2, double alpha_L) {
795  double d_error_f, d_cc, d_dc, d_con;
796 
797  auto const d_rcut = r_cut_iL * box_size;
798  auto const d_rcut2 = Utils::sqr(d_rcut);
799  auto const d_rcut4 = Utils::sqr(d_rcut2);
800 
801  auto const d_a2 = Utils::sqr(alpha_L) / Utils::sqr(box_size);
802 
803  auto const d_c = sum_q2 * exp(-d_a2 * d_rcut2);
804 
805  d_cc = 4. * Utils::sqr(d_a2) * Utils::sqr(d_rcut2) + 6. * d_a2 * d_rcut2 + 3.;
806 
807  d_dc = 8. * Utils::int_pow<3>(d_a2) * Utils::int_pow<3>(d_rcut2) +
808  20. * Utils::sqr(d_a2) * d_rcut4 + 30. * d_a2 * d_rcut2 + 15.;
809 
810  d_con = 1. / sqrt(Utils::int_pow<3>(box_size) * Utils::sqr(d_a2) * d_rcut *
811  Utils::sqr(d_rcut4) * static_cast<double>(n_c_part));
812 
813  d_error_f = d_c * d_con *
814  sqrt((13. / 6.) * Utils::sqr(d_cc) +
815  (2. / 15.) * Utils::sqr(d_dc) - (13. / 15.) * d_cc * d_dc);
816 
817  return d_error_f;
818 }

820 /** Using bisection, find the root of a function "func-tuned_accuracy/sqrt(2.)"
821  * known to lie between x1 and x2. The root, returned as rtbis, will be
822  * refined until its accuracy is \f$\pm\f$ @p xacc.
823  */
824 double dp3m_rtbisection(double box_size, double r_cut_iL, int n_c_part,
825  double sum_q2, double x1, double x2, double xacc,
826  double tuned_accuracy) {
827  constexpr int JJ_RTBIS_MAX = 40;
828 
829  auto const constant = tuned_accuracy / Utils::sqrt_2();
830 
831  auto const f1 =
832  dp3m_real_space_error(box_size, r_cut_iL, n_c_part, sum_q2, x1) -
833  constant;
834  auto const f2 =
835  dp3m_real_space_error(box_size, r_cut_iL, n_c_part, sum_q2, x2) -
836  constant;
837  if (f1 * f2 >= 0.0) {
838  throw std::runtime_error(
839  "Root must be bracketed for bisection in dp3m_rtbisection");
840  }
841  // Orient the search dx, and set rtb to x1 or x2 ...
842  double dx;
843  double rtb = f1 < 0.0 ? (dx = x2 - x1, x1) : (dx = x1 - x2, x2);
844  for (int j = 1; j <= JJ_RTBIS_MAX; j++) {
845  auto const xmid = rtb + (dx *= 0.5);
846  auto const fmid =
847  dp3m_real_space_error(box_size, r_cut_iL, n_c_part, sum_q2, xmid) -
848  constant;
849  if (fmid <= 0.0)
850  rtb = xmid;
851  if (fabs(dx) < xacc || fmid == 0.0)
852  return rtb;
853  }
854  throw std::runtime_error("Too many bisections in dp3m_rtbisection");
855 }
