74/** External particle forces */
75static ParticleForce external_force(Particle const &p) {
76 ParticleForce f = {};
77
78#ifdef ESPRESSO_EXTERNAL_FORCES
79 f.f += p.ext_force();
80#ifdef ESPRESSO_ROTATION
81 f.torque += p.ext_torque();
82#endif
83#endif
84
85#ifdef ESPRESSO_ENGINE
86 // apply a swimming force in the direction of
87 // the particle's orientation axis
88 if (p.swimming().swimming and !p.swimming().is_engine_force_on_fluid) {
89 f.f += p.swimming().f_swim * p.calc_director();
90 }
91#endif
92
93 return f;
94}

96/** Combined force initialization and Langevin noise application */
97void init_forces_and_thermostat(System::System const &system) {
98#ifdef ESPRESSO_CALIPER
99 CALI_CXX_MARK_FUNCTION;
100#endif
101
102 auto &cell_structure = *system.cell_structure;
103 auto const &propagation = *system.propagation;
104 auto const &thermostat = *system.thermostat;
105 auto const kT = thermostat.kT;
106 auto const time_step = system.get_time_step();
107
108 // Check if Langevin thermostat is active
109 bool const langevin_active =
110 thermostat.langevin &&
111 (propagation.used_propagations &
112 (PropagationMode::TRANS_LANGEVIN | PropagationMode::ROT_LANGEVIN));
113
114 // Single pass over all local particles
115 cell_structure.for_each_local_particle([&](Particle &p) {
116 // Initialize force with external forces
117 p.force_and_torque() = external_force(p);
118
119 // Apply Langevin noise if thermostat is active
120 if (langevin_active) {
121 auto const &langevin = *thermostat.langevin;
122 if (propagation.should_propagate_with(p, PropagationMode::TRANS_LANGEVIN))
123 p.force() += friction_thermo_langevin(langevin, p, time_step, kT);
124#ifdef ESPRESSO_ROTATION
125 if (propagation.should_propagate_with(p, PropagationMode::ROT_LANGEVIN))
126 p.torque() += convert_vector_body_to_space(
127 p, friction_thermo_langevin_rotation(langevin, p, time_step, kT));
128#endif
129 }
130 });
131#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
132 cell_structure.reset_local_force();
133#endif
134
135 // Initialize ghost forces (unchanged)
136 init_forces_ghosts(cell_structure);
137}

144static void force_capping(CellStructure &cell_structure, double force_cap) {
145 if (force_cap > 0.) {
146 auto const force_cap_sq = Utils::sqr(force_cap);
147 cell_structure.for_each_local_particle(
148 [&force_cap, &force_cap_sq](Particle &p) {
149 auto const force_sq = p.force().norm2();
150 if (force_sq > force_cap_sq) {
151 p.force() *= force_cap / std::sqrt(force_sq);
152 }
153 });
154 }
155}

164void System::System::calculate_forces() {
165#ifdef ESPRESSO_CALIPER
166 CALI_CXX_MARK_FUNCTION;
167#endif
168#ifdef ESPRESSO_CUDA
169#ifdef ESPRESSO_CALIPER
170 CALI_MARK_BEGIN("copy_particles_to_GPU");
171#endif
172 gpu.update();
173#ifdef ESPRESSO_CALIPER
174 CALI_MARK_END("copy_particles_to_GPU");
175#endif
176#endif // ESPRESSO_CUDA
177
178#ifdef ESPRESSO_COLLISION_DETECTION
179 collision_detection->clear_queue();
180 auto const collision_detection_cutoff = collision_detection->cutoff();
181#else
182 auto const collision_detection_cutoff = inactive_cutoff;
183#endif
184 bond_breakage->clear_queue();
185 auto particles = cell_structure->local_particles();
186#ifdef ESPRESSO_NPT
187 if (propagation->used_propagations & PropagationMode::TRANS_LANGEVIN_NPT) {
188 // reset virial part of instantaneous pressure
189 npt_inst_pressure->p_vir = Utils::Vector3d{};
190 }
191#endif
192#ifdef ESPRESSO_DIPOLE_FIELD_TRACKING
193 // reset dipole field
194 reinit_dip_fld(*cell_structure);
195#endif
196
197 // Use combined function instead of two separate calls
198
199 auto const elc_kernel = coulomb.pair_force_elc_kernel();
200 auto const coulomb_kernel = coulomb.pair_force_kernel();
201 auto const dipoles_kernel = dipoles.pair_force_kernel();
202 auto const coulomb_u_kernel = coulomb.pair_energy_kernel();
203 auto *const virial = get_npt_virial();
204
205 // interaction kernel is defined
206 auto bond_kernel = [coulomb_kernel_ptr = get_ptr(coulomb_kernel),
207 &bonded_ias = *bonded_ias,
208 &bond_breakage = *bond_breakage, virial,
209 &box_geo = *box_geo](Particle &p1, int bond_id,
210 std::span<Particle *> partners) {
211 return add_bonded_force(p1, bond_id, partners, bonded_ias, bond_breakage,
212 box_geo, virial, coulomb_kernel_ptr);
213 };
214
215 VerletCriterion<> const verlet_criterion{*this,
216 cell_structure->get_verlet_skin(),
217 get_interaction_range(),
218 coulomb.cutoff(),
219 dipoles.cutoff(),
220 collision_detection_cutoff};
221
222#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
223#ifdef ESPRESSO_CALIPER
224 CALI_MARK_BEGIN("convert particles AoS to SoA");
225#endif
226 update_cabana_state(*cell_structure, verlet_criterion,
227 get_interaction_range(), propagation->integ_switch);
228#ifdef ESPRESSO_CALIPER
229 CALI_MARK_END("convert particles AoS to SoA");
230#endif
231#endif
232#ifdef ESPRESSO_ELECTROSTATICS
233 if (coulomb.impl->extension) {
234 update_icc_particles();
235 }
236#endif // ESPRESSO_ELECTROSTATICS
237 init_forces_and_thermostat(*this);
238 calc_long_range_forces(particles);
239
240#ifdef ESPRESSO_SHARED_MEMORY_PARALLELISM
241#ifdef ESPRESSO_CALIPER
242 CALI_MARK_BEGIN("parallel short range");
243#endif
244 using execution_space = Kokkos::DefaultExecutionSpace;
245 auto const &unique_particles = cell_structure->get_unique_particles();
246 auto const &local_force = cell_structure->get_local_force();
247#ifdef ESPRESSO_ROTATION
248 auto const &local_torque = cell_structure->get_local_torque();
249#endif
250#ifdef ESPRESSO_NPT
251 auto const &local_virial = cell_structure->get_local_virial();
252#endif
253 auto const &aosoa = cell_structure->get_aosoa();
254
255 ForcesKernel first_neighbor_kernel(
256 *bonded_ias, *nonbonded_ias, get_ptr(coulomb_kernel),
257 get_ptr(dipoles_kernel), get_ptr(elc_kernel), get_ptr(coulomb_u_kernel),
258 *thermostat, *box_geo, unique_particles, local_force,
259#ifdef ESPRESSO_ROTATION
260 local_torque,
261#endif
262#ifdef ESPRESSO_NPT
263 virial, local_virial,
264#endif
265 aosoa);
266
267 cabana_short_range(bond_kernel, first_neighbor_kernel, *cell_structure,
268 get_interaction_range(), bonded_ias->maximal_cutoff(),
269 verlet_criterion, propagation->integ_switch);
270 // Force and Torque reduction
271 int num_threads = execution_space().concurrency();
272 Kokkos::RangePolicy<execution_space> policy(std::size_t{0},
273 unique_particles.size());
274 Kokkos::parallel_for("reduction", policy,
275 [&local_force,
276#ifdef ESPRESSO_ROTATION
277 &local_torque,
278#endif
279 &unique_particles, num_threads](std::size_t const i) {
280 Utils::Vector3d force{};
281#ifdef ESPRESSO_ROTATION
282 Utils::Vector3d torque{};
283#endif
284 for (int tid = 0; tid < num_threads; ++tid) {
285 force[0] += local_force(i, tid, 0);
286 force[1] += local_force(i, tid, 1);
287 force[2] += local_force(i, tid, 2);
288#ifdef ESPRESSO_ROTATION
289 torque[0] += local_torque(i, tid, 0);
290 torque[1] += local_torque(i, tid, 1);
291 torque[2] += local_torque(i, tid, 2);
292#endif
293 }
294 unique_particles.at(i)->force() += force;
295#ifdef ESPRESSO_ROTATION
296 unique_particles.at(i)->torque() += torque;
297#endif
298 });
299 Kokkos::fence();
300
301#ifdef ESPRESSO_NPT
302 if (virial) {
303 for (int tid = 0; tid < num_threads; ++tid) {
304 (*virial)[0] += local_virial(tid, 0);
305 (*virial)[1] += local_virial(tid, 1);
306 (*virial)[2] += local_virial(tid, 2);
307 }
308 }
309#endif
310
311#ifdef ESPRESSO_COLLISION_DETECTION
312 auto collision_kernel = [&collision_detection = *collision_detection](
313 Particle const &p1, Particle const &p2,
314 Distance const &d) {
315 if (not collision_detection.is_off()) {
316 collision_detection.detect_collision(p1, p2, d.dist2);
317 }
318 };
319 cell_structure->non_bonded_loop(collision_kernel, verlet_criterion);
320#endif
321
322#ifdef ESPRESSO_CALIPER
323 CALI_MARK_END("parallel short range");
324#endif
325
326#else // ESPRESSO_SHARED_MEMORY_PARALLELISM
327
328 auto pair_kernel = [coulomb_kernel_ptr = get_ptr(coulomb_kernel),
329 dipoles_kernel_ptr = get_ptr(dipoles_kernel),
330 elc_kernel_ptr = get_ptr(elc_kernel),
331 coulomb_u_kernel_ptr = get_ptr(coulomb_u_kernel),
332 &nonbonded_ias = *nonbonded_ias,
333 &thermostat = *thermostat, &bonded_ias = *bonded_ias,
334 virial,
335#ifdef ESPRESSO_COLLISION_DETECTION
336 &collision_detection = *collision_detection,
337#endif
338 &box_geo = *box_geo](Particle &p1, Particle &p2,
339 Distance const &d) {
340 auto const &ia_params = nonbonded_ias.get_ia_param(p1.type(), p2.type());
341 add_non_bonded_pair_force(
342 p1, p2, d.vec21, sqrt(d.dist2), d.dist2, p1.q() * p2.q(), ia_params,
343 thermostat, box_geo, bonded_ias, virial, coulomb_kernel_ptr,
344 dipoles_kernel_ptr, elc_kernel_ptr, coulomb_u_kernel_ptr);
345#ifdef ESPRESSO_COLLISION_DETECTION
346 if (not collision_detection.is_off()) {
347 collision_detection.detect_collision(p1, p2, d.dist2);
348 }
349#endif
350 };
351
352 short_range_loop(bond_kernel, pair_kernel, *cell_structure, maximal_cutoff(),
353 bonded_ias->maximal_cutoff(), verlet_criterion);
354
355#endif // ESPRESSO_SHARED_MEMORY_PARALLELISM
356
357 constraints->add_forces(particles, get_sim_time());
358 oif_global->calculate_forces();
359
360 // Must be done here. Forces need to be ghost-communicated
361 immersed_boundaries->volume_conservation(*cell_structure);
362
363 if (thermostat->lb and (propagation->used_propagations &
364 PropagationMode::TRANS_LB_MOMENTUM_EXCHANGE)) {
365 lb_couple_particles();
366 }
367
368#ifdef ESPRESSO_CUDA
369#ifdef ESPRESSO_CALIPER
370 CALI_MARK_BEGIN("copy_forces_from_GPU");
371#endif
372 gpu.copy_forces_to_host(particles, this_node);
373
374#ifdef ESPRESSO_DIPOLE_FIELD_TRACKING
375 gpu.copy_dip_fld_to_host(particles, this_node);
376#endif
377
378#ifdef ESPRESSO_CALIPER
379 CALI_MARK_END("copy_forces_from_GPU");
380#endif
381#endif // ESPRESSO_CUDA
382
383#ifdef ESPRESSO_VIRTUAL_SITES_RELATIVE
384 if (propagation->used_propagations &
385 (PropagationMode::TRANS_VS_RELATIVE | PropagationMode::ROT_VS_RELATIVE)) {
386 vs_relative_back_transfer_forces_and_torques(*cell_structure);
387 }
388#endif
389
390 // Communication step: ghost forces
391 cell_structure->ghosts_reduce_forces();
392
393 // should be pretty late, since it needs to zero out the total force
394 comfixed->apply(particles);
395
396 // Needs to be the last one to be effective
397 force_capping(*cell_structure, force_cap);
398
399 // mark that forces are now up-to-date
400 propagation->recalc_forces = false;
401}
