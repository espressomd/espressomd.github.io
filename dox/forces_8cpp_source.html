65 /** External particle forces */
66 inline ParticleForce external_force(Particle const &p) {
67  ParticleForce f = {};
68 
69 #ifdef EXTERNAL_FORCES
70  f.f += p.ext_force();
71 #ifdef ROTATION
72  f.torque += p.ext_torque();
73 #endif
74 #endif
75 
76 #ifdef ENGINE
77  // apply a swimming force in the direction of
78  // the particle's orientation axis
79  if (p.swimming().swimming) {
80  f.f += p.swimming().f_swim * p.calc_director();
81  }
82 #endif
83 
84  return f;
85 }

87 inline ParticleForce thermostat_force(Particle const &p, double time_step,
88  double kT) {
89  extern LangevinThermostat langevin;
90  if (!(thermo_switch & THERMO_LANGEVIN)) {
91  return {};
92  }
93 
94 #ifdef ROTATION
95  return {friction_thermo_langevin(langevin, p, time_step, kT),
96  p.can_rotate() ? convert_vector_body_to_space(
97  p, friction_thermo_langevin_rotation(
98  langevin, p, time_step, kT))
99  : Utils::Vector3d{}};
100 #else
101  return friction_thermo_langevin(langevin, p, time_step, kT);
102 #endif
103 }

111 static void init_forces(const ParticleRange &particles,
112  const ParticleRange &ghost_particles, double time_step,
113  double kT) {
114  ESPRESSO_PROFILER_CXX_MARK_FUNCTION;
115  /* The force initialization depends on the used thermostat and the
116  thermodynamic ensemble */
117 
118 #ifdef NPT
119  npt_reset_instantaneous_virials();
120 #endif
121 
122  /* initialize forces with Langevin thermostat forces
123  or zero depending on the thermostat
124  set torque to zero for all and rescale quaternions
125  */
126  for (auto &p : particles) {
127  p.f = init_real_particle_force(p, time_step, kT);
128  }
129 
130  /* initialize ghost forces with zero
131  set torque to zero for all and rescale quaternions
132  */
133  for (auto &p : ghost_particles) {
134  p.f = init_ghost_force(p);
135  }
136 }

144 void force_calc(CellStructure &cell_structure, double time_step, double kT) {
145  ESPRESSO_PROFILER_CXX_MARK_FUNCTION;
146 
147  auto &espresso_system = EspressoSystemInterface::Instance();
148  espresso_system.update();
149 
150 #ifdef COLLISION_DETECTION
151  prepare_local_collision_queue();
152 #endif
153  BondBreakage::clear_queue();
154  auto particles = cell_structure.local_particles();
155  auto ghost_particles = cell_structure.ghost_particles();
156 #ifdef ELECTROSTATICS
157  if (electrostatics_extension) {
158  if (auto icc = boost::get<std::shared_ptr<ICCStar>>(
159  electrostatics_extension.get_ptr())) {
160  (**icc).iteration(cell_structure, particles, ghost_particles);
161  }
162  }
163 #endif
164  init_forces(particles, ghost_particles, time_step, kT);
165 
166  calc_long_range_forces(particles);
167 
168  auto const elc_kernel = Coulomb::pair_force_elc_kernel();
169  auto const coulomb_kernel = Coulomb::pair_force_kernel();
170  auto const dipoles_kernel = Dipoles::pair_force_kernel();
171 
172 #ifdef ELECTROSTATICS
173  auto const coulomb_cutoff = Coulomb::cutoff();
174 #else
175  auto const coulomb_cutoff = INACTIVE_CUTOFF;
176 #endif
177 
178 #ifdef DIPOLES
179  auto const dipole_cutoff = Dipoles::cutoff();
180 #else
181  auto const dipole_cutoff = INACTIVE_CUTOFF;
182 #endif
183 
184  short_range_loop(
185  [coulomb_kernel_ptr = coulomb_kernel.get_ptr()](
186  Particle &p1, int bond_id, Utils::Span<Particle *> partners) {
187  return add_bonded_force(p1, bond_id, partners, coulomb_kernel_ptr);
188  },
189  [coulomb_kernel_ptr = coulomb_kernel.get_ptr(),
190  dipoles_kernel_ptr = dipoles_kernel.get_ptr(),
191  elc_kernel_ptr = elc_kernel.get_ptr()](Particle &p1, Particle &p2,
192  Distance const &d) {
193  add_non_bonded_pair_force(p1, p2, d.vec21, sqrt(d.dist2), d.dist2,
194  coulomb_kernel_ptr, dipoles_kernel_ptr,
195  elc_kernel_ptr);
196 #ifdef COLLISION_DETECTION
197  if (collision_params.mode != CollisionModeType::OFF)
198  detect_collision(p1, p2, d.dist2);
199 #endif
200  },
201  maximal_cutoff(n_nodes), maximal_cutoff_bonded(),
202  VerletCriterion<>{skin, interaction_range(), coulomb_cutoff,
203  dipole_cutoff, collision_detection_cutoff()});
204 
205  Constraints::constraints.add_forces(particles, get_sim_time());
206 
207  if (max_oif_objects) {
208  // There are two global quantities that need to be evaluated:
209  // object's surface and object's volume.
210  for (int i = 0; i < max_oif_objects; i++) {
211  auto const area_volume = calc_oif_global(i, cell_structure);
212  if (fabs(area_volume[0]) < 1e-100 && fabs(area_volume[1]) < 1e-100)
213  break;
214  add_oif_global_forces(area_volume, i, cell_structure);
215  }
216  }
217 
218  // Must be done here. Forces need to be ghost-communicated
219  immersed_boundaries.volume_conservation(cell_structure);
220 
221  lb_lbcoupling_calc_particle_lattice_ia(thermo_virtual, particles,
222  ghost_particles, time_step);
223 
224 #ifdef CUDA
225  copy_forces_from_GPU(particles, this_node);
226 #endif
227 
228 // VIRTUAL_SITES distribute forces
229 #ifdef VIRTUAL_SITES
230  virtual_sites()->back_transfer_forces_and_torques();
231 #endif
232 
233  // Communication Step: ghost forces
234  cell_structure.ghosts_reduce_forces();
235 
236  // should be pretty late, since it needs to zero out the total force
237  comfixed.apply(comm_cart, particles);
238 
239  // Needs to be the last one to be effective
240  forcecap_cap(particles);
241 
242  // mark that forces are now up-to-date
243  recalc_forces = false;
244 }
