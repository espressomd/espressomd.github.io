99 void on_particle_type_change(int p_id, int type) {
100  if (type_list_enable) {
101  // check if the particle exists already and the type is changed, then remove
102  // it from the list which contains it
103  auto const &cur_par = get_particle_data(p_id);
104  int prev_type = cur_par.type();
105  if (prev_type != type) {
106  // particle existed before so delete it from the list
107  remove_id_from_map(p_id, prev_type);
108  }
109  add_id_to_type_map(p_id, type);
110  }

172 /**
173  * @brief Get multiple particles at once.
174  *
175  * *WARNING* Particles are returned in an arbitrary order.
176  *
177  * @param ids The ids of the particles that should be returned.
178  *
179  * @returns The particle list.
180  */
181 static std::vector<Particle> mpi_get_particles(Utils::Span<const int> ids) {
182  mpi_call(mpi_get_particles_local);
183  /* Return value */
184  std::vector<Particle> parts(ids.size());
185 
186  /* Group ids per node */
187  static std::vector<std::vector<int>> node_ids(comm_cart.size());
188  for (auto &per_node : node_ids) {
189  per_node.clear();
190  }
191 
192  for (auto const &p_id : ids) {
193  auto const p_node = get_particle_node(p_id);
194 
195  if (p_node != this_node)
196  node_ids[p_node].push_back(p_id);
197  }
198 
199  /* Distributed ids to the nodes */
200  {
201  static std::vector<int> ignore;
202  boost::mpi::scatter(comm_cart, node_ids, ignore, 0);
203  }
204 
205  /* Copy local particles */
206  std::transform(node_ids[this_node].cbegin(), node_ids[this_node].cend(),
207  parts.begin(), [](int p_id) {
208  assert(cell_structure.get_local_particle(p_id));
209  return *cell_structure.get_local_particle(p_id);
210  });
211 
212  static std::vector<int> node_sizes(comm_cart.size());
213  std::transform(
214  node_ids.cbegin(), node_ids.cend(), node_sizes.begin(),
215  [](std::vector<int> const &ids) { return static_cast<int>(ids.size()); });
216 
217  Utils::Mpi::gatherv(comm_cart, parts.data(), static_cast<int>(parts.size()),
218  parts.data(), node_sizes.data(), 0);
219 
220  return parts;
221 }

223 void prefetch_particle_data(Utils::Span<const int> in_ids) {
224  /* Nothing to do on a single node. */
225  // NOLINTNEXTLINE(clang-analyzer-core.NonNullParamChecker)
226  if (comm_cart.size() == 1)
227  return;
228 
229  static std::vector<int> ids;
230  ids.clear();
231 
232  boost::algorithm::copy_if(in_ids, std::back_inserter(ids), [](int id) {
233  return (get_particle_node(id) != this_node) && particle_fetch_cache.has(id);
234  });
235 
236  /* Don't prefetch more particles than fit the cache. */
237  if (ids.size() > particle_fetch_cache.max_size())
238  ids.resize(particle_fetch_cache.max_size());
239 
240  /* Fetch the particles... */
241  for (auto &p : mpi_get_particles(ids)) {
242  auto id = p.id();
243  particle_fetch_cache.put(id, std::move(p));
244  }
245 }

267 static void mpi_who_has() {
268  mpi_call(mpi_who_has_local);
269 
270  auto local_particles = cell_structure.local_particles();
271 
272  static std::vector<int> n_parts;
273  boost::mpi::gather(comm_cart, static_cast<int>(local_particles.size()),
274  n_parts, 0);
275 
276  static std::vector<int> pdata;
277  max_seen_pid = -1;
278 
279  /* then fetch particle locations */
280  for (int pnode = 0; pnode < n_nodes; pnode++) {
281  if (pnode == this_node) {
282  for (auto const &p : local_particles) {
283  particle_node[p.id()] = this_node;
284  max_seen_pid = std::max(max_seen_pid, p.id());
285  }
286  } else if (n_parts[pnode] > 0) {
287  pdata.resize(n_parts[pnode]);
288  comm_cart.recv(pnode, some_tag, pdata);
289  for (int i = 0; i < n_parts[pnode]; i++) {
290  particle_node[pdata[i]] = pnode;
291  max_seen_pid = std::max(max_seen_pid, pdata[i]);
292  }
293  }
294  }

322 /**
323  * @brief Calculate the largest particle id.
324  * Traversing the @ref particle_node to find the largest particle id
325  * scales with O(N). Consider using the cached value in @ref max_seen_pid
326  * if possible. This function is only necessary when the cached value is
327  * invalidated, for example when removing the particle which has the
328  * largest id.
329  */
330 static int calculate_max_seen_id() {
331  return boost::accumulate(particle_node, -1,
332  [](int max, const std::pair<int, int> &kv) {
333  return std::max(max, kv.first);
334  });

465 void remove_particle(int p_id) {
466  auto const &cur_par = get_particle_data(p_id);
467  if (type_list_enable) {
468  // remove particle from its current type_list
469  int type = cur_par.type();
470  remove_id_from_map(p_id, type);
471  }
472 
473  particle_node[p_id] = -1;
474  mpi_call_all(mpi_remove_particle_local, p_id);
475  particle_node.erase(p_id);
476  if (p_id == max_seen_pid) {
477  --max_seen_pid;
478  // if there is a gap (i.e. there is no particle with id max_seen_pid - 1,
479  // then the cached value is invalidated and has to be recomputed (slow)
480  if (particle_node.count(max_seen_pid) == 0 or
481  particle_node[max_seen_pid] == -1) {
482  max_seen_pid = calculate_max_seen_id();
483  }
484  }
485 }
