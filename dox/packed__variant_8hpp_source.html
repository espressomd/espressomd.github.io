37  * This assigns every ObjectHandle a unique id.

39 inline ObjectId object_id(const ObjectHandle *p) {
40  // NOLINTNEXTLINE(bugprone-sizeof-expression)
41  static_assert(sizeof(const ObjectHandle *) <= sizeof(ObjectId), "");
42  // Use the pointer value as the unique identifier.
43  // This function is only called on the head node.
44  return reinterpret_cast<ObjectId>(p);
45 }

63 /**
64  * @brief Visitor that converts a Variant to a PackedVariant.
65  *
66  * While packing, keeps track of all the ObjectRef values that
67  * were encountered and stores them. This also keeps the
68  * referees alive if there are no other owners.
69  */
70 struct PackVisitor : boost::static_visitor<PackedVariant> {
71 private:
72  mutable std::unordered_map<ObjectId, ObjectRef> m_objects;
73 
74 public:
75  /** @brief Map of objects whose references were replaced by ids. */
76  auto const &objects() const { return m_objects; }
77 
78  /* For the vector, we recurse into each element. */
79  auto operator()(const std::vector<Variant> &vec) const {
80  std::vector<PackedVariant> ret(vec.size());
81 
82  boost::transform(vec, ret.begin(), [this](const Variant &v) {
83  return boost::apply_visitor(*this, v);
84  });
85 
86  return ret;
87  }
88 
89  /* For the map, we recurse into each element. */
90  template <typename K>
91  auto operator()(const std::unordered_map<K, Variant> &map) const {
92  std::unordered_map<K, PackedVariant> ret{};
93 
94  for (auto const &it : map) {
95  ret.insert({it.first, boost::apply_visitor(*this, it.second)});
96  }
97 
98  return ret;
99  }
100 
101  /* For object references we store the object reference, and
102  * replace it by just an id. */
103  PackedVariant operator()(const ObjectRef &so_ptr) const {
104  auto const oid = object_id(so_ptr.get());
105  m_objects[oid] = so_ptr;
106 
107  return oid;
108  }
109 
110  /* Regular value are just verbatim copied into the result. */
111  template <class T> PackedVariant operator()(T &&val) const {
112  return std::forward<T>(val);
113  }
114 };

116 /**
117  * @brief Visitor that converts a PackedVariant to a Variant.
118  *
119  * ObjectId are replaced according to the provided object map.
120  */
121 struct UnpackVisitor : boost::static_visitor<Variant> {
122  std::unordered_map<ObjectId, ObjectRef> const &objects;
123 
124  explicit UnpackVisitor(std::unordered_map<ObjectId, ObjectRef> const &objects)
125  : objects(objects) {}
126 
127  /* For the vector, we recurse into each element. */
128  auto operator()(const std::vector<PackedVariant> &vec) const {
129  std::vector<Variant> ret(vec.size());
130 
131  boost::transform(vec, ret.begin(), [this](const PackedVariant &v) {
132  return boost::apply_visitor(*this, v);
133  });
134 
135  return ret;
136  }
137 
138  /* For the map, we recurse into each element. */
139  template <typename K>
140  auto operator()(const std::unordered_map<K, PackedVariant> &map) const {
141  std::unordered_map<K, Variant> ret{};
142 
143  for (auto const &it : map) {
144  ret.insert({it.first, boost::apply_visitor(*this, it.second)});
145  }
146 
147  return ret;
148  }
149 
150  /* Regular value are just verbatim copied into the result. */
151  template <class T> Variant operator()(T &&val) const {
152  return std::forward<T>(val);
153  }
154 
155  /* For object id's they are replaced by references according to the map. */
156  Variant operator()(const ObjectId &id) const { return objects.at(id); }
157 };

185 /**
186  * @brief Pack a VariantMap.
187  *
188  * Applies @ref pack to every value in the
189  * input map.
190  */
191 inline PackedMap pack(const VariantMap &v) {
192  PackedMap ret(v.size());
193 
194  boost::transform(v, ret.begin(), [](auto const &kv) {
195  return std::pair<std::string, PackedVariant>{kv.first, pack(kv.second)};

207 inline VariantMap

210  VariantMap ret;
