96 void HybridDecomposition::resort(bool global,
97  std::vector<ParticleChange> &diff) {
98  ParticleList displaced_parts;
99 
100  /* Check for n_square type particles in regular decomposition */
101  for (auto &c : m_regular_decomposition.local_cells()) {
102  for (auto it = c->particles().begin(); it != c->particles().end();) {
103  /* Particle is in the right decomposition, i.e. has no n_square type */
104  if (not is_n_square_type(it->type())) {
105  std::advance(it, 1);
106  continue;
107  }
108 
109  /* else remove from current cell ... */
110  auto p = std::move(*it);
111  it = c->particles().erase(it);
112  diff.emplace_back(ModifiedList{c->particles()});
113  diff.emplace_back(RemovedParticle{p.id()});
114 
115  /* ... and insert into a n_square cell */
116  auto const first_local_cell = m_n_square.get_local_cells()[0];
117  first_local_cell->particles().insert(std::move(p));
118  diff.emplace_back(ModifiedList{first_local_cell->particles()});
119  }
120 
121  /* Now check for regular decomposition type particles in n_square */
122  for (auto &c : m_n_square.local_cells()) {
123  for (auto it = c->particles().begin(); it != c->particles().end();) {
124  /* Particle is of n_square type */
125  if (is_n_square_type(it->type())) {
126  std::advance(it, 1);
127  continue;
128  }
129 
130  /* else remove from current cell ... */
131  auto p = std::move(*it);
132  it = c->particles().erase(it);
133  diff.emplace_back(ModifiedList{c->particles()});
134  diff.emplace_back(RemovedParticle{p.id()});
135 
136  /* ... and insert in regular decomposition */
137  auto const target_cell = particle_to_cell(p);
138  /* if particle belongs to this node insert it into correct cell */
139  if (target_cell != nullptr) {
140  target_cell->particles().insert(std::move(p));
141  diff.emplace_back(ModifiedList{target_cell->particles()});
142  }
143  /* otherwise just put into regular decomposition */
144  else {
145  auto first_local_cell = m_regular_decomposition.get_local_cells()[0];
146  first_local_cell->particles().insert(std::move(p));
147  diff.emplace_back(ModifiedList{first_local_cell->particles()});
148  }
149  }
150  }
151  }
152 
153  /* now resort into correct cells within the respective decompositions */
154  m_regular_decomposition.resort(global, diff);
155  m_n_square.resort(global, diff);
156 
157  /* basically do CellStructure::ghost_count() */
158  ghost_communicator(exchange_ghosts_comm(), GHOSTTRANS_PARTNUM);
159 
160  /* basically do CellStructure::ghost_update(unsigned data_parts) */
161  ghost_communicator(exchange_ghosts_comm(),
162  map_data_parts(global_ghost_flags()));
163 }

165 std::size_t HybridDecomposition::count_particles(
166  std::vector<Cell *> const &local_cells) const {
167  std::size_t count_local = 0;
168  std::size_t count_global = 0;
169  for (auto const &cell : local_cells) {
170  count_local += cell->particles().size();
171  }
172  boost::mpi::reduce(m_comm, count_local, count_global, std::plus<>{}, 0);
173  return count_global;
