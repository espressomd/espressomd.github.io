39 void lb_tracers_add_particle_force_to_fluid(CellStructure &cell_structure,
40  BoxGeometry const &box_geo,
41  LocalBox const &local_box,
42  LB::Solver &lb, double time_step) {
43  if (lb_sanity_checks(lb)) {
44  return;
45  }
46  auto const agrid = lb.get_agrid();
47  auto const to_lb_units = 1. / agrid;
48 
49  // Distribute summed-up forces from physical particles to ghosts
50  init_forces_ghosts(cell_structure.ghost_particles());
51  cell_structure.update_ghosts_and_resort_particle(Cells::DATA_PART_FORCE);
52 
53  // Keep track of ghost particles (ids) that have already been coupled
54  LB::CouplingBookkeeping bookkeeping{cell_structure};
55  // Apply particle forces to the LB fluid at particle positions.
56  // For physical particles, also set particle velocity = fluid velocity.
57  for (auto const &particle_range :
58  {cell_structure.local_particles(), cell_structure.ghost_particles()}) {
59  for (auto const &p : particle_range) {
60  if (!LB::is_tracer(p))
61  continue;
62  if (bookkeeping.should_be_coupled(p)) {
63  for (auto const &pos :
64  positions_in_halo(p.pos(), box_geo, local_box, agrid)) {
65  add_md_force(lb, pos * to_lb_units, p.force(), time_step);
66  }
67  }
68  }
69  }
70 
71  // Clear ghost forces to avoid double counting later
72  init_forces_ghosts(cell_structure.ghost_particles());
73 }

75 void lb_tracers_propagate(CellStructure &cell_structure, LB::Solver const &lb,
76  double time_step) {
77  if (lb_sanity_checks(lb)) {
78  return;
79  }
80  auto const verlet_skin = cell_structure.get_verlet_skin();
81  auto const verlet_skin_sq = verlet_skin * verlet_skin;
82 
83  // Advect particles
84  for (auto &p : cell_structure.local_particles()) {
85  if (!LB::is_tracer(p))
86  continue;
87  p.v() = lb.get_coupling_interpolated_velocity(p.pos());
88  for (unsigned int i = 0u; i < 3u; i++) {
89  if (!p.is_fixed_along(i)) {
90  p.pos()[i] += p.v()[i] * time_step;
91  }
92  }
93  // Verlet list update check
94  if ((p.pos() - p.pos_at_last_verlet_update()).norm2() > verlet_skin_sq) {
95  cell_structure.set_resort_particles(Cells::RESORT_LOCAL);
96  }
97  }
98 }
