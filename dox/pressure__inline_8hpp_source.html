47/** Calculate non-bonded energies between a pair of particles.
48 * @param p1 pointer to particle 1.
49 * @param p2 pointer to particle 2.
50 * @param d vector between p1 and p2.
51 * @param dist distance between p1 and p2.
52 * @param ia_params non-bonded interaction kernels.
53 * @param dipoles Magnetostatics solver.
54 * @param bonded_ias bonded interaction kernels.
55 * @param kernel_forces Coulomb force kernel.
56 * @param kernel_pressure Coulomb pressure kernel.
57 * @param[in,out] obs_pressure pressure observable.
58 */
59inline void add_non_bonded_pair_virials(
60 Particle const &p1, Particle const &p2, Utils::Vector3d const &d,
61 double dist, IA_parameters const &ia_params,
62 [[maybe_unused]] BondedInteractionsMap const &bonded_ias,
63 Dipoles::Solver const &dipoles,
64 Coulomb::ShortRangeForceKernel::kernel_type const *kernel_forces,
65 Coulomb::ShortRangePressureKernel::kernel_type const *kernel_pressure,
66 Observable_stat &obs_pressure) {
67#ifdef ESPRESSO_EXCLUSIONS
68 if (do_nonbonded(p1, p2))
69#endif
70 {
71 auto f = calc_non_central_force(p1, p2, ia_params, d, dist).f;
72 f += calc_central_radial_force(ia_params, d, dist);
73#ifdef ESPRESSO_THOLE
74 f +=
75 thole_pair_force(p1, p2, ia_params, d, dist, bonded_ias, kernel_forces);
76#endif
77 auto const stress = Utils::tensor_product(d, f);
78 obs_pressure.add_non_bonded_contribution(p1.type(), p2.type(), p1.mol_id(),
79 p2.mol_id(), flatten(stress));
80 }
81
82#ifdef ESPRESSO_ELECTROSTATICS
83 if (!obs_pressure.coulomb.empty() and kernel_pressure != nullptr) {
84 /* real space Coulomb */
85 auto const p_coulomb = (*kernel_pressure)(p1.q() * p2.q(), d, dist);
86
87 for (std::size_t i = 0u; i < 3u; i++) {
88 for (std::size_t j = 0u; j < 3u; j++) {
89 obs_pressure.coulomb[i * 3u + j] += p_coulomb(i, j);
90 }
91 }
92 }
93#endif // ESPRESSO_ELECTROSTATICS
94
95#ifdef ESPRESSO_DIPOLES
96 /* real space magnetic dipole-dipole */
97 if (dipoles.impl->solver) {
98 fprintf(stderr, "calculating pressure for magnetostatics which doesn't "
99 "have it implemented\n");
100 }
101#endif // ESPRESSO_DIPOLES
102}

104inline std::optional<Utils::Matrix<double, 3, 3>>
105calc_bonded_virial_pressure_tensor(
106 Bonded_IA_Parameters const &iaparams, Particle const &p1,
107 Particle const &p2, BoxGeometry const &box_geo,
108 Coulomb::ShortRangeForceKernel::kernel_type const *kernel) {
109 auto const dx = box_geo.get_mi_vector(p1.pos(), p2.pos());
110 auto const pair_force = calc_bond_pair_force(iaparams, p1, p2, dx, kernel);
111 std::optional<Utils::Matrix<double, 3, 3>> pressure{std::nullopt};
112 if (pair_force) {
113 pressure = Utils::tensor_product(*pair_force, dx);
114 }
115 return pressure;
116}

118inline std::optional<Utils::Matrix<double, 3, 3>>
119calc_bonded_three_body_pressure_tensor(Bonded_IA_Parameters const &iaparams,
120 Particle const &p1, Particle const &p2,
121 Particle const &p3,
122 BoxGeometry const &box_geo) {
123 if (std::holds_alternative<AngleHarmonicBond>(iaparams) or
124 std::holds_alternative<AngleCosineBond>(iaparams) or
125#ifdef ESPRESSO_TABULATED
126 std::holds_alternative<TabulatedAngleBond>(iaparams) or
127#endif
128 std::holds_alternative<AngleCossquareBond>(iaparams)) {
129 auto const dx21 = -box_geo.get_mi_vector(p1.pos(), p2.pos());
130 auto const dx31 = box_geo.get_mi_vector(p3.pos(), p1.pos());
131
132 auto const result =
133 calc_bonded_three_body_force(iaparams, box_geo, p1, p2, p3);
134 if (result) {
135 Utils::Vector3d force2, force3;
136 std::tie(std::ignore, force2, force3) = result.value();
137
138 return Utils::tensor_product(force2, dx21) +
139 Utils::tensor_product(force3, dx31);
140 }
141 } else {
142 runtimeWarningMsg() << "Unsupported bond type " +
143 std::to_string(iaparams.index()) +
144 " in pressure calculation.";
145 return Utils::Matrix<double, 3, 3>{};
146 }
147
148 return {};
149}

151inline std::optional<Utils::Matrix<double, 3, 3>> calc_bonded_pressure_tensor(
152 Bonded_IA_Parameters const &iaparams, Particle const &p1,
153 std::span<Particle *> partners, BoxGeometry const &box_geo,
154 Coulomb::ShortRangeForceKernel::kernel_type const *kernel) {
155 switch (number_of_partners(iaparams)) {
156 case 1:
157 return calc_bonded_virial_pressure_tensor(iaparams, p1, *partners[0],
158 box_geo, kernel);
159 case 2:
160 return calc_bonded_three_body_pressure_tensor(iaparams, p1, *partners[0],
161 *partners[1], box_geo);
162 default:
163 runtimeWarningMsg() << "Unsupported bond type " +
164 std::to_string(iaparams.index()) +
165 " in pressure calculation.";
166 return Utils::Matrix<double, 3, 3>{};
167 }
168}

170/** Calculate kinetic pressure (aka energy) for one particle.
171 * @param[in] p1 particle for which to calculate pressure
172 * @param[out] obs_pressure pressure observable
173 */
174inline void add_kinetic_virials(Particle const &p1,
175 Observable_stat &obs_pressure) {
176 if (p1.is_virtual())
177 return;
178
179 /* kinetic pressure */
180 for (std::size_t k = 0u; k < 3u; k++)
181 for (std::size_t l = 0u; l < 3u; l++)
182 obs_pressure.kinetic_lin[k * 3u + l] += p1.v()[k] * p1.v()[l] * p1.mass();
183}
