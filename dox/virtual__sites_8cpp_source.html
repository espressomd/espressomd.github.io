58 /** Calculate the rotation quaternion and distance between two particles */
59 static std::tuple<Utils::Quaternion<double>, double>
60 calculate_vs_relate_to_params(Particle const &p_current,
61  Particle const &p_relate_to) {
62  // get the distance between the particles
63  Utils::Vector3d d = box_geo.get_mi_vector(p_current.pos(), p_relate_to.pos());
64 
65  // Check if the distance between virtual and non-virtual particles is larger
66  // than minimum global cutoff. If so, warn user.
67  auto const dist = d.norm();
68  auto const min_global_cut = get_min_global_cut();
69  if (dist > min_global_cut && n_nodes > 1) {
70  runtimeErrorMsg()
71  << "Warning: The distance between virtual and non-virtual particle ("
72  << dist << ") is larger than the minimum global cutoff ("
73  << min_global_cut << "). This may lead to incorrect simulations under "
74  << "certain conditions. Adjust the property system.min_global_cut to "
75  << "increase the minimum cutoff.";
76  }
77 
78  // Now, calculate the quaternions which specify the angle between
79  // the director of the particle we relate to and the vector
80  // (particle_we_relate_to - this_particle)
81  // The vs_relative implementation later obtains the director by multiplying
82  // the quaternions representing the orientation of the real particle
83  // with those in the virtual particle. The resulting quaternion is then
84  // converted to a director.
85  // We have quat_(real particle) * quat_(virtual particle)
86  // = quat_(obtained from desired director)
87  // Resolving this for the quat_(virtual particle)
88 
89  Utils::Quaternion<double> quat;
90  // If the distance between real & virtual particle is 0 we just set the
91  // relative orientation to {1 0 0 0}, as it is irrelevant but needs to be
92  // a valid quaternion
93  if (dist == 0) {
94  quat = Utils::Quaternion<double>::identity();
95  } else {
96  d.normalize();
97 
98  // Obtain quaternion from desired director
99  Utils::Quaternion<double> quat_director =
100  Utils::convert_director_to_quaternion(d);
101 
102  // Define quaternion as described above
103  auto relate_to_quat = p_relate_to.quat();
104  quat = Utils::Quaternion<double>{
105  {{{Utils::dot(relate_to_quat, quat_director),
106  -quat_director[0] * relate_to_quat[1] +
107  quat_director[1] * relate_to_quat[0] +
108  quat_director[2] * relate_to_quat[3] -
109  quat_director[3] * relate_to_quat[2],
110  relate_to_quat[1] * quat_director[3] +
111  relate_to_quat[0] * quat_director[2] -
112  relate_to_quat[3] * quat_director[1] -
113  relate_to_quat[2] * quat_director[0],
114  quat_director[3] * relate_to_quat[0] -
115  relate_to_quat[3] * quat_director[0] +
116  relate_to_quat[2] * quat_director[1] -
117  relate_to_quat[1] * quat_director[2]}}}};
118  quat /= relate_to_quat.norm2();
119 
120  // Verify result
121  Utils::Quaternion<double> qtemp = relate_to_quat * quat;
122  for (int i = 0; i < 4; i++)
123  if (fabs(qtemp[i] - quat_director[i]) > 1E-9)
124  fprintf(stderr, "vs_relate_to: component %d: %f instead of %f\n", i,
125  qtemp[i], quat_director[i]);
126  }
127  return std::make_tuple(quat, dist);
128 }

130 void local_vs_relate_to(Particle &p_current, Particle const &p_relate_to) {
131  // Set the particle id of the particle we want to relate to, the distance
132  // and the relative orientation
133  p_current.vs_relative().to_particle_id = p_relate_to.id();
134  std::tie(p_current.vs_relative().rel_orientation,
135  p_current.vs_relative().distance) =
136  calculate_vs_relate_to_params(p_current, p_relate_to);
137 }

139 void vs_relate_to(int part_num, int relate_to) {
140  if (part_num == relate_to) {
141  throw std::invalid_argument("A virtual site cannot relate to itself");
142  }
143  // Get the data for the particle we act on and the one we want to relate
144  // it to.
145  auto const &p_current = get_particle_data(part_num);
146  auto const &p_relate_to = get_particle_data(relate_to);
147 
148  Utils::Quaternion<double> quat;
149  double dist;
150  std::tie(quat, dist) = calculate_vs_relate_to_params(p_current, p_relate_to);
151 
152  // Set the particle id of the particle we want to relate to, the distance
153  // and the relative orientation
154  set_particle_vs_relative(part_num, relate_to, dist, quat);
155  set_particle_virtual(part_num, true);
156 }
