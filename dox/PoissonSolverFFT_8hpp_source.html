45template <typename FloatType> class PoissonSolverFFT : public PoissonSolver {
46private:
47 template <typename T> FloatType FloatType_c(T t) {
48 return numeric_cast<FloatType>(t);
49 }
50
51 domain_decomposition::BlockDataID m_potential_field_id;
52
53 using PotentialField = GhostLayerField<FloatType, 1>;
54
55 std::shared_ptr<fft::FourierTransform<PotentialField>> m_ft;
56 std::shared_ptr<blockforest::StructuredBlockForest> m_blocks;
57
58 using FullCommunicator =
59 blockforest::communication::UniformBufferedScheme<stencil::D3Q27>;
60 std::shared_ptr<FullCommunicator> m_full_communication;
61
62public:
63 PoissonSolverFFT(std::shared_ptr<LatticeWalberla> lattice,
64 double permittivity)
65 : PoissonSolver(std::move(lattice), permittivity) {
66 m_blocks = get_lattice().get_blocks();
67
68 Vector3<uint_t> dim(m_blocks->getNumberOfXCells(),
69 m_blocks->getNumberOfYCells(),
70 m_blocks->getNumberOfZCells());
71 auto const greens = [dim](uint_t x, uint_t y, uint_t z) -> real_t {
72 if (x == 0u && y == 0u && z == 0u)
73 return 0.;
74 return -0.5 /
75 (std::cos(2. * std::numbers::pi * real_c(x) / real_c(dim[0])) +
76 std::cos(2. * std::numbers::pi * real_c(y) / real_c(dim[1])) +
77 std::cos(2. * std::numbers::pi * real_c(z) / real_c(dim[2])) -
78 3.) /
79 real_c(dim[0] * dim[1] * dim[2]);
80 };
81
82 m_potential_field_id = field::addToStorage<PotentialField>(
83 get_lattice().get_blocks(), "potential field", 0.0, field::fzyx,
84 get_lattice().get_ghost_layers());
85
86 m_ft = std::make_shared<fft::FourierTransform<PotentialField>>(
87 m_blocks, m_potential_field_id, greens);
88
89 m_full_communication =
90 std::make_shared<FullCommunicator>(get_lattice().get_blocks());
91 m_full_communication->addPackInfo(
92 std::make_shared<field::communication::PackInfo<PotentialField>>(
93 m_potential_field_id));
94 }
95 ~PoissonSolverFFT() override = default;
96
97 void reset_charge_field() override {
98 // the FFT-solver re-uses the potential field for the charge
99 auto const potential_id = walberla::BlockDataID(get_potential_field_id());
100
101 for (auto &block : *get_lattice().get_blocks()) {
102 auto field = block.template getData<PotentialField>(potential_id);
103 WALBERLA_FOR_ALL_CELLS_XYZ(field, field->get(x, y, z) = 0.;)
104 }
105 }
106
107 void add_charge_to_field(std::size_t id, double valency,
108 bool is_double_precision) override {
109 auto const factor = FloatType_c(valency) / FloatType_c(get_permittivity());
110 // the FFT-solver re-uses the potential field for the charge
111 const auto charge_id = walberla::BlockDataID(get_potential_field_id());
112 const auto density_id = walberla::BlockDataID(id);
113 for (auto &block : *get_lattice().get_blocks()) {
114 auto charge_field = block.template getData<PotentialField>(charge_id);
115 if (is_double_precision) {
116 auto density_field =
117 block.template getData<walberla::GhostLayerField<double, 1>>(
118 density_id);
119 WALBERLA_FOR_ALL_CELLS_XYZ(
120 charge_field, charge_field->get(x, y, z) +=
121 factor * FloatType_c(density_field->get(x, y, z));)
122 } else {
123 auto density_field =
124 block.template getData<walberla::GhostLayerField<float, 1>>(
125 density_id);
126 WALBERLA_FOR_ALL_CELLS_XYZ(
127 charge_field, charge_field->get(x, y, z) +=
128 factor * FloatType_c(density_field->get(x, y, z));)
129 }
130 }
131 }
132
133 [[nodiscard]] std::size_t get_potential_field_id() const noexcept override {
134 return static_cast<std::size_t>(m_potential_field_id);
135 }
136
137 void solve() override {
138 (*m_ft)();
139 ghost_communication();
140 integrate_vtk_writers();
141 }

143 [[nodiscard]] std::optional<double>
144 get_node_potential(Utils::Vector3i const &node,
145 bool consider_ghosts = false) override {
146 auto bc = get_block_and_cell(get_lattice(), node, consider_ghosts);
147
148 if (!bc || (get_potential_field_id() == 0))
149 return std::nullopt;
150
151 auto const potential_field =
152 bc->block->template getData<PotentialField>(m_potential_field_id);
153 return {double_c(
154 walberla::ek::accessor::Scalar::get(potential_field, bc->cell))};
155 }
156
157 [[nodiscard]] std::vector<double>
158 get_slice_potential(Utils::Vector3i const &lower_corner,
159 Utils::Vector3i const &upper_corner) const override {
160 std::vector<double> out;
161 uint_t values_size = 0;
162 auto const &lattice = get_lattice();
163 if (auto const ci = get_interval(lattice, lower_corner, upper_corner)) {
164 out = std::vector<double>(ci->numCells());
165 for (auto &block : *lattice.get_blocks()) {
166 auto const block_offset = lattice.get_block_corner(block, true);
167 if (auto const bci = get_block_interval(
168 lattice, lower_corner, upper_corner, block_offset, block)) {
169 auto const potential_field =
170 block.template getData<PotentialField>(m_potential_field_id);
171 auto const values = ek::accessor::Scalar::get(potential_field, *bci);
172 assert(values.size() == bci->numCells());
173 values_size += bci->numCells();
174 auto kernel = [&values, &out](unsigned const block_index,
175 unsigned const local_index,
176 Utils::Vector3i const &node) {
177 out[local_index] = double_c(values[block_index]);
178 };
179
180 copy_block_buffer(*bci, *ci, block_offset, lower_corner, kernel);
181 }
182 }
183 assert(values_size == ci->numCells());
184 }
185 return out;
186 }

209 template <typename OutputType = float>
210 class PotentialVTKWriter : public VTKWriter<PotentialField, 1u, OutputType> {
211 public:
212 using Base = VTKWriter<PotentialField, 1u, OutputType>;
213 using Base::Base;
214 using Base::evaluate;
215
216 protected:
217 OutputType evaluate(cell_idx_t const x, cell_idx_t const y,
218 cell_idx_t const z, cell_idx_t const) override {
219 WALBERLA_ASSERT_NOT_NULLPTR(this->m_field);
220 auto const potential =
221 walberla::ek::accessor::Scalar::get(this->m_field, {x, y, z});
222 return numeric_cast<OutputType>(this->m_conversion * potential);
223 }
224 };

230 auto const unit_conversion = FloatType_c(units.at("potential"));

233 }
234 }

236 void integrate_vtk_writers() override {
237 for (auto const &it : m_vtk_auto) {
238 auto &vtk_handle = it.second;
239 if (vtk_handle->enabled) {
240 vtk::writeFiles(vtk_handle->ptr)();
241 vtk_handle->execution_count++;
242 }
243 }
244 }

246private:
247 void ghost_communication() { (*m_full_communication)(); }
