44 /** Calculate Morse force factor */
45 inline double morse_pair_force_factor(IA_parameters const &ia_params,
46  double dist) {
47  if (dist < ia_params.morse.cut) {
48  auto const add =
49  exp(-ia_params.morse.alpha * (dist - ia_params.morse.rmin));
50  return -ia_params.morse.eps * 2.0 * ia_params.morse.alpha *
51  (add - Utils::sqr(add)) / dist;
52  }
53  return 0.0;
54 }
55 
56 /** Calculate Morse energy */
57 inline double morse_pair_energy(IA_parameters const &ia_params, double dist) {
58  if (dist < ia_params.morse.cut) {
59  auto const add =
60  exp(-ia_params.morse.alpha * (dist - ia_params.morse.rmin));
61  return ia_params.morse.eps * (Utils::sqr(add) - 2 * add) -
62  ia_params.morse.rest;
63  }
64  return 0.0;
65 }
