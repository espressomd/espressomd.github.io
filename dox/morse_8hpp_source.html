40 /** Calculate Morse force factor */
41 inline double morse_pair_force_factor(IA_parameters const &ia_params,
42  double dist) {
43  if (dist < ia_params.morse.cut) {
44  auto const add =
45  exp(-ia_params.morse.alpha * (dist - ia_params.morse.rmin));
46  return -ia_params.morse.eps * 2.0 * ia_params.morse.alpha *
47  (add - Utils::sqr(add)) / dist;
48  }
49  return 0.0;
50 }
51 
52 /** Calculate Morse energy */
53 inline double morse_pair_energy(IA_parameters const &ia_params, double dist) {
54  if (dist < ia_params.morse.cut) {
55  auto const add =
56  exp(-ia_params.morse.alpha * (dist - ia_params.morse.rmin));
57  return ia_params.morse.eps * (Utils::sqr(add) - 2 * add) -
58  ia_params.morse.rest;
59  }
60  return 0.0;
61 }
