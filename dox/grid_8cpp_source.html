53  Utils::Vector3i im;

71 LocalBox<double> regular_decomposition(const BoxGeometry &box,
72  Utils::Vector3i const &node_pos,
73  Utils::Vector3i const &node_grid_par) {
74  Utils::Vector3d local_length;
75  Utils::Vector3d my_left;
76 
77  for (int i = 0; i < 3; i++) {
78  local_length[i] = box.length()[i] / node_grid_par[i];
79  my_left[i] = node_pos[i] * local_length[i];
80  }
81 
82  Utils::Array<int, 6> boundaries;
83  for (std::size_t dir = 0; dir < 3; dir++) {
84  /* left boundary ? */
85  boundaries[2 * dir] = (node_pos[dir] == 0);
86  /* right boundary ? */
87  boundaries[2 * dir + 1] = -(node_pos[dir] == node_grid_par[dir] - 1);
88  }
89 
90  return {my_left, local_length, boundaries,
91  CellStructureType::CELL_STRUCTURE_REGULAR};
92 }

109 void rescale_boxl(int dir, double d_new) {
110  double scale = (dir - 3) ? d_new * box_geo.length_inv()[dir]
111  : d_new * box_geo.length_inv()[0];
112 
113  /* If shrinking, rescale the particles first. */
114  if (scale <= 1.) {
115  mpi_rescale_particles(dir, scale);
116  }
117 
118  if (dir < 3) {
119  auto box_l = box_geo.length();
120  box_l[dir] = d_new;
121  mpi_set_box_length(box_l);
122  } else {
123  mpi_set_box_length({d_new, d_new, d_new});
124  }
125 
126  if (scale > 1.) {
127  mpi_rescale_particles(dir, scale);
128  }
129 }
