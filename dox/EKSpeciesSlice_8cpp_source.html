38Variant EKSpeciesSlice::do_call_method(std::string const &name,
39 VariantMap const &params) {
40 if (name == "get_slice_size") {
41 return {m_slice_upper_corner - m_slice_lower_corner};
42 }
43 if (name == "get_slice_ranges") {
44 return {std::vector<Variant>{m_slice_lower_corner, m_slice_upper_corner}};
45 }
46 if (name == "get_ek_sip") {
47 return {m_ek_sip};
48 }
49 if (name == "get_value_shape") {
50 auto const name = get_value<std::string>(params, "name");
51 if (not m_shape_val.contains(name)) {
52 context()->parallel_try_catch([&]() {
53 throw std::runtime_error("Unknown EK property '" + name + "'");
54 });
55 }
56 return m_shape_val.at(name);
57 }
58
59 if (not name.starts_with("get_")) {
60 context()->parallel_try_catch(
61 [&]() { ek_throw_if_expired(m_mpi_cart_comm_observer); });
62 }
63
64 // slice getter/setter callback
65 auto const call = [this, params](auto method_ptr,
66 std::vector<int> const &data_dims,
67 double units = 1.) -> Variant {
68 auto &obj = *m_ek_species;
69 if constexpr (std::is_invocable_v<decltype(method_ptr), LatticeModel *,
70 Utils::Vector3i const &,
71 Utils::Vector3i const &>) {
72 return gather_3d(data_dims, obj, method_ptr, units);
73 } else {
74 scatter_3d(params.at("values"), data_dims, obj, method_ptr, units);
75 return {};
76 }
77 };
78
79 if (name == "get_density") {
80 return call(&LatticeModel::get_slice_density, {1}, 1. / m_conv_dens);
81 }
82 if (name == "set_density") {
83 return call(&LatticeModel::set_slice_density, {1}, m_conv_dens);
84 }
85 if (name == "get_flux") {
86 return call(&LatticeModel::get_slice_flux_vector, {3}, 1. / m_conv_flux);
87 }
88 if (name == "get_is_boundary") {
89 return call(&LatticeModel::get_slice_is_boundary, {1});
90 }
91 if (name == "get_flux_at_boundary") {
92 return call(&LatticeModel::get_slice_flux_at_boundary, {1},
93 1. / m_conv_flux);
94 }
95 if (name == "set_flux_at_boundary") {
96 m_ek_sip->flux_boundary_ghost_layer_size_sanity_check();
97 return call(&LatticeModel::set_slice_flux_boundary, {1}, m_conv_flux);
98 }
99 if (name == "get_density_at_boundary") {
100 return call(&LatticeModel::get_slice_density_at_boundary, {1},
101 1. / m_conv_dens);
102 }
103 if (name == "set_density_at_boundary") {
104 return call(&LatticeModel::set_slice_density_boundary, {1}, m_conv_dens);
105 }
106
107 return {};
108}
