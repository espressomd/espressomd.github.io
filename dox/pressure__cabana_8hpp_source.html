51struct PressureBinLayout {
52 std::size_t n_bonded;
53 std::size_t n_types;
54 std::size_t off_bonded = 0;
55 std::size_t off_nb_inter;
56 std::size_t off_nb_intra;
57 std::size_t off_coulomb;
58 std::size_t off_dipolar;
59 std::size_t off_dpd;
60 std::size_t total;
61
62 PressureBinLayout(std::size_t n_bonded_, std::size_t n_types_)
63 : n_bonded(n_bonded_), n_types(n_types_) {
64 auto const n_nb = n_types * (n_types + 1) / 2;
65 off_nb_inter = off_bonded + n_bonded;
66 off_nb_intra = off_nb_inter + n_nb;
67 off_coulomb = off_nb_intra + n_nb;
68 off_dipolar = off_coulomb + 1;
69 off_dpd = off_dipolar + 1;
70 total = off_dpd + 1;
71 }
72
73 KOKKOS_INLINE_FUNCTION
74 std::size_t tensor_offset(std::size_t bin, std::size_t k) const {
75 return bin * 9 + k;
76 }
77
78 KOKKOS_INLINE_FUNCTION
79 std::size_t nb_inter_idx(int t1, int t2) const {
80 return off_nb_inter +
81 Utils::lower_triangular(std::max(t1, t2), std::min(t1, t2));
82 }
83
84 KOKKOS_INLINE_FUNCTION
85 std::size_t nb_intra_idx(int t1, int t2) const {
86 return off_nb_intra +
87 Utils::lower_triangular(std::max(t1, t2), std::min(t1, t2));
88 }
89
90 KOKKOS_INLINE_FUNCTION std::size_t coulomb_idx() const { return off_coulomb; }
91 KOKKOS_INLINE_FUNCTION std::size_t dipolar_idx() const { return off_dipolar; }
92 KOKKOS_INLINE_FUNCTION std::size_t dpd_idx() const { return off_dpd; }
93 KOKKOS_INLINE_FUNCTION std::size_t bonded_idx(int b) const {
94 return off_bonded + static_cast<std::size_t>(b);
95 }
96};

142 KOKKOS_INLINE_FUNCTION
143 void operator()(std::size_t i, std::size_t j) const {
144 auto const pos1 = aosoa.get_vector_at(aosoa.position, i);
145 auto const pos2 = aosoa.get_vector_at(aosoa.position, j);
146 auto const d = box_geo.get_mi_vector(pos1, pos2);
147 auto const dist = d.norm();
148 if (dist > system_max_cutoff)
149 return;
150
151 auto const t1 = aosoa.type(i);
152 auto const t2 = aosoa.type(j);
153 auto const &ia_params = nonbonded_ias.get_ia_param(t1, t2);
154 auto const tid = omp_get_thread_num();
155
156 // --- Non-bonded virial ---
157 if (dist <= ia_params.max_cut) {
158 bool skip = false;
159#ifdef ESPRESSO_EXCLUSIONS
160 if (aosoa.has_exclusion(i) or aosoa.has_exclusion(j)) {
161 skip =
162 not do_nonbonded(*unique_particles.at(i), *unique_particles.at(j));
163 }
164#endif
165 if (not skip) {
166 Utils::Vector3d f = calc_central_radial_force(ia_params, d, dist);
167
168#ifdef ESPRESSO_GAY_BERNE
169 if (gay_berne_active(dist, ia_params)) {
170 auto const dir1 = aosoa.get_vector_at(aosoa.director, i);
171 auto const dir2 = aosoa.get_vector_at(aosoa.director, j);
172 f += calc_non_central_force(dir1, dir2, ia_params, d, dist);
173 }
174#endif
175#ifdef ESPRESSO_THOLE
176 if (thole_active(ia_params, coulomb_f_kernel != nullptr)) {
177 f += thole_pair_force(*unique_particles.at(i),
178 *unique_particles.at(j), ia_params, d, dist,
179 bonded_ias, coulomb_f_kernel);
180 }
181#endif
182
183 auto const stress = Utils::flatten(Utils::tensor_product(d, f));
184 auto const bin = (mol_id_view(i) == mol_id_view(j))
185 ? layout.nb_intra_idx(t1, t2)
186 : layout.nb_inter_idx(t1, t2);
187 for (std::size_t k = 0; k < 9; ++k)
188 local_pressure(tid, layout.tensor_offset(bin, k)) += stress[k];
189
190#ifdef ESPRESSO_DPD
191 if (dpd_active(ia_params, thermo_switch)) {
192 auto const pid1 = aosoa.id(i);
193 auto const pid2 = aosoa.id(j);
194 auto const noise_vec = (ia_params.dpd.radial.pref > 0.0 ||
195 ia_params.dpd.trans.pref > 0.0)
196 ? dpd_noise(*dpd, pid1, pid2)
197 : Utils::Vector3d{};
198 auto const vel1 = aosoa.get_vector_at(aosoa.velocity, i);
199 auto const vel2 = aosoa.get_vector_at(aosoa.velocity, j);
200 auto const v21 = box_geo.velocity_difference(pos1, pos2, vel1, vel2);
201 auto const dist2 = d.norm2();
202 // f_r/f_t: dissipative force from radial/transverse DPD channel
203 auto const f_r =
204 dpd_pair_force(ia_params.dpd.radial, v21, dist, noise_vec);
205 auto const f_t =
206 dpd_pair_force(ia_params.dpd.trans, v21, dist, noise_vec);
207 auto const P = Utils::tensor_product(d / dist2, d);
208 auto const f_d = P * (f_r - f_t) + f_t;
209 auto const s = Utils::flatten(Utils::tensor_product(d, f_d));
210 for (std::size_t k = 0; k < 9; ++k)
211 local_pressure(tid, layout.tensor_offset(layout.dpd_idx(), k)) +=
212 s[k];
213 }
214#endif
215 }
216 }
217
218#ifdef ESPRESSO_ELECTROSTATICS
219 if (coulomb_p_kernel != nullptr) {
220 auto const q1 = aosoa.charge(i), q2 = aosoa.charge(j);
221 if (q1 != 0. and q2 != 0.) {
222 auto const p_c = Utils::flatten((*coulomb_p_kernel)(q1 * q2, d, dist));
223 for (std::size_t k = 0; k < 9; ++k)
224 local_pressure(tid, layout.tensor_offset(layout.coulomb_idx(), k)) +=
225 p_c[k];
226 }
227 }
228#endif
229 }
230};

232static void reduce_cabana_pressure(
233 Kokkos::View<double **, Kokkos::LayoutRight> const &local_pressure,
234 PressureBinLayout const &layout, Observable_stat &obs,
235 [[maybe_unused]] BondedInteractionsMap const &bonded_ias, int n_types) {
236 auto const nthreads = local_pressure.extent(0);
237 auto host =
238 Kokkos::create_mirror_view_and_copy(Kokkos::HostSpace{}, local_pressure);
239
240 auto sum_tensor = [&](std::size_t bin, std::span<double> dest) {
241 for (std::size_t k = 0; k < 9; ++k) {
242 double s = 0.;
243 for (std::size_t t = 0; t < nthreads; ++t)
244 s += host(t, layout.tensor_offset(bin, k));
245 dest[k] += s;
246 }
247 };
248
249 for (int b = 0; b < int(layout.n_bonded); ++b)
250 sum_tensor(layout.bonded_idx(b), obs.bonded_contribution(b));
251
252 for (int t1 = 0; t1 < n_types; ++t1)
253 for (int t2 = 0; t2 <= t1; ++t2)
254 sum_tensor(layout.nb_inter_idx(t1, t2),
255 obs.non_bonded_inter_contribution(t1, t2));
256
257 for (int t1 = 0; t1 < n_types; ++t1)
258 for (int t2 = 0; t2 <= t1; ++t2)
259 sum_tensor(layout.nb_intra_idx(t1, t2),
260 obs.non_bonded_intra_contribution(t1, t2));
261
262 if (!obs.coulomb.empty())
263 sum_tensor(layout.coulomb_idx(), obs.coulomb.subspan(0, 9));
264 if (!obs.dipolar.empty())
265 sum_tensor(layout.dipolar_idx(), obs.dipolar.subspan(0, 9));
266 if (!obs.dpd.empty())
267 sum_tensor(layout.dpd_idx(), obs.dpd);
268}
