77 auto find_bond_id(Variant const &v) const {
78 auto &system = get_system();
79 if (is_type<int>(v)) {
80 auto const bond_id = get_value<int>(v);
81 std::optional<int> retval = std::nullopt;
82 if (system.bonded_ias->contains(bond_id)) {
83 retval = bond_id;
84 }
85 return retval;
86 }
87 auto obj = get_value<std::shared_ptr<Interactions::BondedInteraction>>(v);
88 return system.bonded_ias->find_bond_id(obj->bonded_ia());
89 }

162 void do_construct(VariantMap const &params) override {
163 m_params = std::make_unique<VariantMap>(params);
164 if (params.empty()) {
165 (*m_params)["mode"] = std::string("off");
166 } else {
167 // Assume we are reloading from a checkpoint file.
168 // This else branch can be removed once issue #4483 is fixed.
169 m_params = std::make_unique<VariantMap>();
170 auto const name = get_value<std::string>(params, "mode");
171 check_mode_name(name);
172 for (auto const &name :
173 cd_mode_to_parameters.at(cd_name_to_mode.at(name))) {
174 (*m_params)[name] = params.at(name);
175 }
176 (*m_params)["mode"] = params.at("mode");
177 }
178 }
179
180 Variant do_call_method(const std::string &name,
181 const VariantMap &params) override {
182 if (name == "set_params") {
183 context()->parallel_try_catch([this, &params]() {
184 auto const backup = std::make_shared<::CollisionDetection>(*m_handle);
185 auto &system = get_system();
186 try {
187 // check provided parameters
188 check_input_parameters(params);
189 // set parameters
190 for (auto const &kv : params) {
191 do_set_parameter(get_value<std::string>(kv.first), kv.second);
192 }
193 // sanitize parameters and calculate derived parameters
194 m_handle->initialize();
195 return none;
196 } catch (...) {
197 // restore original parameters and re-throw exception
198 m_handle = system.collision_detection = backup;
199 m_handle->initialize();
200 throw;
201 }
202 });
203 }
204 if (name == "params_for_mode") {
205 auto const name = get_value<std::string>(params, "mode");
206 check_mode_name(name);
207 auto const mode = cd_name_to_mode.at(name);
208 return make_vector_of_variants(cd_mode_to_parameters.at(mode));
209 }
210 if (name == "get_bond_by_id") {
211 if (not context()->is_head_node()) {
212 return {};
213 }
214 return m_bonded_ias.lock()->call_method("get_bond", params);
215 }
216 return none;
217 }

230 void on_bind_system(::System::System &system) override {
231 m_handle = system.collision_detection;
232 m_handle->bind_system(m_system.lock());
233 if (not m_params->empty()) {
234 do_call_method("set_params", *m_params);
235 }
236 m_params.reset();
237 }
238
239 void check_input_parameters(VariantMap const &params) const {
240 auto const name = get_value<std::string>(params, "mode");
241 check_mode_name(name);
242 auto const mode = cd_name_to_mode.at(name);
243 auto const expected_params = cd_mode_to_parameters.at(mode);
244 auto const expected_param_names =
245 std::set<std::string>{expected_params.begin(), expected_params.end()};
246 std::set<std::string> input_parameter_names = {};
247 for (auto const &kv : params) {
248 auto const &param_name = kv.first;
249 if (not expected_param_names.contains(param_name)) {
250 // throw if parameter is unknown
251 std::ignore = get_parameter(param_name);
252 // throw if parameter is known but doesn't match the selected mode
253 throw std::runtime_error("Parameter '" + param_name + "' is not " +
254 "required for mode '" + name + "'");
255 }
256 input_parameter_names.insert(param_name);
257 }
258 for (auto const &param_name : expected_param_names) {
259 if (not input_parameter_names.contains(param_name)) {
260 throw std::runtime_error("Parameter '" + param_name + "' is " +
261 "required for mode '" + name + "'");
262 }
263 }
264 }
