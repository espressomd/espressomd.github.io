49template <typename FloatType>
50void FFTBackendLegacy<FloatType>::update_mesh_data() {
51 auto const mesh_size_ptr = fft->get_mesh_size();
52 auto const mesh_start_ptr = fft->get_mesh_start();
53 for (auto i = 0u; i < 3u; ++i) {
54 mesh.size[i] = mesh_size_ptr[i];
55 mesh.start[i] = mesh_start_ptr[i];
56 }
57 mesh.stop = mesh.start + mesh.size;
58 mesh.ks_scalar = std::span(ks_mesh);
59 mesh.rs_scalar = std::span(rs_mesh);
60 for (auto i = 0u; i < 3u; ++i) {
61 mesh.rs_fields[i] = std::span(rs_mesh_fields[i]);
62 }
63}

69template <typename FloatType> void FFTBackendLegacy<FloatType>::init_fft() {
70 auto ca_mesh_size = fft->initialize_fft(
71 ::comm_cart, local_mesh.dim, local_mesh.margin, params.mesh,
72 params.mesh_off, mesh.ks_pnum, ::communicator.node_grid);
73 rs_mesh.resize(ca_mesh_size);
74 if (dipolar) {
75 ks_mesh.resize(ca_mesh_size);
76 }
77 for (auto &rs_mesh_field : rs_mesh_fields) {
78 rs_mesh_field.resize(ca_mesh_size);
79 }
80 update_mesh_data();
81}
