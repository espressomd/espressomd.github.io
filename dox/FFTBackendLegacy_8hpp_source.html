43/**
44 * @brief Historic FFT backend based on FFTW3.
45 * The 3D FFT is split into three 1D FFTs.
46 */
47template <typename FloatType>
48class FFTBackendLegacy : public FFTBackend<FloatType> {
49 static_assert(std::is_same_v<FloatType, float> or
50 std::is_same_v<FloatType, double>,
51 "FFTW only implements float and double");
52 bool dipolar;
53 std::unique_ptr<fft::fft_data_struct<FloatType>> fft;
54 /** @brief k-space mesh (local) for k-space calculations. */
55 std::vector<FloatType> ks_mesh;
56 /** @brief real-space mesh (local) for CA/FFT. */
57 fft::vector<FloatType> rs_mesh;
58 /** @brief real-space mesh (local) for the electric or dipolar field. */
59 std::array<fft::vector<FloatType>, 3> rs_mesh_fields;
60 p3m_send_mesh<FloatType> mesh_comm;
61 using FFTBackend<FloatType>::params;
62 using FFTBackend<FloatType>::mesh;
63 using FFTBackend<FloatType>::local_mesh;
64 using FFTBackend<FloatType>::check_complex_residuals;
65
66public:
67 FFTBackendLegacy(p3m_data_struct_fft<FloatType> &obj, bool dipolar);
68 ~FFTBackendLegacy() override;
69 void init_fft() override;
70 void init_halo() override;
71 void perform_scalar_fwd_fft() override;
72 void perform_vector_fwd_fft() override;
73 void perform_scalar_back_fft() override;
74 void perform_vector_back_fft() override;
75 void perform_scalar_halo_gather() override;
76 void perform_vector_halo_gather() override;
77 void perform_scalar_halo_spread() override;
78 void perform_vector_halo_spread() override;
79 void update_mesh_data();
80
81 /**
82 * @brief Index helpers for reciprocal space.
83 * After the FFT the data is in order YZX, which
84 * means that Y is the slowest changing index.
85 */
86 std::tuple<int, int, int> get_permutations() const override {
87 constexpr static int KX = 2;
88 constexpr static int KY = 0;
89 constexpr static int KZ = 1;
90 return {KX, KY, KZ};
91 }
92};
