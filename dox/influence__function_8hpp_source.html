42/**
43 * @brief Calculate the aliasing sums for the optimal influence function.
44 *
45 * This implements Eq. 30 of @cite cerda08d, which can be used
46 * for monopole and dipole P3M by choosing the appropriate S factor.
47 * @f[
48 * \tilde{G}_{\text{opt}} ( k ) = \frac{\sum_{m \in \mathbb{Z}^{3}} [ [ \tilde{
49 * \left ( D \right ) } ( k ) \cdot i k_{m} ]^{S} ( \hat{U} ( k_{m} ) )^{2}
50 * \hat{\phi} ( k_{m} ) ]}{[ \tilde{ \left ( D \right ) } ( k ) ]^{2 S} [
51 * \sum_{m \in \mathbb{Z}^{3}} ( \hat{U} ( k_{m} ) )^{2} ]^{2}}
52 * @f]
53 *
54 * Eq 8.29 in Hockney:
55 * @f[
56 * G_{\text{opt}}(\mathbf{k}) = \frac{\hat{\mathbf{D}}\sum_n
57 * \hat{\mathbf{R}}^\ast \hat{U}^2}{|\hat{\mathbf{D}}|^2\sum_n
58 * \hat{U}^2\sum_{n'}^{'} \hat{U}^2}
59 * @f]
60 *
61 * The full equation is:
62 * @f{align*}{
63 * G_{\text{opt}}(\vec{n}, S, \text{cao}) &=
64 * \displaystyle\frac{4\pi}{\sum_\vec{m} U^2}
65 * \sum_\vec{m}U^2
66 * \displaystyle\frac{\left(\vec{k}\odot\vec{k}_m\right)^S}
67 * {|\vec{k}_m|^2}
68 * e^{-1/(2\alpha)^2|\vec{k}_m|^2} \\
69 *
70 * U &= \operatorname{det}\left[I_3\cdot\operatorname{sinc}\left(
71 * \frac{\vec{k}_m\odot\vec{a}}{2\pi}\right)\right]^{\text{cao}} \\
72 *
73 * \vec{k} &= \frac{2\pi}{\vec{N}\odot\vec{a}}\vec{s}[\vec{n}] \\
74 *
75 * \vec{k}_m &= \frac{2\pi}{\vec{N}\odot\vec{a}}
76 * \left(\vec{s}[\vec{n}]+\vec{m}\odot\vec{N}\right)
77 * @f}
78 *
79 * with @f$ I_3 @f$ the 3x3 identity matrix, @f$ \vec{N} @f$ the global mesh
80 * size in k-space coordinates, @f$ \vec{a} @f$ the mesh size, @f$ \vec{m} @f$
81 * the Brillouin zone coordinates, @f$ \odot @f$ the Hadamard product.
82 *
83 * @tparam S Order of the differential operator (0 for potential, 1 for E-field)
84 * @tparam m Number of Brillouin zones that contribute to the aliasing sums
85 *
86 * @param params P3M parameters
87 * @param k k-vector to evaluate the function for.
88 * @return The optimal influence function for a specific k-vector.
89 */
90template <std::size_t S, std::size_t m>
91double G_opt(P3MParameters const &params, Utils::Vector3d const &k) {
92
93 auto const k2 = k.norm2();
94 if (k2 == 0.) {
95 return 0.;
96 }
97
98 auto constexpr exp_limit = 30.;
99 auto constexpr m_start = Utils::Vector3i::broadcast(-m);
100 auto constexpr m_stop = Utils::Vector3i::broadcast(m + 1);
101 auto const cao = params.cao;
102 auto const exponent_prefactor = Utils::sqr(1. / (2. * params.alpha));
103 auto const wavevector = (2. * std::numbers::pi) / params.a;
104 auto const wavevector_i = 1. / wavevector;
105 auto indices = Utils::Vector3i{};
106 auto km = Utils::Vector3d{};
107 auto fnm = Utils::Vector3d{};
108 auto numerator = 0.;
109 auto denominator = 0.;
110
111 for_each_3d(
112 m_start, m_stop, indices,
113 [&]() {
114 auto const U2 = std::pow(Utils::product(fnm), 2 * cao);
115 auto const km2 = km.norm2();
116 auto const exp_term = exponent_prefactor * km2;
117 if (exp_term < exp_limit) {
118 auto const f3 = std::exp(-exp_term) * (4. * std::numbers::pi / km2);
119 numerator += U2 * f3 * Utils::int_pow<S>(k * km);
120 }
121 denominator += U2;
122 },
123 [&](unsigned dim, int n) {
124 km[dim] = k[dim] + n * wavevector[dim];
125 fnm[dim] = math::sinc(km[dim] * wavevector_i[dim]);
126 });
127
128 if (numerator == 0. and denominator != 0.) {
129 // avoid FPE issues involving subnormal numbers
130 return 0.;
131 }
132 return numerator / (Utils::int_pow<S>(k2) * Utils::sqr(denominator));
133}

135/**
136 * @brief Map influence function over a grid.
137 *
138 * This evaluates the optimal influence function @ref G_opt
139 * over a regular grid of k vectors, and returns the values as a vector.
140 *
141 * @tparam S Order of the differential operator (0 for potential, 1 for E-field)
142 * @tparam m Number of Brillouin zones that contribute to the aliasing sums
143 *
144 * @param params P3M parameters.
145 * @param n_start Lower left corner of the grid in k-space.
146 * @param n_stop Upper right corner of the grid in k-space.
147 * @param inv_box_l Inverse box length.
148 * @return Values of G_opt at regular grid points.
149 */
150template <typename FloatType, std::size_t S, std::size_t m,
151 Utils::MemoryOrder memory_order>
152std::vector<FloatType> grid_influence_function(
153 P3MParameters const &params, Utils::Vector3i const &n_start,
154 Utils::Vector3i const &n_stop, Utils::Vector3d const &inv_box_l) {
155
156 auto const shifts = calc_p3m_mesh_shift(params.mesh);
157 auto const size = n_stop - n_start;
158
159 /* The influence function grid */
160 auto g = std::vector<FloatType>(Utils::product(size), FloatType(0));
161
162 /* Skip influence function calculation in tuning mode,
163 the results need not be correct for timing. */
164 if (params.tuning) {
165 return g;
166 }
167
168 auto const wavevector = (2. * std::numbers::pi) * inv_box_l;
169 auto const half_mesh = params.mesh / 2;
170 auto indices = Utils::Vector3i{};
171
172 for_each_3d(n_start, n_stop, indices, [&]() {
173 if ((indices[0u] % half_mesh[0u] != 0) or
174 (indices[1u] % half_mesh[1u] != 0) or
175 (indices[2u] % half_mesh[2u] != 0)) {
176 auto const k =
177 Utils::Vector3d{{shifts[0u][indices[0u]] * wavevector[0u],
178 shifts[1u][indices[1u]] * wavevector[1u],
179 shifts[2u][indices[2u]] * wavevector[2u]}};
180 auto const index =
181 Utils::get_linear_index<memory_order>(indices - n_start, size);
182 g[index] = FloatType(G_opt<S, m>(params, k));
183 }
184 });
185
186 return g;
187}
