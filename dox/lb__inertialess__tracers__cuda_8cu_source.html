64 __global__ void
65 ForcesIntoFluid_Kernel(const IBM_CUDA_ParticleDataInput *const particle_input,
66  std::size_t number_of_particles,
67  LB_node_force_density_gpu node_f,
68  const LB_parameters_gpu *const paraP) {
69  const unsigned int particleIndex = blockIdx.y * gridDim.x * blockDim.x +
70  blockDim.x * blockIdx.x + threadIdx.x;
71  const LB_parameters_gpu &para = *paraP;
72 
73  if (particleIndex < number_of_particles &&
74  particle_input[particleIndex].is_virtual) {
75  // MD to LB units: mass is not affected, length are scaled by agrid, times
76  // by para.tau
77  const float factor = 1 / para.agrid * para.tau * para.tau;
78  const float particleForce[3] = {particle_input[particleIndex].f[0] * factor,
79  particle_input[particleIndex].f[1] * factor,
80  particle_input[particleIndex].f[2] *
81  factor};
82  const float pos[3] = {particle_input[particleIndex].pos[0],
83  particle_input[particleIndex].pos[1],
84  particle_input[particleIndex].pos[2]};
85 
86  // First part is the same as for interpolation --> merge into a single
87  // function
88  float temp_delta[6];
89  float delta[8];
90  int my_left[3];
91  unsigned int node_index[8];
92  for (int i = 0; i < 3; ++i) {
93  const float scaledpos = pos[i] / para.agrid - 0.5f;
94  my_left[i] = static_cast<int>(floorf(scaledpos));
95  temp_delta[3 + i] = scaledpos - static_cast<float>(my_left[i]);
96  temp_delta[i] = 1.f - temp_delta[3 + i];
97  }
98 
99  delta[0] = temp_delta[0] * temp_delta[1] * temp_delta[2];
100  delta[1] = temp_delta[3] * temp_delta[1] * temp_delta[2];
101  delta[2] = temp_delta[0] * temp_delta[4] * temp_delta[2];
102  delta[3] = temp_delta[3] * temp_delta[4] * temp_delta[2];
103  delta[4] = temp_delta[0] * temp_delta[1] * temp_delta[5];
104  delta[5] = temp_delta[3] * temp_delta[1] * temp_delta[5];
105  delta[6] = temp_delta[0] * temp_delta[4] * temp_delta[5];
106  delta[7] = temp_delta[3] * temp_delta[4] * temp_delta[5];
107 
108  // modulo for negative numbers is strange at best, shift to make sure we are
109  // positive
110  auto const x = static_cast<unsigned int>(my_left[0] + para.dim[0]);
111  auto const y = static_cast<unsigned int>(my_left[1] + para.dim[1]);
112  auto const z = static_cast<unsigned int>(my_left[2] + para.dim[2]);
113 
114  node_index[0] = x % para.dim[0] + para.dim[0] * (y % para.dim[1]) +
115  para.dim[0] * para.dim[1] * (z % para.dim[2]);
116  node_index[1] = (x + 1) % para.dim[0] + para.dim[0] * (y % para.dim[1]) +
117  para.dim[0] * para.dim[1] * (z % para.dim[2]);
118  node_index[2] = x % para.dim[0] + para.dim[0] * ((y + 1) % para.dim[1]) +
119  para.dim[0] * para.dim[1] * (z % para.dim[2]);
120  node_index[3] = (x + 1) % para.dim[0] +
121  para.dim[0] * ((y + 1) % para.dim[1]) +
122  para.dim[0] * para.dim[1] * (z % para.dim[2]);
123  node_index[4] = x % para.dim[0] + para.dim[0] * (y % para.dim[1]) +
124  para.dim[0] * para.dim[1] * ((z + 1) % para.dim[2]);
125  node_index[5] = (x + 1) % para.dim[0] + para.dim[0] * (y % para.dim[1]) +
126  para.dim[0] * para.dim[1] * ((z + 1) % para.dim[2]);
127  node_index[6] = x % para.dim[0] + para.dim[0] * ((y + 1) % para.dim[1]) +
128  para.dim[0] * para.dim[1] * ((z + 1) % para.dim[2]);
129  node_index[7] = (x + 1) % para.dim[0] +
130  para.dim[0] * ((y + 1) % para.dim[1]) +
131  para.dim[0] * para.dim[1] * ((z + 1) % para.dim[2]);
132 
133  for (int i = 0; i < 8; ++i) {
134  // Atomic add is essential because this runs in parallel!
135  atomicAdd(&(node_f.force_density[node_index[i]][0]),
136  (particleForce[0] * delta[i]));
137  atomicAdd(&(node_f.force_density[node_index[i]][1]),
138  (particleForce[1] * delta[i]));
139  atomicAdd(&(node_f.force_density[node_index[i]][2]),
140  (particleForce[2] * delta[i]));
141  }
142  }
143 }

145 __global__ void ParticleVelocitiesFromLB_Kernel(
146  LB_nodes_gpu n_curr,
147  const IBM_CUDA_ParticleDataInput *const particles_input,
148  std::size_t number_of_particles,
149  IBM_CUDA_ParticleDataOutput *const particles_output,
150  LB_node_force_density_gpu node_f, const float *const lb_boundary_velocity,
151  const LB_parameters_gpu *const paraP) {
152 
153  const unsigned int particleIndex = blockIdx.y * gridDim.x * blockDim.x +
154  blockDim.x * blockIdx.x + threadIdx.x;
155 
156  const LB_parameters_gpu &para = *paraP;
157 
158  if (particleIndex < number_of_particles &&
159  particles_input[particleIndex].is_virtual) {
160 
161  // Get position
162  float pos[3] = {particles_input[particleIndex].pos[0],
163  particles_input[particleIndex].pos[1],
164  particles_input[particleIndex].pos[2]};
165  float v[3] = {0};
166 
167  // This part is copied from get_interpolated_velocity
168  // + we add the force + we consider boundaries
169 
170  float temp_delta[6];
171  float delta[8];
172  int my_left[3];
173  unsigned int node_index[8];
174  Utils::Array<float, 4> mode;
175 #pragma unroll
176  for (int i = 0; i < 3; ++i) {
177  const float scaledpos = pos[i] / para.agrid - 0.5f;
178  my_left[i] = static_cast<int>(floorf(scaledpos));
179  temp_delta[3 + i] = scaledpos - static_cast<float>(my_left[i]);
180  temp_delta[i] = 1.f - temp_delta[3 + i];
181  }
182 
183  delta[0] = temp_delta[0] * temp_delta[1] * temp_delta[2];
184  delta[1] = temp_delta[3] * temp_delta[1] * temp_delta[2];
185  delta[2] = temp_delta[0] * temp_delta[4] * temp_delta[2];
186  delta[3] = temp_delta[3] * temp_delta[4] * temp_delta[2];
187  delta[4] = temp_delta[0] * temp_delta[1] * temp_delta[5];
188  delta[5] = temp_delta[3] * temp_delta[1] * temp_delta[5];
189  delta[6] = temp_delta[0] * temp_delta[4] * temp_delta[5];
190  delta[7] = temp_delta[3] * temp_delta[4] * temp_delta[5];
191 
192  // modulo for negative numbers is strange at best, shift to make sure we are
193  // positive
194  auto const x = static_cast<unsigned int>(my_left[0] + para.dim[0]);
195  auto const y = static_cast<unsigned int>(my_left[1] + para.dim[1]);
196  auto const z = static_cast<unsigned int>(my_left[2] + para.dim[2]);
197 
198  node_index[0] = x % para.dim[0] + para.dim[0] * (y % para.dim[1]) +
199  para.dim[0] * para.dim[1] * (z % para.dim[2]);
200  node_index[1] = (x + 1) % para.dim[0] + para.dim[0] * (y % para.dim[1]) +
201  para.dim[0] * para.dim[1] * (z % para.dim[2]);
202  node_index[2] = x % para.dim[0] + para.dim[0] * ((y + 1) % para.dim[1]) +
203  para.dim[0] * para.dim[1] * (z % para.dim[2]);
204  node_index[3] = (x + 1) % para.dim[0] +
205  para.dim[0] * ((y + 1) % para.dim[1]) +
206  para.dim[0] * para.dim[1] * (z % para.dim[2]);
207  node_index[4] = x % para.dim[0] + para.dim[0] * (y % para.dim[1]) +
208  para.dim[0] * para.dim[1] * ((z + 1) % para.dim[2]);
209  node_index[5] = (x + 1) % para.dim[0] + para.dim[0] * (y % para.dim[1]) +
210  para.dim[0] * para.dim[1] * ((z + 1) % para.dim[2]);
211  node_index[6] = x % para.dim[0] + para.dim[0] * ((y + 1) % para.dim[1]) +
212  para.dim[0] * para.dim[1] * ((z + 1) % para.dim[2]);
213  node_index[7] = (x + 1) % para.dim[0] +
214  para.dim[0] * ((y + 1) % para.dim[1]) +
215  para.dim[0] * para.dim[1] * ((z + 1) % para.dim[2]);
216 
217  for (int i = 0; i < 8; ++i) {
218  double local_rho;
219  double local_j[3];
220 #ifdef LB_BOUNDARIES_GPU
221  if (n_curr.boundary[node_index[i]]) {
222  // Boundary node
223  auto const boundary_index =
224  static_cast<int>(n_curr.boundary[node_index[i]]);
225 
226  // lb_boundary_velocity is given in MD units --> convert to LB and
227  // reconvert back at the end of this function
228  local_rho = para.rho;
229  local_j[0] =
230  para.rho * lb_boundary_velocity[3 * (boundary_index - 1) + 0];
231  local_j[1] =
232  para.rho * lb_boundary_velocity[3 * (boundary_index - 1) + 1];
233  local_j[2] =
234  para.rho * lb_boundary_velocity[3 * (boundary_index - 1) + 2];
235 
236  } else
237 #endif
238  {
239  calc_mass_and_momentum_mode(mode, n_curr, node_index[i]);
240  local_rho = para.rho + mode[0];
241 
242  // Add the +f/2 contribution!!
243  local_j[0] = mode[1] + node_f.force_density_buf[node_index[i]][0] / 2.f;
244  local_j[1] = mode[2] + node_f.force_density_buf[node_index[i]][1] / 2.f;
245  local_j[2] = mode[3] + node_f.force_density_buf[node_index[i]][2] / 2.f;
246  }
247 
248  // Interpolate velocity
249  v[0] += static_cast<float>(delta[i] * local_j[0] / local_rho);
250  v[1] += static_cast<float>(delta[i] * local_j[1] / local_rho);
251  v[2] += static_cast<float>(delta[i] * local_j[2] / local_rho);
252  }
253 
254  // Rescale and store output
255  particles_output[particleIndex].v[0] = v[0] * para.agrid / para.tau;
256  particles_output[particleIndex].v[1] = v[1] * para.agrid / para.tau;
257  particles_output[particleIndex].v[2] = v[2] * para.agrid / para.tau;
258  }
259 }

261 __global__ void ResetLBForces_Kernel(LB_node_force_density_gpu node_f,
262  const LB_parameters_gpu *const paraP) {
263 
264  const std::size_t index = blockIdx.y * gridDim.x * blockDim.x +
265  blockDim.x * blockIdx.x + threadIdx.x;
266  const LB_parameters_gpu &para = *paraP;
267 
268  if (index < para.number_of_nodes) {
269  const float force_factor = powf(para.agrid, 2) * para.tau * para.tau;
270  if (para.external_force_density) {
271  node_f.force_density[index][0] = para.ext_force_density[0] * force_factor;
272  node_f.force_density[index][1] = para.ext_force_density[1] * force_factor;
273  node_f.force_density[index][2] = para.ext_force_density[2] * force_factor;
274  } else {
275  node_f.force_density[index] = {};
276  }
277  }

280 /** Transfer particle forces into the LB fluid.
281  * Called from @ref integrate.
282  * This must be the first CUDA-IBM function to be called because it also does
283  * some initialization.
284  */
285 void IBM_ForcesIntoFluid_GPU(ParticleRange const &particles, int this_node) {
286  // This function does
287  // (1) Gather forces from all particles via MPI
288  // (2) Copy forces to the GPU
289  // (3) interpolate on the LBM grid and spread forces
290 
291  auto const numParticles = gpu_get_particle_pointer().size();
292 
293  // Storage only needed on head node
294  if (this_node == 0 &&
295  (IBM_ParticleDataInput_host.empty() || !IBM_initialized ||
296  numParticles != IBM_numParticlesCache))
297  InitCUDA_IBM(numParticles);
298 
299  // We gather particle positions and forces from all nodes
300  IBM_cuda_mpi_get_particles(particles);
301 
302  // GPU only on head node
303  if (this_node == 0 && numParticles > 0) {
304 
305  // Copy data to device
306  cuda_safe_mem(cudaMemcpy(IBM_ParticleDataInput_device,
307  IBM_ParticleDataInput_host.data(),
308  numParticles * sizeof(IBM_CUDA_ParticleDataInput),
309  cudaMemcpyHostToDevice));
310 
311  // Kernel call for spreading the forces on the LB grid
312  dim3 dim_grid = calculate_dim_grid(static_cast<unsigned>(numParticles), 4,
313  threads_per_block);
314  KERNELCALL(ForcesIntoFluid_Kernel, dim_grid, threads_per_block,
315  IBM_ParticleDataInput_device, numParticles, node_f, para_gpu);
316  }
317 }

319 void InitCUDA_IBM(std::size_t const numParticles) {
320 
321  // Check if we have to delete
322  if (!IBM_ParticleDataInput_host.empty()) {
323  IBM_ParticleDataInput_host.clear();
324  IBM_ParticleDataOutput_host.clear();
325  cuda_safe_mem(cudaFree(IBM_ParticleDataInput_device));
326  cuda_safe_mem(cudaFree(IBM_ParticleDataOutput_device));
327  cuda_safe_mem(cudaFree(lb_boundary_velocity_IBM));
328  }
329 
330  // Back and forth communication of positions and velocities
331  IBM_ParticleDataInput_host.resize(numParticles);
332  IBM_ParticleDataOutput_host.resize(numParticles);
333  cuda_safe_mem(cudaMalloc((void **)&IBM_ParticleDataInput_device,
334  numParticles * sizeof(IBM_CUDA_ParticleDataInput)));
335  cuda_safe_mem(cudaMalloc((void **)&IBM_ParticleDataOutput_device,
336  numParticles * sizeof(IBM_CUDA_ParticleDataOutput)));
337 
338  // Use LB parameters
339  lb_get_para_pointer(&para_gpu);
340 
341  // Copy boundary velocities to the GPU
342  // First put them into correct format
343 #ifdef LB_BOUNDARIES_GPU
344  auto *host_lb_boundary_velocity =
345  new float[3 * (LBBoundaries::lbboundaries.size() + 1)];
346 
347  for (int n = 0; n < LBBoundaries::lbboundaries.size(); n++) {
348  host_lb_boundary_velocity[3 * n + 0] =
349  static_cast<float>(LBBoundaries::lbboundaries[n]->velocity()[0]);
350  host_lb_boundary_velocity[3 * n + 1] =
351  static_cast<float>(LBBoundaries::lbboundaries[n]->velocity()[1]);
352  host_lb_boundary_velocity[3 * n + 2] =
353  static_cast<float>(LBBoundaries::lbboundaries[n]->velocity()[2]);
354  }
355 
356  host_lb_boundary_velocity[3 * LBBoundaries::lbboundaries.size() + 0] = 0.0f;
357  host_lb_boundary_velocity[3 * LBBoundaries::lbboundaries.size() + 1] = 0.0f;
358  host_lb_boundary_velocity[3 * LBBoundaries::lbboundaries.size() + 2] = 0.0f;
359 
360  cuda_safe_mem(
361  cudaMalloc((void **)&lb_boundary_velocity_IBM,
362  3 * LBBoundaries::lbboundaries.size() * sizeof(float)));
363  cuda_safe_mem(
364  cudaMemcpy(lb_boundary_velocity_IBM, host_lb_boundary_velocity,
365  3 * LBBoundaries::lbboundaries.size() * sizeof(float),
366  cudaMemcpyHostToDevice));
367 
368  delete[] host_lb_boundary_velocity;
369 #endif
370 
371  IBM_numParticlesCache = numParticles;
372  IBM_initialized = true;
373 }
