72 void recalc_prefactors(double time_step) {
73 pref1_com = gamma_com;
74 pref2_com = std::sqrt(24. * gamma_com / time_step * temp_com);

76 pref2_dist = std::sqrt(24. * gamma_distance / time_step * temp_distance);
