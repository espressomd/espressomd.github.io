27 /**
28  * Template specialization for `Particle`. The traits mechanism is used to get
29  * indirect access to particle properties. This helps making the implementation
30  * of observables independent of the particle type.
31  */
32 template <> struct traits<Particle> {
33  auto position(Particle const &p) const { return p.pos(); }
34  auto velocity(Particle const &p) const { return p.v(); }
35  auto force(Particle const &p) const { return p.force(); }
36  auto mass(Particle const &p) const {
37 #ifdef VIRTUAL_SITES
38  // we exclude virtual particles since their mass does not have a meaning
39  if (p.is_virtual())
40  return decltype(p.mass()){};
41 #endif
42  return p.mass();
43  }
44  auto charge(Particle const &p) const { return p.q(); }
45  auto dipole_moment(Particle const &p) const {
46 #if defined(ROTATION) && defined(DIPOLES)
47  return p.calc_dip();
48 #else
49  return Utils::Vector3d{};
50 #endif
51  }
52  auto velocity_body(Particle const &p) const {
53 #ifdef ROTATION
54  return convert_vector_space_to_body(p, p.v());
55 #else
56  return Utils::Vector3d{};
57 #endif
58  }
59  auto angular_velocity(Particle const &p) const {
60 #ifdef ROTATION
61  return convert_vector_body_to_space(p, p.omega());
62 #else
63  return Utils::Vector3d{};
64 #endif
65  }
66  auto angular_velocity_body(Particle const &p) const {
67 #ifdef ROTATION
68  return p.omega();
69 #else
70  return Utils::Vector3d{};
71 #endif
72  }
73 };
