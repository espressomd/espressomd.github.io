38class ShapeBasedConstraint : public Constraint {
39public:
40 ShapeBasedConstraint()
41 : m_constraint(std::make_shared<::Constraints::ShapeBasedConstraint>(
42 ::System::get_system())),
43 m_shape(nullptr) {
44 add_parameters({{"only_positive", m_constraint->only_positive()},
45 {"penetrable", m_constraint->penetrable()},
46 {"particle_type",
47 [this](Variant const &value) {
48 m_constraint->set_type(get_value<int>(value));
49 },
50 [this]() { return m_constraint->type(); }},
51 {"shape",
52 [this](Variant const &value) {
53 m_shape =
54 get_value<std::shared_ptr<Shapes::Shape>>(value);
55 if (m_shape) {
56 m_constraint->set_shape(m_shape->shape());
57 }
58 },
59 [this]() { return m_shape; }},
60 {"particle_velocity", m_constraint->velocity()}});
61 }
62
63 Variant do_call_method(std::string const &name, VariantMap const &) override {
64 if (name == "total_force") {
65 return shape_based_constraint()->total_force();
66 }
67 if (name == "min_dist") {
68 auto const &system = ::System::get_system();
69 return shape_based_constraint()->min_dist(
70 *system.box_geo, system.cell_structure->local_particles());
71 }
72 if (name == "total_normal_force") {
73 return shape_based_constraint()->total_normal_force();
74 }
75
76 return none;
77 }
78
79 std::shared_ptr<::Constraints::Constraint> constraint() override {
80 return std::static_pointer_cast<::Constraints::Constraint>(m_constraint);
81 }
82 std::shared_ptr<const ::Constraints::Constraint> constraint() const override {
83 return std::static_pointer_cast<::Constraints::Constraint>(m_constraint);
84 }
85 std::shared_ptr<::Constraints::ShapeBasedConstraint>
86 shape_based_constraint() const {
87 return m_constraint;
88 }
89
90private:
91 /* The actual constraint */
92 std::shared_ptr<::Constraints::ShapeBasedConstraint> m_constraint;
93
94 /* Keep a reference to the shape */
95 std::shared_ptr<Shapes::Shape> m_shape;
96};
