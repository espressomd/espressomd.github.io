33 Morse_Parameters::Morse_Parameters(double eps, double alpha, double rmin,
34  double cutoff)
35  : eps{eps}, alpha{alpha}, rmin{rmin}, cut{cutoff} {
36  if (eps < 0.) {
37  throw std::domain_error("Morse parameter 'eps' has to be >= 0");
38  }
39  if (cutoff < 0.) {
40  throw std::domain_error("Morse parameter 'cutoff' has to be >= 0");
41  }
42  auto const add1 = std::exp(-2.0 * alpha * (cut - rmin));
43  auto const add2 = 2.0 * std::exp(-alpha * (cut - rmin));
44  rest = eps * (add1 - add2);
45 }
