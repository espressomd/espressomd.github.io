60 /**
61  * @brief Get pairs of particles that are closer than a distance and fulfill a
62  * filter criterion.
63  *
64  * It uses link_cell to get pairs out of the cellsystem
65  * by a simple distance criterion and
66  *
67  * Pairs are sorted so that first.id < second.id
68  */
69 template <class Filter>
70 std::vector<std::pair<int, int>> get_pairs_filtered(double const distance,
71  Filter filter) {
72  std::vector<std::pair<int, int>> ret;
73  on_observable_calc();
74  auto const cutoff2 = distance * distance;
75  auto pair_kernel = [&ret, &cutoff2, &filter](Particle const &p1,
76  Particle const &p2,
77  Distance const &d) {
78  if (d.dist2 < cutoff2 and filter(p1) and filter(p2))
79  ret.emplace_back(p1.id(), p2.id());
80  };
81 
82  cell_structure.non_bonded_loop(pair_kernel);
83 
84  /* Sort pairs */
85  for (auto &pair : ret) {
86  if (pair.first > pair.second)
87  std::swap(pair.first, pair.second);
88  }
89 
90  return ret;
91 }

108 static void search_distance_sanity_check(double const distance) {
109  /* get_pairs_filtered() finds pairs via the non_bonded_loop. The maximum
110  * finding range is therefore limited by the decomposition that is used.
111  */
112  auto const range = *boost::min_element(cell_structure.max_range());
113  if (distance > range) {
114  throw std::domain_error("pair search distance " + std::to_string(distance) +
115  " bigger than the decomposition range " +
116  std::to_string(range));
117  }
118 }
119 static void search_neighbors_sanity_check(double const distance) {
120  search_distance_sanity_check(distance);
121  if (cell_structure.decomposition_type() ==
122  CellStructureType::CELL_STRUCTURE_HYBRID) {
123  throw std::runtime_error("Cannot search for neighbors in the hybrid "
124  "decomposition cell system");
125  }
126 }
127 } // namespace detail

129 boost::optional<std::vector<int>>
130 mpi_get_short_range_neighbors_local(int const pid, double const distance,
131  bool run_sanity_checks) {
132 
133  if (run_sanity_checks) {
134  detail::search_neighbors_sanity_check(distance);
135  }
136  on_observable_calc();
137 
138  auto const p = cell_structure.get_local_particle(pid);
139  if (not p or p->is_ghost()) {
140  return {};
141  }
142 
143  std::vector<int> ret;
144  auto const cutoff2 = distance * distance;
145  auto kernel = [&ret, cutoff2](Particle const &p1, Particle const &p2,
146  Utils::Vector3d const &vec) {
147  if (vec.norm2() < cutoff2) {
148  ret.emplace_back(p2.id());
149  }
150  };
151  cell_structure.run_on_particle_short_range_neighbors(*p, kernel);
152  return ret;
153 }

172 std::vector<std::pair<int, int>>
173 get_pairs_of_types(double const distance, std::vector<int> const &types) {
174  detail::search_distance_sanity_check(distance);
175  auto pairs = get_pairs_filtered(distance, [types](Particle const &p) {
176  return std::any_of(types.begin(), types.end(),
177  // NOLINTNEXTLINE(bugprone-exception-escape)
178  [p](int const type) { return p.type() == type; });
179  });
180  Utils::Mpi::gather_buffer(pairs, comm_cart);
181  return pairs;
182 }

203 void cells_re_init(CellStructureType new_cs) {
204  switch (new_cs) {
205  case CellStructureType::CELL_STRUCTURE_REGULAR:
206  cell_structure.set_regular_decomposition(comm_cart, interaction_range(),
207  box_geo, local_geo);
208  break;
209  case CellStructureType::CELL_STRUCTURE_NSQUARE:
210  cell_structure.set_atom_decomposition(comm_cart, box_geo, local_geo);
211  break;
212  case CellStructureType::CELL_STRUCTURE_HYBRID: {
213  /* Get current HybridDecomposition to extract n_square_types */
214  auto &current_hybrid_decomposition =
215  dynamic_cast<HybridDecomposition const &>(
216  Utils::as_const(cell_structure).decomposition());
217  cell_structure.set_hybrid_decomposition(
218  comm_cart, current_hybrid_decomposition.get_cutoff_regular(), box_geo,
219  local_geo, current_hybrid_decomposition.get_n_square_types());
220  break;
221  }
222  default:
223  throw std::runtime_error("Unknown cell system type");
224  }
225 
226  on_cell_structure_change();
227 }

238 void cells_update_ghosts(unsigned data_parts) {
239  /* data parts that are only updated on resort */
240  auto constexpr resort_only_parts =
241  Cells::DATA_PART_PROPERTIES | Cells::DATA_PART_BONDS;
242 
243  auto const global_resort =
244  boost::mpi::all_reduce(comm_cart, cell_structure.get_resort_particles(),
245  std::bit_or<unsigned>());
246 
247  if (global_resort != Cells::RESORT_NONE) {
248  int global = (global_resort & Cells::RESORT_GLOBAL)
249  ? CELL_GLOBAL_EXCHANGE
250  : CELL_NEIGHBOR_EXCHANGE;
251 
252  /* Resort cell system */
253  cell_structure.resort_particles(global, box_geo);
254  cell_structure.ghosts_count();
255  cell_structure.ghosts_update(data_parts);
256 
257  /* Add the ghost particles to the index if we don't already
258  * have them. */
259  for (auto &p : cell_structure.ghost_particles()) {
260  if (cell_structure.get_local_particle(p.id()) == nullptr) {
261  cell_structure.update_particle_index(p.id(), &p);
262  }
263  }
264 
265  /* Particles are now sorted */
266  cell_structure.clear_resort_particles();
267  } else {
268  /* Communication step: ghost information */
269  cell_structure.ghosts_update(data_parts & ~resort_only_parts);
270  }
271 }
