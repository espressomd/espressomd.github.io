42 template <typename T, typename BoundaryClass> class BoundaryHandling {
43 private:
44  /** Flag for domain cells, i.e. all cells. */
45  FlagUID const Domain_flag{"domain"};
46  /** Flag for boundary cells. */
47  FlagUID const Boundary_flag{"boundary"};

49  /** Container for the map between cells and values. */
50  class DynamicValueCallback {
51  public:
52  DynamicValueCallback() {
53  m_value_boundary = std::make_shared<std::unordered_map<Cell, T>>();
54  }
55 
56  [[nodiscard]] T operator()(
57  Cell const &local,
58  std::shared_ptr<blockforest::StructuredBlockForest> const &blocks,
59  IBlock &block) const {
60  Cell global;
61  blocks->transformBlockLocalToGlobalCell(global, block, local);
62  return get_value(global);
63  }
64 
65  void set_node_boundary_value(Utils::Vector3i const &node, T const &val) {
66  auto const global = Cell(node[0], node[1], node[2]);
67  (*m_value_boundary)[global] = val;
68  }
69 
70  void unset_node_boundary_value(Utils::Vector3i const &node) {
71  auto const global = Cell(node[0], node[1], node[2]);
72  assert(m_value_boundary->count(global));
73  m_value_boundary->erase(global);
74  }
75 
76  [[nodiscard]] auto
77  get_node_boundary_value(Utils::Vector3i const &node) const {
78  auto const global = Cell(node[0], node[1], node[2]);
79  return get_value(global);
80  }
81 
82  bool node_is_boundary(Utils::Vector3i const &node) const {
83  auto const global = Cell(node[0], node[1], node[2]);
84  return m_value_boundary->count(global) != 0;
85  }
86 
87  private:
88  std::shared_ptr<std::unordered_map<Cell, T>> m_value_boundary;
89  static constexpr T default_value{};
90 
91  [[nodiscard]] T get_value(Cell const &cell) const {
92  if (m_value_boundary->count(cell) == 0) {
93  return default_value;
94  }
95  return m_value_boundary->at(cell);
96  }
97  };

99  [[nodiscard]] inline auto get_flag_field_and_flag(IBlock *block) const {
100  auto const flag_field =
101  block->template uncheckedFastGetData<FlagField>(m_flag_field_id);
102  auto const boundary_flag = flag_field->getFlag(Boundary_flag);
103  return std::make_tuple(flag_field, boundary_flag);
104  }
105 
106 public:
107  using value_type = T;
108  using FlagField = field::FlagField<uint8_t>;
109 
110  BoundaryHandling(std::shared_ptr<StructuredBlockForest> blocks,
111  BlockDataID value_field_id, BlockDataID flag_field_id)
112  : m_blocks(std::move(blocks)), m_flag_field_id(flag_field_id),
113  m_callback(DynamicValueCallback()), m_pending_changes(false) {
114  // reinitialize the flag field
115  for (auto b = m_blocks->begin(); b != m_blocks->end(); ++b) {
116  flag_reset_kernel(&*b);
117  }
118  // instantiate the boundary sweep
119  std::function callback = m_callback;
120  m_boundary =
121  std::make_shared<BoundaryClass>(m_blocks, value_field_id, callback);
122  }
123 
124  void operator()(IBlock *block) { (*m_boundary)(block); }
125 
126  [[nodiscard]] bool node_is_boundary(Utils::Vector3i const &node) const {
127  return m_callback.node_is_boundary(node);
128  }
129 
130  [[nodiscard]] auto
131  get_node_value_at_boundary(Utils::Vector3i const &node) const {

133  }

135  void set_node_value_at_boundary(Utils::Vector3i const &node, T const &v,
136  BlockAndCell const &bc) {
137  auto [flag_field, boundary_flag] = get_flag_field_and_flag(bc.block);
138  m_callback.set_node_boundary_value(node, v);
139  flag_field->addFlag(bc.cell, boundary_flag);
140  m_pending_changes = true;
141  }
142 
143  void unpack_node(Utils::Vector3i const &node, T const &v) {
144  m_callback.set_node_boundary_value(node, v);
145  }
146 
147  void remove_node_from_boundary(Utils::Vector3i const &node,
148  BlockAndCell const &bc) {
149  auto [flag_field, boundary_flag] = get_flag_field_and_flag(bc.block);
150  m_callback.unset_node_boundary_value(node);
151  flag_field->removeFlag(bc.cell, boundary_flag);
152  m_pending_changes = true;
153  }

155  /** Assign boundary conditions to boundary cells. */
156  void boundary_update() {
157  if (m_pending_changes) {
158  m_boundary->template fillFromFlagField<FlagField>(
159  m_blocks, m_flag_field_id, Boundary_flag, Domain_flag);
160  m_pending_changes = false;
161  }
162  }
163 
164  std::tuple<int, int, int> block_dims(IBlock const &block) const {
165  auto const field = block.template getData<FlagField>(m_flag_field_id);
166  return {static_cast<int>(field->xSize()), static_cast<int>(field->ySize()),
167  static_cast<int>(field->zSize())};
168  }
169 
170 private:
171  std::shared_ptr<StructuredBlockForest> m_blocks;
172  BlockDataID m_flag_field_id;
173  DynamicValueCallback m_callback;
174  std::shared_ptr<BoundaryClass> m_boundary;
175  bool m_pending_changes;
176 
177  /** Register flags and reset all cells. */
178  void flag_reset_kernel(IBlock *const block) {
179  auto flag_field = block->template getData<FlagField>(m_flag_field_id);
180  // register flags
181  if (!flag_field->flagExists(Domain_flag))
182  flag_field->registerFlag(Domain_flag);
183  if (!flag_field->flagExists(Boundary_flag))
184  flag_field->registerFlag(Boundary_flag);
185  // mark all cells as domain cells and fluid cells
186  auto domain_flag = flag_field->getFlag(Domain_flag);
187  auto boundary_flag = flag_field->getFlag(Boundary_flag);
188  for (auto it = flag_field->begin(); it != flag_field->end(); ++it) {
189  flag_field->addFlag(it.x(), it.y(), it.z(), domain_flag);
190  flag_field->removeFlag(it.x(), it.y(), it.z(), boundary_flag);
191  }
192  }
193 };
