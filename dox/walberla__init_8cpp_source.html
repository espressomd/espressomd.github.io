43void mpi_init() {
44 assert(::walberla_mpi_env == nullptr);
45 assert(::walberla_mpi_comm == nullptr);
46 assert(::walberla_mpi_cart_comm_observer == nullptr);
47 int argc = 0;
48 char **argv = nullptr;
49 ::walberla_mpi_env = std::make_shared<walberla::mpi::Environment>(argc, argv);
50 ::walberla_mpi_comm = walberla::mpi::MPIManager::instance();
51 ::walberla_mpi_cart_comm_observer = std::make_shared<int>(0);
52}
53
54void mpi_reinit(int const *const cart_topol) {
55 assert(::walberla_mpi_env.use_count() >= 1);
56 assert(::walberla_mpi_comm.use_count() >= 1);
57 assert(::walberla_mpi_cart_comm_observer.use_count() == 1);
58 ::walberla_mpi_comm->resetMPI();
59 ::walberla_mpi_comm->createCartesianComm(cart_topol[0], cart_topol[1],
60 cart_topol[2], true, true, true);
61 ::walberla_mpi_cart_comm_observer = std::make_shared<int>(0);
62}
63
64void mpi_deinit() {
65 assert(::walberla_mpi_env.use_count() >= 1);
66 assert(::walberla_mpi_comm.use_count() >= 1);
67 assert(::walberla_mpi_cart_comm_observer.use_count() == 1);
68 ::walberla_mpi_env.reset();
69 ::walberla_mpi_comm.reset();
70 ::walberla_mpi_cart_comm_observer.reset();
71}

73std::unique_ptr<ResourceManager> get_vtk_dependent_resources() {
74 auto vtk_dependencies = std::make_unique<ResourceManager>();
75 // waLBerla MPI communicator (singleton)
76 vtk_dependencies->acquire_lock(::walberla_mpi_comm);
77 // waLBerla MPI environment (destructor depends on the MPI communicator)
78 vtk_dependencies->acquire_lock(::walberla_mpi_env);
79 return vtk_dependencies;
80}

82ResourceObserver get_mpi_cart_comm_observer() {
83 assert(walberla_mpi_cart_comm_observer.use_count() <= 1);
84 return walberla_mpi_cart_comm_observer;
85}
