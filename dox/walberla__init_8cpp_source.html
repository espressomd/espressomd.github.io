36void mpi_init() {
37 assert(::walberla_mpi_env == nullptr);
38 assert(::walberla_mpi_comm == nullptr);
39 int argc = 0;
40 char **argv = nullptr;
41 ::walberla_mpi_env = std::make_shared<walberla::mpi::Environment>(argc, argv);
42 ::walberla_mpi_comm = walberla::MPIManager::instance();
43}
44
45void mpi_deinit() {
46 assert(::walberla_mpi_env.use_count() >= 1);
47 assert(::walberla_mpi_comm.use_count() >= 1);
48 ::walberla_mpi_env.reset();
49 ::walberla_mpi_comm.reset();
50}
