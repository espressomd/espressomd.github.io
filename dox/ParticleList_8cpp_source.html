55 static void set_bonds(ParticleHandle &p, Variant const &bonds) {
56  auto const bond_list_flat = get_value<std::vector<std::vector<int>>>(bonds);
57  for (auto const &bond_flat : bond_list_flat) {
58  auto const bond_id = bond_flat[0];
59  auto const part_id =
60  std::vector<int>{bond_flat.begin() + 1, bond_flat.end()};
61  p.call_method("add_bond",
62  {{"bond_id", bond_id}, {"part_id", std::move(part_id)}});
63  }
64 }

66 std::string ParticleList::get_internal_state() const {
67  auto const p_ids = get_particle_ids();
68  std::vector<std::string> object_states(p_ids.size());
69 
70  boost::transform(p_ids, object_states.begin(), [this](auto const p_id) {
71  auto p_obj =
72  context()->make_shared("Particles::ParticleHandle", {{"id", p_id}});
73  auto &p_handle = dynamic_cast<ParticleHandle &>(*p_obj);
74  auto const packed_state = p_handle.serialize();
75  // custom particle serialization
76  auto state = Utils::unpack<ObjectState>(packed_state);
77  state.name = "Particles::ParticleHandle";
78  auto const bonds_view = p_handle.call_method("get_bonds_view", {});
79  state.params.emplace_back(std::string{"bonds"}, pack(bonds_view));
80 #ifdef EXCLUSIONS
81  auto const exclusions = p_handle.call_method("get_exclusions", {});
82  state.params.emplace_back(std::string{"exclusions"}, pack(exclusions));
83 #endif // EXCLUSIONS
84  state.params.emplace_back(std::string{"__cpt_sentinel"}, pack(None{}));
85  return Utils::pack(state);
86  });
87 
88  return Utils::pack(object_states);
89 }

91 void ParticleList::set_internal_state(std::string const &state) {
92  auto const object_states = Utils::unpack<std::vector<std::string>>(state);
93 #ifdef EXCLUSIONS
94  std::unordered_map<int, Variant> exclusions = {};
95 #endif // EXCLUSIONS
96  std::unordered_map<int, Variant> bonds = {};
97 
98  for (auto const &packed_object : object_states) {
99  auto state = Utils::unpack<ObjectState>(packed_object);
100  VariantMap params = {};
101  for (auto const &kv : state.params) {
102  params[kv.first] = unpack(kv.second, {});
103  }
104  auto const p_id = get_value<int>(params.at("id"));
105  bonds[p_id] = params.extract("bonds").mapped();
106 #ifdef EXCLUSIONS
107  exclusions[p_id] = params.extract("exclusions").mapped();
108 #endif // EXCLUSIONS
109  context()->make_shared("Particles::ParticleHandle", params);
110  }
111 
112  for (auto const p_id : get_particle_ids()) {
113  auto p_obj =
114  context()->make_shared("Particles::ParticleHandle", {{"id", p_id}});
115  auto &p_handle = dynamic_cast<ParticleHandle &>(*p_obj);
116  set_bonds(p_handle, bonds[p_id]);
117 #ifdef EXCLUSIONS
118  set_exclusions(p_handle, exclusions[p_id]);
119 #endif // EXCLUSIONS
120  }
121 }

124 /**
125  * @brief Use the bond topology to automatically add exclusions between
126  * particles that are up to @c n_bonds_max bonds apart in a chain.
127  */
128 static void auto_exclusions(boost::mpi::communicator const &comm,
129  int const n_bonds_max) {
130  // bookkeeping of particle exclusions, with their n-bond distance
131  std::unordered_map<int, std::vector<std::pair<int, int>>> partners;
132  std::vector<int> bonded_pairs;
133 
134  auto &system = ::System::get_system();
135  auto &cell_structure = *system.cell_structure;
136 
137  // determine initial connectivity
138  for (auto const &p : cell_structure.local_particles()) {
139  auto const pid1 = p.id();
140  for (auto const bond : p.bonds()) {
141  if (bond.partner_ids().size() == 1) {
142  auto const pid2 = bond.partner_ids()[0];
143  if (pid1 != pid2) {
144  bonded_pairs.emplace_back(pid1);
145  bonded_pairs.emplace_back(pid2);
146  }
147  }
148  }
149  }
150 
151  Utils::Mpi::gather_buffer(bonded_pairs, comm);
152 
153  if (comm.rank() == 0) {
154  auto const add_partner = [&partners](int pid1, int pid2, int n_bonds) {
155  if (pid2 == pid1)
156  return;
157  for (auto const &partner : partners[pid1])
158  if (partner.first == pid2)
159  return;
160  partners[pid1].emplace_back(pid2, n_bonds);
161  };
162 
163  for (auto it = bonded_pairs.begin(); it != bonded_pairs.end(); it += 2) {
164  add_partner(it[0], it[1], 1);
165  add_partner(it[1], it[0], 1);
166  }
167 
168  // determine transient connectivity
169  for (int iteration = 1; iteration < n_bonds_max; iteration++) {
170  std::vector<int> pids;
171  for (auto const &kv : partners) {
172  pids.emplace_back(kv.first);
173  }
174  for (auto const pid1 : pids) {
175  // loop over partners (counter-based loops due to iterator invalidation)
176  // NOLINTNEXTLINE(modernize-loop-convert)
177  for (int i = 0; i < partners[pid1].size(); ++i) {
178  auto const [pid2, dist21] = partners[pid1][i];
179  if (dist21 > n_bonds_max)
180  continue;
181  // loop over all partners of the partner
182  // NOLINTNEXTLINE(modernize-loop-convert)
183  for (int j = 0; j < partners[pid2].size(); ++j) {
184  auto const [pid3, dist32] = partners[pid2][j];
185  auto const dist31 = dist32 + dist21;
186  if (dist31 > n_bonds_max)
187  continue;
188  add_partner(pid1, pid3, dist31);
189  add_partner(pid3, pid1, dist31);
190  }
191  }
192  }
193  }
194  }
195 
196  boost::mpi::broadcast(comm, partners, 0);
197  for (auto const &kv : partners) {
198  auto const pid1 = kv.first;
199  auto const &partner_list = kv.second;
200  for (auto const &partner : partner_list) {
201  auto const pid2 = partner.first;
202  if (auto p1 = cell_structure.get_local_particle(pid1)) {
203  add_exclusion(*p1, pid2);
204  }
205  if (auto p2 = cell_structure.get_local_particle(pid2)) {
206  add_exclusion(*p2, pid1);
207  }
208  }
209  }
210  system.on_particle_change();
211 }
212 #endif // EXCLUSIONS

214 Variant ParticleList::do_call_method(std::string const &name,
215  VariantMap const &params) {
216 #ifdef EXCLUSIONS
217  if (name == "auto_exclusions") {
218  auto const distance = get_value<int>(params, "distance");
219  auto_exclusions(context()->get_comm(), distance);
220  return {};
221  }
222 #endif // EXCLUSIONS
223  if (name == "get_highest_particle_id") {
224  return get_maximal_particle_id();
225  }
226  if (name == "clear") {
227  remove_all_particles();
228  return {};
229  }
230  if (not context()->is_head_node()) {
231  return {};
232  }
233  if (name == "get_n_part") {
234  return get_n_part();
235  }
236  if (name == "get_particle_ids") {
237  return get_particle_ids();
238  }
239  if (name == "particle_exists") {
240  return particle_exists(get_value<int>(params, "p_id"));
241  }
242  if (name == "add_particle") {
243  assert(params.count("bonds") == 0);
244  auto obj = context()->make_shared("Particles::ParticleHandle", params);
245  auto &p_handle = dynamic_cast<ParticleHandle &>(*obj);
246 #ifdef EXCLUSIONS
247  if (params.count("exclusions")) {
248  set_exclusions(p_handle, params.at("exclusions"));
249  }
250 #endif // EXCLUSIONS
251  return p_handle.get_parameter("id");
252  }
253  return {};
254 }
