81inline ParticleForce calc_central_radial_force(IA_parameters const &ia_params,
82 Utils::Vector3d const &d,
83 double const dist) {
84
85 ParticleForce pf{};
86 auto force_factor = 0.;
87/* Lennard-Jones */
88#ifdef ESPRESSO_LENNARD_JONES
89 force_factor += lj_pair_force_factor(ia_params, dist);
90#endif
91/* WCA */
92#ifdef ESPRESSO_WCA
93 force_factor += wca_pair_force_factor(ia_params, dist);
94#endif
95/* Lennard-Jones generic */
96#ifdef ESPRESSO_LENNARD_JONES_GENERIC
97 force_factor += ljgen_pair_force_factor(ia_params, dist);
98#endif
99/* smooth step */
100#ifdef ESPRESSO_SMOOTH_STEP
101 force_factor += SmSt_pair_force_factor(ia_params, dist);
102#endif
103/* Hertzian force */
104#ifdef ESPRESSO_HERTZIAN
105 force_factor += hertzian_pair_force_factor(ia_params, dist);
106#endif
107/* Gaussian force */
108#ifdef ESPRESSO_GAUSSIAN
109 force_factor += gaussian_pair_force_factor(ia_params, dist);
110#endif
111/* BMHTF NaCl */
112#ifdef ESPRESSO_BMHTF_NACL
113 force_factor += BMHTF_pair_force_factor(ia_params, dist);
114#endif
115/* Buckingham*/
116#ifdef ESPRESSO_BUCKINGHAM
117 force_factor += buck_pair_force_factor(ia_params, dist);
118#endif
119/* Morse*/
120#ifdef ESPRESSO_MORSE
121 force_factor += morse_pair_force_factor(ia_params, dist);
122#endif
123/*soft-sphere potential*/
124#ifdef ESPRESSO_SOFT_SPHERE
125 force_factor += soft_pair_force_factor(ia_params, dist);
126#endif
127/*hat potential*/
128#ifdef ESPRESSO_HAT
129 force_factor += hat_pair_force_factor(ia_params, dist);
130#endif
131/* Lennard-Jones cosine */
132#ifdef ESPRESSO_LJCOS
133 force_factor += ljcos_pair_force_factor(ia_params, dist);
134#endif
135/* Lennard-Jones cosine */
136#ifdef ESPRESSO_LJCOS2
137 force_factor += ljcos2_pair_force_factor(ia_params, dist);
138#endif
139/* tabulated */
140#ifdef ESPRESSO_TABULATED
141 force_factor += tabulated_pair_force_factor(ia_params, dist);
142#endif
143 pf.f += force_factor * d;
144 return pf;
145}

148 Particle const &p2,
149 IA_parameters const &ia_params,
150 Utils::Vector3d const &d,

161inline ParticleForce calc_opposing_force(ParticleForce const &pf,
162 Utils::Vector3d const &d) {
163 ParticleForce out{-pf.f};
164#ifdef ESPRESSO_ROTATION
165 // if torque is a null vector, the opposing torque is a null vector too
166 // (this check guards from returning a small yet non-null opposing
167 // torque due to numerical imprecision)
168 if (pf.torque[0] != 0. || pf.torque[1] != 0. || pf.torque[2] != 0.) {
169 out.torque = -(pf.torque + vector_product(d, pf.f));
170 }
171#endif
172 return out;
173}

178inline void add_non_bonded_pair_force_with_p(
179 Particle &p1, Particle &p2, ParticleForce &pf, ParticleForce &p1f_asym,
180 ParticleForce &p2f_asym, Utils::Vector3d const &d, double dist,
181 double dist2, double q1q2, IA_parameters const &ia_params,
182 [[maybe_unused]] bool do_nonbonded_flag,
183 Thermostat::Thermostat const &thermostat, BoxGeometry const &box_geo,
184 [[maybe_unused]] BondedInteractionsMap const &bonded_ias,
185 [[maybe_unused]] Utils::Vector3d *const virial,
186 Coulomb::ShortRangeForceKernel::kernel_type const *coulomb_kernel,
187 Dipoles::ShortRangeForceKernel::kernel_type const *dipoles_kernel,
188 Coulomb::ShortRangeForceCorrectionsKernel::kernel_type const *elc_kernel,
189 Coulomb::ShortRangeEnergyKernel::kernel_type const *coulomb_u_kernel) {
190
191 /***********************************************/
192 /* non-bonded pair potentials */
193 /***********************************************/
194
195 if (dist < ia_params.max_cut) {
196#ifdef ESPRESSO_EXCLUSIONS
197 if (do_nonbonded_flag) {
198#endif
199#ifdef ESPRESSO_THOLE
200 pf.f += thole_pair_force(p1, p2, ia_params, d, dist, bonded_ias,
201 coulomb_kernel);
202#endif
203 pf += calc_non_central_force(p1, p2, ia_params, d, dist);
204#ifdef ESPRESSO_EXCLUSIONS
205 }
206#endif
207 }
208
209 /*********************************************************************/
210 /* everything before this contributes to the virial pressure in NpT, */
211 /* but nothing afterwards, since the contribution to pressure from */
212 /* electrostatic is calculated by energy */
213 /*********************************************************************/
214#ifdef ESPRESSO_NPT
215 if (virial) {
216 *virial += hadamard_product(pf.f, d);
217 }
218#endif // ESPRESSO_NPT
219
220 /***********************************************/
221 /* short-range electrostatics */
222 /***********************************************/
223
224#ifdef ESPRESSO_ELECTROSTATICS
225 // real-space electrostatic charge-charge interaction
226 if (q1q2 != 0. and coulomb_kernel != nullptr) {
227 pf.f += (*coulomb_kernel)(q1q2, d, dist);
228#ifdef ESPRESSO_NPT
229 if (virial) {
230 (*virial)[0] += (*coulomb_u_kernel)(p1.pos(), p2.pos(), q1q2, d, dist);
231 }
232#endif // ESPRESSO_NPT
233 if (elc_kernel) {
234 (*elc_kernel)(p1.pos(), p2.pos(), p1f_asym, p2f_asym, q1q2);
235 }
236 }
237#endif // ESPRESSO_ELECTROSTATICS
238
239 /***********************************************/
240 /* thermostat */
241 /***********************************************/
242
243 /* The inter dpd force should not be part of the virial */
244#ifdef ESPRESSO_DPD
245 if (thermostat.thermo_switch & THERMO_DPD) {
246 auto const force =
247 dpd_pair_force(p1.pos(), p1.v(), p1.id(), p2.pos(), p2.v(), p2.id(),
248 *thermostat.dpd, box_geo, ia_params, d, dist, dist2);
249 pf += force;
250 }
251#endif
252
253 /***********************************************/
254 /* short-range magnetostatics */
255 /***********************************************/
256
257#ifdef ESPRESSO_DIPOLES
258 // real-space magnetic dipole-dipole
259 if (dipoles_kernel) {
260 auto const d1d2 = p1.dipm() * p2.dipm();
261 if (d1d2 != 0.) {
262 pf +=
263 (*dipoles_kernel)(d1d2, p1.calc_dip(), p2.calc_dip(), d, dist, dist2);
264 }
265 }
266#endif
267}

269/** Calculate non-bonded forces between a pair of particles and update their
270 * forces and torques.
271 * @param[in,out] p1 particle 1.
272 * @param[in,out] p2 particle 2.
273 * @param[in] d vector between @p p1 and @p p2.
274 * @param[in] dist distance between @p p1 and @p p2.
275 * @param[in] dist2 distance squared between @p p1 and @p p2.
276 * @param[in] q1q2 charge x charge between @p p1 and @p p2.
277 * @param[in] ia_params non-bonded interaction kernels.
278 * @param[in] thermostat thermostat.
279 * @param[in] box_geo box geometry.
280 * @param[in] bonded_ias bonded interaction kernels.
281 * @param[out] virial NpT virial.
282 * @param[in] coulomb_kernel Coulomb force kernel.
283 * @param[in] dipoles_kernel Dipolar force kernel.
284 * @param[in] elc_kernel ELC force correction kernel.
285 * @param[in] coulomb_u_kernel Coulomb energy kernel.
286 */
287inline void add_non_bonded_pair_force(
288 Particle &p1, Particle &p2, Utils::Vector3d const &d, double dist,
289 double dist2, double q1q2, IA_parameters const &ia_params,
290 Thermostat::Thermostat const &thermostat, BoxGeometry const &box_geo,
291 [[maybe_unused]] BondedInteractionsMap const &bonded_ias,
292 [[maybe_unused]] Utils::Vector3d *const virial,
293 Coulomb::ShortRangeForceKernel::kernel_type const *coulomb_kernel,
294 Dipoles::ShortRangeForceKernel::kernel_type const *dipoles_kernel,
295 Coulomb::ShortRangeForceCorrectionsKernel::kernel_type const *elc_kernel,
296 Coulomb::ShortRangeEnergyKernel::kernel_type const *coulomb_u_kernel) {
297
298 ParticleForce pf{};
299 ParticleForce p1f_asym{};
300 ParticleForce p2f_asym{};
301
302#ifdef ESPRESSO_EXCLUSIONS
303 auto const do_nonbonded_flag = do_nonbonded(p1, p2);
304#else
305 auto constexpr do_nonbonded_flag = true;
306#endif
307
308 if (dist < ia_params.max_cut) {
309#ifdef ESPRESSO_EXCLUSIONS
310 if (do_nonbonded_flag) {
311#endif
312 pf += calc_central_radial_force(ia_params, d, dist);
313#ifdef ESPRESSO_EXCLUSIONS
314 }
315#endif
316 }
317
318 add_non_bonded_pair_force_with_p(
319 p1, p2, pf, p1f_asym, p2f_asym, d, dist, dist2, q1q2, ia_params,
320 do_nonbonded_flag, thermostat, box_geo, bonded_ias, virial,
321 coulomb_kernel, dipoles_kernel, elc_kernel, coulomb_u_kernel);
322
323 /***********************************************/
324 /* add total non-bonded forces to particles */
325 /***********************************************/
326
327 p1.force_and_torque() += pf + p1f_asym;
328 p2.force_and_torque() += calc_opposing_force(pf, d) + p2f_asym;
329}

331/** Compute the bonded interaction force between particle pairs.
332 *
333 * @param[in] iaparams Bonded parameters for the interaction.
334 * @param[in] p1 First particle.
335 * @param[in] p2 Second particle.
336 * @param[in] dx Vector between @p p1 and @p p2.
337 * @param[in] kernel Coulomb force kernel.
338 */
339inline std::optional<Utils::Vector3d> calc_bond_pair_force(
340 Bonded_IA_Parameters const &iaparams, Particle const &p1,
341 Particle const &p2, Utils::Vector3d const &dx,
342 Coulomb::ShortRangeForceKernel::kernel_type const *kernel) {
343 if (auto const *iap = std::get_if<FeneBond>(&iaparams)) {
344 return iap->force(dx);
345 }
346 if (auto const *iap = std::get_if<HarmonicBond>(&iaparams)) {
347 return iap->force(dx);
348 }
349 if (auto const *iap = std::get_if<QuarticBond>(&iaparams)) {
350 return iap->force(dx);
351 }
352#ifdef ESPRESSO_ELECTROSTATICS
353 if (auto const *iap = std::get_if<BondedCoulomb>(&iaparams)) {
354 return iap->force(p1.q() * p2.q(), dx);
355 }
356 if (auto const *iap = std::get_if<BondedCoulombSR>(&iaparams)) {
357 return iap->force(dx, *kernel);
358 }
359#endif
360#ifdef ESPRESSO_BOND_CONSTRAINT
361 if (std::get_if<RigidBond>(&iaparams)) {
362 return Utils::Vector3d{};
363 }
364#endif
365#ifdef ESPRESSO_TABULATED
366 if (auto const *iap = std::get_if<TabulatedDistanceBond>(&iaparams)) {
367 return iap->force(dx);
368 }
369#endif
370 if (std::get_if<VirtualBond>(&iaparams)) {
371 return Utils::Vector3d{};
372 }
373 throw BondUnknownTypeError();
374}

376inline bool add_bonded_two_body_force(
377 Bonded_IA_Parameters const &iaparams, BoxGeometry const &box_geo,
378 Particle &p1, Particle &p2, [[maybe_unused]] Utils::Vector3d *const virial,
379 Coulomb::ShortRangeForceKernel::kernel_type const *kernel) {
380 auto const dx = box_geo.get_mi_vector(p1.pos(), p2.pos());
381
382 if (auto const *iap = std::get_if<ThermalizedBond>(&iaparams)) {
383 auto result = iap->forces(p1, p2, dx);
384 if (result) {
385 auto const &forces = result.value();
386
387 p1.force() += std::get<0>(forces);
388 p2.force() += std::get<1>(forces);
389
390 return false;
391 }
392 } else {
393 auto result = calc_bond_pair_force(iaparams, p1, p2, dx, kernel);
394 if (result) {
395 p1.force() += result.value();
396 p2.force() -= result.value();
397
398#ifdef ESPRESSO_NPT
399 if (virial) {
400 *virial += hadamard_product(result.value(), dx);
401 }
402#endif
403 return false;
404 }
405 }
406 return true;
407}

427 return iap->forces(vec1, vec2);
428 }
429#endif
430 if (auto const *iap = std::get_if<IBMTriel>(&iaparams)) {
431 return iap->calc_forces(vec1, vec2);
432 }
433 throw BondUnknownTypeError();

436inline bool add_bonded_three_body_force(Bonded_IA_Parameters const &iaparams,
437 BoxGeometry const &box_geo,
438 Particle &p1, Particle &p2,
439 Particle &p3) {
440 if (std::get_if<OifGlobalForcesBond>(&iaparams)) {

442 }
443 auto const result =

446 auto const &forces = result.value();
447
448 p1.force() += std::get<0>(forces);
449 p2.force() += std::get<1>(forces);
450 p3.force() += std::get<2>(forces);
451
452 return false;
453 }
454 return true;

457inline std::optional<std::tuple<Utils::Vector3d, Utils::Vector3d,
458 Utils::Vector3d, Utils::Vector3d>>
459calc_bonded_four_body_force(Bonded_IA_Parameters const &iaparams,
460 BoxGeometry const &box_geo, Particle const &p1,
461 Particle const &p2, Particle const &p3,
462 Particle const &p4) {
463 if (auto const *iap = std::get_if<OifLocalForcesBond>(&iaparams)) {
464 return iap->calc_forces(box_geo, p1, p2, p3, p4);
465 }
466 if (auto const *iap = std::get_if<IBMTribend>(&iaparams)) {
467 return iap->calc_forces(box_geo, p1, p2, p3, p4);
468 }
469 // note: particles in a dihedral bond are ordered as p2-p1-p3-p4
470 auto const v12 = box_geo.get_mi_vector(p1.pos(), p2.pos());
471 auto const v23 = box_geo.get_mi_vector(p3.pos(), p1.pos());
472 auto const v34 = box_geo.get_mi_vector(p4.pos(), p3.pos());
473 if (auto const *iap = std::get_if<DihedralBond>(&iaparams)) {
474 return iap->forces(v12, v23, v34);
475 }
476#ifdef ESPRESSO_TABULATED
477 if (auto const *iap = std::get_if<TabulatedDihedralBond>(&iaparams)) {
478 return iap->forces(v12, v23, v34);
479 }
480#endif
481 throw BondUnknownTypeError();
482}

504inline bool
505add_bonded_force(Particle &p1, int bond_id, std::span<Particle *> partners,
506 BondedInteractionsMap const &bonded_ia_params,
507 BondBreakage::BondBreakage &bond_breakage,
508 BoxGeometry const &box_geo,
509 [[maybe_unused]] Utils::Vector3d *const virial,
510 Coulomb::ShortRangeForceKernel::kernel_type const *kernel) {
511
512 // Consider for bond breakage
513 if (partners.size() == 1u) { // pair bonds
514 auto d = box_geo.get_mi_vector(p1.pos(), partners[0]->pos()).norm();
515 if (bond_breakage.check_and_handle_breakage(
516 p1.id(), {{partners[0]->id(), std::nullopt}}, bond_id, d)) {
517 return false;
518 }
519 }
520 if (partners.size() == 2u) { // angle bond
521 auto d =
522 box_geo.get_mi_vector(partners[0]->pos(), partners[1]->pos()).norm();
523 if (bond_breakage.check_and_handle_breakage(
524 p1.id(), {{partners[0]->id(), partners[1]->id()}}, bond_id, d)) {
525 return false;
526 }
527 }
528
529 auto const &iaparams = *bonded_ia_params.at(bond_id);
530
531 switch (number_of_partners(iaparams)) {
532 case 0:
533 return false;
534 case 1:
535 return add_bonded_two_body_force(iaparams, box_geo, p1, *partners[0],
536 virial, kernel);
537 case 2:
538 return add_bonded_three_body_force(iaparams, box_geo, p1, *partners[0],
539 *partners[1]);
540 case 3:
541 return add_bonded_four_body_force(iaparams, box_geo, p1, *partners[0],
542 *partners[1], *partners[2]);
543 default:
544 throw BondInvalidSizeError{number_of_partners(iaparams)};
545 }
