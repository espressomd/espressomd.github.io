80 inline ParticleForce calc_central_radial_force(Particle const &p1,
81  Particle const &p2,
82  IA_parameters const &ia_params,
83  Utils::Vector3d const &d,
84  double const dist) {
85 
86  ParticleForce pf{};
87  double force_factor = 0;
88 /* Lennard-Jones */
89 #ifdef LENNARD_JONES
90  force_factor += lj_pair_force_factor(ia_params, dist);
91 #endif
92 /* WCA */
93 #ifdef WCA
94  force_factor += wca_pair_force_factor(ia_params, dist);
95 #endif
96 /* Lennard-Jones generic */
97 #ifdef LENNARD_JONES_GENERIC
98  force_factor += ljgen_pair_force_factor(ia_params, dist);
99 #endif
100 /* smooth step */
101 #ifdef SMOOTH_STEP
102  force_factor += SmSt_pair_force_factor(ia_params, dist);
103 #endif
104 /* Hertzian force */
105 #ifdef HERTZIAN
106  force_factor += hertzian_pair_force_factor(ia_params, dist);
107 #endif
108 /* Gaussian force */
109 #ifdef GAUSSIAN
110  force_factor += gaussian_pair_force_factor(ia_params, dist);
111 #endif
112 /* BMHTF NaCl */
113 #ifdef BMHTF_NACL
114  force_factor += BMHTF_pair_force_factor(ia_params, dist);
115 #endif
116 /* Buckingham*/
117 #ifdef BUCKINGHAM
118  force_factor += buck_pair_force_factor(ia_params, dist);
119 #endif
120 /* Morse*/
121 #ifdef MORSE
122  force_factor += morse_pair_force_factor(ia_params, dist);
123 #endif
124 /*soft-sphere potential*/
125 #ifdef SOFT_SPHERE
126  force_factor += soft_pair_force_factor(ia_params, dist);
127 #endif
128 /*hat potential*/
129 #ifdef HAT
130  force_factor += hat_pair_force_factor(ia_params, dist);
131 #endif
132 /* Lennard-Jones cosine */
133 #ifdef LJCOS
134  force_factor += ljcos_pair_force_factor(ia_params, dist);
135 #endif
136 /* Lennard-Jones cosine */
137 #ifdef LJCOS2
138  force_factor += ljcos2_pair_force_factor(ia_params, dist);
139 #endif
140 /* tabulated */
141 #ifdef TABULATED
142  force_factor += tabulated_pair_force_factor(ia_params, dist);
143 #endif
144  pf.f += force_factor * d;
145  return pf;
146 }

176 inline ParticleForce calc_opposing_force(ParticleForce const &pf,
177  Utils::Vector3d const &d) {
178  ParticleForce out{-pf.f};
179 #ifdef ROTATION
180  // if torque is a null vector, the opposing torque is a null vector too
181  // (this check guards from returning a small yet non-null opposing
182  // torque due to numerical imprecision)
183  if (pf.torque[0] != 0. || pf.torque[1] != 0. || pf.torque[2] != 0.) {
184  out.torque = -(pf.torque + vector_product(d, pf.f));
185  }
186 #endif
187  return out;
188 }

190 /** Calculate non-bonded forces between a pair of particles and update their
191  * forces and torques.
192  * @param[in,out] p1 particle 1.
193  * @param[in,out] p2 particle 2.
194  * @param[in] d vector between @p p1 and @p p2.
195  * @param[in] dist distance between @p p1 and @p p2.
196  * @param[in] dist2 distance squared between @p p1 and @p p2.
197  * @param[in] ia_params non-bonded interaction kernels.
198  * @param[in] thermostat thermostat.
199  * @param[in] box_geo box geometry.
200  * @param[in] coulomb_kernel Coulomb force kernel.
201  * @param[in] dipoles_kernel Dipolar force kernel.
202  * @param[in] elc_kernel ELC force correction kernel.
203  */
204 inline void add_non_bonded_pair_force(
205  Particle &p1, Particle &p2, Utils::Vector3d const &d, double dist,
206  double dist2, IA_parameters const &ia_params,
207  Thermostat::Thermostat const &thermostat, BoxGeometry const &box_geo,
208  Coulomb::ShortRangeForceKernel::kernel_type const *coulomb_kernel,
209  Dipoles::ShortRangeForceKernel::kernel_type const *dipoles_kernel,
210  Coulomb::ShortRangeForceCorrectionsKernel::kernel_type const *elc_kernel) {
211 
212  ParticleForce pf{};
213 
214  /***********************************************/
215  /* non-bonded pair potentials */
216  /***********************************************/
217 
218  if (dist < ia_params.max_cut) {
219 #ifdef EXCLUSIONS
220  if (do_nonbonded(p1, p2)) {
221 #endif
222  pf += calc_central_radial_force(p1, p2, ia_params, d, dist);
223  pf += calc_central_radial_charge_force(p1, p2, ia_params, d, dist,
224  coulomb_kernel);
225  pf += calc_non_central_force(p1, p2, ia_params, d, dist);
226 #ifdef EXCLUSIONS
227  }
228 #endif
229  }
230 
231  /***********************************************/
232  /* short-range electrostatics */
233  /***********************************************/
234 
235 #ifdef ELECTROSTATICS
236  // real-space electrostatic charge-charge interaction
237  auto const q1q2 = p1.q() * p2.q();
238  if (q1q2 != 0. and coulomb_kernel != nullptr) {
239  pf.f += (*coulomb_kernel)(q1q2, d, dist);
240 #ifdef P3M
241  if (elc_kernel)
242  (*elc_kernel)(p1, p2, q1q2);
243 #endif // P3M
244  }
245 #endif // ELECTROSTATICS
246 
247  /*********************************************************************/
248  /* everything before this contributes to the virial pressure in NpT, */
249  /* but nothing afterwards */
250  /*********************************************************************/
251 #ifdef NPT
252  npt_add_virial_force_contribution(pf.f, d);
253 #endif
254 
255  /***********************************************/
256  /* thermostat */
257  /***********************************************/
258 
259  /* The inter dpd force should not be part of the virial */
260 #ifdef DPD
261  if (thermostat.thermo_switch & THERMO_DPD) {
262  auto const force = dpd_pair_force(p1, p2, *thermostat.dpd, box_geo,
263  ia_params, d, dist, dist2);
264  p1.force() += force;
265  p2.force() -= force;
266  }
267 #endif
268 
269  /***********************************************/
270  /* short-range magnetostatics */
271  /***********************************************/
272 
273 #ifdef DIPOLES
274  // real-space magnetic dipole-dipole
275  if (dipoles_kernel) {
276  pf += (*dipoles_kernel)(p1, p2, d, dist, dist2);
277  }
278 #endif
279 
280  /***********************************************/
281  /* add total non-bonded forces to particles */
282  /***********************************************/
283 
284  p1.force_and_torque() += pf;
285  p2.force_and_torque() += calc_opposing_force(pf, d);
286 }

288 /** Compute the bonded interaction force between particle pairs.
289  *
290  * @param[in] p1 First particle.
291  * @param[in] p2 Second particle.
292  * @param[in] iaparams Bonded parameters for the interaction.
293  * @param[in] dx Vector between @p p1 and @p p2.
294  * @param[in] kernel Coulomb force kernel.
295  */
296 inline boost::optional<Utils::Vector3d> calc_bond_pair_force(
297  Particle const &p1, Particle const &p2,
298  Bonded_IA_Parameters const &iaparams, Utils::Vector3d const &dx,
299  Coulomb::ShortRangeForceKernel::kernel_type const *kernel) {
300  if (auto const *iap = boost::get<FeneBond>(&iaparams)) {
301  return iap->force(dx);
302  }
303  if (auto const *iap = boost::get<HarmonicBond>(&iaparams)) {
304  return iap->force(dx);
305  }
306  if (auto const *iap = boost::get<QuarticBond>(&iaparams)) {
307  return iap->force(dx);
308  }
309 #ifdef ELECTROSTATICS
310  if (auto const *iap = boost::get<BondedCoulomb>(&iaparams)) {
311  return iap->force(p1.q() * p2.q(), dx);
312  }
313  if (auto const *iap = boost::get<BondedCoulombSR>(&iaparams)) {
314  return iap->force(dx, *kernel);
315  }
316 #endif
317 #ifdef BOND_CONSTRAINT
318  if (boost::get<RigidBond>(&iaparams)) {
319  return Utils::Vector3d{};
320  }
321 #endif
322 #ifdef TABULATED
323  if (auto const *iap = boost::get<TabulatedDistanceBond>(&iaparams)) {
324  return iap->force(dx);
325  }
326 #endif
327  if (boost::get<VirtualBond>(&iaparams)) {
328  return Utils::Vector3d{};
329  }
330  throw BondUnknownTypeError();
331 }

333 inline bool add_bonded_two_body_force(
334  Bonded_IA_Parameters const &iaparams, Particle &p1, Particle &p2,
335  BoxGeometry const &box_geo,
336  Coulomb::ShortRangeForceKernel::kernel_type const *kernel) {
337  auto const dx = box_geo.get_mi_vector(p1.pos(), p2.pos());
338 
339  if (auto const *iap = boost::get<ThermalizedBond>(&iaparams)) {
340  auto result = iap->forces(p1, p2, dx);
341  if (result) {
342  auto const &forces = result.get();
343 
344  p1.force() += std::get<0>(forces);
345  p2.force() += std::get<1>(forces);
346 
347  return false;
348  }
349  } else {
350  auto result = calc_bond_pair_force(p1, p2, iaparams, dx, kernel);
351  if (result) {
352  p1.force() += result.get();
353  p2.force() -= result.get();
354 
355 #ifdef NPT
356  npt_add_virial_force_contribution(result.get(), dx);
357 #endif
358  return false;
359  }
360  }
361  return true;
362 }

459 inline bool
460 add_bonded_force(Particle &p1, int bond_id, Utils::Span<Particle *> partners,
461  BondBreakage::BondBreakage &bond_breakage,
462  BoxGeometry const &box_geo,
463  Coulomb::ShortRangeForceKernel::kernel_type const *kernel) {
464 
465  // Consider for bond breakage
466  if (partners.size() == 1) { // pair bonds
467  auto d = box_geo.get_mi_vector(p1.pos(), partners[0]->pos()).norm();
468  if (bond_breakage.check_and_handle_breakage(
469  p1.id(), {{partners[0]->id(), std::nullopt}}, bond_id, d)) {
470  return false;
471  }
472  }
473  if (partners.size() == 2) { // angle bond
474  auto d =
475  box_geo.get_mi_vector(partners[0]->pos(), partners[1]->pos()).norm();
476  if (bond_breakage.check_and_handle_breakage(
477  p1.id(), {{partners[0]->id(), partners[1]->id()}}, bond_id, d)) {
478  return false;
479  }
480  }
481 
482  auto const &iaparams = *bonded_ia_params.at(bond_id);
483 
484  switch (number_of_partners(iaparams)) {
485  case 0:
486  return false;
487  case 1:
488  return add_bonded_two_body_force(iaparams, p1, *partners[0], box_geo,
489  kernel);
490  case 2:
491  return add_bonded_three_body_force(iaparams, box_geo, p1, *partners[0],
492  *partners[1]);
493  case 3:
494  return add_bonded_four_body_force(iaparams, box_geo, p1, *partners[0],
495  *partners[1], *partners[2]);
496  default:
497  throw BondInvalidSizeError{number_of_partners(iaparams)};
498  }
