from .script_interface import script_interface_register, ScriptInterfaceHelper


@script_interface_register
class _Version(ScriptInterfaceHelper):
    _so_name = "CodeInfo::CodeVersion"
    _so_creation_policy = "LOCAL"
    _so_bind_methods = (
        "version",
        "version_major",
        "version_minor",
        "version_friendly",
        "git_branch",
        "git_commit",
        "git_state",
    )

def version():
    """Get the full version of ESPResSo as a tuple of integers."""
    return tuple(_Version().version())

def git_state():
    """
    Git state of the build if known, otherwise empty string. State is "CLEAN" if
    the repository was not changed from :func:`git_commit()`, "DIRTY" otherwise.
    """
    return _Version().git_state()
